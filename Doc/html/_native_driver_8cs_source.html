189  public void Open()
190  {
191  // connect to one of our specified hosts
192  try
193  {
194 #if !CF
195  if (Settings.ConnectionProtocol == MySqlConnectionProtocol.SharedMemory)
196  {
197  SharedMemoryStream str = new SharedMemoryStream(Settings.SharedMemoryName);
198  str.Open(Settings.ConnectionTimeout);
199  baseStream = str;
200  }
201  else
202  {
203 #endif
204  string pipeName = Settings.PipeName;
205  if (Settings.ConnectionProtocol != MySqlConnectionProtocol.NamedPipe)
206  pipeName = null;
207  StreamCreator sc = new StreamCreator(Settings.Server, Settings.Port, pipeName,
208  Settings.Keepalive, this.Version);
209 #if !CF
210  if (Settings.IncludeSecurityAsserts)
211  MySqlSecurityPermission.CreatePermissionSet(false).Assert();
212 #endif
213  baseStream = sc.GetStream(Settings.ConnectionTimeout);
214 #if !CF
215  }
216 #endif
217  }
218  catch (System.Security.SecurityException)
219  {
220  throw;
221  }
222  catch (Exception ex)
223  {
224  throw new MySqlException(Resources.UnableToConnectToHost,
225  (int)MySqlErrorCode.UnableToConnectToHost, ex);
226  }
227  if (baseStream == null)
228  throw new MySqlException(Resources.UnableToConnectToHost,
229  (int)MySqlErrorCode.UnableToConnectToHost);
230 
231  int maxSinglePacket = 255 * 255 * 255;
232  stream = new MySqlStream(baseStream, Encoding, false);
233 
234  stream.ResetTimeout((int)Settings.ConnectionTimeout * 1000);
235 
236  // read off the welcome packet and parse out it's values
237  packet = stream.ReadPacket();
238  int protocol = packet.ReadByte();
239  string versionString = packet.ReadString();
240  version = DBVersion.Parse(versionString);
241  if (!version.isAtLeast(5, 0, 0))
242  throw new NotSupportedException(Resources.ServerTooOld);
243  threadId = packet.ReadInteger(4);
244  encryptionSeed = packet.ReadString();
245 
246  maxSinglePacket = (256 * 256 * 256) - 1;
247 
248  // read in Server capabilities if they are provided
249  ClientFlags serverCaps = 0;
250  if (packet.HasMoreData)
251  serverCaps = (ClientFlags)packet.ReadInteger(2);
252 
253  /* New protocol with 16 bytes to describe server characteristics */
254  owner.ConnectionCharSetIndex = (int)packet.ReadByte();
255 
256  serverStatus = (ServerStatusFlags)packet.ReadInteger(2);
257 
258  // Since 5.5, high bits of server caps are stored after status.
259  // Previously, it was part of reserved always 0x00 13-byte filler.
260  uint serverCapsHigh = (uint)packet.ReadInteger(2);
261  serverCaps |= (ClientFlags)(serverCapsHigh << 16);
262 
263  packet.Position += 11;
264  string seedPart2 = packet.ReadString();
265  encryptionSeed += seedPart2;
266 
267  string authenticationMethod = "";
268  if ((serverCaps & ClientFlags.PLUGIN_AUTH) != 0)
269  {
270  authenticationMethod = packet.ReadString();
271  }
272  else
273  {
274  // Some MySql versions like 5.1, don't give name of plugin, default to native password.
275  authenticationMethod = "mysql_native_password";
276  }
277 
278  // based on our settings, set our connection flags
279  SetConnectionFlags(serverCaps);
280 
281  packet.Clear();
282  packet.WriteInteger((int)connectionFlags, 4);
283 
284 #if !CF
285  if ((serverCaps & ClientFlags.SSL) == 0)
286  {
287  if ((Settings.SslMode != MySqlSslMode.None)
288  && (Settings.SslMode != MySqlSslMode.Preferred))
289  {
290  // Client requires SSL connections.
291  string message = String.Format(Resources.NoServerSSLSupport,
292  Settings.Server);
293  throw new MySqlException(message);
294  }
295  }
296  else if (Settings.SslMode != MySqlSslMode.None)
297  {
298  stream.SendPacket(packet);
299  StartSSL();
300  packet.Clear();
301  packet.WriteInteger((int)connectionFlags, 4);
302  }
303 #endif
304 
305  packet.WriteInteger(maxSinglePacket, 4);
306  packet.WriteByte(8);
307  packet.Write(new byte[23]);
308 
309  Authenticate(authenticationMethod, false);
310 
311  // if we are using compression, then we use our CompressedStream class
312  // to hide the ugliness of managing the compression
313  if ((connectionFlags & ClientFlags.COMPRESS) != 0)
314  stream = new MySqlStream(baseStream, Encoding, true);
315 
316  // give our stream the server version we are connected to.
317  // We may have some fields that are read differently based
318  // on the version of the server we are connected to.
319  packet.Version = version;
320  stream.MaxBlockSize = maxSinglePacket;
321  }

331  private X509CertificateCollection GetClientCertificates()
332  {
333  X509CertificateCollection certs = new X509CertificateCollection();
334 
335  // Check for file-based certificate
336  if (Settings.CertificateFile != null)
337  {
338  if (!Version.isAtLeast(5, 1, 0))
339  throw new MySqlException(Properties.Resources.FileBasedCertificateNotSupported);
340 
341  X509Certificate2 clientCert = new X509Certificate2(Settings.CertificateFile,
342  Settings.CertificatePassword);
343  certs.Add(clientCert);
344  return certs;
345  }
346 
347  if (Settings.CertificateStoreLocation == MySqlCertificateStoreLocation.None)
348  return certs;
349 
350  StoreLocation location =
351  (Settings.CertificateStoreLocation == MySqlCertificateStoreLocation.CurrentUser) ?
352  StoreLocation.CurrentUser : StoreLocation.LocalMachine;
353 
354  // Check for store-based certificate
355  X509Store store = new X509Store(StoreName.My, location);
356  store.Open(OpenFlags.ReadOnly | OpenFlags.OpenExistingOnly);
357 
358 
359  if (Settings.CertificateThumbprint == null)
360  {
361  // Return all certificates from the store.
362  certs.AddRange(store.Certificates);
363  return certs;
364  }
365 
366  // Find certificate with given thumbprint
367  certs.AddRange(store.Certificates.Find(X509FindType.FindByThumbprint,
368  Settings.CertificateThumbprint, true));
369 
370  if (certs.Count == 0)
371  {
372  throw new MySqlException("Certificate with Thumbprint " +
373  Settings.CertificateThumbprint + " not found");
374  }
375  return certs;
376  }

392  private bool ServerCheckValidation(object sender, X509Certificate certificate,
393  X509Chain chain, SslPolicyErrors sslPolicyErrors)
394  {
395  if (sslPolicyErrors == SslPolicyErrors.None)
396  return true;
397 
398  if (Settings.SslMode == MySqlSslMode.Preferred ||
399  Settings.SslMode == MySqlSslMode.Required)
400  {
401  //Tolerate all certificate errors.
402  return true;
403  }
404 
405  if (Settings.SslMode == MySqlSslMode.VerifyCA &&
406  sslPolicyErrors == SslPolicyErrors.RemoteCertificateNameMismatch)

410  }
411 
412  return false;

426  private void SetConnectionFlags(ClientFlags serverCaps)
427  {
428  // allow load data local infile
429  ClientFlags flags = ClientFlags.LOCAL_FILES;
430 
431  if (!Settings.UseAffectedRows)
432  flags |= ClientFlags.FOUND_ROWS;
433 
434  flags |= ClientFlags.PROTOCOL_41;
435  // Need this to get server status values
436  flags |= ClientFlags.TRANSACTIONS;
437 
438  // user allows/disallows batch statements
439  if (Settings.AllowBatch)
440  flags |= ClientFlags.MULTI_STATEMENTS;
441 
442  // We always allow multiple result sets
443  flags |= ClientFlags.MULTI_RESULTS;
444 
445  // if the server allows it, tell it that we want long column info
446  if ((serverCaps & ClientFlags.LONG_FLAG) != 0)
447  flags |= ClientFlags.LONG_FLAG;
448 
449  // if the server supports it and it was requested, then turn on compression
450  if ((serverCaps & ClientFlags.COMPRESS) != 0 && Settings.UseCompression)
451  flags |= ClientFlags.COMPRESS;
452 
453  flags |= ClientFlags.LONG_PASSWORD; // for long passwords
454 
455  // did the user request an interactive session?
456  if (Settings.InteractiveSession)
457  flags |= ClientFlags.INTERACTIVE;
458 
459  // if the server allows it and a database was specified, then indicate
460  // that we will connect with a database name
461  if ((serverCaps & ClientFlags.CONNECT_WITH_DB) != 0 &&
462  Settings.Database != null && Settings.Database.Length > 0)
463  flags |= ClientFlags.CONNECT_WITH_DB;
464 
465  // if the server is requesting a secure connection, then we oblige
466  if ((serverCaps & ClientFlags.SECURE_CONNECTION) != 0)
467  flags |= ClientFlags.SECURE_CONNECTION;
468 
469 #if !CF
470  // if the server is capable of SSL and the user is requesting SSL
471  if ((serverCaps & ClientFlags.SSL) != 0 && Settings.SslMode != MySqlSslMode.None)
472  flags |= ClientFlags.SSL;
473 #endif
474 
475  // if the server supports output parameters, then we do too
476  if ((serverCaps & ClientFlags.PS_MULTI_RESULTS) != 0)
477  flags |= ClientFlags.PS_MULTI_RESULTS;
478 
479  if ((serverCaps & ClientFlags.PLUGIN_AUTH) != 0)
480  flags |= ClientFlags.PLUGIN_AUTH;
481 
482  // if the server supports connection attributes
483  if ((serverCaps & ClientFlags.CONNECT_ATTRS) != 0)
484  flags |= ClientFlags.CONNECT_ATTRS;
485 
486  if ((serverCaps & ClientFlags.CAN_HANDLE_EXPIRED_PASSWORD) != 0)
487  flags |= ClientFlags.CAN_HANDLE_EXPIRED_PASSWORD;
488 
489  connectionFlags = flags;
490  }

522  public void SendQuery(MySqlPacket queryPacket)
523  {
524  warnings = 0;
525  queryPacket.Buffer[4] = (byte)DBCmd.QUERY;
526  ExecutePacket(queryPacket);
527  // the server will respond in one of several ways with the first byte indicating
528  // the type of response.
529  // 0 == ok packet. This indicates non-select queries
530  // 0xff == error packet. This is handled in stream.OpenPacket
531  // > 0 = number of columns in select query
532  // We don't actually read the result here since a single query can generate
533  // multiple resultsets and we don't want to duplicate code. See ReadResult
534  // Instead we set our internal server status flag to indicate that we have a query waiting.
535  // This flag will be maintained by ReadResult
536  serverStatus |= ServerStatusFlags.AnotherQuery;
537  }

636  {

639  long len = 0;
