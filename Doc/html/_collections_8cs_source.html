102  public static void ProcessReachableCollection(IPersistentCollection collection, CollectionType type, object entity, ISessionImplementor session)
103  {
104  collection.Owner = entity;
105  CollectionEntry ce = session.PersistenceContext.GetCollectionEntry(collection);
106 
107  if (ce == null)
108  {
109  // refer to comment in StatefulPersistenceContext.addCollection()
110  throw new HibernateException(string.Format("Found two representations of same collection: {0}", type.Role));
111  }
112 
113  // The CollectionEntry.isReached() stuff is just to detect any silly users
114  // who set up circular or shared references between/to collections.
115  if (ce.IsReached)
116  {
117  // We've been here before
118  throw new HibernateException(string.Format("Found shared references to a collection: {0}", type.Role));
119  }
120  ce.IsReached = true;
121 
122  ISessionFactoryImplementor factory = session.Factory;
123  ICollectionPersister persister = factory.GetCollectionPersister(type.Role);
124  ce.CurrentPersister = persister;
125  ce.CurrentKey = type.GetKeyOfOwner(entity, session); //TODO: better to pass the id in as an argument?
126 
127  if (log.IsDebugEnabled)
128  {
129  log.Debug("Collection found: " +
130  MessageHelper.InfoString(persister, ce.CurrentKey, factory) + ", was: " +
131  MessageHelper.InfoString(ce.LoadedPersister, ce.LoadedKey, factory) +
132  (collection.WasInitialized ? " (initialized)" : " (uninitialized)"));
133  }
134 
135  PrepareCollectionForUpdate(collection, ce, session.EntityMode, factory);
136  }

138  private static void PrepareCollectionForUpdate(IPersistentCollection collection, CollectionEntry entry, EntityMode entityMode, ISessionFactoryImplementor factory)
139  {
140  //1. record the collection role that this collection is referenced by
141  //2. decide if the collection needs deleting/creating/updating (but don't actually schedule the action yet)
142 
143  if (entry.IsProcessed)
144  throw new AssertionFailure("collection was processed twice by flush()");
145 
146  entry.IsProcessed = true;
147 
148  ICollectionPersister loadedPersister = entry.LoadedPersister;
149  ICollectionPersister currentPersister = entry.CurrentPersister;
150  if (loadedPersister != null || currentPersister != null)
151  {
152  // it is or was referenced _somewhere_
153  bool ownerChanged = loadedPersister != currentPersister ||
154  !currentPersister.KeyType.IsEqual(entry.LoadedKey, entry.CurrentKey, entityMode, factory);
155 
156  if (ownerChanged)
157  {
158  // do a check
159  bool orphanDeleteAndRoleChanged = loadedPersister != null &&
160  currentPersister != null && loadedPersister.HasOrphanDelete;
161 
162  if (orphanDeleteAndRoleChanged)
163  {
164  throw new HibernateException("Don't change the reference to a collection with cascade=\"all-delete-orphan\": " + loadedPersister.Role);
165  }
166 
167  // do the work
168  if (currentPersister != null)
169  {
170  entry.IsDorecreate = true; // we will need to create new entries
171  }
172 
173  if (loadedPersister != null)
174  {
175  entry.IsDoremove = true; // we will need to remove ye olde entries
176  if (entry.IsDorecreate)
177  {
178  log.Debug("Forcing collection initialization");
179  collection.ForceInitialization(); // force initialize!
180  }
181  }
182  }
183  else if (collection.IsDirty)
184  {
185  // else if it's elements changed
186  entry.IsDoupdate = true;
187  }
188  }
189  }
