77  public LocationInfo(Type callerStackBoundaryDeclaringType)
78  {
79  // Initialize all fields
80  m_className = NA;
81  m_fileName = NA;
82  m_lineNumber = NA;
83  m_methodName = NA;
84  m_fullInfo = NA;
85 
86 #if !NETCF
87  if (callerStackBoundaryDeclaringType != null)
88  {
89  try
90  {
91  StackTrace st = new StackTrace(true);
92  int frameIndex = 0;
93 
94  // skip frames not from fqnOfCallingClass
95  while (frameIndex < st.FrameCount)
96  {
97  StackFrame frame = st.GetFrame(frameIndex);
98  if (frame != null && frame.GetMethod().DeclaringType == callerStackBoundaryDeclaringType)
99  {
100  break;
101  }
102  frameIndex++;
103  }
104 
105  // skip frames from fqnOfCallingClass
106  while (frameIndex < st.FrameCount)
107  {
108  StackFrame frame = st.GetFrame(frameIndex);
109  if (frame != null && frame.GetMethod().DeclaringType != callerStackBoundaryDeclaringType)
110  {
111  break;
112  }
113  frameIndex++;
114  }
115 
116  if (frameIndex < st.FrameCount)
117  {
118  // take into account the frames we skip above
119  int adjustedFrameCount = st.FrameCount - frameIndex;
120  ArrayList stackFramesList = new ArrayList(adjustedFrameCount);
121  m_stackFrames = new StackFrame[adjustedFrameCount];
122  for (int i=frameIndex; i < st.FrameCount; i++)
123  {
124  stackFramesList.Add(st.GetFrame(i));
125  }
126 
127  stackFramesList.CopyTo(m_stackFrames, 0);
128 
129  // now frameIndex is the first 'user' caller frame
130  StackFrame locationFrame = st.GetFrame(frameIndex);
131 
132  if (locationFrame != null)
133  {
134  System.Reflection.MethodBase method = locationFrame.GetMethod();
135 
136  if (method != null)
137  {
138  m_methodName = method.Name;
139  if (method.DeclaringType != null)
140  {
141  m_className = method.DeclaringType.FullName;
142  }
143  }
144  m_fileName = locationFrame.GetFileName();
145  m_lineNumber = locationFrame.GetFileLineNumber().ToString(System.Globalization.NumberFormatInfo.InvariantInfo);
146 
147  // Combine all location info
148  m_fullInfo = m_className + '.' + m_methodName + '(' + m_fileName + ':' + m_lineNumber + ')';
149  }
150  }
151  }
152  catch(System.Security.SecurityException)
153  {
154  // This security exception will occur if the caller does not have
155  // some undefined set of SecurityPermission flags.
156  LogLog.Debug(declaringType, "Security exception while trying to get caller stack frame. Error Ignored. Location Information Not Available.");
157  }
158  }
159 #endif
160  }
