383  public override bool ChangePassword(string username, string oldPassword, string newPassword)
384  {
385  // this will return false if the username doesn't exist
386  if (!(ValidateUser(username, oldPassword)))
387  return false;
388 
389  ValidatePasswordEventArgs args = new ValidatePasswordEventArgs(username, newPassword, true);
390  OnValidatingPassword(args);
391  if (args.Cancel)
392  {
393  if (!(args.FailureInformation == null))
394  throw args.FailureInformation;
395  else
396  throw new ProviderException(Resources.ChangePasswordCanceled);
397  }
398 
399  // validate the password according to current guidelines
400  if (!ValidatePassword(newPassword, "newPassword", true))
401  return false;
402 
403  try
404  {
405  using (MySqlConnection connection = new MySqlConnection(connectionString))
406  {
407  connection.Open();
408 
409  // retrieve the existing key and format for this user
410  string passwordKey;
411  MembershipPasswordFormat passwordFormat;
412  int userId = GetUserId(connection, username);
413 
414  GetPasswordInfo(connection, userId, out passwordKey, out passwordFormat);
415 
416  MySqlCommand cmd = new MySqlCommand(
417  @"UPDATE my_aspnet_membership
418  SET Password = @pass, LastPasswordChangedDate = @lastPasswordChangedDate
419  WHERE userId=@userId", connection);
420  cmd.Parameters.AddWithValue("@pass",
421  EncodePassword(newPassword, passwordKey, passwordFormat));
422  cmd.Parameters.AddWithValue("@lastPasswordChangedDate", DateTime.Now);
423  cmd.Parameters.AddWithValue("@userId", userId);
424  return cmd.ExecuteNonQuery() > 0;
425  }
426  }
427  catch (MySqlException e)
428  {

498  public override MembershipUser CreateUser(string username, string password,
499  string email, string passwordQuestion, string passwordAnswer,
500  bool isApproved, object providerUserKey, out MembershipCreateStatus status)
501  {
502  ValidatePasswordEventArgs Args = new ValidatePasswordEventArgs(username, password, true);
503  OnValidatingPassword(Args);
504  if (Args.Cancel)
505  {
506  status = MembershipCreateStatus.InvalidPassword;
507  return null;
508  }
509  if (RequiresUniqueEmail && !String.IsNullOrEmpty(GetUserNameByEmail(email)))
510  {
511  status = MembershipCreateStatus.DuplicateEmail;
512  return null;
513  }
514 
515  ValidateQA(passwordQuestion, passwordAnswer);
516 
517  // now try to validate the password
518  if (!ValidatePassword(password, "password", false))
519  {
520  status = MembershipCreateStatus.InvalidPassword;
521  return null;
522  }
523 
524  // now check to see if we already have a member by this name
525  MembershipUser u = GetUser(username, false);
526  if (u != null)
527  {
528  status = MembershipCreateStatus.DuplicateUserName;
529  return null;
530  }
531 
532  string passwordKey = GetPasswordKey();
533  DateTime createDate = DateTime.Now;
534  MySqlTransaction transaction = null;
535 
536  using (MySqlConnection connection = new MySqlConnection(connectionString))
537  {
538  try
539  {
540  connection.Open();
541  transaction = connection.BeginTransaction();
542 
543  // either create a new user or fetch the existing user id
544  long userId = SchemaManager.CreateOrFetchUserId(connection, username,
545  app.EnsureId(connection), true);
546 
547  MySqlCommand cmd = new MySqlCommand(
548  @"INSERT INTO my_aspnet_membership
549  VALUES(@userId, @email, @comment, @password, @passwordKey,
550  @passwordFormat, @passwordQuestion, @passwordAnswer,
551  @isApproved, @lastActivityDate, @lastLoginDate,
552  @lastPasswordChangedDate, @creationDate,
553  @isLockedOut, @lastLockedOutDate, @failedPasswordAttemptCount,
554  @failedPasswordAttemptWindowStart, @failedPasswordAnswerAttemptCount,
555  @failedPasswordAnswerAttemptWindowStart)",
556  connection);
557  cmd.Parameters.AddWithValue("@userId", userId);
558  cmd.Parameters.AddWithValue("@email", email);
559  cmd.Parameters.AddWithValue("@comment", "");
560  cmd.Parameters.AddWithValue("@password",
561  EncodePassword(password, passwordKey, PasswordFormat));
562  cmd.Parameters.AddWithValue("@passwordKey", passwordKey);
563  cmd.Parameters.AddWithValue("@passwordFormat", PasswordFormat);
564  cmd.Parameters.AddWithValue("@passwordQuestion", passwordQuestion);
565  cmd.Parameters.AddWithValue("@passwordAnswer",
566  EncodePassword(passwordAnswer, passwordKey, PasswordFormat));
567  cmd.Parameters.AddWithValue("@isApproved", isApproved);
568  cmd.Parameters.AddWithValue("@lastActivityDate", createDate);
569  cmd.Parameters.AddWithValue("@lastLoginDate", createDate);
570  cmd.Parameters.AddWithValue("@lastPasswordChangedDate", createDate);
571  cmd.Parameters.AddWithValue("@creationDate", createDate);
572  cmd.Parameters.AddWithValue("@isLockedOut", false);
573  cmd.Parameters.AddWithValue("@lastLockedOutDate", createDate);
574  cmd.Parameters.AddWithValue("@failedPasswordAttemptCount", 0);
575  cmd.Parameters.AddWithValue("@failedPasswordAttemptWindowStart", createDate);
576  cmd.Parameters.AddWithValue("@failedPasswordAnswerAttemptCount", 0);
577  cmd.Parameters.AddWithValue("@failedPasswordAnswerAttemptWindowStart", createDate);
578 
579  int recAdded = cmd.ExecuteNonQuery();
580  if (recAdded > 0)
581  status = MembershipCreateStatus.Success;
582  else
583  status = MembershipCreateStatus.UserRejected;
584  transaction.Commit();
585  }
586  catch (MySqlException e)
587  {
588  if (WriteExceptionsToEventLog)
589  WriteToEventLog(e, "CreateUser");
590  status = MembershipCreateStatus.ProviderError;
591  if (transaction != null)
592  transaction.Rollback();
593  return null;
594  }
595  }
596 
597  return GetUser(username, false);

701  public override string GetPassword(string username, string answer)
702  {
703  if (!EnablePasswordRetrieval)
704  throw new ProviderException(Resources.PasswordRetrievalNotEnabled);
705 
706  try
707  {
708  using (MySqlConnection connection = new MySqlConnection(connectionString))
709  {
710  connection.Open();
711 
712  int userId = GetUserId(connection, username);
713  if (-1 == userId)
714  throw new ProviderException("Username not found.");
715 
716  string sql = @"SELECT Password, PasswordAnswer, PasswordKey, PasswordFormat,
717  IsLockedOut FROM my_aspnet_membership WHERE userId=@userId";
718  MySqlCommand cmd = new MySqlCommand(sql, connection);
719  cmd.Parameters.AddWithValue("@userId", userId);
720 
721  using (MySqlDataReader reader = cmd.ExecuteReader(CommandBehavior.SingleRow))
722  {
723  reader.Read();
724  if (reader.GetBoolean("IsLockedOut"))
725  throw new MembershipPasswordException(Resources.UserIsLockedOut);
726 
727  string password = reader.GetString("Password");
728  string passwordAnswer = reader.GetValue(reader.GetOrdinal("PasswordAnswer")).ToString();
729  string passwordKey = reader.GetString("PasswordKey");
730  MembershipPasswordFormat format = (MembershipPasswordFormat)reader.GetInt32(3);
731  reader.Close();
732 
733  if (RequiresQuestionAndAnswer &&
734  !(CheckPassword(answer, passwordAnswer, passwordKey, format)))
735  {
736  UpdateFailureCount(userId, "PasswordAnswer", connection);
737  throw new MembershipPasswordException(Resources.IncorrectPasswordAnswer);
738  }
739  if (PasswordFormat == MembershipPasswordFormat.Encrypted)
740  {
741  password = UnEncodePassword(password, format);
742  }
743  return password;
744  }
745  }
746  }
747  catch (MySqlException e)
748  {
749  if (WriteExceptionsToEventLog)
750  WriteToEventLog(e, "GetPassword");
751  throw new ProviderException(exceptionMessage, e);
752  }

918  {

921 
922  try

925  {
926  connection.Open();
927 
928  // fetch the userid first
929  int userId = GetUserId(connection, username);
930  if (-1 == userId)
931  throw new ProviderException(Resources.UsernameNotFound);
932 
933  if (answer == null && RequiresQuestionAndAnswer)
934  {
935  UpdateFailureCount(userId, "PasswordAnswer", connection);
936  throw new ProviderException(Resources.PasswordRequiredForReset);
937  }
938 
939  string newPassword = Membership.GeneratePassword(newPasswordLength, MinRequiredNonAlphanumericCharacters);
940  ValidatePasswordEventArgs Args = new ValidatePasswordEventArgs(username, newPassword, true);
941  OnValidatingPassword(Args);
942  if (Args.Cancel)
943  {
944  if (!(Args.FailureInformation == null))
945  throw Args.FailureInformation;
946  else
947  throw new MembershipPasswordException(Resources.PasswordResetCanceledNotValid);
948  }
949 
950  MySqlCommand cmd = new MySqlCommand(@"SELECT PasswordAnswer,
951  PasswordKey, PasswordFormat, IsLockedOut
952  FROM my_aspnet_membership WHERE userId=@userId", connection);
953  cmd.Parameters.AddWithValue("@userId", userId);
954 
955  string passwordKey = String.Empty;
956  MembershipPasswordFormat format;
957  using (MySqlDataReader reader = cmd.ExecuteReader(CommandBehavior.SingleRow))
958  {
959  reader.Read();
960  if (reader.GetBoolean("IsLockedOut"))
961  throw new MembershipPasswordException(Resources.UserIsLockedOut);
962 
963  object passwordAnswer = reader.GetValue(reader.GetOrdinal("PasswordAnswer"));
964  passwordKey = reader.GetString("PasswordKey");
965  format = (MembershipPasswordFormat)reader.GetByte("PasswordFormat");
966  reader.Close();
967
