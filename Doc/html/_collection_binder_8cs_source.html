142  private void BindCollection(ICollectionPropertiesMapping collectionMapping, Mapping.Collection model, string className,
143  string path, System.Type containingType, IDictionary<string, MetaAttribute> inheritedMetas)
144  {
145  // ROLENAME
146  model.Role = path;
147 
148  model.IsInverse = collectionMapping.Inverse;
149  model.IsMutable = collectionMapping.Mutable;
150  model.IsOptimisticLocked = collectionMapping.OptimisticLock;
151  model.OrderBy = collectionMapping.OrderBy;
152  model.Where = collectionMapping.Where;
153  if (collectionMapping.BatchSize.HasValue)
154  model.BatchSize = collectionMapping.BatchSize.Value;
155 
156  // PERSISTER
157  if (!string.IsNullOrEmpty(collectionMapping.PersisterQualifiedName))
158  {
159  model.CollectionPersisterClass = ClassForNameChecked(collectionMapping.PersisterQualifiedName, mappings,
160  "could not instantiate collection persister class: {0}");
161  }
162 
163  if(!string.IsNullOrEmpty(collectionMapping.CollectionType))
164  {
165  TypeDef typeDef = mappings.GetTypeDef(collectionMapping.CollectionType);
166  if (typeDef != null)
167  {
168  model.TypeName = typeDef.TypeClass;
169  model.TypeParameters = typeDef.Parameters;
170  }
171  else
172  {
173  model.TypeName = FullQualifiedClassName(collectionMapping.CollectionType, mappings);
174  }
175  }
176 
177  // FETCH STRATEGY
178  InitOuterJoinFetchSetting(collectionMapping, model);
179 
180  // LAZINESS
181  InitLaziness(collectionMapping, model);
182 
183  var oneToManyMapping = collectionMapping.ElementRelationship as HbmOneToMany;
184  if (oneToManyMapping != null)
185  {
186  var oneToMany = new OneToMany(model.Owner);
187  model.Element = oneToMany;
188  BindOneToMany(oneToManyMapping, oneToMany);
189  //we have to set up the table later!! yuck
190  }
191  else
192  {
193  //TABLE
194  string tableName = !string.IsNullOrEmpty(collectionMapping.Table)
195  ? mappings.NamingStrategy.TableName(collectionMapping.Table)
196  : mappings.NamingStrategy.PropertyToTableName(className, path);
197 
198  string schema = string.IsNullOrEmpty(collectionMapping.Schema) ? mappings.SchemaName : collectionMapping.Schema;
199  string catalog = string.IsNullOrEmpty(collectionMapping.Catalog) ? mappings.CatalogName : collectionMapping.Catalog;
200 
201  // TODO NH : add schema-action to the xsd
202  model.CollectionTable = mappings.AddTable(schema, catalog, tableName, collectionMapping.Subselect, false, "all");
203 
204  log.InfoFormat("Mapping collection: {0} -> {1}", model.Role, model.CollectionTable.Name);
205  }
206 
207  //SORT
208  var sortedAtt = collectionMapping.Sort;
209  // unsorted, natural, comparator.class.name
210  if (string.IsNullOrEmpty(sortedAtt) || sortedAtt.Equals("unsorted"))
211  model.IsSorted = false;
212  else
213  {
214  model.IsSorted = true;
215  if (!sortedAtt.Equals("natural"))
216  {
217  string comparatorClassName = FullQualifiedClassName(sortedAtt, mappings);
218  model.ComparerClassName = comparatorClassName;
219  }
220  }
221 
222  //ORPHAN DELETE (used for programmer error detection)
223  var cascadeAtt = collectionMapping.Cascade;
224  if (!string.IsNullOrEmpty(cascadeAtt) && cascadeAtt.IndexOf("delete-orphan") >= 0)
225  model.HasOrphanDelete = true;
226 
227  // GENERIC
228  bool? isGeneric = collectionMapping.Generic;
229  System.Type collectionType = null;
230  if (!isGeneric.HasValue && containingType != null)
231  {
232  collectionType = GetPropertyType(containingType, collectionMapping.Name, collectionMapping.Access);
233  isGeneric = collectionType.IsGenericType;
234  }
235 
236  model.IsGeneric = isGeneric.GetValueOrDefault();
237 
238  if (model.IsGeneric)
239  {
240  // Determine the generic arguments using reflection
241  if (collectionType == null)
242  collectionType = GetPropertyType(containingType, collectionMapping.Name, collectionMapping.Access);
243  System.Type[] genericArguments = collectionType.GetGenericArguments();
244  model.GenericArguments = genericArguments;
245  }
246 
247  // CUSTOM SQL
248  HandleCustomSQL(collectionMapping, model);
249  if (collectionMapping.SqlLoader != null)
250  model.LoaderName = collectionMapping.SqlLoader.queryref;
251 
252  new FiltersBinder(model, Mappings).Bind(collectionMapping.Filters);
253 
254  var key = collectionMapping.Key;
255  if (key != null)
256  model.ReferencedPropertyName = key.propertyref;
257  }

259  private void InitLaziness(ICollectionPropertiesMapping collectionMapping, Mapping.Collection fetchable)
260  {
261  var lazyMapping = collectionMapping.Lazy;
262  if(!lazyMapping.HasValue)
263  {
264  fetchable.IsLazy = mappings.DefaultLazy;
265  fetchable.ExtraLazy = false;
266  return;
267  }
268 
269  switch (lazyMapping.Value)
270  {
271  case HbmCollectionLazy.True:
272  fetchable.IsLazy = true;
273  break;
274  case HbmCollectionLazy.False:
275  fetchable.IsLazy = false;
276  fetchable.ExtraLazy = false;
277  break;
278  case HbmCollectionLazy.Extra:
279  fetchable.IsLazy = true;
280  fetchable.ExtraLazy = true;
281  break;
282  default:
283  throw new ArgumentOutOfRangeException();
284  }
285  }

287  private void InitOuterJoinFetchSetting(ICollectionPropertiesMapping collectionMapping, Mapping.Collection model)
288  {
289  FetchMode fetchStyle = FetchMode.Default;
290 
291  if (!collectionMapping.FetchMode.HasValue)
292  {
293  if (collectionMapping.OuterJoin.HasValue)
294  {
295  // use old (HB 2.1) defaults if outer-join is specified
296  switch (collectionMapping.OuterJoin.Value)
297  {
298  case HbmOuterJoinStrategy.Auto:
299  fetchStyle = FetchMode.Default;
300  break;
301  case HbmOuterJoinStrategy.True:
302  fetchStyle = FetchMode.Join;
303  break;
304  case HbmOuterJoinStrategy.False:
305  fetchStyle = FetchMode.Select;
306  break;
307  default:
308  throw new ArgumentOutOfRangeException();
309  }
310  }
311  }
312  else
313  {
314  switch (collectionMapping.FetchMode.Value)
315  {
316  case HbmCollectionFetchMode.Select:
317  fetchStyle = FetchMode.Select;
318  break;
319  case HbmCollectionFetchMode.Join:
320  fetchStyle = FetchMode.Join;
321  break;
322  case HbmCollectionFetchMode.Subselect:
323  fetchStyle = FetchMode.Select;
324  model.IsSubselectLoadable = true;
325  model.Owner.HasSubselectLoadableCollections = true;
326  break;
327  default:
328  throw new ArgumentOutOfRangeException();
329  }
330  }
331 
332  model.FetchMode = fetchStyle;
333  }

650  else if ((indexManyToManyMapping = mapMapping.Item as HbmIndexManyToMany) != null)
651  {
652  var manyToOne = new ManyToOne(model.CollectionTable);
653  BindIndexManyToMany(indexManyToManyMapping, manyToOne, IndexedCollection.DefaultIndexColumnName, model.IsOneToMany);
654  model.Index = manyToOne;
655  }
656  else if ((mapKeyManyToManyMapping = mapMapping.Item as HbmMapKeyManyToMany) != null)
657  {
658  var manyToOne = new ManyToOne(model.CollectionTable);
659  BindMapKeyManyToMany(mapKeyManyToManyMapping, manyToOne, IndexedCollection.DefaultIndexColumnName, model.IsOneToMany);
660  model.Index = manyToOne;
661  }
662  else if ((compositeIndexMapping = mapMapping.Item as HbmCompositeIndex) != null)
663  {
664  var component = new Component(model);
665  BindComponent(compositeIndexMapping, component, null, null, model.Role + ".index", model.IsOneToMany, inheritedMetas);
666  model.Index = component;
667  }
668  else if ((compositeMapKeyMapping = mapMapping.Item as HbmCompositeMapKey) != null)
669  {
670  var component = new Component(model);
671  BindComponent(compositeMapKeyMapping, component, null, null, model.Role + ".index", model.IsOneToMany, inheritedMetas);
672  model.Index = component;
673  }
674  else if ((indexManyToAnyMapping = mapMapping.Item as HbmIndexManyToAny) != null)
675  {
676  var any = new Any(model.CollectionTable);
677  BindIndexManyToAny(indexManyToAnyMapping, any, model.IsOneToMany);
678  model.Index = any;
679  }
680 
681  bool indexIsFormula = model.Index.ColumnIterator.Any(x=> x.IsFormula);
682  if (NeedBackref(model) && !indexIsFormula)
683  {
684  string entityName = ((OneToMany)model.Element).ReferencedEntityName;
685  PersistentClass referenced = mappings.GetClass(entityName);
686  var ib = new IndexBackref();
687  ib.Name = '_' + model.OwnerEntityName + "." + mapMapping.Name + "IndexBackref";
688  ib.IsUpdateable = false;
689  ib.IsSelectable = false;
690  ib.CollectionRole = model.Role;
691  ib.EntityName = model.Owner.EntityName;
692  ib.Value = model.Index;
693  referenced.AddProperty(ib);
694  }
695  }

731  private void BindCollectionSecondPass(ICollectionPropertiesMapping collectionMapping, Mapping.Collection model,
732  IDictionary<string, PersistentClass> persistentClasses, IDictionary<string, MetaAttribute> inheritedMetas)
733  {
734  if (model.IsOneToMany)
735  {
736  var oneToMany = (OneToMany)model.Element;
737  string associatedEntityName = oneToMany.ReferencedEntityName;
738  PersistentClass persistentClass;
739  if (persistentClasses.TryGetValue(associatedEntityName, out persistentClass) == false)
740  throw new MappingException("Association references unmapped class: " + associatedEntityName);
741  oneToMany.AssociatedClass = persistentClass;
742  model.CollectionTable = persistentClass.Table;
743  if (model.IsInverse && persistentClass.JoinClosureSpan > 0)
744  {
745  // NH: bidirectional one-to-many with a class splitted in more tables; have to find in which table is the inverse side
746  foreach (var joined in persistentClass.JoinClosureIterator)
747  {
748  if (collectionMapping.Key.Columns.Select(x=> x.name).All(x => joined.Table.ColumnIterator.Select(jc=> jc.Name).Contains(x)))
749  {
750  model.CollectionTable = joined.Table;
751  break;
752  }
753  }
754  }
755 
756  if (log.IsInfoEnabled)
757  log.Info("mapping collection: " + model.Role + " -> " + model.CollectionTable.Name);
758  }
759 
760  //CHECK
761  if (!string.IsNullOrEmpty(collectionMapping.Check))
762  {
763  model.CollectionTable.AddCheckConstraint(collectionMapping.Check);
764  }
765  BindKey(collectionMapping.Key, model);
766 
767  //contained elements:
768  HbmCompositeElement compositeElementMapping;
769  HbmElement elementMapping;
770  HbmManyToAny manyToAnyMapping;
771  HbmManyToMany manyToManyMapping;
772 
773  if ((elementMapping = collectionMapping.ElementRelationship as HbmElement) != null)
774  {
775  BindElement(elementMapping, model);
776  }
777  else if ((manyToManyMapping = collectionMapping.ElementRelationship as HbmManyToMany) != null)
778  {
779  BindManyToMany(manyToManyMapping, model);
780  }
781  else if ((compositeElementMapping = collectionMapping.ElementRelationship as HbmCompositeElement) != null)
782  {
783  BindCompositeElement(compositeElementMapping, model, inheritedMetas);
784  }
785  else if ((manyToAnyMapping = collectionMapping.ElementRelationship as HbmManyToAny) != null)
786  {
787  BindManyToAny(manyToAnyMapping, model);
788  }
789 
790  BindCache(collectionMapping.Cache, model);
791 
792  if (NeedBackref(model))
793  {
794  // for non-inverse one-to-many, with a not-null fk, add a backref!
795  string entityName = ((OneToMany)model.Element).ReferencedEntityName;
796  PersistentClass referenced = mappings.GetClass(entityName);
797  var prop = new Backref();
798  prop.Name = '_' + model.OwnerEntityName + "." + collectionMapping.Name + "Backref";
799  prop.IsUpdateable = false;
800  prop.IsSelectable = false;
801  prop.CollectionRole = model.Role;
802  prop.EntityName = model.Owner.EntityName;
803  prop.Value = model.Key;
804  referenced.AddProperty(prop);
805  }
806  }

886  private void BindKey(HbmKey keyMapping, Mapping.Collection model)
887  {
888  if (keyMapping == null)
889  {
890  return;
891  }
892  string propRef = model.ReferencedPropertyName;
893  IKeyValue keyValue;
894  if (propRef == null)
895  {
896  keyValue = model.Owner.Identifier;
897  }
898  else
899  {
900  keyValue = (IKeyValue) model.Owner.GetProperty(propRef).Value;
901  }
902  var key = new DependantValue(model.CollectionTable, keyValue)
903  {IsCascadeDeleteEnabled = keyMapping.ondelete == HbmOndelete.Cascade};
904 
905  new ValuePropertyBinder(key, Mappings).BindSimpleValue(keyMapping, Mapping.Collection.DefaultKeyColumnName, model.IsOneToMany);
906 
907  if (key.Type.ReturnedClass.IsArray)
908  throw new MappingException("illegal use of an array as an identifier (arrays don't reimplement Equals)");
909  model.Key = key;
910 
911  key.SetNullable(!keyMapping.IsNullable.HasValue || keyMapping.IsNullable.Value);
912  key.SetUpdateable(!keyMapping.IsUpdatable.HasValue || keyMapping.IsUpdatable.Value);
913  BindForeignKey(keyMapping.foreignkey, key);
914  }

916  private void BindManyToManySubelements(HbmManyToMany manyToManyMapping, Mapping.Collection collection)
917  {
918  // Bind the where
919  string whereCondition = string.IsNullOrEmpty(manyToManyMapping.where) ? null : manyToManyMapping.where;
920  collection.ManyToManyWhere = whereCondition;
921 
922  // Bind the order-by
923  string orderFragment = string.IsNullOrEmpty(manyToManyMapping.orderby) ? null : manyToManyMapping.orderby;
924  collection.ManyToManyOrdering = orderFragment;
925 
926  // Bind the filters
927  var filters = manyToManyMapping.filter;
928  var hasFilters = filters != null && filters.Length > 0;
929 
930  if ((hasFilters || whereCondition != null) && collection.FetchMode == FetchMode.Join
931  && collection.Element.FetchMode != FetchMode.Join)
932  {
933  throw new MappingException(
934  string.Format(
935  "many-to-many defining filter or where without join fetching not valid within collection using join fetching [{0}]",
936  collection.Role));
937 
938  }
939 
940  new FiltersBinder(collection, Mappings).Bind(filters, collection.AddManyToManyFilter);
941  }
