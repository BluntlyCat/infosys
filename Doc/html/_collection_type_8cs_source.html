235  public object GetCollection(object key, ISessionImplementor session, object owner)
236  {
237  ICollectionPersister persister = GetPersister(session);
238  IPersistenceContext persistenceContext = session.PersistenceContext;
239  EntityMode entityMode = session.EntityMode;
240 
241  if (entityMode == EntityMode.Xml && !isEmbeddedInXML)
242  {
243  return UnfetchedCollection;
244  }
245 
246  // check if collection is currently being loaded
247  IPersistentCollection collection = persistenceContext.LoadContexts.LocateLoadingCollection(persister, key);
248  if (collection == null)
249  {
250  // check if it is already completely loaded, but unowned
251  collection = persistenceContext.UseUnownedCollection(new CollectionKey(persister, key, entityMode));
252  if (collection == null)
253  {
254  // create a new collection wrapper, to be initialized later
255  collection = Instantiate(session, persister, key);
256  collection.Owner = owner;
257 
258  persistenceContext.AddUninitializedCollection(persister, collection, key);
259 
260  // some collections are not lazy:
261  if (InitializeImmediately(entityMode))
262  {
263  session.InitializeCollection(collection, false);
264  }
265  else if (!persister.IsLazy)
266  {
267  persistenceContext.AddNonLazyCollection(collection);
268  }
269 
270  if (HasHolder(entityMode))
271  {
272  session.PersistenceContext.AddCollectionHolder(collection);
273  }
274  }
275  }
276  collection.Owner = owner;
277  return collection.GetValue();
278  }

332  IDictionary copyCache)
333  {

364  public virtual object ReplaceElements(object original, object target, object owner, IDictionary copyCache,
365  ISessionImplementor session)
366  {
367  // TODO: does not work for EntityMode.DOM4J yet!
368  object result = target;
369  Clear(result);
370 
371  // copy elements into newly empty target collection
372  IType elemType = GetElementType(session.Factory);
373  IEnumerable iter = (IEnumerable)original;
374  foreach (object obj in iter)
375  {
376  Add(result, elemType.Replace(obj, null, session, owner, copyCache));
377  }
378 
379  // if the original is a PersistentCollection, and that original
380  // was not flagged as dirty, then reset the target's dirty flag
381  // here after the copy operation.
382  // One thing to be careful of here is a "bare" original collection
383  // in which case we should never ever ever reset the dirty flag
384  // on the target because we simply do not know...
385  IPersistentCollection originalPc = original as IPersistentCollection;
386  IPersistentCollection resultPc = result as IPersistentCollection;
387  if(originalPc != null && resultPc!=null)
388  {
389  if (!originalPc.IsDirty)
390  resultPc.ClearDirty();
391  }
392 
393  return result;
394  }
