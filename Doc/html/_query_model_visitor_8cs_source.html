124  public override void VisitMainFromClause(MainFromClause fromClause, QueryModel queryModel)
125  {
126  var querySourceName = VisitorParameters.QuerySourceNamer.GetName(fromClause);
127  var hqlExpressionTree = HqlGeneratorExpressionTreeVisitor.Visit(fromClause.FromExpression, VisitorParameters);
128 
129  _hqlTree.AddFromClause(_hqlTree.TreeBuilder.Range(hqlExpressionTree, _hqlTree.TreeBuilder.Alias(querySourceName)));
130 
131  // apply any result operators that were rewritten
132  if (RewrittenOperatorResult != null)
133  {
134  CurrentEvaluationType = RewrittenOperatorResult.EvaluationType;
135  foreach (ResultOperatorBase rewrittenOperator in RewrittenOperatorResult.RewrittenOperators)
136  {
137  VisitResultOperator(rewrittenOperator, queryModel, -1);
138  }
139  }
140 
141  base.VisitMainFromClause(fromClause, queryModel);
142  }

144  public override void VisitAdditionalFromClause(AdditionalFromClause fromClause, QueryModel queryModel, int index)
145  {
146  var querySourceName = VisitorParameters.QuerySourceNamer.GetName(fromClause);
147 
148  var joinClause = fromClause as NhJoinClause;
149  if (joinClause != null)
150  {
151  VisitNhJoinClause(querySourceName, joinClause);
152  }
153  else if (fromClause.FromExpression is MemberExpression)
154  {
155  // It's a join
156  _hqlTree.AddFromClause(
157  _hqlTree.TreeBuilder.Join(
158  HqlGeneratorExpressionTreeVisitor.Visit(fromClause.FromExpression, VisitorParameters).AsExpression(),
159  _hqlTree.TreeBuilder.Alias(querySourceName)));
160  }
161  else
162  {
163  // TODO - exact same code as in MainFromClause; refactor this out
164  _hqlTree.AddFromClause(
165  _hqlTree.TreeBuilder.Range(

174  private void VisitNhJoinClause(string querySourceName, NhJoinClause joinClause)
175  {
176  var expression = HqlGeneratorExpressionTreeVisitor.Visit(joinClause.FromExpression, VisitorParameters).AsExpression();
177  var alias = _hqlTree.TreeBuilder.Alias(querySourceName);
178 
179  HqlTreeNode hqlJoin;
180  if (joinClause.IsInner)
181  {
182  hqlJoin = _hqlTree.TreeBuilder.Join(expression, @alias);
183  }
184  else
185  {
186  hqlJoin = _hqlTree.TreeBuilder.LeftJoin(expression, @alias);
187  }
188 
189  foreach (var withClause in joinClause.Restrictions)
190  {
191  var booleanExpression = HqlGeneratorExpressionTreeVisitor.Visit(withClause.Predicate, VisitorParameters).AsBooleanExpression();
192  hqlJoin.AddChild(_hqlTree.TreeBuilder.With(booleanExpression));
193  }
194 
195  _hqlTree.AddFromClause(hqlJoin);
196  }
