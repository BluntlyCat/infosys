1022  virtual protected PatternParser CreatePatternParser(string pattern)
1023  {
1024  PatternParser patternParser = new PatternParser(pattern);
1025 
1026  // Add all the builtin patterns
1027  foreach(DictionaryEntry entry in s_globalRulesRegistry)
1028  {
1029  ConverterInfo converterInfo = new ConverterInfo();
1030  converterInfo.Name = (string)entry.Key;
1031  converterInfo.Type = (Type)entry.Value;
1032  patternParser.PatternConverters[entry.Key] = converterInfo;
1033  }
1034  // Add the instance patterns
1035  foreach(DictionaryEntry entry in m_instanceRulesRegistry)
1036  {
1037  patternParser.PatternConverters[entry.Key] = entry.Value;
1038  }
1039 
1040  return patternParser;
1041  }

1061  override public void ActivateOptions()
1062  {
1063  m_head = CreatePatternParser(m_pattern).Parse();
1064 
1065  PatternConverter curConverter = m_head;
1066  while(curConverter != null)
1067  {
1068  PatternLayoutConverter layoutConverter = curConverter as PatternLayoutConverter;
1069  if (layoutConverter != null)
1070  {
1071  if (!layoutConverter.IgnoresException)
1072  {
1073  // Found converter that handles the exception
1074  this.IgnoresException = false;
1075 
1076  break;
1077  }
1078  }
1079  curConverter = curConverter.Next;
1080  }
1081  }
