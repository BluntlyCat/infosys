288  public bool AcquireLock()
289  {
290  bool ret=false;
291  lock(this)
292  {
293  if (m_lockLevel==0)
294  {
295  // If lock is already acquired, nop
296  m_realStream=m_lockingModel.AcquireLock();
297  }
298  if (m_realStream!=null)
299  {
300  m_lockLevel++;
301  ret=true;
302  }
303  }
304  return ret;
305  }
306 
307  public void ReleaseLock()
308  {
309  lock(this)
310  {
311  m_lockLevel--;
312  if (m_lockLevel==0)
313  {
314  // If already unlocked, nop
315  m_lockingModel.ReleaseLock();
316  m_realStream=null;
317  }
318  }
