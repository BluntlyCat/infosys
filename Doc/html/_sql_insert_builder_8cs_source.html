116  public virtual SqlString ToSqlString()
117  {
118  // 5 = "INSERT INTO", tableName, " (" , ") VALUES (", and ")"
119  int initialCapacity = 5;
120 
121  // 2 = the first column is just the columnName and columnValue
122  initialCapacity += 2;
123 
124  // eachColumn after the first one is 4 because of the ", ", columnName
125  // and the ", " columnValue
126  if (columns.Count > 0)
127  {
128  initialCapacity += ((columns.Count - 1) * 4);
129  }
130 
131  if (!string.IsNullOrEmpty(comment))
132  initialCapacity++;
133 
134  SqlStringBuilder sqlBuilder = new SqlStringBuilder(initialCapacity + 2);
135  if (!string.IsNullOrEmpty(comment))
136  {
137  sqlBuilder.Add("/* " + comment + " */ ");
138  }
139 
140  sqlBuilder.Add("INSERT INTO ")
141  .Add(tableName);
142 
143  if (columns.Count == 0)
144  {
145  sqlBuilder.Add(" ").Add(factory.Dialect.NoColumnsInsertString);
146  }
147  else
148  {
149  sqlBuilder.Add(" (");
150 
151  // do we need a comma before we add the column to the INSERT list
152  // when we get started the first column doesn't need one.
153  bool commaNeeded = false;
154  foreach (string columnName in columns.Keys)
155  {
156  // build up the column list
157  if (commaNeeded)
158  sqlBuilder.Add(StringHelper.CommaSpace);
159  commaNeeded = true;
160 
161  sqlBuilder.Add(columnName);
162  }
163 
164  sqlBuilder.Add(") VALUES (");
165 
166  commaNeeded = false;
167  foreach (object obj in columns.Values)
168  {
169  if (commaNeeded)
170  sqlBuilder.Add(StringHelper.CommaSpace);
171  commaNeeded = true;
172 
173  SqlType param = obj as SqlType;
174  if (param != null)
175  sqlBuilder.Add(Parameter.Placeholder);
176  else
177  sqlBuilder.Add((string) obj);
178  }
179 
180  sqlBuilder.Add(")");
181  }
182 
183  if (log.IsDebugEnabled)
184  {
185  if (initialCapacity < sqlBuilder.Count)
186  {
187  log.Debug(
188  "The initial capacity was set too low at: " + initialCapacity + " for the InsertSqlBuilder " +
189  "that needed a capacity of: " + sqlBuilder.Count + " for the table " + tableName);
190  }
191  else if (initialCapacity > 16 && ((float) initialCapacity / sqlBuilder.Count) > 1.2)
192  {
193  log.Debug(
194  "The initial capacity was set too high at: " + initialCapacity + " for the InsertSqlBuilder " +
195  "that needed a capacity of: " + sqlBuilder.Count + " for the table " + tableName);
196  }
197  }
198 
199  return sqlBuilder.ToSqlString();
200  }
