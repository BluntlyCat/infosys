38  method.DeclaringType != null &&

233  public override HqlTreeNode BuildHql(MethodInfo method, Expression targetObject, ReadOnlyCollection<Expression> arguments, HqlTreeBuilder treeBuilder, IHqlExpressionVisitor visitor)
234  {
235  string trimWhere;
236  if (method.Name == "TrimStart")
237  trimWhere = "leading";
238  else if (method.Name == "TrimEnd")
239  trimWhere = "trailing";
240  else
241  trimWhere = "both";
242 
243  string trimChars = "";
244  if (method.GetParameters().Length > 0)
245  foreach (char c in (char[])((ConstantExpression)arguments[0]).Value)
246  trimChars += c;
247 
248 
249  if (trimChars == "")
250  {
251  return treeBuilder.MethodCall("trim", treeBuilder.Ident(trimWhere), treeBuilder.Ident("from"), visitor.Visit(targetObject).AsExpression());
252  }
253  else
254  {
255  return treeBuilder.MethodCall("trim", treeBuilder.Ident(trimWhere), treeBuilder.Constant(trimChars), treeBuilder.Ident("from"), visitor.Visit(targetObject).AsExpression());
