625  public string[] GetImplementors(string entityOrClassName)
626  {
627  string[] knownMap;
628  if (entityNameImplementorsMap.TryGetValue(entityOrClassName, out knownMap))
629  {
630  return knownMap;
631  }
632  System.Type clazz = null;
633 
634  // NH Different implementation for performance: a class without at least a namespace sure can't be found by reflection
635  if (entityOrClassName.IndexOf('.') > 0)
636  {
637  IEntityPersister checkPersister;
638  // NH Different implementation: we have better performance checking, first of all, if we know the class
639  // and take the System.Type directly from the persister (className have high probability to be entityName at least using Criteria or Linq)
640  if (entityPersisters.TryGetValue(entityOrClassName, out checkPersister))
641  {
642  if (!checkPersister.EntityMetamodel.HasPocoRepresentation)
643  {
644  // we found the persister but it is a dynamic entity without class
645  knownMap = new[] { entityOrClassName };
646  entityNameImplementorsMap[entityOrClassName] = knownMap;
647  return knownMap;
648  }
649  // NH : take care with this because we are forcing the Poco EntityMode
650  clazz = checkPersister.GetMappedClass(EntityMode.Poco);
651  }
652 
653  if (clazz == null)
654  {
655  try
656  {
657  clazz = ReflectHelper.ClassForFullNameOrNull(entityOrClassName);
658  }
659  catch (Exception)
660  {
661  clazz = null;
662  }
663  }
664  }
665 
666  if (clazz == null)
667  {
668  // try to get the class from imported names
669  string importedName = GetImportedClassName(entityOrClassName);
670  if (importedName != null)
671  {
672  clazz = System.Type.GetType(importedName, false);
673  }
674  }
675 
676  if (clazz == null)
677  {
678  knownMap = new[] { entityOrClassName };
679  entityNameImplementorsMap[entityOrClassName] = knownMap;
680  return knownMap; //for a dynamic-class
681  }
682 
683  var results = new List<string>();
684  foreach (var q in entityPersisters.Values.OfType<IQueryable>())
685  {
686  string registeredEntityName = q.EntityName;
687  // NH: as entity-name we are using the FullName but in HQL we allow just the Name, the class is mapped even when its FullName match the entity-name
688  bool isMappedClass = entityOrClassName.Equals(registeredEntityName) || clazz.FullName.Equals(registeredEntityName);
689  if (q.IsExplicitPolymorphism)
690  {
691  if (isMappedClass)
692  {
693  knownMap = new[] { registeredEntityName };
694  entityNameImplementorsMap[entityOrClassName] = knownMap;
695  return knownMap; // NOTE EARLY EXIT
696  }
697  }
698  else
699  {
700  if (isMappedClass)
701  {
702  results.Add(registeredEntityName);
703  }
704  else
705  {
706  if (IsMatchingImplementor(entityOrClassName, clazz, q))
707  {
708  bool assignableSuperclass;
709  if (q.IsInherited)
710  {
711  System.Type mappedSuperclass = GetEntityPersister(q.MappedSuperclass).GetMappedClass(EntityMode.Poco);
712  assignableSuperclass = clazz.IsAssignableFrom(mappedSuperclass);
713  }
714  else
715  {
716  assignableSuperclass = false;
717  }
718  if (!assignableSuperclass)
719  {
720  results.Add(registeredEntityName);
721  }
722  }
723  }
724  }
725  }
726  knownMap = results.ToArray();
727  entityNameImplementorsMap[entityOrClassName] = knownMap;
728  return knownMap;
729  }

731  private static bool IsMatchingImplementor(string entityOrClassName, System.Type entityClass, IQueryable implementor)
732  {
733  var implementorClass = implementor.GetMappedClass(EntityMode.Poco);
734  if (implementorClass == null)
735  {
736  return false;
737  }
738  if (entityClass == implementorClass)
739  {
740  // It is possible to have multiple mappings for the same entity class, but with different entity names.
741  // When querying for a specific entity name, we should only return entities for the requested entity name
742  // and not return entities for any other entity names that may map to the same entity class.
743  bool isEntityName = !entityOrClassName.Equals(entityClass.FullName);
744  return !isEntityName || entityOrClassName.Equals(implementor.EntityName);
745  }
746  return entityClass.IsAssignableFrom(implementorClass);
747  }

1139  private IDictionary<string, HibernateException> CheckNamedQueries()
1140  {
1141  IDictionary<string, HibernateException> errors = new Dictionary<string, HibernateException>();
1142 
1143  // Check named HQL queries
1144  log.Debug("Checking " + namedQueries.Count + " named HQL queries");
1145  foreach (var entry in namedQueries)
1146  {
1147  string queryName = entry.Key;
1148  NamedQueryDefinition qd = entry.Value;
1149  // this will throw an error if there's something wrong.
1150  try
1151  {
1152  log.Debug("Checking named query: " + queryName);
1153  //TODO: BUG! this currently fails for named queries for non-POJO entities
1154  queryPlanCache.GetHQLQueryPlan(qd.QueryString.ToQueryExpression(), false, new CollectionHelper.EmptyMapClass<string, IFilter>());
1155  }
1156  catch (QueryException e)
1157  {
1158  errors[queryName] = e;
1159  }
1160  catch (MappingException e)
1161  {
1162  errors[queryName] = e;
1163  }
1164  }
1165 
1166  log.Debug("Checking " + namedSqlQueries.Count + " named SQL queries");
1167  foreach (KeyValuePair<string, NamedSQLQueryDefinition> entry in namedSqlQueries)
1168  {
1169  string queryName = entry.Key;
1170  NamedSQLQueryDefinition qd = entry.Value;
1171  // this will throw an error if there's something wrong.
1172  try
1173  {
1174  log.Debug("Checking named SQL query: " + queryName);
1175  // TODO : would be really nice to cache the spec on the query-def so as to not have to re-calc the hash;
1176  // currently not doable though because of the resultset-ref stuff...
1177  NativeSQLQuerySpecification spec;
1178  if (qd.ResultSetRef != null)
1179  {
1180  ResultSetMappingDefinition definition = sqlResultSetMappings[qd.ResultSetRef];
1181  if (definition == null)
1182  {
1183  throw new MappingException("Unable to find resultset-ref definition: " + qd.ResultSetRef);
1184  }
1185  spec = new NativeSQLQuerySpecification(qd.QueryString, definition.GetQueryReturns(), qd.QuerySpaces);
1186  }
1187  else
1188  {
1189  spec = new NativeSQLQuerySpecification(qd.QueryString, qd.QueryReturns, qd.QuerySpaces);
1190  }
1191  queryPlanCache.GetNativeSQLQueryPlan(spec);
1192  }
1193  catch (QueryException e)
1194  {
1195  errors[queryName] = e;
1196  }
1197  catch (MappingException e)
1198  {
1199  errors[queryName] = e;
1200  }
1201  }
1202 
1203  return errors;
