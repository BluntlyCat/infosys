46  public SharedMemory(string name, IntPtr size)
47  {
48  fileMapping = NativeMethods.OpenFileMapping(FILE_MAP_WRITE, false,
49  name);
50  if (fileMapping == IntPtr.Zero)
51  {
52  throw new MySqlException("Cannot open file mapping " + name);
53  }
54  view = NativeMethods.MapViewOfFile(fileMapping, FILE_MAP_WRITE, 0, 0, size);

109  public SharedMemoryStream(string memName)
110  {
111  memoryName = memName;

114  public void Open(uint timeOut)
115  {
116  if (connectionClosed != null)
117  {
118  Debug.Assert(false, "Connection is already open");
119  }
120 
121  GetConnectNumber(timeOut);
122  SetupEvents();
123  }
124 
125  public override void Close()
126  {
127  if (connectionClosed != null)
128  {
129  bool isClosed = connectionClosed.WaitOne(0);
130  if (!isClosed)
131  {
132  connectionClosed.Set();
133  connectionClosed.Close();
134  }
135  connectionClosed = null;
136  EventWaitHandle[] handles = { serverRead, serverWrote, clientRead, clientWrote };
137 
138  for (int i = 0; i < handles.Length; i++)
139  {
140  if (handles[i] != null)
141  handles[i].Close();
142  }
143  if (data != null)
144  {
145  data.Dispose();
146  data = null;
147  }
148  }
149  }
150 
151  private void GetConnectNumber(uint timeOut)
152  {
153  EventWaitHandle connectRequest;
154  try
155  {
156  connectRequest =
157  EventWaitHandle.OpenExisting(memoryName + "_CONNECT_REQUEST");
158 
159  }
160  catch (Exception)
161  {
162  // If server runs as service, its shared memory is global
163  // And if connector runs in user session, it needs to prefix
164  // shared memory name with "Global\"
165  string prefixedMemoryName = @"Global\" + memoryName;
166  connectRequest =
167  EventWaitHandle.OpenExisting(prefixedMemoryName + "_CONNECT_REQUEST");
168  memoryName = prefixedMemoryName;
169  }
170  EventWaitHandle connectAnswer =
171  EventWaitHandle.OpenExisting(memoryName + "_CONNECT_ANSWER");
172  using (SharedMemory connectData =
173  new SharedMemory(memoryName + "_CONNECT_DATA", (IntPtr)4))
174  {
175  // now start the connection
176  if (!connectRequest.Set())
177  throw new MySqlException("Failed to open shared memory connection");
178  if (!connectAnswer.WaitOne((int)(timeOut * 1000), false))
179  throw new MySqlException("Timeout during connection");
180  connectNumber = Marshal.ReadInt32(connectData.View);
181  }
182  }
183 
184 
185  private void SetupEvents()
186  {
187  string prefix = memoryName + "_" + connectNumber;
188  data = new SharedMemory(prefix + "_DATA", (IntPtr)BUFFERLENGTH);
189  serverWrote = EventWaitHandle.OpenExisting(prefix + "_SERVER_WROTE");
190  serverRead = EventWaitHandle.OpenExisting(prefix + "_SERVER_READ");
191  clientWrote = EventWaitHandle.OpenExisting(prefix + "_CLIENT_WROTE");
192  clientRead = EventWaitHandle.OpenExisting(prefix + "_CLIENT_READ");
193  connectionClosed = EventWaitHandle.OpenExisting(prefix + "_CONNECTION_CLOSED");
194 
195  // tell the server we are ready
196  serverRead.Set();
197  }

234  public override int Read(byte[] buffer, int offset, int count)
235  {
236  int timeLeft = readTimeout;
237  WaitHandle[] waitHandles = { serverWrote, connectionClosed };
238  LowResolutionStopwatch stopwatch = new LowResolutionStopwatch();
239  while (bytesLeft == 0)
240  {
241  stopwatch.Start();
242  int index = WaitHandle.WaitAny(waitHandles, timeLeft);
243  stopwatch.Stop();
244  if (index == WaitHandle.WaitTimeout)
245  throw new TimeoutException("Timeout when reading from shared memory");
246 
247  if (waitHandles[index] == connectionClosed)
248  throw new MySqlException("Connection to server lost", true, null);
249 
250  if (readTimeout != System.Threading.Timeout.Infinite)
251  {
252  timeLeft = readTimeout - (int)stopwatch.ElapsedMilliseconds;
253  if (timeLeft < 0)
254  throw new TimeoutException("Timeout when reading from shared memory");
255  }
256 
257  bytesLeft = Marshal.ReadInt32(data.View);
258  position = 4;
259  }
260 
261  int len = Math.Min(count, bytesLeft);
262  long baseMem = data.View.ToInt64() + position;
263 
264  for (int i = 0; i < len; i++, position++)
265  buffer[offset + i] = Marshal.ReadByte((IntPtr)(baseMem + i));
266 
267  bytesLeft -= len;
268  if (bytesLeft == 0)
269  clientRead.Set();
270 
271  return len;
272  }

279  public override void Write(byte[] buffer, int offset, int count)
280  {
281  int leftToDo = count;
282  int buffPos = offset;
283  WaitHandle[] waitHandles = { serverRead, connectionClosed };
284  LowResolutionStopwatch stopwatch = new LowResolutionStopwatch();
285  int timeLeft = writeTimeout;
286 
287  while (leftToDo > 0)
288  {
289  stopwatch.Start();
290  int index = WaitHandle.WaitAny(waitHandles, timeLeft);
291  stopwatch.Stop();
292 
293  if (waitHandles[index] == connectionClosed)
294  throw new MySqlException("Connection to server lost", true, null);
295 
296  if (index == WaitHandle.WaitTimeout)
297  throw new TimeoutException("Timeout when reading from shared memory");
298 
299  if (writeTimeout != System.Threading.Timeout.Infinite)
300  {
301  timeLeft = writeTimeout - (int)stopwatch.ElapsedMilliseconds;
302  if (timeLeft < 0)
303  throw new TimeoutException("Timeout when writing to shared memory");
304  }
305  int bytesToDo = Math.Min(leftToDo, BUFFERLENGTH);
306  long baseMem = data.View.ToInt64() + 4;
307  Marshal.WriteInt32(data.View, bytesToDo);
308  Marshal.Copy(buffer, buffPos, (IntPtr)baseMem, bytesToDo);
309  buffPos += bytesToDo;
310  leftToDo -= bytesToDo;
311  if (!clientWrote.Set())
312  throw new MySqlException("Writing to shared memory failed");
313  }
314  }
