81  public LoggingEvent Append(LoggingEvent loggingEvent)
82  {
83  if (loggingEvent == null)
84  {
85  throw new ArgumentNullException("loggingEvent");
86  }
87 
88  lock(this)
89  {
90  // save the discarded event
91  LoggingEvent discardedLoggingEvent = m_events[m_last];
92 
93  // overwrite the last event position
94  m_events[m_last] = loggingEvent;
95  if (++m_last == m_maxSize)
96  {
97  m_last = 0;
98  }
99 
100  if (m_numElems < m_maxSize)
101  {
102  m_numElems++;
103  }
104  else if (++m_first == m_maxSize)
105  {
106  m_first = 0;
107  }
108 
109  if (m_numElems < m_maxSize)
110  {
111  // Space remaining
112  return null;
113  }
114  else
115  {
116  // Buffer is full and discarding an event
117  return discardedLoggingEvent;

221  {
222  lock(this)
223  {

225  {

229  {
230  return; // nothing to do
231  }
232 
233  LoggingEvent[] temp = new LoggingEvent[newSize];
234 
235  int loopLen = (newSize < m_numElems) ? newSize : m_numElems;
236 
237  for(int i = 0; i < loopLen; i++)
238  {
239  temp[i] = m_events[m_first];
240  m_events[m_first] = null;
241 
242  if (++m_first == m_numElems)
243  {
244  m_first = 0;
245  }
246  }
247 
248  m_events = temp;
249  m_first = 0;
250  m_numElems = loopLen;
251  m_maxSize = newSize;
252 
253  if (loopLen == newSize)
