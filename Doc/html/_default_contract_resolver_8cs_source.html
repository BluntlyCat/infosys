223  public virtual JsonContract ResolveContract(Type type)
224  {
225  if (type == null)
226  throw new ArgumentNullException("type");
227 
228  JsonContract contract;
229  ResolverContractKey key = new ResolverContractKey(GetType(), type);
230  Dictionary<ResolverContractKey, JsonContract> cache = GetCache();
231  if (cache == null || !cache.TryGetValue(key, out contract))
232  {
233  contract = CreateContract(type);
234 
235  // avoid the possibility of modifying the cache dictionary while another thread is accessing it
236  lock (_typeContractCacheLock)
237  {
238  cache = GetCache();
239  Dictionary<ResolverContractKey, JsonContract> updatedCache =
240  (cache != null)
241  ? new Dictionary<ResolverContractKey, JsonContract>(cache)
242  : new Dictionary<ResolverContractKey, JsonContract>();
243  updatedCache[key] = contract;
244 
245  UpdateCache(updatedCache);
246  }
247  }
248 
249  return contract;

269  .Where(m => !ReflectionUtils.IsIndexedProperty(m)).ToList();

272

290  serializableMembers.Add(member);

292  else

294  // add members that are explicitly marked with JsonProperty/DataMember attribute

297  serializableMembers.Add(member);
298 #if !NET20

300  serializableMembers.Add(member);
301 #endif

304  }
305  }
306  }

399  private ExtensionDataSetter GetExtensionDataForType(Type type)
400  {
401  ExtensionDataSetter extensionDataSetter = null;
402 
403  foreach (Type baseType in GetClassHierarchyForType(type))
404  {
405  IList<MemberInfo> members = new List<MemberInfo>();
406  members.AddRange(baseType.GetProperties(BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.Instance | BindingFlags.DeclaredOnly));
407  members.AddRange(baseType.GetFields(BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.Instance | BindingFlags.DeclaredOnly));
408 
409  foreach (MemberInfo member in members)
410  {
411  MemberTypes memberType = member.MemberType();
412  if (memberType != MemberTypes.Property && memberType != MemberTypes.Field)
413  continue;
414 
415  // last instance of attribute wins on type if there are multiple
416  if (!member.IsDefined(typeof(JsonExtensionDataAttribute), false))
417  continue;
418 
419  Type t = ReflectionUtils.GetMemberUnderlyingType(member);
420 
421  Type dictionaryType;
422  if (ReflectionUtils.ImplementsGenericDefinition(t, typeof(IDictionary<,>), out dictionaryType))
423  {
424  Type keyType = dictionaryType.GetGenericArguments()[0];
425  Type valueType = dictionaryType.GetGenericArguments()[1];
426 
427  // change type to a class if it is the base interface so it can be instantiated if needed
428  if (ReflectionUtils.IsGenericDefinition(t, typeof(IDictionary<,>)))
429  t = typeof(Dictionary<,>).MakeGenericType(keyType, valueType);
430 
431  if (keyType.IsAssignableFrom(typeof(string)) && valueType.IsAssignableFrom(typeof(JToken)))
432  {
433  MethodInfo addMethod = t.GetMethod("Add", new [] { keyType, valueType });
434  Func<object, object> getExtensionDataDictionary = JsonTypeReflector.ReflectionDelegateFactory.CreateGet<object>(member);
435  Action<object, object> setExtensionDataDictionary = JsonTypeReflector.ReflectionDelegateFactory.CreateSet<object>(member);
436  Func<object> createExtensionDataDictionary = JsonTypeReflector.ReflectionDelegateFactory.CreateDefaultConstructor<object>(t);
437  MethodCall<object, object> setExtensionDataDictionaryValue = JsonTypeReflector.ReflectionDelegateFactory.CreateMethodCall<object>(addMethod);
438 
439  extensionDataSetter = (o, key, value) =>
440  {
441  object dictionary = getExtensionDataDictionary(o);
442  if (dictionary == null)
443  {
444  dictionary = createExtensionDataDictionary();
445  setExtensionDataDictionary(o, dictionary);
446  }
447 
448  setExtensionDataDictionaryValue(dictionary, key, value);
449  };
450 
451  continue;
452  }
453  }
454 
455  throw new JsonException("Invalid extension data attribute on '{0}'. Member '{1}' type must implement IDictionary<string, JToken>.".FormatWith(CultureInfo.InvariantCulture, GetClrTypeFullName(member.DeclaringType), member.Name));
456  }
457  }
458 
459  return extensionDataSetter;
460  }

495  {

499 
500  foreach (ParameterInfo parameterInfo in constructorParameters)
501  {
502  // it is possible to generate a ParameterInfo with a null name using Reflection.Emit

504  JsonProperty matchingMemberProperty = (parameterInfo.Name != null) ? memberProperties.GetClosestMatchProperty(parameterInfo.Name) : null;
505 
506  // type must match as well as name
507  if (matchingMemberProperty != null && matchingMemberProperty.PropertyType != parameterInfo.ParameterType)
508  matchingMemberProperty = null;
509 
510  JsonProperty property = CreatePropertyFromConstructorParameter(matchingMemberProperty, parameterInfo);
511 
512  if (property != null)
513  {
514  parameterCollection.AddProperty(property);
515  }
516  }
517 
518  return parameterCollection;

542  property.Converter = property.Converter ?? matchingMemberProperty.Converter;

628  contract.OnSerializingCallbacks.AddRange(onSerializing);

630  }

653  private void GetCallbackMethodsForType(Type type, out List<SerializationCallback> onSerializing, out List<SerializationCallback> onSerialized, out List<SerializationCallback> onDeserializing, out List<SerializationCallback> onDeserialized, out List<SerializationErrorCallback> onError)
654  {
655  onSerializing = null;
656  onSerialized = null;
657  onDeserializing = null;
658  onDeserialized = null;
659  onError = null;
660 
661  foreach (Type baseType in GetClassHierarchyForType(type))
662  {
663  // while we allow more than one OnSerialized total, only one can be defined per class
664  MethodInfo currentOnSerializing = null;
665  MethodInfo currentOnSerialized = null;
666  MethodInfo currentOnDeserializing = null;
667  MethodInfo currentOnDeserialized = null;
668  MethodInfo currentOnError = null;
669 
670  foreach (MethodInfo method in baseType.GetMethods(BindingFlags.NonPublic | BindingFlags.Public | BindingFlags.Instance | BindingFlags.DeclaredOnly))
671  {
672  // compact framework errors when getting parameters for a generic method
673  // lame, but generic methods should not be callbacks anyway
674  if (method.ContainsGenericParameters)
675  continue;
676 
677  Type prevAttributeType = null;
678  ParameterInfo[] parameters = method.GetParameters();
679 
680  if (IsValidCallback(method, parameters, typeof(OnSerializingAttribute), currentOnSerializing, ref prevAttributeType))
681  {
682  onSerializing = onSerializing ?? new List<SerializationCallback>();
683  onSerializing.Add(JsonContract.CreateSerializationCallback(method));
684  currentOnSerializing = method;
685  }
686  if (IsValidCallback(method, parameters, typeof(OnSerializedAttribute), currentOnSerialized, ref prevAttributeType))
687  {
688  onSerialized = onSerialized ?? new List<SerializationCallback>();
689  onSerialized.Add(JsonContract.CreateSerializationCallback(method));
690  currentOnSerialized = method;
691  }
692  if (IsValidCallback(method, parameters, typeof(OnDeserializingAttribute), currentOnDeserializing, ref prevAttributeType))
693  {
694  onDeserializing = onDeserializing ?? new List<SerializationCallback>();
695  onDeserializing.Add(JsonContract.CreateSerializationCallback(method));
696  currentOnDeserializing = method;
697  }
698  if (IsValidCallback(method, parameters, typeof(OnDeserializedAttribute), currentOnDeserialized, ref prevAttributeType))
699  {
700  onDeserialized = onDeserialized ?? new List<SerializationCallback>();
701  onDeserialized.Add(JsonContract.CreateSerializationCallback(method));
702  currentOnDeserialized = method;
703  }
704  if (IsValidCallback(method, parameters, typeof(OnErrorAttribute), currentOnError, ref prevAttributeType))
705  {
706  onError = onError ?? new List<SerializationErrorCallback>();
707  onError.Add(JsonContract.CreateSerializationErrorCallback(method));
708  currentOnError = method;
709  }
710  }
711  }

714  private List<Type> GetClassHierarchyForType(Type type)
715  {
716  List<Type> ret = new List<Type>();

718  Type current = type;
719  while (current != null && current != typeof(object))
720  {
721  ret.Add(current);
722  current = current.BaseType();
723  }
724 
725  // Return the class list in order of simple => complex
726  ret.Reverse();
727  return ret;

939  private static bool IsValidCallback(MethodInfo method, ParameterInfo[] parameters, Type attributeType, MethodInfo currentCallback, ref Type prevAttributeType)
940  {
941  if (!method.IsDefined(attributeType, false))
942  return false;
943 
944  if (currentCallback != null)
945  throw new JsonException("Invalid attribute. Both '{0}' and '{1}' in type '{2}' have '{3}'.".FormatWith(CultureInfo.InvariantCulture, method, currentCallback, GetClrTypeFullName(method.DeclaringType), attributeType));
946 
947  if (prevAttributeType != null)
948  throw new JsonException("Invalid Callback. Method '{3}' in type '{2}' has both '{0}' and '{1}'.".FormatWith(CultureInfo.InvariantCulture, prevAttributeType, attributeType, GetClrTypeFullName(method.DeclaringType), method));
949 
950  if (method.IsVirtual)
951  throw new JsonException("Virtual Method '{0}' of type '{1}' cannot be marked with '{2}' attribute.".FormatWith(CultureInfo.InvariantCulture, method, GetClrTypeFullName(method.DeclaringType), attributeType));
952 
953  if (method.ReturnType != typeof(void))
954  throw new JsonException("Serialization Callback '{1}' in type '{0}' must return void.".FormatWith(CultureInfo.InvariantCulture, GetClrTypeFullName(method.DeclaringType), method));
955 
956  if (attributeType == typeof(OnErrorAttribute))
957  {
958  if (parameters == null || parameters.Length != 2 || parameters[0].ParameterType != typeof(StreamingContext) || parameters[1].ParameterType != typeof(ErrorContext))
959  throw new JsonException("Serialization Error Callback '{1}' in type '{0}' must have two parameters of type '{2}' and '{3}'.".FormatWith(CultureInfo.InvariantCulture, GetClrTypeFullName(method.DeclaringType), method, typeof(StreamingContext), typeof(ErrorContext)));
960  }
961  else
962  {
963  if (parameters == null || parameters.Length != 1 || parameters[0].ParameterType != typeof(StreamingContext))
964  throw new JsonException("Serialization Callback '{1}' in type '{0}' must have a single parameter of type '{2}'.".FormatWith(CultureInfo.InvariantCulture, GetClrTypeFullName(method.DeclaringType), method, typeof(StreamingContext)));
965  }
966 
967  prevAttributeType = attributeType;
968 
969  return true;
970  }
