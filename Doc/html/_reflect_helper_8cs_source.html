40  private static bool OverrideMethod(System.Type clazz, string methodName, System.Type[] parametersTypes)
41  {
42  try
43  {
44  MethodInfo method = !clazz.IsInterface
45  ? clazz.GetMethod(methodName, parametersTypes)
46  : GetMethodFromInterface(clazz, methodName, parametersTypes);
47  if (method == null)
48  {
49  return false;
50  }
51  else
52  {
53  // make sure that the DeclaringType is not System.Object - if that is the
54  // declaring type then there is no override.
55  return !(method.DeclaringType == typeof(object));
56  }
57  }
58  catch (AmbiguousMatchException)
59  {
60  // an ambigious match means that there is an override and it
61  // can't determine which one to use.
62  return true;
63  }
64  }

257  {
258  try

261  System.Type type = System.Type.GetType(name.ToString());
262 
263  if (type != null)
264  {
265  return type;
266  }
267 
268  if (name.Assembly == null)
269  {

272  log.Warn(message);

274  return null;

282  return null;

286 
287  if (type == null)

289  log.Warn("Could not load type " + name + ".");
290  return null;
291  }
292 
293  return type;
294  }
295  catch (Exception e)
296  {
297  if (log.IsErrorEnabled)

502  private static MethodInfo SafeGetMethod(System.Type type, MethodInfo method, System.Type[] tps)
503  {
504  const BindingFlags bindingFlags = BindingFlags.Instance | BindingFlags.Public | BindingFlags.NonPublic;
505 
506  List<System.Type> typesToSearch = new List<System.Type>();
507  MethodInfo foundMethod = null;
508 
509  try
510  {
511  typesToSearch.Add(type);
512 
513  if (type.IsInterface)
514  {
515  // Methods on parent interfaces are not actually inherited
516  // by child interfaces, so we have to use GetInterfaces to
517  // identify any parent interfaces that may contain the
518  // method implementation
519  System.Type[] inheritedInterfaces = type.GetInterfaces();
520  typesToSearch.AddRange(inheritedInterfaces);
521  }
522 
523  foreach (System.Type typeToSearch in typesToSearch)
524  {
525  MethodInfo result = typeToSearch.GetMethod(method.Name, bindingFlags, null, tps, null);
526  if (result != null)
527  {
528  foundMethod = result;
529  break;
530  }
531  }
532  }
533  catch (Exception)
534  {
535  throw;
536  }
537 
538  return foundMethod;

709  public static bool IsMethodOf(this MethodInfo source, System.Type realDeclaringType)
710  {
711  if (source == null)
712  {
713  throw new ArgumentNullException("source");
714  }
715  if (realDeclaringType == null)
716  {
717  throw new ArgumentNullException("realDeclaringType");
718  }
719  var methodDeclaringType = source.DeclaringType;
720  if(realDeclaringType == methodDeclaringType)
721  {
722  return true;
723  }
724  if (methodDeclaringType.IsGenericType && !methodDeclaringType.IsGenericTypeDefinition &&
725  realDeclaringType == methodDeclaringType.GetGenericTypeDefinition())
726  {
727  return true;
728  }
729  if (realDeclaringType.IsInterface)
730  {
731  var declaringTypeInterfaces = methodDeclaringType.GetInterfaces();
732  if(declaringTypeInterfaces.Contains(realDeclaringType))
733  {
734  var methodsMap = methodDeclaringType.GetInterfaceMap(realDeclaringType);
735  if(methodsMap.TargetMethods.Contains(source))
736  {
737  return true;
738  }
739  }
740  if (realDeclaringType.IsGenericTypeDefinition)
741  {
742  bool implements = declaringTypeInterfaces
743  .Where(t => t.IsGenericType && t.GetGenericTypeDefinition() == realDeclaringType)
744  .Select(implementedGenericInterface => methodDeclaringType.GetInterfaceMap(implementedGenericInterface))
745  .Any(methodsMap => methodsMap.TargetMethods.Contains(source));
746  if (implements)
747  {
748  return true;
749  }
750  }
751  }
752  return false;
753  }
