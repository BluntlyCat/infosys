416  private static int GetAfterSelectInsertPoint(SqlString sql)
417  {
418  // Assume no common table expressions with the statement.
419  if (sql.StartsWithCaseInsensitive("select distinct"))
420  {
421  return 15;
422  }
423  else if (sql.StartsWithCaseInsensitive("select"))
424  {
425  return 6;
426  }
427  return 0;
428  }
429 
430  public override SqlString GetLimitString(SqlString sql, SqlString offset, SqlString limit)
431  {
432  // SQL Anywhere 11 uses SELECT TOP n START AT m [ select list items ]
433  // for LIMIT/OFFSET support. Does not support a limit of zero.
434 
435  // FIXME - Add support for where offset is set, but limit is not.
436 
437  int insertionPoint = GetAfterSelectInsertPoint(sql);
438 
439  if (insertionPoint > 0)
440  {
441  SqlStringBuilder limitBuilder = new SqlStringBuilder();
442  limitBuilder.Add("select");
443  if (insertionPoint > 6)
444  {
445  limitBuilder.Add(" distinct ");
446  }
447  limitBuilder.Add(" top ");
448  limitBuilder.Add(limit);
449  if (offset != null)
450  {
451  limitBuilder.Add(" start at ");
452  limitBuilder.Add(offset);
453  }
454  limitBuilder.Add(sql.Substring(insertionPoint));
455  return limitBuilder.ToSqlString();
456  }
457  else
458  {
459  return sql; // unchanged
460  }
461  }
