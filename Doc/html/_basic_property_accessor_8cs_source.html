79  internal static BasicGetter GetGetterOrNull(System.Type type, string propertyName)
80  {
81  if (type == typeof(object) || type == null)
82  {
83  // the full inheritance chain has been walked and we could
84  // not find the Property get
85  return null;
86  }
87 
88  PropertyInfo property =
89  type.GetProperty(propertyName,
90  BindingFlags.Instance | BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.DeclaredOnly);
91 
92  if (property != null && property.CanRead)
93  {
94  return new BasicGetter(type, property, propertyName);
95  }
96  else
97  {
98  // recursively call this method for the base Type
99  BasicGetter getter = GetGetterOrNull(type.BaseType, propertyName);
100 
101  // didn't find anything in the base class - check to see if there is
102  // an explicit interface implementation.
103  if (getter == null)
104  {
105  System.Type[] interfaces = type.GetInterfaces();
106  for (int i = 0; getter == null && i < interfaces.Length; i++)
107  {
108  getter = GetGetterOrNull(interfaces[i], propertyName);
109  }
110  }
111  return getter;
112  }
113  }

124  internal static BasicSetter GetSetterOrNull(System.Type type, string propertyName)
125  {
126  if (type == typeof(object) || type == null)
127  {
128  // the full inheritance chain has been walked and we could
129  // not find the Property get
130  return null;
131  }
132 
133  BindingFlags bindingFlags = BindingFlags.Instance | BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.DeclaredOnly;
134 
135  if (type.IsValueType)
136  {
137  // the BindingFlags.IgnoreCase is important here because if type is a struct, the GetProperty method does
138  // not ignore case by default. If type is a class, it _does_ ignore case... we're better off explicitly
139  // stating that casing should be ignored so we get the same behavior for both structs and classes
140  bindingFlags = bindingFlags | BindingFlags.IgnoreCase;
141  }
142 
143  PropertyInfo property = type.GetProperty(propertyName, bindingFlags);
144 
145  if (property != null && property.CanWrite)
146  {
147  return new BasicSetter(type, property, propertyName);
148  }
149 
150  // recursively call this method for the base Type
151  BasicSetter setter = GetSetterOrNull(type.BaseType, propertyName);
152 
153  // didn't find anything in the base class - check to see if there is
154  // an explicit interface implementation.
155  if (setter == null)
156  {
157  System.Type[] interfaces = type.GetInterfaces();
158  for (int i = 0; setter == null && i < interfaces.Length; i++)
159  {
160  setter = GetSetterOrNull(interfaces[i], propertyName);
161  }
162  }
163  return setter;
164  }
