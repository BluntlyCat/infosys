152  public override int Read(byte[] buffer, int offset, int count)
153  {
154  if (buffer == null)
155  throw new ArgumentNullException("buffer", Resources.BufferCannotBeNull);
156  if (offset < 0 || offset >= buffer.Length)
157  throw new ArgumentOutOfRangeException("offset", Resources.OffsetMustBeValid);
158  if ((offset + count) > buffer.Length)
159  throw new ArgumentException(Resources.BufferNotLargeEnough, "buffer");
160 
161  if (inPos == maxInPos)
162  PrepareNextPacket();
163 
164  int countToRead = Math.Min(count, maxInPos - inPos);
165  int countRead;
166  if (zInStream != null)
167  countRead = zInStream.read(buffer, offset, countToRead);
168  else
169  countRead = baseStream.Read(buffer, offset, countToRead);
170  inPos += countRead;
171 
172  // release the weak reference
173  if (inPos == maxInPos)
174  {
175  zInStream = null;
176  if (!Platform.IsMono())
177  {
178  inBufferRef = new WeakReference(inBuffer, false);
179  inBuffer = null;
180  }
181  }
182 
183  return countRead;
184  }

186  private void PrepareNextPacket()
187  {
188  MySqlStream.ReadFully(baseStream, lengthBytes, 0, 7);
189  int compressedLength = lengthBytes[0] + (lengthBytes[1] << 8) + (lengthBytes[2] << 16);
190  // lengthBytes[3] is seq
191  int unCompressedLength = lengthBytes[4] + (lengthBytes[5] << 8) +
192  (lengthBytes[6] << 16);
193 
194  if (unCompressedLength == 0)
195  {
196  unCompressedLength = compressedLength;
197  zInStream = null;
198  }
199  else
200  {
201  ReadNextPacket(compressedLength);
202  MemoryStream ms = new MemoryStream(inBuffer);
203  zInStream = new ZInputStream(ms);
204  zInStream.maxInput = compressedLength;
205  }
206 
207  inPos = 0;
208  maxInPos = unCompressedLength;
209  }

221  private MemoryStream CompressCache()
222  {
223  // small arrays almost never yeild a benefit from compressing
224  if (cache.Length < 50)
225  return null;
226 
227  byte[] cacheBytes = cache.GetBuffer();
228  MemoryStream compressedBuffer = new MemoryStream();
229  ZOutputStream zos = new ZOutputStream(compressedBuffer, zlibConst.Z_DEFAULT_COMPRESSION);
230  zos.Write(cacheBytes, 0, (int)cache.Length);
231  zos.finish();
232 
233  // if the compression hasn't helped, then just return null
234  if (compressedBuffer.Length >= cache.Length)
235  return null;
236  return compressedBuffer;
237  }

239  private void CompressAndSendCache()
240  {
241  long compressedLength, uncompressedLength;
242 
243  // we need to save the sequence byte that is written
244  byte[] cacheBuffer = cache.GetBuffer();
245  byte seq = cacheBuffer[3];
246  cacheBuffer[3] = 0;
247 
248  // first we compress our current cache
249  MemoryStream compressedBuffer = CompressCache();
250 
251  // now we set our compressed and uncompressed lengths
252  // based on if our compression is going to help or not
253  MemoryStream memStream;
254 
255  if (compressedBuffer == null)
256  {
257  compressedLength = cache.Length;
258  uncompressedLength = 0;
259  memStream = cache;
260  }
261  else
262  {
263  compressedLength = compressedBuffer.Length;
264  uncompressedLength = cache.Length;
265  memStream = compressedBuffer;
266  }
267 
268  // Make space for length prefix (7 bytes) at the start of output
269  long dataLength = memStream.Length;
270  int bytesToWrite = (int)dataLength + 7;
271  memStream.SetLength(bytesToWrite);
272 
273  byte[] buffer = memStream.GetBuffer();
274  Array.Copy(buffer, 0, buffer, 7, (int)dataLength);
275 
276  // Write length prefix
277  buffer[0] = (byte)(compressedLength & 0xff);
278  buffer[1] = (byte)((compressedLength >> 8) & 0xff);
279  buffer[2] = (byte)((compressedLength >> 16) & 0xff);
280  buffer[3] = seq;
281  buffer[4] = (byte)(uncompressedLength & 0xff);
282  buffer[5] = (byte)((uncompressedLength >> 8) & 0xff);
283  buffer[6] = (byte)((uncompressedLength >> 16) & 0xff);
284 
285  baseStream.Write(buffer, 0, bytesToWrite);
286  baseStream.Flush();
287  cache.SetLength(0);
288  if (compressedBuffer != null)
289  {
290 #if !CF
291  compressedBuffer.Dispose();
292 #else
293  System.Reflection.MethodInfo dynMethod = cache.GetType().GetMethod("Dispose",
294  System.Reflection.BindingFlags.NonPublic | System.Reflection.BindingFlags.Instance);
295  dynMethod.Invoke(this, new object[0]);
296 #endif
297  }
298  }

300  public override void Flush()
301  {

307  private bool InputDone()
308  {
309  // if we have not done so yet, see if we can calculate how many bytes we are expecting
310  if (cache.Length < 4) return false;
311  byte[] buf = cache.GetBuffer();
312  int expectedLen = buf[0] + (buf[1] << 8) + (buf[2] << 16);
313  if (cache.Length < (expectedLen + 4)) return false;
314  return true;
315  }
