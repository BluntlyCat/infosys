251  public int Load()
252  {
253  bool openedConnection = false;
254 
255  if (Connection == null)
256  throw new InvalidOperationException(Resources.ConnectionNotSet);
257 
258  // next we open up the connetion if it is not already open
259  if (connection.State != ConnectionState.Open)
260  {
261  openedConnection = true;
262  connection.Open();
263  }
264 
265  try
266  {
267  string sql = BuildSqlCommand();
268  MySqlCommand cmd = new MySqlCommand(sql, Connection);
269  cmd.CommandTimeout = Timeout;
270  return cmd.ExecuteNonQuery();
271  }
272  finally
273  {
274  if (openedConnection)
275  connection.Close();
276  }
277  }

279  private string BuildSqlCommand()
280  {
281  StringBuilder sql = new StringBuilder("LOAD DATA ");
282  if (Priority == MySqlBulkLoaderPriority.Low)
283  sql.Append("LOW_PRIORITY ");
284  else if (Priority == MySqlBulkLoaderPriority.Concurrent)
285  sql.Append("CONCURRENT ");
286 
287  if (Local)
288  sql.Append("LOCAL ");
289  sql.Append("INFILE ");
290  if (Path.DirectorySeparatorChar == '\\')
291  sql.AppendFormat("'{0}' ", FileName.Replace(@"\", @"\\"));
292  else
293  sql.AppendFormat("'{0}' ", FileName);
294 
295  if (ConflictOption == MySqlBulkLoaderConflictOption.Ignore)
296  sql.Append("IGNORE ");
297  else if (ConflictOption == MySqlBulkLoaderConflictOption.Replace)
298  sql.Append("REPLACE ");
299 
300  sql.AppendFormat("INTO TABLE {0} ", TableName);
301 
302  if (CharacterSet != null)
303  sql.AppendFormat("CHARACTER SET {0} ", CharacterSet);
304 
305  StringBuilder optionSql = new StringBuilder(String.Empty);
306  if (FieldTerminator != defaultFieldTerminator)
307  optionSql.AppendFormat("TERMINATED BY '{0}' ", FieldTerminator);
308  if (FieldQuotationCharacter != Char.MinValue)
309  optionSql.AppendFormat("{0} ENCLOSED BY '{1}' ",
310  FieldQuotationOptional ? "OPTIONALLY" : "", FieldQuotationCharacter);
311  if (EscapeCharacter != defaultEscapeCharacter &&
312  EscapeCharacter != Char.MinValue)
313  optionSql.AppendFormat("ESCAPED BY '{0}' ", EscapeCharacter);
314  if (optionSql.Length > 0)
315  sql.AppendFormat("FIELDS {0}", optionSql.ToString());
316 
317  optionSql = new StringBuilder(String.Empty);
318  if (LinePrefix != null && LinePrefix.Length > 0)
319  optionSql.AppendFormat("STARTING BY '{0}' ", LinePrefix);
320  if (LineTerminator != defaultLineTerminator)
321  optionSql.AppendFormat("TERMINATED BY '{0}' ", LineTerminator);
322  if (optionSql.Length > 0)
323  sql.AppendFormat("LINES {0}", optionSql.ToString());
324 
325  if (NumberOfLinesToSkip > 0)
326  sql.AppendFormat("IGNORE {0} LINES ", NumberOfLinesToSkip);
327 
328  if (Columns.Count > 0)
329  {
330  sql.Append("(");
331  sql.Append(Columns[0]);
332  for (int i = 1; i < Columns.Count; i++)
333  sql.AppendFormat(",{0}", Columns[i]);
334  sql.Append(") ");
335  }
336 
337  if (Expressions.Count > 0)
338  {
339  sql.Append("SET ");
340  sql.Append(Expressions[0]);
341  for (int i = 1; i < Expressions.Count; i++)
342  sql.AppendFormat(",{0}", Expressions[i]);
343  }
344 
345  return sql.ToString();
346  }
347  }
