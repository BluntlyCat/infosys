220  this.Hostname,
221  hadoopLog);
222 
223  var mail = new EmailNotifier(dbManager);

655  private SshCommand RunCommand(string command)
656  {
657  using (this.client = new SshClient(this.sshConnectionInfo))
658  {
659  SshCommand sshCommand;
660 
661  try
662  {
663  this.client.Connect();
664  Log.InfoFormat(Properties.Resources.NUTCH_CONTROLLER_CLIENT_CONNECT, this.Hostname);
665 
666  sshCommand = this.client.RunCommand(command);
667  Log.InfoFormat(Properties.Resources.NUTCH_CONTROLLER_CLIENT_COMMAND_EXECUTED, command, this.Hostname);
668  }
669  catch (SocketException se)
670  {
671  Log.ErrorFormat(
672  Properties.Resources.NUTCH_CONTROLLER_SOCKET_EXCEPTION,
673  this.Hostname,
674  se);
675 
676  throw;
677  }
678  catch (SshAuthenticationException ae)
679  {
680  Log.ErrorFormat(
681  Properties.Resources.NUTCH_CONTROLLER_CLIENT_AUTHENTICATION_EXCEPTION,
682  this.Hostname,
683  ae);
684 
685  throw;
686  }
687  catch (SshConnectionException ce)
688  {
689  Log.ErrorFormat(
690  Properties.Resources.NUTCH_CONTROLLER_CLIENT_CONNECTION_EXEPTION,
691  this.Hostname,
692  ce);
693 
694  throw;
695  }
696  catch (SshOperationTimeoutException toe)
697  {
698  Log.ErrorFormat(
699  Properties.Resources.NUTCH_CONTROLLER_CLIENT_SSH_TIMEOUT_EXCEPTION,
700  this.Hostname,
701  toe);
702 
703  throw;
704  }
705  catch (Exception e)
706  {
707  Log.ErrorFormat(Properties.Resources.LOG_COMMON_ERROR, e);
708 
709  throw;
710  }
711  finally
712  {
713  this.Disconnect();
714  }
715 
716  return sshCommand;
717  }
718  }
