153  string continents = "from Continent";
154  int results = s.CreateQuery(continents).List().Count;
155  QueryStatistics continentStats = stats.GetQueryStatistics(continents);
156  Assert.IsNotNull(continentStats, "stats were null");
157  Assert.AreEqual(1, continentStats.ExecutionCount, "unexpected execution count");
158  Assert.AreEqual(results, continentStats.ExecutionRowCount, "unexpected row count");
159  var maxTime = continentStats.ExecutionMaxTime;
160  Assert.AreEqual(maxTime, stats.QueryExecutionMaxTime);
161  Assert.AreEqual( continents, stats.QueryExecutionMaxTimeQueryString );
162 
163  IEnumerable itr = s.CreateQuery(continents).Enumerable();
164  // Enumerable() should increment the execution count
165  Assert.AreEqual(2, continentStats.ExecutionCount, "unexpected execution count");
166  // but should not effect the cumulative row count
167  Assert.AreEqual(results, continentStats.ExecutionRowCount, "unexpected row count");
168  NHibernateUtil.Close(itr);
169  tx.Commit();
170  s.Close();
171 
172  // explicitly check that statistics for "split queries" get collected
173  // under the original query
174  stats.Clear();
175  s = OpenSession();

178  results = s.CreateQuery(localities).List().Count;
179  QueryStatistics localityStats = stats.GetQueryStatistics(localities);
180  Assert.IsNotNull(localityStats, "stats were null");
181  // ...one for each split query
182  Assert.AreEqual(2, localityStats.ExecutionCount, "unexpected execution count");
183  Assert.AreEqual(results, localityStats.ExecutionRowCount, "unexpected row count");
184  maxTime = localityStats.ExecutionMaxTime;
185  Assert.AreEqual(maxTime, stats.QueryExecutionMaxTime);
186  Assert.AreEqual( localities, stats.QueryExecutionMaxTimeQueryString );
187  tx.Commit();
188  s.Close();
189  Assert.IsFalse(s.IsOpen);
