70  public int AppendLoopOnAppenders(LoggingEvent loggingEvent)
71  {
72  if (loggingEvent == null)
73  {
74  throw new ArgumentNullException("loggingEvent");
75  }
76 
77  // m_appenderList is null when empty
78  if (m_appenderList == null)
79  {
80  return 0;
81  }
82 
83  if (m_appenderArray == null)
84  {
85  m_appenderArray = m_appenderList.ToArray();
86  }
87 
88  foreach(IAppender appender in m_appenderArray)
89  {
90  try
91  {
92  appender.DoAppend(loggingEvent);

94  catch(Exception ex)
95  {

113  public int AppendLoopOnAppenders(LoggingEvent[] loggingEvents)
114  {
115  if (loggingEvents == null)
116  {
117  throw new ArgumentNullException("loggingEvents");
118  }
119  if (loggingEvents.Length == 0)
120  {
121  throw new ArgumentException("loggingEvents array must not be empty", "loggingEvents");
122  }
123  if (loggingEvents.Length == 1)
124  {
125  // Fall back to single event path
126  return AppendLoopOnAppenders(loggingEvents[0]);
127  }
128 
129  // m_appenderList is null when empty
130  if (m_appenderList == null)
131  {
132  return 0;
133  }
134 
135  if (m_appenderArray == null)
136  {
137  m_appenderArray = m_appenderList.ToArray();
138  }
139 
140  foreach(IAppender appender in m_appenderArray)
141  {
142  try
143  {
144  CallAppend(appender, loggingEvents);
145  }
146  catch(Exception ex)
147  {
148  LogLog.Error(declaringType, "Failed to append to appender [" + appender.Name + "]", ex);
149  }
150  }
151  return m_appenderList.Count;
152  }

172  private static void CallAppend(IAppender appender, LoggingEvent[] loggingEvents)
173  {
