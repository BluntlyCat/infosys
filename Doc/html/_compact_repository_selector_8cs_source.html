172  public ILoggerRepository CreateRepository(Assembly assembly, Type repositoryType)
173  {
174  // If the type is not set then use the default type
175  if (repositoryType == null)
176  {
177  repositoryType = m_defaultRepositoryType;
178  }
179 
180  lock(this)
181  {
182  // This method should not throw if the default repository already exists.
183 
184  // First check that the repository does not exist
185  ILoggerRepository rep = m_name2repositoryMap[DefaultRepositoryName] as ILoggerRepository;
186  if (rep == null)
187  {
188  // Must create the repository
189  rep = CreateRepository(DefaultRepositoryName, repositoryType);
190  }
191 
192  return rep;
193  }

219  public ILoggerRepository CreateRepository(string repositoryName, Type repositoryType)
220  {
221  if (repositoryName == null)
222  {
223  throw new ArgumentNullException("repositoryName");
224  }
225 
226  // If the type is not set then use the default type
227  if (repositoryType == null)
228  {
229  repositoryType = m_defaultRepositoryType;
230  }
231 
232  lock(this)
233  {
234  ILoggerRepository rep = null;
235 
236  // First check that the repository does not exist
237  rep = m_name2repositoryMap[repositoryName] as ILoggerRepository;
238  if (rep != null)
239  {
240  throw new LogException("Repository ["+repositoryName+"] is already defined. Repositories cannot be redefined.");
241  }
242  else
243  {
244  LogLog.Debug(declaringType, "Creating repository ["+repositoryName+"] using type ["+repositoryType+"]");
245 
246  // Call the no arg constructor for the repositoryType
247  rep = (ILoggerRepository)Activator.CreateInstance(repositoryType);
248 
249  // Set the name of the repository
250  rep.Name = repositoryName;
251 
252  // Store in map
253  m_name2repositoryMap[repositoryName] = rep;
254 
255  // Notify listeners that the repository has been created
256  OnLoggerRepositoryCreatedEvent(rep);
257  }
258 
259  return rep;
260  }
