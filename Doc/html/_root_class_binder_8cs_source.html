16  public void Bind(HbmClass classSchema, IDictionary<string, MetaAttribute> inheritedMetas)
17  {
18  var rootClass = new RootClass();
19  BindClass(classSchema, rootClass, inheritedMetas);
20  // OPTIMISTIC LOCK MODE
21  rootClass.OptimisticLockMode = classSchema.optimisticlock.ToOptimisticLock();
22 
23  inheritedMetas = GetMetas(classSchema, inheritedMetas, true); // get meta's from <class>
24 
25  //TABLENAME
26  string schema = classSchema.schema ?? mappings.SchemaName;
27  string catalog = classSchema.catalog ?? mappings.CatalogName;
28  string tableName = GetClassTableName(rootClass, classSchema);
29  if (string.IsNullOrEmpty(tableName))
30  {
31  throw new MappingException(
32  string.Format(
33  "Could not determine the name of the table for entity '{0}'; remove the 'table' attribute or assign a value to it.",
34  rootClass.EntityName));
35  }
36 
37  Table table = mappings.AddTable(schema, catalog, tableName, classSchema.Subselect, rootClass.IsAbstract.GetValueOrDefault(), classSchema.schemaaction);
38  ((ITableOwner) rootClass).Table = table;
39 
40  log.InfoFormat("Mapping class: {0} -> {1}", rootClass.EntityName, rootClass.Table.Name);
41 
42  rootClass.IsMutable = classSchema.mutable;
43  rootClass.Where = classSchema.where ?? rootClass.Where;
44 
45  if (classSchema.check != null)
46  table.AddCheckConstraint(classSchema.check);
47 
48  rootClass.IsExplicitPolymorphism = classSchema.polymorphism == HbmPolymorphismType.Explicit;
49 
50  BindCache(classSchema.cache, rootClass);
51  new ClassIdBinder(this).BindId(classSchema.Id, rootClass, table);
52  new ClassCompositeIdBinder(this).BindCompositeId(classSchema.CompositeId, rootClass);
53  new ClassDiscriminatorBinder(rootClass, Mappings).BindDiscriminator(classSchema.discriminator, table);
54  BindTimestamp(classSchema.Timestamp, rootClass, table, inheritedMetas);
55  BindVersion(classSchema.Version, rootClass, table, inheritedMetas);
56 
57  rootClass.CreatePrimaryKey(dialect);

100  private void BindTimestamp(HbmTimestamp timestampSchema, PersistentClass rootClass, Table table, IDictionary<string, MetaAttribute> inheritedMetas)
101  {
102  if (timestampSchema == null)
103  return;
104 
105  string propertyName = timestampSchema.name;
106  var simpleValue = new SimpleValue(table);
107  new ColumnsBinder(simpleValue, Mappings).Bind(timestampSchema.Columns, false,
108  () =>
109  new HbmColumn
110  {name = mappings.NamingStrategy.PropertyToColumnName(propertyName)});
111 
112  if (!simpleValue.IsTypeSpecified)
113  {
114  switch (timestampSchema.source)
115  {
116  case HbmTimestampSource.Vm:
117  simpleValue.TypeName = NHibernateUtil.Timestamp.Name;
118  break;
119  case HbmTimestampSource.Db:
120  simpleValue.TypeName = NHibernateUtil.DbTimestamp.Name;
121  break;
122  default:
123  simpleValue.TypeName = NHibernateUtil.Timestamp.Name;
124  break;
125  }
126  }
127 
128  var property = new Property(simpleValue);
129  BindProperty(timestampSchema, property, inheritedMetas);
130 
131  // for version properties marked as being generated, make sure they are "always"
132  // generated; "insert" is invalid. This is dis-allowed by the schema, but just to make
133  // sure...
134 
135  if (property.Generation == PropertyGeneration.Insert)
136  throw new MappingException("'generated' attribute cannot be 'insert' for versioning property");
137  simpleValue.NullValue = timestampSchema.unsavedvalue == HbmTimestampUnsavedvalue.Null ? null : "undefined";
138  rootClass.Version = property;
139  rootClass.AddProperty(property);
140  }

142  private void BindProperty(HbmTimestamp timestampSchema, Property property, IDictionary<string, MetaAttribute> inheritedMetas)
143  {
144  property.Name = timestampSchema.name;
145 
146  if (property.Value.Type == null)
147  throw new MappingException("could not determine a property type for: " + property.Name);
148 
149  property.PropertyAccessorName = timestampSchema.access ?? mappings.DefaultAccess;
150  property.Cascade = mappings.DefaultCascade;
151  property.IsUpdateable = true;
152  property.IsInsertable = true;
153  property.IsOptimisticLocked = true;
154  property.Generation = Convert(timestampSchema.generated);
155 
156  if (property.Generation == PropertyGeneration.Always ||
157  property.Generation == PropertyGeneration.Insert)
158  {
159  // generated properties can *never* be insertable...
160  if (property.IsInsertable)
161  // insertable simply because that is the user did not specify
162  // anything; just override it
163  property.IsInsertable = false;
164 
165  // properties generated on update can never be updateable...
166  if (property.IsUpdateable && property.Generation == PropertyGeneration.Always)
167  // updateable only because the user did not specify
168  // anything; just override it
169  property.IsUpdateable = false;
170  }
171 
172  property.MetaAttributes = GetMetas(timestampSchema, inheritedMetas);
173 
174  property.LogMapped(log);
175  }
