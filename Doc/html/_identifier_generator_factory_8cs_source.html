295  public static System.Type GetIdentifierGeneratorClass(string strategy, Dialect.Dialect dialect)
296  {
297  System.Type clazz;
298  if ("native".Equals(strategy))
299  {
300  clazz = dialect.NativeIdentifierGeneratorClass;
301  }
302  else if ("identity".Equals(strategy))
303  {
304  clazz = dialect.IdentityStyleIdentifierGeneratorClass;
305  }
306  else
307  {
308  idgenerators.TryGetValue(strategy, out clazz);
309  }
310  try
311  {
312  if (clazz == null)
313  {
314  clazz = ReflectHelper.ClassForName(strategy);
315  }
316  }
317  catch (Exception)
318  {
319  throw new IdentifierGenerationException("Could not interpret id generator strategy: " + strategy);
320  }
