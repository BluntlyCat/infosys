43  private SqlString PageByLimitAndOffset(SqlString offset, SqlString limit)
44  {
45  var queryParser = new MsSqlSelectParser(_sourceQuery);
46  if (queryParser.SelectIndex < 0) return null;
47 
48  var result = new SqlStringBuilder();
49  BuildSelectClauseForPagingQuery(queryParser, limit, result);
50  if (queryParser.IsDistinct)
51  {
52  BuildFromClauseForPagingDistinctQuery(queryParser, result);
53  }
54  else

57  }

62  private static void BuildSelectClauseForPagingQuery(MsSqlSelectParser sqlQuery, SqlString limit, SqlStringBuilder result)
63  {
64  result.Add(sqlQuery.Sql.Substring(0, sqlQuery.SelectIndex));
65  result.Add("SELECT ");
66 
67  if (limit != null)
68  {
69  result.Add("TOP (").Add(limit).Add(") ");
70  }
71  else
72  {
73  // ORDER BY can only be used in subqueries if TOP is also specified.
74  result.Add("TOP (" + int.MaxValue + ") ");
75  }

87  private static void BuildFromClauseForPagingQuery(MsSqlSelectParser sqlQuery, SqlStringBuilder result)
88  {
89  result.Add(" FROM (")
90  .Add(sqlQuery.SelectClause)
91  .Add(", ROW_NUMBER() OVER(ORDER BY ");
92 
93  var orderIndex = 0;
94  foreach (var order in sqlQuery.Orders)
95  {
96  if (orderIndex++ > 0) result.Add(", ");
97  if (order.Column.Name != null)
98  {
99  result.Add(order.Column.Name);
100  }
101  else
102  {
103  result.Add(sqlQuery.Sql.Substring(order.Column.SqlIndex, order.Column.SqlLength).Trim());
104  }
105  if (order.IsDescending) result.Add(" DESC");
106  }
107 
108  if (orderIndex == 0)
109  {
110  result.Add("CURRENT_TIMESTAMP");
111  }

118  private static void BuildFromClauseForPagingDistinctQuery(MsSqlSelectParser sqlQuery, SqlStringBuilder result)
119  {
120  result.Add(" FROM (SELECT *, ROW_NUMBER() OVER(ORDER BY ");
121 
122  int orderIndex = 0;
123  foreach (var order in sqlQuery.Orders)
124  {
125  if (orderIndex++ > 0) result.Add(", ");
126  if (!order.Column.InSelectClause)
127  {
128  throw new HibernateException(
129  "The dialect was unable to perform paging of a statement that requires distinct results, and " +
130  "is ordered by a column that is not included in the result set of the query.");
131  }
132  result.Add("q_.").Add(order.Column.Alias);
133  if (order.IsDescending) result.Add(" DESC");
134  }
135  if (orderIndex == 0)
136  {
137  result.Add("CURRENT_TIMESTAMP");
138  }
139 
140  result.Add(") as __hibernate_sort_row FROM (")
141  .Add(sqlQuery.SelectClause)
142  .Add(" ")
143  .Add(sqlQuery.FromAndWhereClause)
144  .Add(") as q_) as query");
145  }
