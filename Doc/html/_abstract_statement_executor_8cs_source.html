57  protected SqlString GenerateIdInsertSelect(IQueryable persister, string tableAlias, IASTNode whereClause)
58  {
59  var select = new SqlSelectBuilder(Factory);
60  SelectFragment selectFragment = new SelectFragment(Factory.Dialect)
61  .AddColumns(tableAlias, persister.IdentifierColumnNames, persister.IdentifierColumnNames);
62  select.SetSelectClause(selectFragment.ToFragmentString().Substring(2));
63 
64  string rootTableName = persister.TableName;
65  SqlString fromJoinFragment = persister.FromJoinFragment(tableAlias, true, false);
66  select.SetFromClause(rootTableName + " " + tableAlias + fromJoinFragment);
67 
68  var whereJoinFragment = GetWhereJoinFragment(persister, tableAlias);
69 
70  SqlString userWhereClause = SqlString.Empty;
71  if (whereClause.ChildCount != 0)
72  {
73  // If a where clause was specified in the update/delete query, use it to limit the
74  // returned ids here...
75  try
76  {
77  var nodes = new CommonTreeNodeStream(whereClause);
78  var gen = new SqlGenerator(Factory, nodes);
79  gen.whereClause();
80  userWhereClause = gen.GetSQL().Substring(7);
81  }
82  catch (RecognitionException e)
83  {
84  throw new HibernateException("Unable to generate id select for DML operation", e);
85  }
86  if (whereJoinFragment.Length > 0)
87  {
88  whereJoinFragment.Append(" and ");
89  }
90  }
91 
92  select.SetWhereClause(whereJoinFragment + userWhereClause);
93 
94  var insert = new InsertSelect();
95  if (Factory.Settings.IsCommentsEnabled)
96  {
97  insert.SetComment("insert-select for " + persister.EntityName + " ids");
98  }
99  insert.SetTableName(persister.TemporaryIdTableName);
100  insert.SetSelect(select);
101  return insert.ToSqlString();
102  }
