66  public void Open(string path, FileAccess mode, uint timeout)
67  {
68  IntPtr nativeHandle;
69 
70  for (; ; )
71  {
72  NativeMethods.SecurityAttributes security = new NativeMethods.SecurityAttributes();
73  security.inheritHandle = true;
74  security.Length = Marshal.SizeOf(security);
75 
76  nativeHandle = NativeMethods.CreateFile(path, NativeMethods.GENERIC_READ | NativeMethods.GENERIC_WRITE,
77  0, security, NativeMethods.OPEN_EXISTING, NativeMethods.FILE_FLAG_OVERLAPPED, 0);
78 
79  if (nativeHandle != IntPtr.Zero)
80  break;
81 
82  if (Marshal.GetLastWin32Error() != ERROR_PIPE_BUSY)
83  {
84  throw new Win32Exception(Marshal.GetLastWin32Error(),
85  "Error opening pipe");
86  }
87  LowResolutionStopwatch sw = LowResolutionStopwatch.StartNew();
88  bool success = NativeMethods.WaitNamedPipe(path, timeout);
89  sw.Stop();
90  if (!success)
91  {
92  if (timeout < sw.ElapsedMilliseconds ||
93  Marshal.GetLastWin32Error() == ERROR_SEM_TIMEOUT)
94  {
95  throw new TimeoutException("Timeout waiting for named pipe");
96  }
97  else
98  {
99  throw new Win32Exception(Marshal.GetLastWin32Error(),
100  "Error waiting for pipe");
101  }
102  }
103  timeout -= (uint)sw.ElapsedMilliseconds;
104  }
105  handle = new SafeFileHandle(nativeHandle, true);
106  fileStream = new FileStream(handle, mode, 4096, true);
107  }

140  public override int Read(byte[] buffer, int offset, int count)
141  {
142  if (readTimeout == Timeout.Infinite)
143  {
144  return fileStream.Read(buffer, offset, count);
145  }
146  IAsyncResult result = fileStream.BeginRead(buffer, offset, count, null, null);
147  if (result.CompletedSynchronously)
148  return fileStream.EndRead(result);
149 
150  if (!result.AsyncWaitHandle.WaitOne(readTimeout))
151  {
152  CancelIo();
153  throw new TimeoutException("Timeout in named pipe read");
154  }
155  return fileStream.EndRead(result);
156  }
157

159  public override void Write(byte[] buffer, int offset, int count)
160  {
161  if (writeTimeout == Timeout.Infinite)
162  {
163  fileStream.Write(buffer, offset, count);
164  return;
165  }
166  IAsyncResult result = fileStream.BeginWrite(buffer, offset, count, null, null);
167  if (result.CompletedSynchronously)
168  {
169  fileStream.EndWrite(result);
170  }
171 
172  if (!result.AsyncWaitHandle.WaitOne(readTimeout))
173  {
174  CancelIo();
175  throw new TimeoutException("Timeout in named pipe write");
176  }
177  fileStream.EndWrite(result);
178  }

238  internal static Stream Create(string pipeName, string hostname, uint timeout)
239  {
240  string pipePath;
