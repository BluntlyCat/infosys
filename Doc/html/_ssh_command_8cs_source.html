215  public IAsyncResult BeginExecute(AsyncCallback callback, object state)
216  {
217  // Prevent from executing BeginExecute before calling EndExecute
218  if (this._asyncResult != null)
219  {
220  throw new InvalidOperationException("Asynchronous operation is already in progress.");
221  }
222 
223  // Create new AsyncResult object
224  this._asyncResult = new CommandAsyncResult(this)
225  {
226  AsyncWaitHandle = new ManualResetEvent(false),
227  IsCompleted = false,
228  AsyncState = state,
229  };
230 
231  // When command re-executed again, create a new channel
232  if (this._channel != null)
233  {
234  throw new SshException("Invalid operation.");
235  }

241
