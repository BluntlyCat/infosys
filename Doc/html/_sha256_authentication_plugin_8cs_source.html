63  public override object GetPassword()

113  private byte[] GetRsaPassword(string password, byte[] seedBytes)
114  {
115  // Obfuscate the plain text password with the session scramble
116  byte[] ofuscated = GetXor(this.Encoding.GetBytes(password), seedBytes);
117  // Encrypt the password and send it to the server
118  byte[] result = Encrypt(ofuscated, publicKey );
119  return result;
120  }

122  private byte[] GetXor( byte[] src, byte[] pattern )
123  {
124  byte[] src2 = new byte[src.Length + 1];
125  Array.Copy(src, 0, src2, 0, src.Length);
126  src2[src.Length] = 0;
127  byte[] result = new byte[src2.Length];
128  for (int i = 0; i < src2.Length; i++)
129  {
130  result[ i ] = ( byte )( src2[ i ] ^ ( pattern[ i % pattern.Length ] ));
131  }
132  return result;
133  }

136  {

138  c.Init(true, key);
139  byte[] result = c.DoFinal(data);
140  return result;
