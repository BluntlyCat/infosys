185  override public FilterDecision Decide(LoggingEvent loggingEvent)
186  {
187  if (loggingEvent == null)
188  {
189  throw new ArgumentNullException("loggingEvent");
190  }
191 
192  string msg = loggingEvent.RenderedMessage;
193 
194  // Check if we have been setup to filter
195  if (msg == null || (m_stringToMatch == null && m_regexToMatch == null))
196  {
197  // We cannot filter so allow the filter chain
198  // to continue processing
199  return FilterDecision.Neutral;
200  }
201 
202  // Firstly check if we are matching using a regex
203  if (m_regexToMatch != null)
204  {
205  // Check the regex
206  if (m_regexToMatch.Match(msg).Success == false)
207  {
208  // No match, continue processing
209  return FilterDecision.Neutral;
210  }
211 
212  // we've got a match
213  if (m_acceptOnMatch)
214  {
215  return FilterDecision.Accept;
216  }
217  return FilterDecision.Deny;
218  }
219  else if (m_stringToMatch != null)
220  {
221  // Check substring match
222  if (msg.IndexOf(m_stringToMatch) == -1)
223  {
224  // No match, continue processing
225  return FilterDecision.Neutral;
226  }
227 
228  // we've got a match
229  if (m_acceptOnMatch)
230  {
231  return FilterDecision.Accept;
232  }
233  return FilterDecision.Deny;
234  }
235  return FilterDecision.Neutral;
236 
237  }
