85  public IList List(ISessionImplementor session, QueryParameters queryParameters)
86  {
87  // Delegate to the QueryLoader...
88  ErrorIfDML();
89  var query = ( QueryNode ) _sqlAst;
90  bool hasLimit = queryParameters.RowSelection != null && queryParameters.RowSelection.DefinesLimits;
91  bool needsDistincting = ( query.GetSelectClause().IsDistinct || hasLimit ) && ContainsCollectionFetches;
92 
93  QueryParameters queryParametersToUse;
94 
95  if ( hasLimit && ContainsCollectionFetches )
96  {
97  log.Warn( "firstResult/maxResults specified with collection fetch; applying in memory!" );
98  var selection = new RowSelection
99  {
100  FetchSize = queryParameters.RowSelection.FetchSize,
101  Timeout = queryParameters.RowSelection.Timeout
102  };
103  queryParametersToUse = queryParameters.CreateCopyUsing( selection );
104  }
105  else
106  {
107  queryParametersToUse = queryParameters;
108  }
109 
110  IList results = _queryLoader.List(session, queryParametersToUse);
111 
112  if ( needsDistincting )
113  {
114  int includedCount = -1;
115  // NOTE : firstRow is zero-based
116  int first = !hasLimit || queryParameters.RowSelection.FirstRow == RowSelection.NoValue
117  ? 0
118  : queryParameters.RowSelection.FirstRow;
119  int max = !hasLimit || queryParameters.RowSelection.MaxRows == RowSelection.NoValue
120  ? -1
121  : queryParameters.RowSelection.MaxRows;
122 
123  int size = results.Count;
124  var tmp = new List<object>();
125  var distinction = new IdentitySet();
126 
127  for ( int i = 0; i < size; i++ )
128  {
129  object result = results[i];
130  if ( !distinction.Add(result ) )
131  {
132  continue;
133  }
134  includedCount++;
135  if ( includedCount < first )
136  {
137  continue;
138  }
139  tmp.Add( result );
140  // NOTE : ( max - 1 ) because first is zero-based while max is not...
141  if ( max >= 0 && ( includedCount - first ) >= ( max - 1 ) )
142  {
143  break;
144  }
145  }
146 
147  results = tmp;
148  }
149 
150  return results;
151  }

308  private void DoCompile(IDictionary<string, string> replacements, bool shallow, String collectionRole)
309  {
310  // If the query is already compiled, skip the compilation.
311  if ( _compiled )
312  {
313  if ( log.IsDebugEnabled )
314  {
315  log.Debug( "compile() : The query is already compiled, skipping..." );
316  }
317  return;
318  }
319 
320  // Remember the parameters for the compilation.
321  _tokenReplacements = replacements ?? new Dictionary<string, string>(1);
322 
323  _shallowQuery = shallow;
324 
325  try
326  {
327  // PHASE 1 : Analyze the HQL AST, and produce an SQL AST.
328  var translator = Analyze(collectionRole);
329 
330  _sqlAst = translator.SqlStatement;
331 
332  // at some point the generate phase needs to be moved out of here,
333  // because a single object-level DML might spawn multiple SQL DML
334  // command executions.
335  //
336  // Possible to just move the sql generation for dml stuff, but for
337  // consistency-sake probably best to just move responsiblity for
338  // the generation phase completely into the delegates
339  // (QueryLoader/StatementExecutor) themselves. Also, not sure why
340  // QueryLoader currently even has a dependency on this at all; does
341  // it need it? Ideally like to see the walker itself given to the delegates directly...
342 
343  if (_sqlAst.NeedsExecutor)
344  {
345  _statementExecutor = BuildAppropriateStatementExecutor(_sqlAst);
346  }
347  else
348  {
349  // PHASE 2 : Generate the SQL.
350  _generator = new HqlSqlGenerator(_sqlAst, _factory);
351  _generator.Generate();
352 
353  _queryLoader = new QueryLoader(this, _factory, _sqlAst.Walker.SelectClause);
354  }
355 
356  _compiled = true;
357  }
358  catch ( QueryException qe )
359  {
360  qe.QueryString = _queryIdentifier;
361  throw;
362  }
363  catch ( RecognitionException e )
364  {
365  // we do not actually propogate ANTLRExceptions as a cause, so
366  // log it here for diagnostic purposes
367  if ( log.IsInfoEnabled )
368  {
369  log.Info( "converted antlr.RecognitionException", e );
370  }
371  throw QuerySyntaxException.Convert(e, _queryIdentifier);
372  }
373 
374  _enabledFilters = null; //only needed during compilation phase...
375  }

377  private static IStatementExecutor BuildAppropriateStatementExecutor(IStatement statement)
378  {
379  HqlSqlWalker walker = statement.Walker;
380  if (walker.StatementType == HqlSqlWalker.DELETE)
381  {
382  FromElement fromElement = walker.GetFinalFromClause().GetFromElement();
383  IQueryable persister = fromElement.Queryable;
384  if (persister.IsMultiTable)
385  {
386  return new MultiTableDeleteExecutor(statement);
387  }
388  else
389  {
390  return new BasicExecutor(statement, persister);
391  }
392  }
393  else if (walker.StatementType == HqlSqlWalker.UPDATE)
394  {
395  FromElement fromElement = walker.GetFinalFromClause().GetFromElement();
396  IQueryable persister = fromElement.Queryable;
397  if (persister.IsMultiTable)
398  {
399  // even here, if only properties mapped to the "base table" are referenced
400  // in the set and where clauses, this could be handled by the BasicDelegate.
401  // TODO : decide if it is better performance-wise to perform that check, or to simply use the MultiTableUpdateDelegate
402  return new MultiTableUpdateExecutor(statement);
403  }
404  else
405  {
406  return new BasicExecutor(statement, persister);
407  }
408  }
409  else if (walker.StatementType == HqlSqlWalker.INSERT)
410  {
411  return new BasicExecutor(statement, ((InsertStatement)statement).IntoClause.Queryable);
412  }
413  else
414  {
415  throw new QueryException("Unexpected statement type");
416  }
417  }

456  var lex = new HqlLexer(new CaseInsensitiveStringStream(_hql));
457  _tokens = new CommonTokenStream(lex);

491  public void Visit(IASTNode node)
492  {

494  {
495  // we are already processing a dot-structure
496  if (ASTUtil.IsSubtreeChild(_dotRoot, node))

501 
502  // we are now at a new tree level

514  {
515  var expression = ASTUtil.GetPathText(dotStructureRoot);

517  var constant = ReflectHelper.GetConstantValue(expression, _sfi);
518 
519  if (constant != null)

522  dotStructureRoot.Type = HqlSqlWalker.JAVA_CONSTANT;
