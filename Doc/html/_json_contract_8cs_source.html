286  internal JsonContract(Type underlyingType)
287  {
288  ValidationUtils.ArgumentNotNull(underlyingType, "underlyingType");
289 
290  UnderlyingType = underlyingType;
291 
292  IsSealed = underlyingType.IsSealed();
293  IsInstantiable = !(underlyingType.IsInterface() || underlyingType.IsAbstract());
294 
295  IsNullable = ReflectionUtils.IsNullable(underlyingType);
296  NonNullableUnderlyingType = (IsNullable && ReflectionUtils.IsNullableType(underlyingType)) ? Nullable.GetUnderlyingType(underlyingType) : underlyingType;
297 
298  CreatedType = NonNullableUnderlyingType;
299 
300  IsConvertable = ConvertUtils.IsConvertible(NonNullableUnderlyingType);
301  IsEnum = NonNullableUnderlyingType.IsEnum();
302 
303  if (NonNullableUnderlyingType == typeof(byte[]))
304  {
305  InternalReadType = ReadType.ReadAsBytes;
306  }
307  else if (NonNullableUnderlyingType == typeof(int))
308  {
309  InternalReadType = ReadType.ReadAsInt32;
310  }
311  else if (NonNullableUnderlyingType == typeof(decimal))
312  {
313  InternalReadType = ReadType.ReadAsDecimal;
314  }
315  else if (NonNullableUnderlyingType == typeof(string))
316  {
317  InternalReadType = ReadType.ReadAsString;
318  }
319  else if (NonNullableUnderlyingType == typeof(DateTime))
320  {
321  InternalReadType = ReadType.ReadAsDateTime;
322  }
323 #if !NET20
324  else if (NonNullableUnderlyingType == typeof(DateTimeOffset))
325  {
326  InternalReadType = ReadType.ReadAsDateTimeOffset;
327  }
328 #endif
329  else
330  {
331  InternalReadType = ReadType.Read;
332  }
333  }
