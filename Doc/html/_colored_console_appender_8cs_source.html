271  [System.Security.Permissions.SecurityPermission(System.Security.Permissions.SecurityAction.Demand, UnmanagedCode = true)]
272  override protected void Append(log4net.Core.LoggingEvent loggingEvent)
273  {
274  if (m_consoleOutputWriter != null)
275  {
276  IntPtr consoleHandle = IntPtr.Zero;
277  if (m_writeToErrorStream)
278  {
279  // Write to the error stream
280  consoleHandle = GetStdHandle(STD_ERROR_HANDLE);
281  }
282  else
283  {
284  // Write to the output stream
285  consoleHandle = GetStdHandle(STD_OUTPUT_HANDLE);
286  }
287 
288  // Default to white on black
289  ushort colorInfo = (ushort)Colors.White;
290 
291  // see if there is a specified lookup
292  LevelColors levelColors = m_levelMapping.Lookup(loggingEvent.Level) as LevelColors;
293  if (levelColors != null)
294  {
295  colorInfo = levelColors.CombinedColor;
296  }
297 
298  // Render the event to a string
299  string strLoggingMessage = RenderLoggingEvent(loggingEvent);
300 
301  // get the current console color - to restore later
302  CONSOLE_SCREEN_BUFFER_INFO bufferInfo;
303  GetConsoleScreenBufferInfo(consoleHandle, out bufferInfo);
304 
305  // set the console colors
306  SetConsoleTextAttribute(consoleHandle, colorInfo);

308  // Using WriteConsoleW seems to be unreliable.
309  // If a large buffer is written, say 15,000 chars
310  // Followed by a larger buffer, say 20,000 chars
311  // then WriteConsoleW will fail, last error 8
312  // 'Not enough storage is available to process this command.'
313  //
314  // Although the documentation states that the buffer must
315  // be less that 64KB (i.e. 32,000 WCHARs) the longest string
316  // that I can write out a the first call to WriteConsoleW
317  // is only 30,704 chars.
318  //
319  // Unlike the WriteFile API the WriteConsoleW method does not
320  // seem to be able to partially write out from the input buffer.
321  // It does have a lpNumberOfCharsWritten parameter, but this is
322  // either the length of the input buffer if any output was written,
323  // or 0 when an error occurs.
324  //
325  // All results above were observed on Windows XP SP1 running
326  // .NET runtime 1.1 SP1.
327  //
328  // Old call to WriteConsoleW:
329  //
330  // WriteConsoleW(
331  // consoleHandle,
332  // strLoggingMessage,
333  // (UInt32)strLoggingMessage.Length,
334  // out (UInt32)ignoreWrittenCount,
335  // IntPtr.Zero);
336  //
337  // Instead of calling WriteConsoleW we use WriteFile which
338  // handles large buffers correctly. Because WriteFile does not
339  // handle the codepage conversion as WriteConsoleW does we
340  // need to use a System.IO.StreamWriter with the appropriate
341  // Encoding. The WriteFile calls are wrapped up in the
342  // System.IO.__ConsoleStream internal class obtained through
343  // the System.Console.OpenStandardOutput method.
344  //
345  // See the ActivateOptions method below for the code that
346  // retrieves and wraps the stream.
347 
348 
349  // The windows console uses ScrollConsoleScreenBuffer internally to
350  // scroll the console buffer when the display buffer of the console
351  // has been used up. ScrollConsoleScreenBuffer fills the area uncovered
352  // by moving the current content with the background color
353  // currently specified on the console. This means that it fills the
354  // whole line in front of the cursor position with the current
355  // background color.
356  // This causes an issue when writing out text with a non default
357  // background color. For example; We write a message with a Blue
358  // background color and the scrollable area of the console is full.
359  // When we write the newline at the end of the message the console
360  // needs to scroll the buffer to make space available for the new line.
361  // The ScrollConsoleScreenBuffer internals will fill the newly created
362  // space with the current background color: Blue.
363  // We then change the console color back to default (White text on a
364  // Black background). We write some text to the console, the text is
365  // written correctly in White with a Black background, however the
366  // remainder of the line still has a Blue background.
367  //
368  // This causes a disjointed appearance to the output where the background
369  // colors change.
370  //
371  // This can be remedied by restoring the console colors before causing
372  // the buffer to scroll, i.e. before writing the last newline. This does
373  // assume that the rendered message will end with a newline.
374  //
375  // Therefore we identify a trailing newline in the message and don't
376  // write this to the output, then we restore the console color and write
377  // a newline. Note that we must AutoFlush before we restore the console
378  // color otherwise we will have no effect.
379  //
380  // There will still be a slight artefact for the last line of the message
381  // will have the background extended to the end of the line, however this
382  // is unlikely to cause any user issues.
383  //
384  // Note that none of the above is visible while the console buffer is scrollable
385  // within the console window viewport, the effects only arise when the actual
386  // buffer is full and needs to be scrolled.
387 
388  char[] messageCharArray = strLoggingMessage.ToCharArray();
389  int arrayLength = messageCharArray.Length;
390  bool appendNewline = false;
391 
392  // Trim off last newline, if it exists
393  if (arrayLength > 1 && messageCharArray[arrayLength-2] == '\r' && messageCharArray[arrayLength-1] == '\n')
394  {
395  arrayLength -= 2;
396  appendNewline = true;
397  }
398 
399  // Write to the output stream
400  m_consoleOutputWriter.Write(messageCharArray, 0, arrayLength);
401 
402  // Restore the console back to its previous color scheme
403  SetConsoleTextAttribute(consoleHandle, bufferInfo.wAttributes);
404 
405  if (appendNewline)
406  {
407  // Write the newline, after changing the color scheme
408  m_consoleOutputWriter.Write(s_windowsNewline, 0, 2);
409  }
410  }
411  }

441  public override void ActivateOptions()
442  {
443  base.ActivateOptions();
444  m_levelMapping.ActivateOptions();
445 
446  System.IO.Stream consoleOutputStream = null;
447 
448  // Use the Console methods to open a Stream over the console std handle
449  if (m_writeToErrorStream)
450  {
451  // Write to the error stream
452  consoleOutputStream = Console.OpenStandardError();
453  }
454  else
455  {
456  // Write to the output stream
457  consoleOutputStream = Console.OpenStandardOutput();
458  }
459 
460  // Lookup the codepage encoding for the console
461  System.Text.Encoding consoleEncoding = System.Text.Encoding.GetEncoding(GetConsoleOutputCP());
462 
463  // Create a writer around the console stream
464  m_consoleOutputWriter = new System.IO.StreamWriter(consoleOutputStream, consoleEncoding, 0x100);
465 
466  m_consoleOutputWriter.AutoFlush = true;
467 
468  // SuppressFinalize on m_consoleOutputWriter because all it will do is flush
469  // and close the file handle. Because we have set AutoFlush the additional flush
470  // is not required. The console file handle should not be closed, so we don't call
471  // Dispose, Close or the finalizer.
472  GC.SuppressFinalize(m_consoleOutputWriter);
473  }
