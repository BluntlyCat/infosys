24  public MultiTableDeleteExecutor(IStatement statement)

32  var deleteStatement = (DeleteStatement) statement;
33 
34  FromElement fromElement = deleteStatement.FromClause.GetFromElement();
35  string bulkTargetAlias = fromElement.TableAlias;
36  persister = fromElement.Queryable;
37 
38  idInsertSelect = GenerateIdInsertSelect(persister, bulkTargetAlias, deleteStatement.WhereClause);
39  log.Debug("Generated ID-INSERT-SELECT SQL (multi-table delete) : " + idInsertSelect);
40 
41  string[] tableNames = persister.ConstraintOrderedTableNameClosure;
42  string[][] columnNames = persister.ContraintOrderedTableKeyColumnClosure;
43  string idSubselect = GenerateIdSubselect(persister);
44 
45  deletes = new SqlString[tableNames.Length];
46  for (int i = tableNames.Length - 1; i >= 0; i--)
47  {
48  // TODO : an optimization here would be to consider cascade deletes and not gen those delete statements;
49  // the difficulty is the ordering of the tables here vs the cascade attributes on the persisters ->
50  // the table info gotten here should really be self-contained (i.e., a class representation
51  // defining all the needed attributes), then we could then get an array of those
52  SqlDeleteBuilder delete = new SqlDeleteBuilder(Factory.Dialect, Factory)
53  .SetTableName(tableNames[i])
54  .SetWhere("(" + StringHelper.Join(", ", columnNames[i]) + ") IN (" + idSubselect + ")");
55  if (Factory.Settings.IsCommentsEnabled)
56  {
57  delete.SetComment("bulk delete");
58  }
59 
60  deletes[i] = delete.ToSqlString();
61  }
