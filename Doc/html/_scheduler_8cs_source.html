77  public void AddOrgUnitConfig(OrgUnitConfig orgUnitConfig)
78  {
79  if (orgUnitConfig.SchedulerActive)
80  {
81  Log.DebugFormat(Properties.Resources.LOG_SCHEDULER_ADD, orgUnitConfig);
82 
83  this.jobMutex.WaitOne();
84 
85  if (this.jobs.ContainsKey(orgUnitConfig.EntityId))
86  {
87  var job = this.UpdateJob(orgUnitConfig);
88 
89  if (job != null)
90  {
91  this.jobs[orgUnitConfig.EntityId].StopService(true);
92  this.jobs[orgUnitConfig.EntityId] = job;
93  this.jobs[orgUnitConfig.EntityId].StartService();
94  }
95  }
96  else
97  {
98  var job = this.SetNewJob(orgUnitConfig);
99 
100  if (job != null)
101  {
102  this.jobs.Add(orgUnitConfig.EntityId, job);
103  job.StartService();
104  }
105  }
106 
107  this.jobMutex.ReleaseMutex();
108  }
