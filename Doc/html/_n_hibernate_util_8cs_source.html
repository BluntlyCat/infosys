529  public static bool IsPropertyInitialized(object proxy, string propertyName)
530  {
531  object entity;
532  if (proxy.IsProxy())
533  {
534  ILazyInitializer li = ((INHibernateProxy)proxy).HibernateLazyInitializer;
535  if (li.IsUninitialized)
536  {
537  return false;
538  }
539  else
540  {

542  }
543  }
