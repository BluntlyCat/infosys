390  public bool Authenticate(Session session)
391  {
392  var authenticated = AuthenticationResult.Failure;
393 
394  if (session == null)
395  throw new ArgumentNullException("session");
396 
397  session.RegisterMessage("SSH_MSG_USERAUTH_FAILURE");
398  session.RegisterMessage("SSH_MSG_USERAUTH_SUCCESS");
399  session.RegisterMessage("SSH_MSG_USERAUTH_BANNER");
400 
401  session.UserAuthenticationBannerReceived += Session_UserAuthenticationBannerReceived;
402 
403  // Try to authenticate against none
404  var noneAuthenticationMethod = new NoneAuthenticationMethod(this.Username);
405 
406  authenticated = noneAuthenticationMethod.Authenticate(session);
407 
408  var allowedAuthentications = noneAuthenticationMethod.AllowedAuthentications;
409 
410  var triedAuthentications = new List<string>();
411  while (authenticated != AuthenticationResult.Success)
412  {
413  // Find first authentication method
414  var method = this.AuthenticationMethods.Where((a) => allowedAuthentications.Contains(a.Name) && !triedAuthentications.Contains(a.Name)).FirstOrDefault();
415  if (method == null)
416  throw new SshAuthenticationException("No suitable authentication method found to complete authentication.");
417 
418  triedAuthentications.Add(method.Name);
419 
420  authenticated = method.Authenticate(session);
421 
422  if (authenticated == AuthenticationResult.PartialSuccess || (method.AllowedAuthentications != null && method.AllowedAuthentications.Count() < allowedAuthentications.Count()))
423  {
424  // If further authentication is required then continue to try another method
425  allowedAuthentications = method.AllowedAuthentications;
426  continue;
427  }
428 
429  // If authentication Fail, and all the authentication have been tried.
430  if (authenticated == AuthenticationResult.Failure && (triedAuthentications.Count() == allowedAuthentications.Count()))
431  {
432  break;
433  }
434  }
435 
436  session.UserAuthenticationBannerReceived -= Session_UserAuthenticationBannerReceived;
437 
438  session.UnRegisterMessage("SSH_MSG_USERAUTH_FAILURE");
439  session.UnRegisterMessage("SSH_MSG_USERAUTH_SUCCESS");
440  session.UnRegisterMessage("SSH_MSG_USERAUTH_BANNER");
441 
442  this.IsAuthenticated = authenticated == AuthenticationResult.Success;
443 
444  return authenticated == AuthenticationResult.Success;
445  }
