145  private Country LoadCountryWithNativeSQL(Country country, string queryName)
146  {
147  // Ensure country is saved and session cache is empty to force from now on the reload of all
148  // persistence objects from the database.
149  using (ISession session = OpenSession())
150  {
151  using (ITransaction tx = session.BeginTransaction())
152  {
153  session.Save(country);
154  tx.Commit();
155  }
156  }
157  using (ISession session = OpenSession())
158  {
159  return session.GetNamedQuery(queryName).SetString("country_code", country.Code).UniqueResult<Country>();
160  }
