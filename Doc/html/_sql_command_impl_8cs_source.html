98  public void ResetParametersIndexesForTheCommand(int singleSqlParametersOffset)
99  {
100  // a better place could be the Bind of each IParameterSpecification but we have to do it before bind values
101  // in this way the same parameter of a dynamic-filter will be set with two different parameter-names in the same command (when it is a command-set).
102  if (singleSqlParametersOffset < 0)
103  {
104  throw new AssertionFailure("singleSqlParametersOffset < 0 - this indicate a bug in NHibernate ");
105  }
106  // due to IType.NullSafeSet(System.Data.IDbCommand , object, int, ISessionImplementor) the SqlType[] is supposed to be in a certain sequence.
107  // this mean that found the first location of a parameter for the IType span, the others are in secuence
108  foreach (IParameterSpecification specification in Specifications)
109  {
110  string firstParameterId = specification.GetIdsForBackTrack(factory).First();
111  int[] effectiveParameterLocations = SqlQueryParametersList.GetEffectiveParameterLocations(firstParameterId).ToArray();
112  if (effectiveParameterLocations.Length > 0) // Parameters previously present might have been removed from the SQL at a later point.
113  {
114  int firstParamNameIndex = effectiveParameterLocations.First() + singleSqlParametersOffset;
115  foreach (int location in effectiveParameterLocations)
116  {
117  int parameterSpan = specification.ExpectedType.GetColumnSpan(factory);
118  for (int j = 0; j < parameterSpan; j++)
119  {
120  sqlQueryParametersList[location + j].ParameterPosition = firstParamNameIndex + j;
121  }
122  }
123  }
124  }
125  }
