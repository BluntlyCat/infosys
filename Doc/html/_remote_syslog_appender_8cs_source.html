344  protected override void Append(LoggingEvent loggingEvent)
345  {
346  try
347  {
348  System.IO.StringWriter writer = new System.IO.StringWriter(System.Globalization.CultureInfo.InvariantCulture);
349 
350  // Priority
351  int priority = GeneratePriority(m_facility, GetSeverity(loggingEvent.Level));
352  writer.Write('<');
353  writer.Write(priority);
354  writer.Write('>');
355 
356  // Identity
357  if (m_identity != null)
358  {
359  m_identity.Format(writer, loggingEvent);
360  }
361  else
362  {
363  writer.Write(loggingEvent.Domain);
364  }
365  writer.Write(": ");
366 
367  // Message. The message goes after the tag/identity
368  RenderLoggingEvent(writer, loggingEvent);
369 
370  // Grab as a byte array
371  string fullMessage = writer.ToString();
372  Byte [] buffer = this.Encoding.GetBytes(fullMessage.ToCharArray());
373 
374  this.Client.Send(buffer, buffer.Length, this.RemoteEndPoint);
375  }
376  catch (Exception e)
377  {
378  ErrorHandler.Error(
379  "Unable to send logging event to remote syslog " +
380  this.RemoteAddress.ToString() +
381  " on port " +
382  this.RemotePort + ".",
383  e,
384  ErrorCode.WriteFailure);
385  }
386  }

416  virtual protected SyslogSeverity GetSeverity(Level level)
417  {
418  LevelSeverity levelSeverity = m_levelMapping.Lookup(level) as LevelSeverity;
419  if (levelSeverity != null)
420  {
421  return levelSeverity.Severity;
422  }
423 
424  //
425  // Fallback to sensible default values
426  //
427 
428  if (level >= Level.Alert)
429  {
430  return SyslogSeverity.Alert;
431  }
432  else if (level >= Level.Critical)
433  {
434  return SyslogSeverity.Critical;
435  }
436  else if (level >= Level.Error)
437  {
438  return SyslogSeverity.Error;
439  }
440  else if (level >= Level.Warn)
441  {
442  return SyslogSeverity.Warning;
443  }
444  else if (level >= Level.Notice)
445  {
446  return SyslogSeverity.Notice;
447  }
448  else if (level >= Level.Info)
449  {
450  return SyslogSeverity.Informational;
451  }

471  public static int GeneratePriority(SyslogFacility facility, SyslogSeverity severity)
472  {
473  if (facility < SyslogFacility.Kernel || facility > SyslogFacility.Local7)
474  {
475  throw new ArgumentException("SyslogFacility out of range", "facility");
476  }
477 
478  if (severity < SyslogSeverity.Emergency || severity > SyslogSeverity.Debug)
479  {
480  throw new ArgumentException("SyslogSeverity out of range", "severity");
481  }
482 
483  unchecked
484  {
485  return ((int)facility * 8) + (int)severity;
486  }
487  }
