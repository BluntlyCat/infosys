513  private static void ApplySerializerSettings(JsonSerializer serializer, JsonSerializerSettings settings)
514  {
515  if (!CollectionUtils.IsNullOrEmpty(settings.Converters))
516  {
517  // insert settings converters at the beginning so they take precedence
518  // if user wants to remove one of the default converters they will have to do it manually
519  for (int i = 0; i < settings.Converters.Count; i++)
520  {
521  serializer.Converters.Insert(i, settings.Converters[i]);
522  }
523  }
524 
525  // serializer specific
526  if (settings._typeNameHandling != null)
527  serializer.TypeNameHandling = settings.TypeNameHandling;
528  if (settings._typeNameAssemblyFormat != null)
529  serializer.TypeNameAssemblyFormat = settings.TypeNameAssemblyFormat;
530  if (settings._preserveReferencesHandling != null)
531  serializer.PreserveReferencesHandling = settings.PreserveReferencesHandling;
532  if (settings._referenceLoopHandling != null)
533  serializer.ReferenceLoopHandling = settings.ReferenceLoopHandling;
534  if (settings._missingMemberHandling != null)
535  serializer.MissingMemberHandling = settings.MissingMemberHandling;
536  if (settings._objectCreationHandling != null)
537  serializer.ObjectCreationHandling = settings.ObjectCreationHandling;
538  if (settings._nullValueHandling != null)
539  serializer.NullValueHandling = settings.NullValueHandling;
540  if (settings._defaultValueHandling != null)
541  serializer.DefaultValueHandling = settings.DefaultValueHandling;
542  if (settings._constructorHandling != null)
543  serializer.ConstructorHandling = settings.ConstructorHandling;
544  if (settings._context != null)
545  serializer.Context = settings.Context;
546  if (settings._checkAdditionalContent != null)
547  serializer._checkAdditionalContent = settings._checkAdditionalContent;
548 
549  if (settings.Error != null)
550  serializer.Error += settings.Error;
551 
552  if (settings.ContractResolver != null)
553  serializer.ContractResolver = settings.ContractResolver;
554  if (settings.ReferenceResolver != null)
555  serializer.ReferenceResolver = settings.ReferenceResolver;
556  if (settings.TraceWriter != null)
557  serializer.TraceWriter = settings.TraceWriter;
558  if (settings.Binder != null)
559  serializer.Binder = settings.Binder;
560 
561  // reader/writer specific
562  // unset values won't override reader/writer set values
563  if (settings._formatting != null)
564  serializer._formatting = settings._formatting;
565  if (settings._dateFormatHandling != null)
566  serializer._dateFormatHandling = settings._dateFormatHandling;
567  if (settings._dateTimeZoneHandling != null)
568  serializer._dateTimeZoneHandling = settings._dateTimeZoneHandling;
569  if (settings._dateParseHandling != null)
570  serializer._dateParseHandling = settings._dateParseHandling;
571  if (settings._dateFormatStringSet)
572  {
573  serializer._dateFormatString = settings._dateFormatString;
574  serializer._dateFormatStringSet = settings._dateFormatStringSet;
575  }
576  if (settings._floatFormatHandling != null)
577  serializer._floatFormatHandling = settings._floatFormatHandling;
578  if (settings._floatParseHandling != null)
579  serializer._floatParseHandling = settings._floatParseHandling;
580  if (settings._stringEscapeHandling != null)
581  serializer._stringEscapeHandling = settings._stringEscapeHandling;
582  if (settings._culture != null)
583  serializer._culture = settings._culture;
584  if (settings._maxDepthSet)

586  serializer._maxDepth = settings._maxDepth;

666  internal virtual object DeserializeInternal(JsonReader reader, Type objectType)
667  {
668  ValidationUtils.ArgumentNotNull(reader, "reader");
669 
670  // set serialization options onto reader
671  CultureInfo previousCulture = null;
672  if (_culture != null && !_culture.Equals(reader.Culture))
673  {
674  previousCulture = reader.Culture;
675  reader.Culture = _culture;
676  }
677 
678  DateTimeZoneHandling? previousDateTimeZoneHandling = null;
679  if (_dateTimeZoneHandling != null && reader.DateTimeZoneHandling != _dateTimeZoneHandling)
680  {
681  previousDateTimeZoneHandling = reader.DateTimeZoneHandling;
682  reader.DateTimeZoneHandling = _dateTimeZoneHandling.Value;
683  }
684 
685  DateParseHandling? previousDateParseHandling = null;
686  if (_dateParseHandling != null && reader.DateParseHandling != _dateParseHandling)
687  {
688  previousDateParseHandling = reader.DateParseHandling;
689  reader.DateParseHandling = _dateParseHandling.Value;
690  }
691 
692  FloatParseHandling? previousFloatParseHandling = null;
693  if (_floatParseHandling != null && reader.FloatParseHandling != _floatParseHandling)
694  {
695  previousFloatParseHandling = reader.FloatParseHandling;
696  reader.FloatParseHandling = _floatParseHandling.Value;
697  }
698 
699  int? previousMaxDepth = null;
700  if (_maxDepthSet && reader.MaxDepth != _maxDepth)
701  {
702  previousMaxDepth = reader.MaxDepth;
703  reader.MaxDepth = _maxDepth;
704  }
705 
706  TraceJsonReader traceJsonReader = (TraceWriter != null && TraceWriter.LevelFilter >= TraceLevel.Verbose)
707  ? new TraceJsonReader(reader)
708  : null;
709 
710  JsonSerializerInternalReader serializerReader = new JsonSerializerInternalReader(this);
711  object value = serializerReader.Deserialize(traceJsonReader ?? reader, objectType, CheckAdditionalContent);
712 
713  if (traceJsonReader != null)
714  TraceWriter.Trace(TraceLevel.Verbose, "Deserialized JSON: " + Environment.NewLine + traceJsonReader.GetJson(), null);
715 
716  // reset reader back to previous options
717  if (previousCulture != null)
718  reader.Culture = previousCulture;
719  if (previousDateTimeZoneHandling != null)
720  reader.DateTimeZoneHandling = previousDateTimeZoneHandling.Value;
721  if (previousDateParseHandling != null)
722  reader.DateParseHandling = previousDateParseHandling.Value;
723  if (previousFloatParseHandling != null)
724  reader.FloatParseHandling = previousFloatParseHandling.Value;
725  if (_maxDepthSet)
726  reader.MaxDepth = previousMaxDepth;
727 
728  return value;
729  }

879  internal static JsonConverter GetMatchingConverter(IList<JsonConverter> converters, Type objectType)
880  {
881 #if DEBUG
882  ValidationUtils.ArgumentNotNull(objectType, "objectType");
883 #endif
884 
885  if (converters != null)

890 
891  if (converter.CanConvert(objectType))
