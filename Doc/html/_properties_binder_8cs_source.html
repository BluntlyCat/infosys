256  private void BindValueProperty(HbmProperty propertyMapping, Property property)
257  {
258  property.IsUpdateable = propertyMapping.updateSpecified ? propertyMapping.update:true;
259  property.IsInsertable = propertyMapping.insertSpecified ? propertyMapping.insert:true;
260  PropertyGeneration generation;
261  switch (propertyMapping.generated)
262  {
263  case HbmPropertyGeneration.Never:
264  generation = PropertyGeneration.Never;
265  break;
266  case HbmPropertyGeneration.Insert:
267  generation = PropertyGeneration.Insert;
268  break;
269  case HbmPropertyGeneration.Always:
270  generation = PropertyGeneration.Always;
271  break;
272  default:
273  throw new ArgumentOutOfRangeException();
274  }
275  property.Generation = generation;
276 
277  if (generation == PropertyGeneration.Always || generation == PropertyGeneration.Insert)
278  {
279  // generated properties can *never* be insertable...
280  if (propertyMapping.insertSpecified && property.IsInsertable)
281  {
282  // the user specifically supplied insert="true",
283  // which constitutes an illegal combo
284  throw new MappingException("cannot specify both insert=\"true\" and generated=\"" + generation
285  + "\" for property: " + propertyMapping.Name);
286  }
287  else
288  {
289  property.IsInsertable = false;
290  }
291 
292  // properties generated on update can never be updateable...
293  if (propertyMapping.updateSpecified && property.IsUpdateable && generation == PropertyGeneration.Always)
294  {
295  // the user specifically supplied update="true",
296  // which constitutes an illegal combo
297  throw new MappingException("cannot specify both update=\"true\" and generated=\"" + generation
298  + "\" for property: " + propertyMapping.Name);
299  }
300  else
301  {
302  property.IsUpdateable = false;
303  }
304  }
305  }
