91  internal MySqlGeometry(MySqlDbType type, Double xValue, Double yValue, int srid)
92  {
93  this._type = type;
94  this._xValue = xValue;
95  this._yValue = yValue;
96  this._isNull = false;
97  this._srid = srid;
98  this._valBinary = new byte[GEOMETRY_LENGTH];
99 
100  byte[] sridBinary = BitConverter.GetBytes(srid);
101 
102  for (int i = 0; i < sridBinary.Length; i++)
103  _valBinary[i] = sridBinary[i];
104 
105  long xVal = BitConverter.DoubleToInt64Bits(xValue);
106  long yVal = BitConverter.DoubleToInt64Bits(yValue);
107 
108  _valBinary[4] = 1;
109  _valBinary[5] = 1;
110 
111  for (int i = 0; i < 8; i++)
112  {
113  _valBinary[i + 9] = (byte)(xVal & 0xff);
114  xVal >>= 8;
115  }
116 
117  for (int i = 0; i < 8; i++)
118  {
119  _valBinary[i + 17] = (byte)(yVal & 0xff);
120  yVal >>= 8;
121  }
122  }

124  public MySqlGeometry(MySqlDbType type, byte[] val)
125  {
126 
127  if (val == null)
128  throw new ArgumentNullException("val");
129 
130  byte[] buffValue = new byte[val.Length];
131 
132  for (int i = 0; i < val.Length; i++)
133  buffValue[i] = val[i];
134 
135  var xIndex = val.Length == GEOMETRY_LENGTH ? 9 : 5;
136  var yIndex = val.Length == GEOMETRY_LENGTH ? 17 : 13;
137 
138  _valBinary = buffValue;
139  _xValue = BitConverter.ToDouble(val, xIndex);
140  _yValue = BitConverter.ToDouble(val, yIndex);
141  this._srid = val.Length == GEOMETRY_LENGTH ? BitConverter.ToInt32(val, 0) : 0;
142  this._isNull = false;
143  this._type = type;
144  }

188  void IMySqlValue.WriteValue(MySqlPacket packet, bool binary, object val, int length)
189  {
190  byte[] buffToWrite = null;
191 
192  try
193  {
194  buffToWrite = ((MySqlGeometry)val)._valBinary;
195  }
196  catch
197  {
198  buffToWrite = val as Byte[];
199  }
200 
201  if (buffToWrite == null)
202  {
203  MySqlGeometry v = new MySqlGeometry(0, 0);
204  MySqlGeometry.TryParse(val.ToString(), out v);
205  buffToWrite = v._valBinary;
206  }
207 
208  byte[] result = new byte[GEOMETRY_LENGTH];
209 
210  for (int i = 0; i < buffToWrite.Length; i++)
211  {
212  if (buffToWrite.Length < GEOMETRY_LENGTH)
213  result[i + 4] = buffToWrite[i];
214  else
215  result[i] = buffToWrite[i];
216  }
217 
218  packet.WriteStringNoNull("_binary ");
219  packet.WriteByte((byte)'\'');
220  EscapeByteArray(result, GEOMETRY_LENGTH, packet);
221  packet.WriteByte((byte)'\'');
222  }

224  private static void EscapeByteArray(byte[] bytes, int length, MySqlPacket packet)
225  {
226  for (int x = 0; x < length; x++)
227  {
228  byte b = bytes[x];
229  if (b == '\0')
230  {
231  packet.WriteByte((byte)'\\');
232  packet.WriteByte((byte)'0');
233  }
234 
235  else if (b == '\\' || b == '\'' || b == '\"')
236  {
237  packet.WriteByte((byte)'\\');
238  packet.WriteByte(b);
239  }
240  else
241  packet.WriteByte(b);
242  }
243  }

287  public static MySqlGeometry Parse(string value)
288  {
289  if (String.IsNullOrEmpty(value))
290  throw new ArgumentNullException("value");

292  if (!(value.Contains("SRID") || value.Contains("POINT(") || value.Contains("POINT (")))
293  throw new FormatException("String does not contain a valid geometry value");

306  public static bool TryParse(string value, out MySqlGeometry mySqlGeometryValue)
307  {
308  string[] arrayResult = new string[0];
309  string strResult = string.Empty;
310  bool hasX = false;
311  bool hasY = false;
312  Double xVal = 0;
313  Double yVal = 0;
314  int sridValue = 0;
315 
316  try
317  {
318  if (value.Contains(";"))
319  arrayResult = value.Split(';');
320  else
321  strResult = value;
322 
323  if (arrayResult.Length > 1 || strResult != String.Empty)
324  {
325  string point = strResult != String.Empty ? strResult : arrayResult[1];
326  point = point.Replace("POINT (", "").Replace("POINT(", "").Replace(")", "");
327  var coord = point.Split(' ');
328  if (coord.Length > 1)
329  {
330  hasX = Double.TryParse(coord[0], out xVal);
331  hasY = Double.TryParse(coord[1], out yVal);
332  }
333  if (arrayResult.Length >= 1)
334  Int32.TryParse(arrayResult[0].Replace("SRID=", ""), out sridValue);
335  }
336  if (hasX && hasY)
337  {
338  mySqlGeometryValue = new MySqlGeometry(xVal, yVal, sridValue);
339  return true;
340  }
341  }
342  catch
343  { }
344 
345  mySqlGeometryValue = new MySqlGeometry(true);
346  return false;
347  }
