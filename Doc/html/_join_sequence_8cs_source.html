139  public JoinFragment ToJoinFragment(
140  IDictionary<string, IFilter> enabledFilters,
141  bool includeExtraJoins,
142  SqlString withClauseFragment,
143  string withClauseJoinAlias)
144  {
145  QueryJoinFragment joinFragment = new QueryJoinFragment(factory.Dialect, useThetaStyle);
146  if (rootJoinable != null)
147  {
148  joinFragment.AddCrossJoin(rootJoinable.TableName, rootAlias);
149  string filterCondition = rootJoinable.FilterFragment(rootAlias, enabledFilters);
150  // JoinProcessor needs to know if the where clause fragment came from a dynamic filter or not so it
151  // can put the where clause fragment in the right place in the SQL AST. 'hasFilterCondition' keeps track
152  // of that fact.
153  joinFragment.HasFilterCondition = joinFragment.AddCondition(filterCondition);
154  if (includeExtraJoins)
155  {
156  //TODO: not quite sure about the full implications of this!
157  AddExtraJoins(joinFragment, rootAlias, rootJoinable, true);
158  }
159  }
160 
161  IJoinable last = rootJoinable;
162 
163  for (int i = 0; i < joins.Count; i++)
164  {
165  Join join = joins[i];
166  string on = join.AssociationType.GetOnCondition(join.Alias, factory, enabledFilters);
167  SqlString condition = new SqlString();
168  if (last != null &&
169  IsManyToManyRoot(last) &&
170  ((IQueryableCollection)last).ElementType == join.AssociationType)
171  {
172  // the current join represents the join between a many-to-many association table
173  // and its "target" table. Here we need to apply any additional filters
174  // defined specifically on the many-to-many
175  string manyToManyFilter = ((IQueryableCollection)last)
176  .GetManyToManyFilterFragment(join.Alias, enabledFilters);
177  condition = new SqlString("".Equals(manyToManyFilter)
178  ? on
179  : "".Equals(on)
180  ? manyToManyFilter
181  : on + " and " + manyToManyFilter);
182  }
183  else
184  {
185  // NH Different behavior : NH1179 and NH1293
186  // Apply filters in Many-To-One association
187  var enabledForManyToOne = FilterHelper.GetEnabledForManyToOne(enabledFilters);
188  condition = new SqlString(string.IsNullOrEmpty(on) && enabledForManyToOne.Count > 0
189  ? join.Joinable.FilterFragment(join.Alias, enabledForManyToOne)
190  : on);
191  }
192 
193  if (withClauseFragment != null)
194  {
195  if (join.Alias.Equals(withClauseJoinAlias))
196  {
197  condition = condition.Append(" and ").Append(withClauseFragment);
198  }
199  }
200 
201  // NH: the variable "condition" have to be a SqlString because it may contains Parameter instances with BackTrack
202  joinFragment.AddJoin(
203  join.Joinable.TableName,
204  join.Alias,
205  join.LHSColumns,
206  JoinHelper.GetRHSColumnNames(join.AssociationType, factory),
207  join.JoinType,
208  condition
209  );
210  if (includeExtraJoins)
211  {
212  //TODO: not quite sure about the full implications of this!
213  AddExtraJoins(joinFragment, join.Alias, join.Joinable, join.JoinType == JoinType.InnerJoin);
214  }
215  last = join.Joinable;
216  }
217  if (next != null)
218  {
219  joinFragment.AddFragment(next.ToJoinFragment(enabledFilters, includeExtraJoins));
220  }
221  joinFragment.AddCondition(conditions.ToSqlString());
222  if (isFromPart)
223  joinFragment.ClearWherePart();
224  return joinFragment;
225  }
