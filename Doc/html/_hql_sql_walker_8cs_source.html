447  void ProcessQuery(IASTNode select, IASTNode query)
448  {
449  if ( log.IsDebugEnabled ) {
450  log.Debug( "processQuery() : " + query.ToStringTree() );
451  }
452 
453  try {
454  QueryNode qn = ( QueryNode ) query;
455 
456  // Was there an explicit select expression?
457  bool explicitSelect = select != null && select.ChildCount > 0;
458 
459  if ( !explicitSelect ) {
460  // No explicit select expression; render the id and properties
461  // projection lists for every persister in the from clause into
462  // a single 'token node'.
463  //TODO: the only reason we need this stuff now is collection filters,
464  // we should get rid of derived select clause completely!
465  CreateSelectClauseFromFromClause( qn );
466  }
467  else {
468  // Use the explicitly declared select expression; determine the
469  // return types indicated by each select token
470  UseSelectClause( select );
471  }
472 
473  // After that, process the JOINs.
474  // Invoke a delegate to do the work, as this is farily complex.
475  JoinProcessor joinProcessor = new JoinProcessor( this );
476  joinProcessor.ProcessJoins( qn );
477 
478  // Attach any mapping-defined "ORDER BY" fragments
479  foreach (FromElement fromElement in qn.FromClause.GetProjectionList())
480  {
481  if ( fromElement.IsFetch && fromElement.QueryableCollection != null )
482  {
483  // Does the collection referenced by this FromElement
484  // specify an order-by attribute? If so, attach it to
485  // the query's order-by
486  if ( fromElement.QueryableCollection.HasOrdering)
487  {
488  string orderByFragment = fromElement
489  .QueryableCollection
490  .GetSQLOrderByString( fromElement.TableAlias );
491  qn.GetOrderByClause().AddOrderFragment( orderByFragment );
492  }
493  if ( fromElement.QueryableCollection.HasManyToManyOrdering )
494  {
495  string orderByFragment = fromElement.QueryableCollection
496  .GetManyToManyOrderByString( fromElement.TableAlias );
497  qn.GetOrderByClause().AddOrderFragment( orderByFragment );
498  }
499  }
500  }
501  }
502  finally
503  {
504  PopFromClause();
505  }
506  }

793  protected IASTNode LookupProperty(IASTNode dot, bool root, bool inSelect)
794  {
795  DotNode dotNode = (DotNode) dot;
796  FromReferenceNode lhs = dotNode.GetLhs();
797  IASTNode rhs = lhs.NextSibling;
798  switch (rhs.Type)
799  {
800  case ELEMENTS:
801  case INDICES:
802  if (log.IsDebugEnabled)
803  {
804  log.Debug("lookupProperty() " + dotNode.Path + " => " + rhs.Text + "(" + lhs.Path + ")");
805  }
806 
807  CollectionFunction f = (CollectionFunction) rhs;
808  // Re-arrange the tree so that the collection function is the root and the lhs is the path.
809 
810  f.SetFirstChild(lhs);
811  lhs.NextSibling = null;
812  dotNode.SetFirstChild(f);
813 
814  Resolve(lhs); // Don't forget to resolve the argument!
815  f.Resolve(inSelect); // Resolve the collection function now.
816  return f;
817  default:
818  // Resolve everything up to this dot, but don't resolve the placeholders yet.
819  dotNode.ResolveFirstChild();
820  return dotNode;
821  }

1058  {

1060  {
1061  ITreeNodeStream old = input;
1062  input = new CommonTreeNodeStream(adaptor, hqlWithNode);
1063

1065  input = old;
1066

1068  {
1069  log.Debug("handleWithFragment() : " + _printer.ShowAsString(hqlSqlWithNode, "-- with clause --"));
1070  }

1082  sql.whereExpr();

1113  public void Visit(IASTNode node)
1114  {
1115  // todo : currently expects that the individual with expressions apply to the same sql table join.
1116  // This may not be the case for joined-subclass where the property values
1117  // might be coming from different tables in the joined hierarchy. At some
1118  // point we should expand this to support that capability. However, that has
1119  // some difficulties:
1120  // 1) the biggest is how to handle ORs when the individual comparisons are
1121  // linked to different sql joins.
1122  // 2) here we would need to track each comparison individually, along with
1123  // the join alias to which it applies and then pass that information
1124  // back to the FromElement so it can pass it along to the JoinSequence
1125  if ( node is DotNode )
1126  {
1127  DotNode dotNode = ( DotNode ) node;
1128  FromElement fromElement = dotNode.FromElement;
1129  if ( _referencedFromElement != null )
1130  {
1131  if ( fromElement != _referencedFromElement )
1132  {
1133  throw new HibernateException( "with-clause referenced two different from-clause elements" );
1134  }
1135  }
1136  else
1137  {
1138  _referencedFromElement = fromElement;
1139  _joinAlias = ExtractAppliedAlias( dotNode );
1140 
1141  // todo : temporary
1142  // needed because currently persister is the one that
1143  // creates and renders the join fragments for inheritence
1144  // hierarchies...
1145  if ( _joinAlias != _referencedFromElement.TableAlias)
1146  {
1147  throw new InvalidWithClauseException( "with clause can only reference columns in the driving table" );
1148  }
1149  }
1150  }
1151  else if ( node is ParameterNode )
1152  {
1153  ApplyParameterSpecification(((ParameterNode) node).HqlParameterSpecification);
1154  }
1155  else if ( node is IParameterContainer )
1156  {
1157  ApplyParameterSpecifications( ( IParameterContainer ) node );
1158  }
1159  }
