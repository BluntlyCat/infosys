202  public SqlString ToSqlString()
203  {
204  // 4 = the "SELECT", selectClause, "FROM", fromClause are straight strings
205  // plus the number of parts in outerJoinsAfterFrom SqlString.
206  // 1 = the "WHERE"
207  // plus the number of parts in outerJoinsAfterWhere SqlString.
208  // 1 = the whereClause
209  // 2 = the "ORDER BY" and orderByClause
210  var joinAfterFrom = outerJoinsAfterFrom != null ? outerJoinsAfterFrom.Count : 0;
211  var joinAfterWhere = outerJoinsAfterWhere != null ? outerJoinsAfterWhere.Count : 0;
212  int initialCapacity = 4 + joinAfterFrom + 1 + joinAfterWhere + 1 + 2;
213  if (!string.IsNullOrEmpty(comment))
214  initialCapacity++;
215 
216  SqlStringBuilder sqlBuilder = new SqlStringBuilder(initialCapacity + 2);
217  if (!string.IsNullOrEmpty(comment))
218  sqlBuilder.Add("/* " + comment + " */ ");
219 
220  sqlBuilder.Add("SELECT ")
221  .Add(selectClause)
222  .Add(" FROM ")
223  .Add(fromClause);
224 
225  if (SqlStringHelper.IsNotEmpty(outerJoinsAfterFrom))
226  {
227  sqlBuilder.Add(outerJoinsAfterFrom);
228  }
229 
230  if (SqlStringHelper.IsNotEmpty(whereClause) || SqlStringHelper.IsNotEmpty(outerJoinsAfterWhere))
231  {
232  sqlBuilder.Add(" WHERE ");
233  // the outerJoinsAfterWhere needs to come before where clause to properly
234  // handle dynamic filters
235  if (SqlStringHelper.IsNotEmpty(outerJoinsAfterWhere))
236  {
237  sqlBuilder.Add(outerJoinsAfterWhere);
238  if (SqlStringHelper.IsNotEmpty(whereClause))
239  {
240  sqlBuilder.Add(" AND ");
241  }
242  }
243 
244  if (SqlStringHelper.IsNotEmpty(whereClause))
245  {
246  sqlBuilder.Add(whereClause);
247  }
248  }
249 
250  if (StringHelper.IsNotEmpty(groupByClause))
251  {
252  sqlBuilder.Add(" GROUP BY ")
253  .Add(groupByClause);
254  }
255 
256  if(SqlStringHelper.IsNotEmpty(havingClause))
257  {
258  sqlBuilder.Add(" HAVING ")
259  .Add(havingClause);
260  }
261 
262  if (SqlStringHelper.IsNotEmpty(orderByClause))
263  {
264  sqlBuilder.Add(" ORDER BY ")
265  .Add(orderByClause);
266  }
267 
268  if (lockMode != null)
269  {
270  sqlBuilder.Add(Dialect.GetForUpdateString(lockMode));
271  }
272 
273  if (log.IsDebugEnabled)
274  {
275  if (initialCapacity < sqlBuilder.Count)
276  {
277  log.Debug(
278  "The initial capacity was set too low at: " + initialCapacity + " for the SelectSqlBuilder " +
279  "that needed a capacity of: " + sqlBuilder.Count + " for the table " + fromClause);
280  }
281  else if (initialCapacity > 16 && ((float) initialCapacity / sqlBuilder.Count) > 1.2)
282  {
283  log.Debug(
284  "The initial capacity was set too high at: " + initialCapacity + " for the SelectSqlBuilder " +
285  "that needed a capacity of: " + sqlBuilder.Count + " for the table " + fromClause);
286  }
287  }
288 
289  return sqlBuilder.ToSqlString();
290  }
