62  public static DBVersion Parse(string versionString)
63  {
64  int start = 0;
65  int index = versionString.IndexOf('.', start);
66  if (index == -1)
67  throw new MySqlException(Resources.BadVersionFormat);
68  string val = versionString.Substring(start, index - start).Trim();
69  int major = Convert.ToInt32(val, System.Globalization.NumberFormatInfo.InvariantInfo);
70 
71  start = index + 1;
72  index = versionString.IndexOf('.', start);
73  if (index == -1)
74  throw new MySqlException(Resources.BadVersionFormat);
75  val = versionString.Substring(start, index - start).Trim();
76  int minor = Convert.ToInt32(val, System.Globalization.NumberFormatInfo.InvariantInfo);
77 
78  start = index + 1;
79  int i = start;
80  while (i < versionString.Length && Char.IsDigit(versionString, i))
81  i++;
82  val = versionString.Substring(start, i - start).Trim();
83  int build = Convert.ToInt32(val, System.Globalization.NumberFormatInfo.InvariantInfo);
84 
85  return new DBVersion(versionString, major, minor, build);
86  }
