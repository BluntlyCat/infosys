135  public void ProcessConstant(SqlNode constant, bool resolveIdent)
136  {
137  // If the constant is an IDENT, figure out what it means...
138  bool isIdent = (constant.Type == HqlSqlWalker.IDENT || constant.Type == HqlSqlWalker.WEIRD_IDENT);
139 
140  if (resolveIdent && isIdent && IsAlias(constant.Text))
141  {
142  // IDENT is a class alias in the FROM.
143  IdentNode ident = (IdentNode)constant;
144  // Resolve to an identity column.
145  ident.Resolve(false, true);
146  }
147  else
148  {
149  // IDENT might be the name of a class.
150  IQueryable queryable = _walker.SessionFactoryHelper.FindQueryableUsingImports(constant.Text);
151  if (isIdent && queryable != null)
152  {
153  constant.Text = queryable.DiscriminatorSQLValue;
154  }
155  // Otherwise, it's a literal.
156  else
157  {
158  ProcessLiteral(constant);
159  }
160  }

180  private static string GetLiteralValue(string text, int type)
181  {
182  string literalValue = text;
183  if (type == HqlSqlWalker.NUM_FLOAT)
184  {
185  if (literalValue.EndsWith("f", StringComparison.OrdinalIgnoreCase))
186  {
187  return literalValue.Substring(0, literalValue.Length - 1);
188  }
189  }
190  else if (type == HqlSqlWalker.NUM_DOUBLE)
191  {
192  if (literalValue.EndsWith("d", StringComparison.OrdinalIgnoreCase))
193  {
194  return literalValue.Substring(0, literalValue.Length - 1);
195  }
196  }
197  else if (type == HqlSqlWalker.NUM_DECIMAL)
198  {
199  if (literalValue.EndsWith("m", StringComparison.OrdinalIgnoreCase))
200  {
201  return literalValue.Substring(0, literalValue.Length - 1);
202  }
203  }
204  return literalValue;
205  }

207  private static string DetermineIntegerRepresentation(string text, int type)
208  {
209  // prevent the fisrt-exception as possible
210  var literalValue = text;
211  bool hasLongSpec = literalValue.EndsWith("l", StringComparison.OrdinalIgnoreCase);
212  if (hasLongSpec)
213  {
214  literalValue = literalValue.Substring(0, literalValue.Length - 1);
215  }
216  try
217  {
218  if (type == HqlSqlWalker.NUM_INT && literalValue.Length <= 10 || hasLongSpec)
219  {
220  try
221  {
222  return int.Parse(literalValue).ToString();
223  }
224  catch (FormatException)
225  {
226  log.Info("could not format incoming text [" + text
227  + "] as a NUM_INT; assuming numeric overflow and attempting as NUM_LONG");
228  }
229  catch (OverflowException)
230  {
231  log.Info("could not format incoming text [" + text
232  + "] as a NUM_INT; assuming numeric overflow and attempting as NUM_LONG");
233  }
234  }
235 
236  return long.Parse(literalValue).ToString();
237  }
238  catch (Exception t)
239  {
240  throw new HibernateException("Could not parse literal [" + text + "] as integer", t);
241  }
242  }

309  }

342  }
