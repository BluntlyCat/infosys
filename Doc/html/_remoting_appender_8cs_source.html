226  private void BeginAsyncSend()
227  {
228  // The work queue is not empty
229  m_workQueueEmptyEvent.Reset();
230 
231  // Increment the queued count
232  Interlocked.Increment(ref m_queuedCallbackCount);
233  }

238  private void EndAsyncSend()
239  {
240  // Decrement the queued count
241  if (Interlocked.Decrement(ref m_queuedCallbackCount) <= 0)
242  {
243  // If the work queue is empty then set the event
244  m_workQueueEmptyEvent.Set();
245  }
246  }
