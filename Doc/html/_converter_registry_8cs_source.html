213  private static object GetConverterFromAttribute(Type destinationType)
214  {
215  // Look for an attribute on the destination type
216  object[] attributes = destinationType.GetCustomAttributes(typeof(TypeConverterAttribute), true);
217  if (attributes != null && attributes.Length > 0)
218  {
219  TypeConverterAttribute tcAttr = attributes[0] as TypeConverterAttribute;
220  if (tcAttr != null)
221  {
222  Type converterType = SystemInfo.GetTypeFromString(destinationType, tcAttr.ConverterTypeName, false, true);
223  return CreateConverterInstance(converterType);
224  }
225  }
226 
227  // Not found converter using attributes
228  return null;
229  }

246  private static object CreateConverterInstance(Type converterType)
247  {
248  if (converterType == null)
249  {
250  throw new ArgumentNullException("converterType", "CreateConverterInstance cannot create instance, converterType is null");
251  }
252 
253  // Check type is a converter
254  if (typeof(IConvertFrom).IsAssignableFrom(converterType) || typeof(IConvertTo).IsAssignableFrom(converterType))
255  {
256  try
257  {
258  // Create the type converter
259  return Activator.CreateInstance(converterType);
260  }
261  catch(Exception ex)
262  {
263  LogLog.Error(declaringType, "Cannot CreateConverterInstance of type ["+converterType.FullName+"], Exception in call to Activator.CreateInstance", ex);
264  }
265  }
266  else
267  {
268  LogLog.Error(declaringType, "Cannot CreateConverterInstance of type ["+converterType.FullName+"], type does not implement IConvertFrom or IConvertTo");
269  }
270  return null;
271  }
