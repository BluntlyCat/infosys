153  protected override int DoUpdateRows(object id, IPersistentCollection collection, ISessionImplementor session)
154  {
155  // we finish all the "removes" first to take care of possible unique
156  // constraints and so that we can take better advantage of batching
157  try
158  {
159  const int offset = 0;
160  int count = 0;
161 
162  if (RowDeleteEnabled)
163  {
164  bool useBatch = true;
165  IDbCommand st = null;
166  // update removed rows fks to null
167  try
168  {
169  int i = 0;
170  IEnumerable entries = collection.Entries(this);
171  IExpectation expectation = Expectations.None;
172 
173  foreach (object entry in entries)
174  {
175  if (collection.NeedsUpdating(entry, i, ElementType))
176  {
177  // will still be issued when it used to be null
178  if (st == null)
179  {
180  SqlCommandInfo sql = SqlDeleteRowString;
181  if (DeleteCallable)
182  {
183  expectation = Expectations.AppropriateExpectation(DeleteCheckStyle);
184  useBatch = expectation.CanBeBatched;
185  st = useBatch
186  ? session.Batcher.PrepareBatchCommand(SqlDeleteRowString.CommandType, sql.Text, SqlDeleteRowString.ParameterTypes)
187  : session.Batcher.PrepareCommand(SqlDeleteRowString.CommandType, sql.Text, SqlDeleteRowString.ParameterTypes);
188  //offset += expectation.prepare(st);
189  }
190  else
191  {
192  st = session.Batcher.PrepareBatchCommand(SqlDeleteRowString.CommandType, sql.Text, SqlDeleteRowString.ParameterTypes);
193  }
194  }
195 
196  int loc = WriteKey(st, id, offset, session);
197  WriteElementToWhere(st, collection.GetSnapshotElement(entry, i), loc, session);
198  if (useBatch)
199  {
200  session.Batcher.AddToBatch(expectation);
201  }
202  else
203  {
204  expectation.VerifyOutcomeNonBatched(session.Batcher.ExecuteNonQuery(st), st);
205  }
206  count++;
207  }
208  i++;
209  }
210  }
211  catch (Exception e)
212  {
213  if (useBatch)
214  {
215  session.Batcher.AbortBatch(e);
216  }
217  throw;
218  }
219  finally
220  {
221  if (!useBatch && st != null)
222  {
223  session.Batcher.CloseCommand(st, null);
224  }
225  }
226  }
227 
228  if (RowInsertEnabled)
229  {
230  IExpectation expectation = Expectations.AppropriateExpectation(InsertCheckStyle);
231  //bool callable = InsertCallable;
232  bool useBatch = expectation.CanBeBatched;
233  SqlCommandInfo sql = SqlInsertRowString;
234  IDbCommand st = null;
235 
236  // now update all changed or added rows fks
237  try
238  {
239  int i = 0;
240  IEnumerable entries = collection.Entries(this);
241  foreach (object entry in entries)
242  {
243  if (collection.NeedsUpdating(entry, i, ElementType))
244  {
245  if (useBatch)
246  {
247  if (st == null)
248  {
249  st = session.Batcher.PrepareBatchCommand(SqlInsertRowString.CommandType, sql.Text, SqlInsertRowString.ParameterTypes);
250  }
251  }
252  else
253  {
254  st = session.Batcher.PrepareCommand(SqlInsertRowString.CommandType, sql.Text, SqlInsertRowString.ParameterTypes);
255  }
256 
257  //offset += expectation.Prepare(st, Factory.ConnectionProvider.Driver);
258  int loc = WriteKey(st, id, offset, session);
259  if (HasIndex && !indexContainsFormula)
260  {
261  loc = WriteIndexToWhere(st, collection.GetIndex(entry, i, this), loc, session);
262  }
263  WriteElementToWhere(st, collection.GetElement(entry), loc, session);
264  if (useBatch)
265  {
266  session.Batcher.AddToBatch(expectation);
267  }
268  else
269  {
270  expectation.VerifyOutcomeNonBatched(session.Batcher.ExecuteNonQuery(st), st);
271  }
272  count++;
273  }
274  i++;
275  }
276  }
277  catch (Exception e)
278  {
279  if (useBatch)
280  {
281  session.Batcher.AbortBatch(e);
282  }
283  throw;
284  }
285  finally
286  {
287  if (!useBatch && st != null)
288  {
289  session.Batcher.CloseCommand(st, null);
290  }
291  }
292  }
293  return count;
294  }
295  catch (DbException sqle)
296  {
297  throw ADOExceptionHelper.Convert(SQLExceptionConverter, sqle, "could not update collection rows: " + MessageHelper.InfoString(this, id));
298  }

314  protected override SelectFragment GenerateSelectFragment(string alias, string columnSuffix)
315  {
316  /*
317  NH-1747 FIX
318  Note: the fix for NH-1747 may cause some problem when the same ColumnName is used to represent two different properties in the same hierarchy
319  In all our test it happened only for NHibernate.Test.Legacy.MultiTableTest.MultiTableCollections
320  The mapping Multi.hbm.xml had a column named "parent" to represent the Key of a set for the relation Lower->Top and SubMulti->SubMulti everything in the same hierarchy
321  NH TODO: Have to add an early check of same ColumnName used to represent two relations.
322  */
323  var ojl = (IOuterJoinLoadable)ElementPersister;
324  var selectFragment = new SelectFragment(Dialect).SetSuffix(columnSuffix);
325  var columnNames = KeyColumnNames;
326  var columnAliases = KeyColumnAliases;
327  for (int i = 0; i < columnNames.Length; i++)
328  {
329  var column = columnNames[i];
330  var tableAlias = ojl.GenerateTableAliasForColumn(alias, column);
331  selectFragment.AddColumn(tableAlias, column, columnAliases[i]);
332  }
333  return selectFragment;
334  }
