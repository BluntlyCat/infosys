39  public static string[] ConcreteQueries(string query, ISessionFactoryImplementor factory)
40  {
41  //scan the query string for class names appearing in the from clause and replace
42  //with all persistent implementors of the class/interface, returning multiple
43  //query strings (make sure we don't pick up a class in the select clause!)
44 
45  //TODO: this is one of the ugliest and most fragile pieces of code in Hibernate....

47  SessionFactoryHelper helper = new SessionFactoryHelper(factory);
48 
49  string[] tokens = StringHelper.Split(StringHelper.WhiteSpace + "(),", query, true);
50  if (tokens.Length == 0)
51  {
52  return new String[] {query}; // just especially for the trivial collection filter
53  }
54  var placeholders = new List<object>();
55  var replacements = new List<object>();
56  StringBuilder templateQuery = new StringBuilder(40);
57  int count = 0;
58  string last = null;
59  int nextIndex = 0;
60  string next = null;
61 
62  templateQuery.Append(tokens[0]);
63  bool isSelectClause = StringHelper.EqualsCaseInsensitive("select", tokens[0]);
64 
65  for (int i = 1; i < tokens.Length; i++)
66  {
67  //update last non-whitespace token, if necessary
68  if (!ParserHelper.IsWhitespace(tokens[i - 1]))
69  {
70  last = tokens[i - 1].ToLowerInvariant();
71  }
72 
73  // select-range is terminated by declaration of "from"
74  isSelectClause = !StringHelper.EqualsCaseInsensitive("from", tokens[i]);
75 
76  string token = tokens[i];
77  if (!ParserHelper.IsWhitespace(token) || last == null)
78  {
79  //scan for next non-whitespace token
80  if (nextIndex <= i)
81  {
82  for (nextIndex = i + 1; nextIndex < tokens.Length; nextIndex++)
83  {
84  next = tokens[nextIndex].ToLowerInvariant();
85  if (!ParserHelper.IsWhitespace(next))
86  {
87  break;
88  }
89  }
90  }
91 
92  // TODO H3.2 Different behavior
93  // NHb: This block is not an exactly port from H3.2 but a port from previous implementation of QueryTranslator
94  if (((last != null && beforeClassTokens.Contains(last)) &&
95  (next == null || !notAfterClassTokens.Contains(next))) ||
96  ParserHelper.EntityClass.Equals(last))
97  {
98  System.Type clazz = helper.GetImportedClass(token);
99  if (clazz != null)
100  {
101  string[] implementors = factory.GetImplementors(clazz.FullName);
102  string placeholder = "$clazz" + count++ + "$";
103 
104  if (implementors != null)
105  {
106  placeholders.Add(placeholder);
107  replacements.Add(implementors);
108  }
109  token = placeholder; //Note this!!
110  }
111  }
112  }
113  templateQuery.Append(token);
114  }
115  string[] results =
116  StringHelper.Multiply(templateQuery.ToString(), placeholders.GetEnumerator(), replacements.GetEnumerator());
117  if (results.Length == 0)
118  {
119  log.Warn("no persistent classes found for query class: " + query);
120  }
121  return results;
122  }
