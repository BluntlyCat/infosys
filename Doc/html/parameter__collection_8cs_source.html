167  protected override DbParameter GetParameter(string parameterName)
168  {
169  int index = IndexOf(parameterName);
170  if (index < 0)
171  {
172  // check to see if the user has added the parameter without a
173  // parameter marker. If so, kindly tell them what they did.
174  if (parameterName.StartsWith("@", StringComparison.Ordinal) ||
175  parameterName.StartsWith("?", StringComparison.Ordinal))
176  {
177  string newParameterName = parameterName.Substring(1);
178  index = IndexOf(newParameterName);
179  if (index != -1)
180  return (DbParameter)items[index];
181  }
182  throw new ArgumentException("Parameter '" + parameterName + "' not found in the collection.");
183  }
184  return (DbParameter)items[index];
185  }

421  private MySqlParameter InternalAdd(MySqlParameter value, int index)
422  {
423  if (value == null)
424  throw new ArgumentException("The MySqlParameterCollection only accepts non-null MySqlParameter type objects.", "value");
425 
426  // if the parameter is unnamed, then assign a default name
427  if (String.IsNullOrEmpty(value.ParameterName))
428  value.ParameterName = String.Format("Parameter{0}", GetNextIndex());
429 
430  // make sure we don't already have a parameter with this name
431  if (IndexOf(value.ParameterName) >= 0)
432  {
433  throw new MySqlException(
434  String.Format(Resources.ParameterAlreadyDefined, value.ParameterName));
435  }
436  else
437  {
438  string inComingName = value.ParameterName;
439  if (inComingName[0] == '@' || inComingName[0] == '?')
440  inComingName = inComingName.Substring(1, inComingName.Length - 1);
441  if (IndexOf(inComingName) >= 0)
442  throw new MySqlException(
443  String.Format(Resources.ParameterAlreadyDefined, value.ParameterName));
444  }
445 
446  if (index == -1)
447  {
448  items.Add(value);
449  index = items.Count - 1;
450  }
451  else
452  {
453  items.Insert(index, value);
454  AdjustHashes(index, true);
455  }
456 
457  indexHashCS.Add(value.ParameterName, index);
458  indexHashCI.Add(value.ParameterName, index);
459 
460  value.Collection = this;
461  return value;
462  }

464  private int GetNextIndex()
465  {
466  int index = Count+1;
467 
468  while (true)
469  {
470  string name = "Parameter" + index.ToString();
471  if (!indexHashCI.Contains(name)) break;
472  index++;
473  }
474  return index;
475  }

477  private static void AdjustHash(Hashtable hash, string parameterName, int keyIndex, bool addEntry)
478  {
479  if (!hash.ContainsKey(parameterName)) return;
480  int index = (int)hash[parameterName];
481  if (index < keyIndex) return;
482  hash[parameterName] = addEntry ? ++index : --index;
483  }

491  private void AdjustHashes(int keyIndex, bool addEntry)
492  {
493  for (int i = 0; i < Count; i++)
494  {
495  string name = (items[i] as MySqlParameter).ParameterName;
496  AdjustHash(indexHashCI, name, keyIndex, addEntry);
497  AdjustHash(indexHashCS, name, keyIndex, addEntry);
498  }

501  private MySqlParameter GetParameterFlexibleInternal(string baseName)
502  {
503  int index = IndexOf(baseName);
504  if (-1 == index)
505  index = IndexOf("?" + baseName);
506  if (-1 == index)
507  index = IndexOf("@" + baseName);
508  if (-1 != index)
509  return this[index];
510  return null;
511  }

513  internal MySqlParameter GetParameterFlexible(string parameterName, bool throwOnNotFound)
514  {
515  string baseName = parameterName;
516  MySqlParameter p = GetParameterFlexibleInternal(baseName);
517  if (p != null) return p;
518 
519  if (parameterName.StartsWith("@", StringComparison.Ordinal) || parameterName.StartsWith("?", StringComparison.Ordinal))
520  baseName = parameterName.Substring(1);
521  p = GetParameterFlexibleInternal(baseName);
522  if (p != null) return p;
523 
524  if (throwOnNotFound)
525  throw new ArgumentException("Parameter '" + parameterName + "' not found in the collection.");
526  return null;
527  }
