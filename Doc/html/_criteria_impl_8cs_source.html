575  private void CloneSubcriteria(CriteriaImpl clone)
576  {
577  //we need to preserve the parent criteria, we rely on the ordering when creating the
578  //subcriterias initially here, so we don't need to make more than a single pass
579  Dictionary<ICriteria, ICriteria> newParents = new Dictionary<ICriteria, ICriteria>();
580  newParents[this] = clone;
581 
582  foreach (Subcriteria subcriteria in IterateSubcriteria())
583  {
584  ICriteria currentParent;
585  if (!newParents.TryGetValue(subcriteria.Parent, out currentParent))
586  {
587  throw new AssertionFailure(
588  "Could not find parent for subcriteria in the previous subcriteria. If you see this error, it is a bug");
589  }
590  Subcriteria clonedSubCriteria =
591  new Subcriteria(clone, currentParent, subcriteria.Path, subcriteria.Alias, subcriteria.JoinType, subcriteria.WithClause);
592  clonedSubCriteria.SetLockMode(subcriteria.LockMode);
593  newParents[subcriteria] = clonedSubCriteria;
594  }
595 
596  // remap the orders
597  foreach (OrderEntry orderEntry in IterateOrderings())
598  {
599  ICriteria currentParent;
600  if (!newParents.TryGetValue(orderEntry.Criteria, out currentParent))
601  {
602  throw new AssertionFailure(
603  "Could not find parent for order in the previous criteria. If you see this error, it is a bug");
604  }
605  currentParent.AddOrder(orderEntry.Order);
606  }
607 
608  // remap the restrictions to appropriate criterias
609  foreach (CriterionEntry criterionEntry in criteria)
610  {
611  ICriteria currentParent;
612  if (!newParents.TryGetValue(criterionEntry.Criteria, out currentParent))
613  {
614  throw new AssertionFailure(
615  "Could not find parent for restriction in the previous criteria. If you see this error, it is a bug.");
616  }
617 
618  currentParent.Add(criterionEntry.Criterion);
619  }
620  }
