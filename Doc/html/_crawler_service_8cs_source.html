128  protected override void Run()
129  {
130  while (this.Running)
131  {
132  if (Console.KeyAvailable)
133  {
134  ConsoleKeyInfo keyInfo = Console.ReadKey(true);
135  Log.InfoFormat(Properties.Resources.WEB_CRAWLER_GOT_USERINPUT, keyInfo.Key);
136 
137  switch (keyInfo.Key)
138  {
139  case ConsoleKey.Q:
140  Log.Info(Properties.Resources.WEB_CRAWLER_EXITED_BY_USER);
141  this.ShutdownCrawler();
142  break;
143 
144  case ConsoleKey.S:
145  this.StopServices();
146  break;
147 
148  default:
149  Log.Info(Properties.Resources.WEB_CRAWLER_UNKOWN_USERINPUT);
150  break;
151  }
152  }
153 
154  Thread.Sleep(1000);
155  }
