267  protected override void WriteIndent()

689  private void WriteIntegerValue(long value)
690  {
691  EnsureWriteBuffer();
692 
693  if (value >= 0 && value <= 9)
694  {
695  _writer.Write((char)('0' + value));
696  }

708  private void WriteIntegerValue(ulong uvalue)
709  {
710  EnsureWriteBuffer();
711 
712  if (uvalue <= 9)
713  {
714  _writer.Write((char)('0' + uvalue));
715  }
716  else
717  {
718  int totalLength = MathUtils.IntLength(uvalue);
719  int length = 0;
720 
721  do
722  {
723  _writeBuffer[totalLength - ++length] = (char)('0' + (uvalue % 10));
724  uvalue /= 10;
725  } while (uvalue != 0);
726 
727  _writer.Write(_writeBuffer, 0, length);
728  }
729  }
