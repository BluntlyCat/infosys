413  public void SetPermissions(short mode)
414  {
415  if (mode < 0 || mode > 999)
416  {
417  throw new ArgumentOutOfRangeException("mode");
418  }
419 
420  var modeBytes = mode.ToString(CultureInfo.InvariantCulture).PadLeft(3, '0').ToArray();
421 
422  var permission = (modeBytes[0] & 0x0F) * 8 * 8 + (modeBytes[1] & 0x0F) * 8 + (modeBytes[2] & 0x0F);
423 
424  this.OwnerCanRead = (permission & S_IRUSR) == S_IRUSR;
425  this.OwnerCanWrite = (permission & S_IWUSR) == S_IWUSR;
