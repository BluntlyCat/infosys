222  private ZLibStatus HuftBuild(int[] b, // code lengths in bits (all assumed <= BMAX)
223  int bindex,
224  int n, // number of codes (assumed <= 288)
225  int s, // number of simple-valued codes (0..s-1)
226  int[] d, // list of base values for non-simple codes
227  int[] e, // list of extra bits for non-simple codes
228  int[] t, // result: starting table
229  int[] m, // maximum lookup bits, returns actual
230  int[] hp,// space for trees
231  int[] hn,// hufts used in space
232  int[] v // working area: values in order of bit length
233  )
234  {
235  // Given a list of code lengths and a maximum table size, make a set of
236  // tables to decode that set of codes. Return Z_OK on success, Z_BUF_ERROR
237  // if the given code set is incomplete (the tables are still built in this
238  // case), Z_DATA_ERROR if the input is invalid (an over-subscribed set of
239  // lengths), or Z_MEM_ERROR if not enough memory.
240 
241  int a; // counter for codes of length k
242  int f; // i repeats in table every f entries
243  int g; // maximum code length
244  int h; // table level
245  int i; // counter, current code
246  int j; // counter
247  int k; // number of bits in current code
248  int l; // bits per table (returned in m)
249  int mask; // (1 << w) - 1, to avoid cc -O bug on HP
250  int p; // pointer into c[], b[], or v[]
251  int q; // points to current table
252  int w; // bits before this table == (l * h)
253  int xp; // pointer into x
254  int y; // number of dummy codes added
255  int z; // number of entries in current table
256 
257  // Generate counts for each bit length
258 
259  p = 0; i = n;
260  do
261  {
262  c[b[bindex + p]]++; p++; i--; // assume all entries <= BMAX
263  } while (i != 0);
264 
265  if (c[0] == n)
266  { // null input--all zero length codes
267  t[0] = -1;
268  m[0] = 0;
269  return ZLibStatus.Z_OK;
270  }
271 
272  // Find minimum and maximum length, bound *m by those
273  l = m[0];
274  for (j = 1; j <= BMAX; j++)
275  if (c[j] != 0) break;
276  k = j; // minimum code length
277  if (l < j)
278  {
279  l = j;
280  }
281  for (i = BMAX; i != 0; i--)
282  {
283  if (c[i] != 0) break;
284  }
285  g = i; // maximum code length
286  if (l > i)
287  {
288  l = i;
289  }
290  m[0] = l;
291 
292  // Adjust last length count to fill out codes, if needed
293  for (y = 1 << j; j < i; j++, y <<= 1)
294  {
295  if ((y -= c[j]) < 0)
296  {
297  return ZLibStatus.Z_DATA_ERROR;
298  }
299  }
300  if ((y -= c[i]) < 0)
301  {
302  return ZLibStatus.Z_DATA_ERROR;
303  }
304  c[i] += y;
305 
306  // Generate starting offsets into the value table for each length
307  x[1] = j = 0;
308  p = 1; xp = 2;
309  while (--i != 0)
310  { // note that i == g from above
311  x[xp] = (j += c[p]);
312  xp++;
313  p++;
314  }
315 
316  // Make a table of values in order of bit lengths
317  i = 0; p = 0;
318  do
319  {
320  if ((j = b[bindex + p]) != 0)
321  {
322  v[x[j]++] = i;
323  }
324  p++;
325  }
326  while (++i < n);
327  n = x[g]; // set n to length of v
328 
329  // Generate the Huffman codes and for each, make the table entries
330  x[0] = i = 0; // first Huffman code is zero
331  p = 0; // grab values in bit order
332  h = -1; // no tables yet--level -1
333  w = -l; // bits decoded == (l * h)
334  u[0] = 0; // just to keep compilers happy
335  q = 0; // ditto
336  z = 0; // ditto
337 
338  // go through the bit lengths (k already is bits in shortest code)
339  for (; k <= g; k++)
340  {
341  a = c[k];
342  while (a-- != 0)
343  {
344  // here i is the Huffman code of length k bits for value *p
345  // make tables up to required level
346  while (k > w + l)
347  {
348  h++;
349  w += l; // previous table always l bits
350  // compute minimum size table less than or equal to l bits
351  z = g - w;
352  z = (z > l) ? l : z; // table size upper limit
353  if ((f = 1 << (j = k - w)) > a + 1)
354  { // try a k-w bit table
355  // too few codes for k-w bit table
356  f -= a + 1; // deduct codes from patterns left
357  xp = k;
358  if (j < z)
359  {
360  while (++j < z)
361  { // try smaller tables up to z bits
362  if ((f <<= 1) <= c[++xp])
363  break; // enough codes to use up j bits
364  f -= c[xp]; // else deduct codes from patterns
365  }
366  }
367  }
368  z = 1 << j; // table entries for j-bit table
369 
370  // allocate new table
371  if (hn[0] + z > MANY)
372  { // (note: doesn't matter for fixed)
373  return ZLibStatus.Z_DATA_ERROR; // overflow of MANY
374  }
375  u[h] = q = /*hp+*/ hn[0]; // DEBUG
376  hn[0] += z;
377 
378  // connect to last table, if there is one
379  if (h != 0)
380  {
381  x[h] = i; // save pattern for backing up
382  r[0] = (byte)j; // bits in this table
383  r[1] = (byte)l; // bits to dump before this table
384  j = i >> (w - l);
385  r[2] = (int)(q - u[h - 1] - j); // offset to this table
386  System.Array.Copy(r, 0, hp, (u[h - 1] + j) * 3, 3); // connect to last table
387  }
388  else
389  {
390  t[0] = q; // first table is returned result
391  }
392  }
393 
394  // set up table entry in r
395  r[1] = (byte)(k - w);
396  if (p >= n)
397  {
398  r[0] = 128 + 64; // out of values--invalid code
399  }
400  else if (v[p] < s)
401  {
402  r[0] = (byte)(v[p] < 256 ? 0 : 32 + 64); // 256 is end-of-block
403  r[2] = v[p++]; // simple code is just the value
404  }
405  else
406  {
407  r[0] = (byte)(e[v[p] - s] + 16 + 64); // non-simple--look up in lists
408  r[2] = d[v[p++] - s];
409  }
410 
411  // fill code-like entries with r
412  f = 1 << (k - w);
413  for (j = i >> w; j < z; j += f)
414  {
415  System.Array.Copy(r, 0, hp, (q + j) * 3, 3);
416  }
417 
418  // backwards increment the k-bit code i
419  for (j = 1 << (k - 1); (i & j) != 0; j >>= 1)
420  {
421  i ^= j;
422  }
423  i ^= j;
424 
425  // backup over finished tables
426  mask = (1 << w) - 1; // needed on HP, cc -O bug
427  while ((i & mask) != x[h])
428  {
429  h--; // don't need to update q
430  w -= l;
431  mask = (1 << w) - 1;
432  }
433  }
434  }
435  // Return Z_BUF_ERROR if we were given an incomplete table
436  return y != 0 && g != 1 ? ZLibStatus.Z_BUF_ERROR : ZLibStatus.Z_OK;
437  }
