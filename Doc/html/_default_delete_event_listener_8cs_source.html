32  public virtual void OnDelete(DeleteEvent @event, ISet<object> transientEntities)
33  {
34  IEventSource source = @event.Session;
35  IPersistenceContext persistenceContext = source.PersistenceContext;
36  object entity = persistenceContext.UnproxyAndReassociate(@event.Entity);
37 
38  EntityEntry entityEntry = persistenceContext.GetEntry(entity);
39  IEntityPersister persister;
40  object id;
41  object version;
42 
43  if (entityEntry == null)
44  {
45  log.Debug("entity was not persistent in delete processing");
46 
47  persister = source.GetEntityPersister(@event.EntityName, entity);
48 
49  if (ForeignKeys.IsTransient(persister.EntityName, entity, null, source))
50  {
51  DeleteTransientEntity(source, entity, @event.CascadeDeleteEnabled, persister, transientEntities);
52  // EARLY EXIT!!!
53  return;
54  }
55  else
56  {
57  PerformDetachedEntityDeletionCheck(@event);
58  }
59 
60  id = persister.GetIdentifier(entity, source.EntityMode);
61 
62  if (id == null)
63  {
64  throw new TransientObjectException("the detached instance passed to delete() had a null identifier");
65  }
66 
67  EntityKey key = source.GenerateEntityKey(id, persister);
68 
69  persistenceContext.CheckUniqueness(key, entity);
70 
71  new OnUpdateVisitor(source, id, entity).Process(entity, persister);
72 
73  version = persister.GetVersion(entity, source.EntityMode);
74 
75  entityEntry = persistenceContext.AddEntity(
76  entity,
77  persister.IsMutable ? Status.Loaded : Status.ReadOnly,
78  persister.GetPropertyValues(entity, source.EntityMode),
79  key,
80  version,
81  LockMode.None,
82  true,
83  persister,
84  false,
85  false);
86  }
87  else
88  {
89  log.Debug("deleting a persistent instance");
90 
91  if (entityEntry.Status == Status.Deleted || entityEntry.Status == Status.Gone)
92  {
93  log.Debug("object was already deleted");
94  return;
95  }
96  persister = entityEntry.Persister;
97  id = entityEntry.Id;
98  version = entityEntry.Version;
99  }
100 
101  if (InvokeDeleteLifecycle(source, entity, persister))
102  return;
103 
104  DeleteEntity(source, entity, entityEntry, @event.CascadeDeleteEnabled, persister, transientEntities);
105 
106  if (source.Factory.Settings.IsIdentifierRollbackEnabled)
107  {
108  persister.ResetIdentifier(entity, id, version, source.EntityMode);
109  }
110  }

169  protected virtual void DeleteEntity(IEventSource session, object entity, EntityEntry entityEntry, bool isCascadeDeleteEnabled, IEntityPersister persister, ISet<object> transientEntities)
170  {
171  if (log.IsDebugEnabled)
172  {
173  log.Debug("deleting " + MessageHelper.InfoString(persister, entityEntry.Id, session.Factory));
174  }
175 
176  IPersistenceContext persistenceContext = session.PersistenceContext;
177 
178  IType[] propTypes = persister.PropertyTypes;
179  object version = entityEntry.Version;
180 
181  object[] currentState;
182  if (entityEntry.LoadedState == null)
183  {
184  //ie. the entity came in from update()
185  currentState = persister.GetPropertyValues(entity, session.EntityMode);
186  }
187  else
188  {
189  currentState = entityEntry.LoadedState;
190  }
191 
192  object[] deletedState = CreateDeletedState(persister, currentState, session);
193  entityEntry.DeletedState = deletedState;
194 
195  session.Interceptor.OnDelete(entity, entityEntry.Id, deletedState, persister.PropertyNames, propTypes);
196 
197  // before any callbacks, etc, so subdeletions see that this deletion happened first
198  persistenceContext.SetEntryStatus(entityEntry, Status.Deleted);
199  EntityKey key = session.GenerateEntityKey(entityEntry.Id, persister);
200 
201  CascadeBeforeDelete(session, persister, entity, entityEntry, transientEntities);
202 
203  new ForeignKeys.Nullifier(entity, true, false, session).NullifyTransientReferences(entityEntry.DeletedState, propTypes);
204  new Nullability(session).CheckNullability(entityEntry.DeletedState, persister, true);
205  persistenceContext.NullifiableEntityKeys.Add(key);
206 
207  // Ensures that containing deletions happen before sub-deletions
208  session.ActionQueue.AddAction(new EntityDeleteAction(entityEntry.Id, deletedState, version, entity, persister, isCascadeDeleteEnabled, session));
209 
210  CascadeAfterDelete(session, persister, entity, transientEntities);
211 
212  // the entry will be removed after the flush, and will no longer
213  // override the stale snapshot
214  // This is now handled by removeEntity() in EntityDeleteAction
215  //persistenceContext.removeDatabaseSnapshot(key);
216  }

218  private object[] CreateDeletedState(IEntityPersister persister, object[] currentState, IEventSource session)
219  {
220  IType[] propTypes = persister.PropertyTypes;

243  protected virtual void CascadeBeforeDelete(IEventSource session, IEntityPersister persister, object entity, EntityEntry entityEntry, ISet<object> transientEntities)
244  {
245  ISessionImplementor si = session;
246  CacheMode cacheMode = si.CacheMode;
247  si.CacheMode = CacheMode.Get;
248  session.PersistenceContext.IncrementCascadeLevel();
249  try
250  {
251  // cascade-delete to collections BEFORE the collection owner is deleted
252  new Cascade(CascadingAction.Delete, CascadePoint.AfterInsertBeforeDelete, session).CascadeOn(persister, entity,
253  transientEntities);
254  }
255  finally
256  {
257  session.PersistenceContext.DecrementCascadeLevel();
