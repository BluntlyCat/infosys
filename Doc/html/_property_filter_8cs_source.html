97  override public FilterDecision Decide(LoggingEvent loggingEvent)
98  {
99  if (loggingEvent == null)
100  {
101  throw new ArgumentNullException("loggingEvent");
102  }
103 
104  // Check if we have a key to lookup the event property value with
105  if (m_key == null)
106  {
107  // We cannot filter so allow the filter chain
108  // to continue processing
109  return FilterDecision.Neutral;
110  }
111 
112  // Lookup the string to match in from the properties using
113  // the key specified.
114  object msgObj = loggingEvent.LookupProperty(m_key);
115 
116  // Use an ObjectRenderer to convert the property value to a string
117  string msg = loggingEvent.Repository.RendererMap.FindAndRender(msgObj);
118 
119  // Check if we have been setup to filter
120  if (msg == null || (m_stringToMatch == null && m_regexToMatch == null))
121  {
122  // We cannot filter so allow the filter chain
123  // to continue processing
124  return FilterDecision.Neutral;
125  }
126 
127  // Firstly check if we are matching using a regex
128  if (m_regexToMatch != null)
129  {
130  // Check the regex
131  if (m_regexToMatch.Match(msg).Success == false)
132  {
133  // No match, continue processing
134  return FilterDecision.Neutral;
135  }
136 
137  // we've got a match
138  if (m_acceptOnMatch)
139  {
140  return FilterDecision.Accept;
141  }
142  return FilterDecision.Deny;
143  }
144  else if (m_stringToMatch != null)
145  {
146  // Check substring match
147  if (msg.IndexOf(m_stringToMatch) == -1)
148  {
149  // No match, continue processing
150  return FilterDecision.Neutral;
151  }
152 
153  // we've got a match
154  if (m_acceptOnMatch)
155  {
156  return FilterDecision.Accept;
157  }
158  return FilterDecision.Deny;
159  }
160  return FilterDecision.Neutral;
161  }
