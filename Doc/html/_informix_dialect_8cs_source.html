368  public override SqlString GetLimitString(SqlString queryString, SqlString offset, SqlString limit)
369  {
370  /*
371  * "SELECT [SKIP x] FIRST y rest-of-sql-statement"
372  */
373 
374  // TODO - Check support for cases where only the offset is specified, but the limit is not. Might need to use int.MaxValue.
375 
376  int insertIndex = GetAfterSelectInsertPoint(queryString);
377 
378  SqlStringBuilder limitFragment = new SqlStringBuilder();
379  if (offset != null)
380  {
381  limitFragment.Add(" skip ");
382  limitFragment.Add(offset);
383  }
384  if (limit != null)
385  {
386  limitFragment.Add(" first ");
387  limitFragment.Add(limit);
388  }
389 
390  return queryString.Insert(insertIndex, limitFragment.ToSqlString());
391  }

459  public class IfxViolatedConstraintExtracter : TemplatedViolatedConstraintNameExtracter
460  {
466  public override string ExtractConstraintName(DbException sqle)
467  {
468  string constraintName = null;
469 
470  var extracter = new ReflectionBasedSqlStateExtracter();
471 
472  int errorCode = extracter.ExtractErrorCode(sqle);
473  if (errorCode == -268)
474  {
475  constraintName = ExtractUsingTemplate("Unique constraint (", ") violated.", sqle.Message);
476  }
477  else if (errorCode == -691)
478  {
479  constraintName = ExtractUsingTemplate("Missing key in referenced table for referential constraint (", ").",
480  sqle.Message);
481  }
482  else if (errorCode == -692)
483  {
484  constraintName = ExtractUsingTemplate("Key value for constraint (", ") is still being referenced.", sqle.Message);
485  }
486 
487  if (constraintName != null)
488  {
489  // strip table-owner because Informix always returns constraint names as "<table-owner>.<constraint-name>"
490  int i = constraintName.IndexOf('.');
491  if (i != -1)
492  {
493  constraintName = constraintName.Substring(i + 1);
494  }
495  }
496  return constraintName;
497  }
498  } ;
