168  virtual public void Format(TextWriter writer, object state)
169  {
170  if (m_min < 0 && m_max == int.MaxValue)
171  {
172  // Formatting options are not in use
173  Convert(writer, state);
174  }
175  else
176  {
177  string msg = null;
178  int len;
179  lock (m_formatWriter)
180  {
181  m_formatWriter.Reset(c_renderBufferMaxCapacity, c_renderBufferSize);
182 
183  Convert(m_formatWriter, state);
184 
185  StringBuilder buf = m_formatWriter.GetStringBuilder();
186  len = buf.Length;
187  if (len > m_max)
188  {
189  msg = buf.ToString(len - m_max, m_max);
190  len = m_max;
191  }
192  else
193  {
194  msg = buf.ToString();
195  }
196  }
197 
198  if (len < m_min)
199  {
200  if (m_leftAlign)
201  {
202  writer.Write(msg);
203  SpacePad(writer, m_min - len);
204  }
205  else
206  {
207  SpacePad(writer, m_min - len);
208  writer.Write(msg);
209  }
210  }
211  else
212  {
213  writer.Write(msg);
214  }
215  }
216  }

232  protected static void SpacePad(TextWriter writer, int length)
233  {
234  while(length >= 32)
235  {
236  writer.Write(SPACES[5]);
237  length -= 32;
238  }
239 
240  for(int i = 4; i >= 0; i--)
241  {
242  if ((length & (1<<i)) != 0)
243  {
244  writer.Write(SPACES[i]);
245  }
246  }
247  }
