460  public override int Read(byte[] buffer, int offset, int count)
461  {
462  int readLen = 0;
463  int tempLen;
464 
465  if (buffer == null)
466  {
467  throw new ArgumentNullException("buffer");
468  }
469  else if (offset < 0)
470  {
471  throw new ArgumentOutOfRangeException("offset");
472  }
473  else if (count < 0)
474  {
475  throw new ArgumentOutOfRangeException("count");
476  }
477  else if ((buffer.Length - offset) < count)
478  {
479  throw new ArgumentException("Invalid array range.");
480  }
481 
482  // Lock down the file stream while we do this.
483  lock (this._lock)
484  {
485  // Set up for the read operation.
486  this.SetupRead();
487 
488  // Read data into the caller's buffer.
489  while (count > 0)
490  {
491  // How much data do we have available in the buffer?
492  tempLen = this._bufferLen - this._bufferPosn;
493  if (tempLen <= 0)
494  {
495  this._bufferPosn = 0;
496 
497  var data = this._session.RequestRead(this._handle, (ulong)this._position, (uint)this._bufferSize);
498 
499  this._bufferLen = data.Length;
500 
501  Buffer.BlockCopy(data, 0, this._buffer, 0, this._bufferLen);
502  this._serverFilePosition = (ulong)this._position;
503 
504  if (this._bufferLen < 0)
505  {
506  this._bufferLen = 0;
507  // TODO: Add SFTP error code or message if possible
508  throw new IOException("Read operation failed.");
509  }
510  else if (this._bufferLen == 0)
511  {
512  break;
513  }
514  else
515  {
516  tempLen = this._bufferLen;
517  }
518  }
519 
520  // Don't read more than the caller wants.
521  if (tempLen > count)
522  {
523  tempLen = count;
524  }
525 
526  // Copy stream data to the caller's buffer.
527  Buffer.BlockCopy(this._buffer, this._bufferPosn, buffer, offset, tempLen);
528 
529  // Advance to the next buffer positions.
530  readLen += tempLen;
531  offset += tempLen;
532  count -= tempLen;
533  this._bufferPosn += tempLen;
534  this._position += tempLen;
535  }
536  }
537 
538  // Return the number of bytes that were read to the caller.
539  return readLen;

603  public override long Seek(long offset, SeekOrigin origin)
604  {
605  long newPosn = -1;
606 
607  // Bail out if this stream is not capable of seeking.
608  if (!this._canSeek)
609  {
610  throw new NotSupportedException("Seek is not supported.");
611  }
612 
613  // Lock down the file stream while we do this.
614  lock (this._lock)
615  {
616  // Bail out if the handle is invalid.
617  if (this._handle == null)
618  {
619  throw new ObjectDisposedException("Stream is closed.");
620  }
621 
622  // Don't do anything if the position won't be moving.
623  if (origin == SeekOrigin.Begin && offset == this._position)
624  {
625  return offset;
626  }
627  else if (origin == SeekOrigin.Current && offset == 0)
628  {
629  return this._position;
630  }
631 
632  this._attributes = this._session.RequestFStat(this._handle);
633 
634  // The behaviour depends upon the read/write mode.
635  if (this._bufferOwnedByWrite)
636  {
637  // Flush the write buffer and then seek.
638  this.FlushWriteBuffer();
639 
640  switch (origin)
641  {
642  case SeekOrigin.Begin:
643  newPosn = offset;
644  break;
645  case SeekOrigin.Current:
646  newPosn = this._position + offset;
647  break;
648  case SeekOrigin.End:
649  newPosn = this._attributes.Size - offset;
650  break;
651  default:
652  break;
653  }
654 
655  if (newPosn == -1)
656  {
657  throw new EndOfStreamException("End of stream.");
658  }
659  this._position = newPosn;
660  this._serverFilePosition = (ulong)newPosn;
661  }
662  else
663  {
664  // Determine if the seek is to somewhere inside
665  // the current read buffer bounds.
666  if (origin == SeekOrigin.Begin)
667  {
668  newPosn = this._position - this._bufferPosn;
669  if (offset >= newPosn && offset <
670  (newPosn + this._bufferLen))
671  {
672  this._bufferPosn = (int)(offset - newPosn);
673  this._position = offset;
674  return this._position;
675  }
676  }
677  else if (origin == SeekOrigin.Current)
678  {
679  newPosn = this._position + offset;
680  if (newPosn >= (this._position - this._bufferPosn) &&
681  newPosn < (this._position - this._bufferPosn + this._bufferLen))
682  {
683  this._bufferPosn =
684  (int)(newPosn - (this._position - this._bufferPosn));
685  this._position = newPosn;
686  return this._position;
687  }
688  }
689 
690  // Abandon the read buffer.
691  this._bufferPosn = 0;
692  this._bufferLen = 0;
693 
694  // Seek to the new position.
695  switch (origin)
696  {
697  case SeekOrigin.Begin:
698  newPosn = offset;
699  break;
700  case SeekOrigin.Current:
701  newPosn = this._position + offset;
702  break;
703  case SeekOrigin.End:
704  newPosn = this._attributes.Size - offset;
705  break;
706  default:
707  break;
708  }
709 
710  if (newPosn < 0)
711  {
712  throw new EndOfStreamException();
713  }
714 
715  this._position = newPosn;
716  }
717  return this._position;
718  }

775  {
776  int tempLen;
777 
778  // Validate the parameters
779  if (buffer == null)
780  {
781  throw new ArgumentNullException("buffer");
782  }
783  else if (offset < 0)
784  {
785  throw new ArgumentOutOfRangeException("offset");
786  }
787  else if (count < 0)
788  {
789  throw new ArgumentOutOfRangeException("count");
790  }
791  else if ((buffer.Length - offset) < count)
792  {
793  throw new ArgumentException("Invalid array range.");
794  }
795 
796  // Lock down the file stream while we do this.
797  lock (this._lock)
798  {
799  // Setup this object for writing.
800  this.SetupWrite();
801 
802  // Write data to the file stream.
803  while (count > 0)
804  {
805  // Determine how many bytes we can write to the buffer.
806  tempLen = this._bufferSize - this._bufferPosn;
807  if (tempLen <= 0)
808  {
809  var data = new byte[this._bufferPosn];
810 
811  Buffer.BlockCopy(this._buffer, 0, data, 0, this._bufferPosn);
812 
813  using (var wait = new AutoResetEvent(false))
814  {
815  this._session.RequestWrite(this._handle, this._serverFilePosition, data, wait);
816  this._serverFilePosition += (ulong)data.Length;
817  }
818 
819  this._bufferPosn = 0;
820  tempLen = this._bufferSize;
821  }
822  if (tempLen > count)
823  {
824  tempLen = count;
825  }
826 
827  // Can we short-cut the internal buffer?
828  if (this._bufferPosn == 0 && tempLen == this._bufferSize)
829  {
830  // Yes: write the data directly to the file.
831  var data = new byte[tempLen];
832 
833  Buffer.BlockCopy(buffer, offset, data, 0, tempLen);
834 
835  using (var wait = new AutoResetEvent(false))
836  {
837  this._session.RequestWrite(this._handle, this._serverFilePosition, data, wait);
838  this._serverFilePosition += (ulong)data.Length;

840  }

849  this._position += tempLen;
850  offset += tempLen;
851  count -= tempLen;
852  }
853 
854  // If the buffer is full, then do a speculative flush now,
855  // rather than waiting for the next call to this method.
856  if (this._bufferPosn >= this._bufferSize)
857  {
858  var data = new byte[this._bufferPosn];
859

861
