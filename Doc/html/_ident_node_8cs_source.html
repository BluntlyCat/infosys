70  public override void ResolveIndex(IASTNode parent)
71  {
72  // An ident node can represent an index expression if the ident
73  // represents a naked property ref
74  // *Note: this makes the assumption (which is currently the case
75  // in the hql-sql grammar) that the ident is first resolved
76  // itself (addrExpr -> resolve()). The other option, if that
77  // changes, is to call resolve from here; but it is
78  // currently un-needed overhead.
79  if (!(IsResolved && _nakedPropertyRef))
80  {
81  throw new InvalidOperationException();
82  }
83 
84  string propertyName = OriginalText;
85  if (!DataType.IsCollectionType)
86  {
87  throw new SemanticException("Collection expected; [" + propertyName + "] does not refer to a collection property");
88  }
89 
90  // TODO : most of below was taken verbatim from DotNode; should either delegate this logic or super-type it
91  CollectionType type = (CollectionType) DataType;
92  string role = type.Role;
93  IQueryableCollection queryableCollection = SessionFactoryHelper.RequireQueryableCollection(role);
94 
95  string columnTableAlias = FromElement.TableAlias;
96 
97  FromElementFactory factory = new FromElementFactory(
98  Walker.CurrentFromClause,
99  FromElement,
100  propertyName,
101  null,
102  FromElement.ToColumns(columnTableAlias, propertyName, false),
103  true
104  );
105 
106  FromElement elem = factory.CreateCollection(queryableCollection, role, JoinType.InnerJoin, false, true);
107  FromElement = elem;
108  Walker.AddQuerySpaces(queryableCollection.CollectionSpaces); // Always add the collection's query spaces.
109  }

111  public override void Resolve(bool generateJoin, bool implicitJoin, string classAlias, IASTNode parent)
112  {
113  if (!IsResolved)
114  {
115  if (Walker.CurrentFromClause.IsFromElementAlias(Text))
116  {
117  if (ResolveAsAlias())
118  {
119  IsResolved = true;
120  // We represent a from-clause alias
121  }
122  }
123  else if (parent != null && parent.Type == HqlSqlWalker.DOT)
124  {
125  DotNode dot = (DotNode)parent;
126  if (parent.GetFirstChild() == this)
127  {
128  if (ResolveAsNakedComponentPropertyRefLhs(dot))
129  {
130  // we are the LHS of the DOT representing a naked comp-prop-ref
131  IsResolved = true;
132  }
133  }
134  else
135  {
136  if (ResolveAsNakedComponentPropertyRefRhs(dot))
137  {
138  // we are the RHS of the DOT representing a naked comp-prop-ref
139  IsResolved = true;
140  }
141  }
142  }
143  else
144  {
145  int result = ResolveAsNakedPropertyRef();
146  if (result == PropertyRef)
147  {
148  // we represent a naked (simple) prop-ref
149  IsResolved = true;
150  }
151  else if (result == ComponentRef)
152  {
153  // EARLY EXIT!!! return so the resolve call explicitly coming from DotNode can
154  // resolve this...
155  return;
156  }
157  }
158 
159  // if we are still not resolved, we might represent a constant.
160  // needed to add this here because the allowance of
161  // naked-prop-refs in the grammar collides with the
162  // definition of literals/constants ("nondeterminism").
163  // TODO: cleanup the grammar so that "processConstants" is always just handled from here
164  if (!IsResolved)
165  {
166  try
167  {
168  Walker.LiteralProcessor.ProcessConstant(this, false);
169  }
170  catch (Exception)
171  {
172  // just ignore it for now, it'll get resolved later...
173  }
174  }
175  }

178  private int ResolveAsNakedPropertyRef()
179  {
180  FromElement fromElement = LocateSingleFromElement();
181 
182  if (fromElement == null)
183  {
184  return Unknown;
185  }
186 
187  IQueryable persister = fromElement.Queryable;
188  if (persister == null)
189  {
190  return Unknown;
191  }
192 
193  IType propertyType = GetNakedPropertyType(fromElement);
194  if (propertyType == null)
195  {
196  // assume this ident's text does *not* refer to a property on the given persister
197  return Unknown;
198  }
199 
200  if ((propertyType.IsComponentType || propertyType.IsAssociationType))
201  {
202  return ComponentRef;
203  }
204 
205  FromElement = fromElement;
206 
207  string property = Text;
208  string[] columns = Walker.IsSelectStatement
209  ? persister.ToColumns(fromElement.TableAlias, property)
210  : persister.ToColumns(property);
211 
212  string text = StringHelper.Join(", ", columns);
213  Text = columns.Length == 1 ? text : "(" + text + ")";
214  Type = HqlSqlWalker.SQL_TOKEN;
215 
216  // these pieces are needed for usage in select clause
217  DataType = propertyType;
218  _nakedPropertyRef = true;
219 
220  return PropertyRef;
221  }

223  private bool ResolveAsNakedComponentPropertyRefLhs(DotNode parent)
224  {
225  FromElement fromElement = LocateSingleFromElement();
226  if (fromElement == null)
227  {
228  return false;
229  }
230 
231  IType componentType = GetNakedPropertyType(fromElement);
232 
233  if (componentType == null)
234  {
235  throw new QueryException("Unable to resolve path [" + parent.Path + "], unexpected token [" + OriginalText + "]");
236  }
237  if (!componentType.IsComponentType)
238  {
239  throw new QueryException("Property '" + OriginalText + "' is not a component. Use an alias to reference associations or collections.");
240  }
241 
242  IType propertyType ; // used to set the type of the parent dot node
243  string propertyPath = Text + "." + NextSibling.Text;
244  try
245  {
246  // check to see if our "propPath" actually
247  // represents a property on the persister
248  propertyType = fromElement.GetPropertyType(Text, propertyPath);
249  }
250  catch (Exception)
251  {
252  // assume we do *not* refer to a property on the given persister
253  return false;
254  }
255 
256  FromElement = fromElement;
257  parent.PropertyPath = propertyPath;
258  parent.DataType = propertyType;
259 
260  return true;
261  }

315  private FromElement LocateSingleFromElement()
316  {
317  IList<IASTNode> fromElements = Walker.CurrentFromClause.GetFromElements();
318  if (fromElements == null || fromElements.Count != 1)
319  {
320  // TODO : should this be an error?
321  return null;
322  }
323 
324  FromElement element = (FromElement)fromElements[0];
325  if (element.ClassAlias != null)
326  {
327  // naked property-refs cannot be used with an aliased from element
328  return null;

333  private bool ResolveAsAlias()
334  {
335  // This is not actually a constant, but a reference to FROM element.
336  FromElement element = Walker.CurrentFromClause.GetFromElement(Text);
337  if (element != null)
338  {
339  FromElement = element;
340  Text = element.GetIdentityColumn();
341  Type = HqlSqlWalker.ALIAS_REF;
342  return true;
