41  public TracingDriver(MySqlConnectionStringBuilder settings)

61  public override void SendQuery(MySqlPacket p)
62  {
63  rowSizeInBytes = 0;
64  string cmdText = Encoding.GetString(p.Buffer, 5, p.Length - 5);
65  string normalized_query = null;
66 
67  if (cmdText.Length > 300)
68  {
69  QueryNormalizer normalizer = new QueryNormalizer();
70  normalized_query = normalizer.Normalize(cmdText);
71  cmdText = cmdText.Substring(0, 300);
72  }
73 
74  base.SendQuery(p);
75 
76  MySqlTrace.TraceEvent(TraceEventType.Information, MySqlTraceEventType.QueryOpened,
77  Resources.TraceQueryOpened, driverId, ThreadID, cmdText);
78  if (normalized_query != null)
79  MySqlTrace.TraceEvent(TraceEventType.Information, MySqlTraceEventType.QueryNormalized,
80  Resources.TraceQueryNormalized, driverId, ThreadID, normalized_query);
81  }

83  protected override int GetResult(int statementId, ref int affectedRows, ref long insertedId)
84  {
85  try
86  {
87  int fieldCount = base.GetResult(statementId, ref affectedRows, ref insertedId);
88  MySqlTrace.TraceEvent(TraceEventType.Information, MySqlTraceEventType.ResultOpened,
89  Resources.TraceResult, driverId, fieldCount, affectedRows, insertedId);
90 
91  return fieldCount;
92  }
93  catch (MySqlException ex)
94  {
95  // we got an error so we report it
96  MySqlTrace.TraceEvent(TraceEventType.Information, MySqlTraceEventType.Error,
97  Resources.TraceOpenResultError, driverId, ex.Number, ex.Message);
98  throw ex;
99  }
100  }

102  public override ResultSet NextResult(int statementId, bool force)
103  {
104  // first let's see if we already have a resultset on this statementId
105  if (activeResult != null)
106  {
107  //oldRS = activeResults[statementId];
108  if (Settings.UseUsageAdvisor)
109  ReportUsageAdvisorWarnings(statementId, activeResult);
110  MySqlTrace.TraceEvent(TraceEventType.Information, MySqlTraceEventType.ResultClosed,
111  Resources.TraceResultClosed, driverId, activeResult.TotalRows, activeResult.SkippedRows,
112  rowSizeInBytes);
113  rowSizeInBytes = 0;
114  activeResult = null;
115  }
116 
117  activeResult = base.NextResult(statementId, force);
118  return activeResult;
119  }

180  public override List<MySqlError> ReportWarnings(MySqlConnection connection)
181  {
182  List<MySqlError> warnings = base.ReportWarnings(connection);
183  foreach (MySqlError warning in warnings)
184  MySqlTrace.TraceEvent(TraceEventType.Warning, MySqlTraceEventType.Warning,
185  Resources.TraceWarning, driverId, warning.Level, warning.Code, warning.Message);

189  private bool AllFieldsAccessed(ResultSet rs)
190  {
191  if (rs.Fields == null || rs.Fields.Length == 0) return true;
192 
193  for (int i = 0; i < rs.Fields.Length; i++)
194  if (!rs.FieldRead(i)) return false;
195  return true;
196  }

198  private void ReportUsageAdvisorWarnings(int statementId, ResultSet rs)
199  {
200  if (!Settings.UseUsageAdvisor) return;
201 
202  if (HasStatus(ServerStatusFlags.NoIndex))
203  MySqlTrace.TraceEvent(TraceEventType.Warning, MySqlTraceEventType.UsageAdvisorWarning,
204  Resources.TraceUAWarningNoIndex, driverId, UsageAdvisorWarningFlags.NoIndex);
205  else if (HasStatus(ServerStatusFlags.BadIndex))
206  MySqlTrace.TraceEvent(TraceEventType.Warning, MySqlTraceEventType.UsageAdvisorWarning,
207  Resources.TraceUAWarningBadIndex, driverId, UsageAdvisorWarningFlags.BadIndex);
208 
209  // report abandoned rows
210  if (rs.SkippedRows > 0)
211  MySqlTrace.TraceEvent(TraceEventType.Warning, MySqlTraceEventType.UsageAdvisorWarning,
212  Resources.TraceUAWarningSkippedRows, driverId, UsageAdvisorWarningFlags.SkippedRows, rs.SkippedRows);
213 
214  // report not all fields accessed
215  if (!AllFieldsAccessed(rs))
216  {
217  StringBuilder notAccessed = new StringBuilder("");
218  string delimiter = "";
219  for (int i = 0; i < rs.Size; i++)
220  if (!rs.FieldRead(i))
221  {
222  notAccessed.AppendFormat("{0}{1}", delimiter, rs.Fields[i].ColumnName);
223  delimiter = ",";
224  }
225  MySqlTrace.TraceEvent(TraceEventType.Warning, MySqlTraceEventType.UsageAdvisorWarning,
226  Resources.TraceUAWarningSkippedColumns, driverId, UsageAdvisorWarningFlags.SkippedColumns,
227  notAccessed.ToString());
228  }
229 
230  // report type conversions if any
231  if (rs.Fields != null)
232  {
233  foreach (MySqlField f in rs.Fields)
234  {
235  StringBuilder s = new StringBuilder();
236  string delimiter = "";
237  foreach (Type t in f.TypeConversions)
238  {
239  s.AppendFormat("{0}{1}", delimiter, t.Name);
240  delimiter = ",";
241  }
242  if (s.Length > 0)
243  MySqlTrace.TraceEvent(TraceEventType.Warning, MySqlTraceEventType.UsageAdvisorWarning,
244  Resources.TraceUAWarningFieldConversion, driverId, UsageAdvisorWarningFlags.FieldConversion,
245  f.ColumnName, s.ToString());
246  }
247  }
248  }
249  }
