186  public override void Close()
187  {
188  if (!isOpen) return;
189 
190  bool shouldCloseConnection = (commandBehavior & CommandBehavior.CloseConnection) != 0;
191  CommandBehavior originalBehavior = commandBehavior;
192 
193  // clear all remaining resultsets
194  try
195  {
196  // Temporarily change to Default behavior to allow NextResult to finish properly.
197  if (!originalBehavior.Equals(CommandBehavior.SchemaOnly))
198  commandBehavior = CommandBehavior.Default;
199  while (NextResult()) { }
200  }
201  catch (MySqlException ex)
202  {
203  // Ignore aborted queries
204  if (!ex.IsQueryAborted)
205  {
206  // ignore IO exceptions.
207  // We are closing or disposing reader, and do not
208  // want exception to be propagated to used. If socket is
209  // is closed on the server side, next query will run into
210  // IO exception. If reader is closed by GC, we also would
211  // like to avoid any exception here.
212  bool isIOException = false;
213  for (Exception exception = ex; exception != null;
214  exception = exception.InnerException)
215  {
216  if (exception is System.IO.IOException)
217  {
218  isIOException = true;
219  break;
220  }
221  }
222  if (!isIOException)
223  {
224  // Ordinary exception (neither IO nor query aborted)
225  throw;
226  }
227  }
228  }
229  catch (System.IO.IOException)
230  {
231  // eat, on the same reason we eat IO exceptions wrapped into
232  // MySqlExceptions reasons, described above.
233  }
234  finally
235  {
236  // always ensure internal reader is null (Bug #55558)
237  connection.Reader = null;
238  commandBehavior = originalBehavior;
239  }
240  // we now give the command a chance to terminate. In the case of
241  // stored procedures it needs to update out and inout parameters
242  command.Close(this);
243  commandBehavior = CommandBehavior.Default;
244 
245  if (this.command.Canceled && connection.driver.Version.isAtLeast(5, 1, 0))
246  {
247  // Issue dummy command to clear kill flag
248  ClearKillFlag();
249  }
250 
251  if (shouldCloseConnection)
252  connection.Close();
253 
254  command = null;
255  connection.IsInUse = false;
256  connection = null;
257  isOpen = false;
258  }

340  public override long GetBytes(int i, long fieldOffset, byte[] buffer, int bufferoffset, int length)
341  {
342  if (i >= FieldCount)
343  Throw(new IndexOutOfRangeException());
344 
345  IMySqlValue val = GetFieldValue(i, false);
346 
347  if (!(val is MySqlBinary) && !(val is MySqlGuid))
348  Throw(new MySqlException("GetBytes can only be called on binary or guid columns"));
349 
350  byte[] bytes = null;
351  if (val is MySqlBinary)
352  bytes = ((MySqlBinary)val).Value;
353  else
354  bytes = ((MySqlGuid)val).Bytes;
355 
356  if (buffer == null)
357  return bytes.Length;
358 
359  if (bufferoffset >= buffer.Length || bufferoffset < 0)
360  Throw(new IndexOutOfRangeException("Buffer index must be a valid index in buffer"));
361  if (buffer.Length < (bufferoffset + length))
362  Throw(new ArgumentException("Buffer is not large enough to hold the requested data"));
363  if (fieldOffset < 0 ||
364  ((ulong)fieldOffset >= (ulong)bytes.Length && (ulong)bytes.Length > 0))
365  Throw(new IndexOutOfRangeException("Data index must be a valid index in the field"));
366 
367  // adjust the length so we don't run off the end
368  if ((ulong)bytes.Length < (ulong)(fieldOffset + length))
369  {
370  length = (int)((ulong)bytes.Length - (ulong)fieldOffset);
371  }
372 
373  Buffer.BlockCopy(bytes, (int)fieldOffset, buffer, (int)bufferoffset, (int)length);
374 
375  return length;
376  }

704  public override DataTable GetSchemaTable()
705  {
706  // Only Results from SQL SELECT Queries
707  // get a DataTable for schema of the result
708  // otherwise, DataTable is null reference
709  if (FieldCount == 0) return null;
710 
711  DataTable dataTableSchema = new DataTable("SchemaTable");
712 
713  dataTableSchema.Columns.Add("ColumnName", typeof(string));
714  dataTableSchema.Columns.Add("ColumnOrdinal", typeof(int));
715  dataTableSchema.Columns.Add("ColumnSize", typeof(int));
716  dataTableSchema.Columns.Add("NumericPrecision", typeof(int));
717  dataTableSchema.Columns.Add("NumericScale", typeof(int));
718  dataTableSchema.Columns.Add("IsUnique", typeof(bool));
719  dataTableSchema.Columns.Add("IsKey", typeof(bool));
720  DataColumn dc = dataTableSchema.Columns["IsKey"];
721  dc.AllowDBNull = true; // IsKey can have a DBNull
722  dataTableSchema.Columns.Add("BaseCatalogName", typeof(string));
723  dataTableSchema.Columns.Add("BaseColumnName", typeof(string));
724  dataTableSchema.Columns.Add("BaseSchemaName", typeof(string));
725  dataTableSchema.Columns.Add("BaseTableName", typeof(string));
726  dataTableSchema.Columns.Add("DataType", typeof(Type));
727  dataTableSchema.Columns.Add("AllowDBNull", typeof(bool));
728  dataTableSchema.Columns.Add("ProviderType", typeof(int));
729  dataTableSchema.Columns.Add("IsAliased", typeof(bool));
730  dataTableSchema.Columns.Add("IsExpression", typeof(bool));
731  dataTableSchema.Columns.Add("IsIdentity", typeof(bool));
732  dataTableSchema.Columns.Add("IsAutoIncrement", typeof(bool));
733  dataTableSchema.Columns.Add("IsRowVersion", typeof(bool));
734  dataTableSchema.Columns.Add("IsHidden", typeof(bool));
735  dataTableSchema.Columns.Add("IsLong", typeof(bool));
736  dataTableSchema.Columns.Add("IsReadOnly", typeof(bool));
737 
738  int ord = 1;
739  for (int i = 0; i < FieldCount; i++)
740  {
741  MySqlField f = resultSet.Fields[i];
742  DataRow r = dataTableSchema.NewRow();
743  r["ColumnName"] = f.ColumnName;
744  r["ColumnOrdinal"] = ord++;
745  r["ColumnSize"] = f.IsTextField ? f.ColumnLength / f.MaxLength : f.ColumnLength;
746  int prec = f.Precision;
747  int pscale = f.Scale;
748  if (prec != -1)
749  r["NumericPrecision"] = (short)prec;
750  if (pscale != -1)
751  r["NumericScale"] = (short)pscale;
752  r["DataType"] = GetFieldType(i);
753  r["ProviderType"] = (int)f.Type;
754  r["IsLong"] = f.IsBlob && f.ColumnLength > 255;
755  r["AllowDBNull"] = f.AllowsNull;
756  r["IsReadOnly"] = false;
757  r["IsRowVersion"] = false;
758  r["IsUnique"] = f.IsUnique;
759  r["IsKey"] = f.IsPrimaryKey;
760  r["IsAutoIncrement"] = f.IsAutoIncrement;
761  r["BaseSchemaName"] = f.DatabaseName;
762  r["BaseCatalogName"] = null;
763  r["BaseTableName"] = f.RealTableName;
764  r["BaseColumnName"] = f.OriginalColumnName;
765 
766  dataTableSchema.Rows.Add(r);
767  }
768 
769  return dataTableSchema;
770  }

812  public override object GetValue(int i)
813  {
814  if (!isOpen)
815  Throw(new Exception("No current query in data reader"));
816  if (i >= FieldCount)
817  Throw(new IndexOutOfRangeException());
818 
819  IMySqlValue val = GetFieldValue(i, false);
820  if (val.IsNull)
821  return DBNull.Value;
822 
823  // if the column is a date/time, then we return a MySqlDateTime
824  // so .ToString() will print '0000-00-00' correctly
825  if (val is MySqlDateTime)
826  {
827  MySqlDateTime dt = (MySqlDateTime)val;
828  if (!dt.IsValidDateTime && connection.Settings.ConvertZeroDateTime)
829  return DateTime.MinValue;
830  else if (connection.Settings.AllowZeroDateTime)
831  return val;
832  else
833  return dt.GetDateTime();
834  }
835  else if (val.Value is string)
836  {
837  Guid guid;
838 
839  Guid.TryParse(val.Value.ToString(), out guid);
840 
841  if(!guid.Equals(Guid.Empty))
842  return guid;
843  }
844 
845  return val.Value;
846  }

936  bool isCaching = command.CommandType == CommandType.TableDirect && command.EnableCaching &&
937  (commandBehavior & CommandBehavior.SequentialAccess) == 0;
938 
939  // this will clear out any unread data
940  if (resultSet != null)
941  {
942  resultSet.Close();
943  if (isCaching)
944  TableCache.AddToCache(command.CommandText, resultSet);
945  }
946 
947  // single result means we only return a single resultset. If we have already
948  // returned one, then we return false
949  // TableDirect is basically a select * from a single table so it will generate
950  // a single result also
951  if (resultSet != null &&
952  ((commandBehavior & CommandBehavior.SingleResult) != 0 || isCaching))
953  return false;
954 
955  // next load up the next resultset if any
956  try
957  {
958  do
959  {
960  resultSet = null;
961  // if we are table caching, then try to retrieve the resultSet from the cache
962  if (isCaching)
963  resultSet = TableCache.RetrieveFromCache(command.CommandText,
964  command.CacheAge);
965 
966  if (resultSet == null)
967  {
968  resultSet = driver.NextResult(Statement.StatementId, false);
969  if (resultSet == null) return false;
970  if (resultSet.IsOutputParameters && command.CommandType == CommandType.StoredProcedure)
971  {
972  StoredProcedure sp = statement as StoredProcedure;
973  sp.ProcessOutputParameters(this);
974  resultSet.Close();
975  if (!sp.ServerProvidingOutputParameters) return false;
976  // if we are using server side output parameters then we will get our ok packet
977  // *after* the output parameters resultset
978  resultSet = driver.NextResult(Statement.StatementId, true);
979  }
980  resultSet.Cached = isCaching;
981  }
982 
983  if (resultSet.Size == 0)
984  {
985  Command.lastInsertedId = resultSet.InsertedId;
986  if (affectedRows == -1)
987  affectedRows = resultSet.AffectedRows;
988  else
989  affectedRows += resultSet.AffectedRows;
990  }
991  } while (resultSet.Size == 0);
992 
993  return true;
994  }
995  catch (MySqlException ex)
996  {
997  if (ex.IsFatal)
998  connection.Abort();
999  if (ex.Number == 0)
1000  throw new MySqlException(Resources.FatalErrorReadingResult, ex);
1001  if ((commandBehavior & CommandBehavior.CloseConnection) != 0)
1002  Close();
1003  throw;
1004  }
1005  }

1077  private void ProcessOutputParameters()
1078  {
1079  // if we are not 5.5 or later or we are not prepared then we are simulating output parameters
1080  // with user variables and they are also string so we have to work some magic with out
1081  // column types before we read the data
1082  if (!driver.SupportsOutputParameters || !command.IsPrepared)
1083  AdjustOutputTypes();
1084 
1085  // now read the output parameters data row
1086  if ((commandBehavior & System.Data.CommandBehavior.SchemaOnly) != 0) return;
1087  resultSet.NextRow(commandBehavior);
1088 
1089  string prefix = "@" + StoredProcedure.ParameterPrefix;
1090 
1091  for (int i = 0; i < FieldCount; i++)
1092  {
1093  string fieldName = GetName(i);
1094  if (fieldName.StartsWith(prefix))
1095  fieldName = fieldName.Remove(0, prefix.Length);
1096  MySqlParameter parameter = command.Parameters.GetParameterFlexible(fieldName, true);
1097  parameter.Value = GetValue(i);
1098  }

1101  private void AdjustOutputTypes()
1102  {
1103  // since MySQL likes to return user variables as strings
1104  // we reset the types of the readers internal value objects
1105  // this will allow those value objects to parse the string based
1106  // return values
1107  for (int i = 0; i < FieldCount; i++)
1108  {
1109  string fieldName = GetName(i);
