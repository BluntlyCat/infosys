693  protected virtual ICollection GetOrphans(ICollection oldElements, ICollection currentElements, string entityName,
694  ISessionImplementor session)
695  {
696  // short-circuit(s)
697  if (currentElements.Count == 0)
698  {
699  // no new elements, the old list contains only Orphans
700  return oldElements;
701  }
702  if (oldElements.Count == 0)
703  {
704  // no old elements, so no Orphans neither
705  return oldElements;
706  }
707 
708  IType idType = session.Factory.GetEntityPersister(entityName).IdentifierType;
709 
710  // create the collection holding the orphans
711  List<object> res = new List<object>();
712 
713  // collect EntityIdentifier(s) of the *current* elements - add them into a HashSet for fast access
714  var currentIds = new HashSet<TypedValue>();
715  foreach (object current in currentElements)
716  {
717  if (current != null && ForeignKeys.IsNotTransient(entityName, current, null, session))
718  {
719  object currentId = ForeignKeys.GetEntityIdentifierIfNotUnsaved(entityName, current, session);
720  currentIds.Add(new TypedValue(idType, currentId, session.EntityMode));
721  }
722  }
723 
724  // iterate over the *old* list
725  foreach (object old in oldElements)
726  {
727  object oldId = ForeignKeys.GetEntityIdentifierIfNotUnsaved(entityName, old, session);
728  if (!currentIds.Contains(new TypedValue(idType, oldId, session.EntityMode)))
729  {
730  res.Add(old);
731  }
732  }
733 
734  return res;
735  }

737  public void IdentityRemove(IList list, object obj, string entityName, ISessionImplementor session)
738  {
