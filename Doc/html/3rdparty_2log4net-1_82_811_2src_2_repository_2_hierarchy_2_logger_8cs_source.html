557  virtual protected void CallAppenders(LoggingEvent loggingEvent)
558  {
559  if (loggingEvent == null)
560  {
561  throw new ArgumentNullException("loggingEvent");
562  }
563 
564  int writes = 0;
565 
566  for(Logger c=this; c != null; c=c.m_parent)
567  {
568  if (c.m_appenderAttachedImpl != null)
569  {
570  // Protected against simultaneous call to addAppender, removeAppender,...
571  c.m_appenderLock.AcquireReaderLock();
572  try
573  {
574  if (c.m_appenderAttachedImpl != null)
575  {
576  writes += c.m_appenderAttachedImpl.AppendLoopOnAppenders(loggingEvent);
577  }
578  }
579  finally
580  {
581  c.m_appenderLock.ReleaseReaderLock();
582  }
583  }
584 
585  if (!c.m_additive)
586  {
587  break;
588  }
589  }
590 
591  // No appenders in hierarchy, warn user only once.
592  //
593  // Note that by including the AppDomain values for the currently running
594  // thread, it becomes much easier to see which application the warning
595  // is from, which is especially helpful in a multi-AppDomain environment
596  // (like IIS with multiple VDIRS). Without this, it can be difficult
597  // or impossible to determine which .config file is missing appender
598  // definitions.
599  //
600  if (!m_hierarchy.EmittedNoAppenderWarning && writes == 0)
601  {
602  LogLog.Debug(declaringType, "No appenders could be found for logger [" + Name + "] repository [" + Repository.Name + "]");
603  LogLog.Debug(declaringType, "Please initialize the log4net system properly.");
604  try
605  {
606  LogLog.Debug(declaringType, " Current AppDomain context information: ");
607  LogLog.Debug(declaringType, " BaseDirectory : " + SystemInfo.ApplicationBaseDirectory);
608 #if !NETCF
609  LogLog.Debug(declaringType, " FriendlyName : " + AppDomain.CurrentDomain.FriendlyName);
610  LogLog.Debug(declaringType, " DynamicDirectory: " + AppDomain.CurrentDomain.DynamicDirectory);
611 #endif
612  }
613  catch(System.Security.SecurityException)
614  {
615  // Insufficient permissions to display info from the AppDomain
616  }
617  m_hierarchy.EmittedNoAppenderWarning = true;
618  }
619  }

700  virtual protected void ForcedLog(LoggingEvent logEvent)
