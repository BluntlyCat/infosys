86  public static string Replace(string template, string placeholder, string replacement, bool wholeWords)
87  {
88  // sometimes a null value will get passed in here -> SqlWhereStrings are a good example
89  if (template == null)
90  {
91  return null;
92  }
93 
94  int loc = template.IndexOf(placeholder);
95  if (loc < 0)
96  {
97  return template;
98  }
99  else
100  {
101  // NH different implementation (NH-1253)
102  string replaceWith = replacement;
103  if(loc + placeholder.Length < template.Length)
104  {
105  string afterPlaceholder = template[loc + placeholder.Length].ToString();
106  //After a token in HQL there can be whitespace, closedparen or comma..
107  if(wholeWords && !(WhiteSpace.Contains(afterPlaceholder) || ClosedParen.Equals(afterPlaceholder) || Comma.Equals(afterPlaceholder)))
108  {
109  //If this is not a full token we don't want to touch it
110  replaceWith = placeholder;
111  }
112  }
113 
114  return
115  new StringBuilder(template.Substring(0, loc)).Append(replaceWith).Append(
116  Replace(template.Substring(loc + placeholder.Length), placeholder, replacement, wholeWords)).ToString();
117  }
118  }

421  public static int CountUnquoted(string str, char character)
422  {
423  if (SingleQuote == character)
424  {
425  throw new ArgumentOutOfRangeException("character", "Unquoted count of quotes is invalid");
426  }
427 
428  // Impl note: takes advantage of the fact that an escaped single quote
429  // embedded within a quote-block can really be handled as two separate
430  // quote-blocks for the purposes of this method...
431  int count = 0;
432  char[] chars = str.ToCharArray();
433  int stringLength = string.IsNullOrEmpty(str) ? 0 : chars.Length;
434  bool inQuote = false;
435  for (int indx = 0; indx < stringLength; indx++)
436  {
437  if (inQuote)
438  {
439  if (SingleQuote == chars[indx])
440  {
441  inQuote = false;
442  }
443  }
444  else if (SingleQuote == chars[indx])
445  {
446  inQuote = true;
447  }
448  else if (chars[indx] == character)
449  {
450  count++;
451  }
452  }

566  {
567  // Remove any generic arguments attached to description
568  int indexOfBacktick = description.IndexOf('`');
569  if (indexOfBacktick > 0)
570  {
