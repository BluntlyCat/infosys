446  protected virtual IList GetResultList(IList results)
447  {
448  var resultCollections = new List<object>(resultCollectionGenericType.Count);
449  for (int i = 0; i < queries.Count; i++)
450  {
451  if (resultCollectionGenericType[i] == typeof(object))
452  {
453  resultCollections.Add(new List<object>());
454  }
455  else
456  {
457  resultCollections.Add(Activator.CreateInstance(typeof(List<>).MakeGenericType(resultCollectionGenericType[i])));
458  }
459  }
460 
461  var multiqueryHolderInstatiator = GetMultiQueryHolderInstatiator();
462  for (int i = 0; i < results.Count; i++)
463  {
464  // First use the transformer of each query transforming each row and then the list
465  // DONE: The behavior when the query has a 'new' instead a transformer is delegated to the Loader
466  var resultList = translators[i].Loader.GetResultList((IList)results[i], Parameters[i].ResultTransformer);
467  // then use the MultiQueryTransformer (if it has some sense...) using, as source, the transformed result.
468  resultList = GetTransformedResults(resultList, multiqueryHolderInstatiator);
469 
470  var queryIndex = translatorQueryMap[i];
471  ArrayHelper.AddAll((IList)resultCollections[queryIndex], resultList);
472  }
473 
474  return resultCollections;
475  }

477  private IList GetTransformedResults(IList source, HolderInstantiator holderInstantiator)
478  {
479  if (!holderInstantiator.IsRequired)
480  {
481  return source;

484  {

502  protected List<object> DoList()
503  {
504  bool statsEnabled = session.Factory.Statistics.IsStatisticsEnabled;
505  var stopWatch = new Stopwatch();
506  if (statsEnabled)
507  {
508  stopWatch.Start();
509  }
510  int rowCount = 0;
511 
512  var results = new List<object>();
513 
514  var hydratedObjects = new List<object>[Translators.Count];
515  List<EntityKey[]>[] subselectResultKeys = new List<EntityKey[]>[Translators.Count];
516  bool[] createSubselects = new bool[Translators.Count];
517 
518  try
519  {
520  using (var reader = resultSetsCommand.GetReader(commandTimeout != RowSelection.NoValue ? commandTimeout : (int?)null))
521  {
522  if (log.IsDebugEnabled)
523  {
524  log.DebugFormat("Executing {0} queries", translators.Count);
525  }
526  for (int i = 0; i < translators.Count; i++)
527  {
528  ITranslator translator = Translators[i];
529  QueryParameters parameter = Parameters[i];
530 
531  int entitySpan = translator.Loader.EntityPersisters.Length;
532  hydratedObjects[i] = entitySpan > 0 ? new List<object>() : null;
533  RowSelection selection = parameter.RowSelection;
534  int maxRows = Loader.Loader.HasMaxRows(selection) ? selection.MaxRows : int.MaxValue;
535  if (!dialect.SupportsLimitOffset || !translator.Loader.UseLimit(selection, dialect))
536  {
537  Loader.Loader.Advance(reader, selection);
538  }
539 
540  LockMode[] lockModeArray = translator.Loader.GetLockModes(parameter.LockModes);
541  EntityKey optionalObjectKey = Loader.Loader.GetOptionalObjectKey(parameter, session);
542 
543  createSubselects[i] = translator.Loader.IsSubselectLoadingEnabled;
544  subselectResultKeys[i] = createSubselects[i] ? new List<EntityKey[]>() : null;
545 
546  translator.Loader.HandleEmptyCollections(parameter.CollectionKeys, reader, session);
547  EntityKey[] keys = new EntityKey[entitySpan]; // we can reuse it each time
548 
549  if (log.IsDebugEnabled)
550  {
551  log.Debug("processing result set");
552  }
553 
554  IList tempResults = new List<object>();
555  int count;
556  for (count = 0; count < maxRows && reader.Read(); count++)
557  {
558  if (log.IsDebugEnabled)
559  {
560  log.Debug("result set row: " + count);
561  }
562 
563  rowCount++;
564  object result = translator.Loader.GetRowFromResultSet(
565  reader, session, parameter, lockModeArray, optionalObjectKey, hydratedObjects[i], keys, true);
566  tempResults.Add(result);
567 
568  if (createSubselects[i])
569  {
570  subselectResultKeys[i].Add(keys);
571  keys = new EntityKey[entitySpan]; //can't reuse in this case
572  }
573  }
574 
575  if (log.IsDebugEnabled)
576  {
577  log.Debug(string.Format("done processing result set ({0} rows)", count));
578  }
579 
580  results.Add(tempResults);
581 
582  if (log.IsDebugEnabled)
583  {
584  log.DebugFormat("Query {0} returned {1} results", i, tempResults.Count);
585  }
586 
587  reader.NextResult();
588  }
589 
590  for (int i = 0; i < translators.Count; i++)
591  {
592  ITranslator translator = translators[i];
593  QueryParameters parameter = parameters[i];
594 
595  translator.Loader.InitializeEntitiesAndCollections(hydratedObjects[i], reader, session, false);
596 
597  if (createSubselects[i])
598  {
599  translator.Loader.CreateSubselects(subselectResultKeys[i], parameter, session);
600  }
601  }
602  }
603  }
604  catch (Exception sqle)
605  {
606  var message = string.Format("Failed to execute multi query: [{0}]", resultSetsCommand.Sql);
607  log.Error(message, sqle);
608  throw ADOExceptionHelper.Convert(session.Factory.SQLExceptionConverter, sqle, "Failed to execute multi query", resultSetsCommand.Sql);
609  }
610 
611  if (statsEnabled)
612  {
613  stopWatch.Stop();
614  session.Factory.StatisticsImplementor.QueryExecuted(string.Format("{0} queries (MultiQuery)", translators.Count), rowCount, stopWatch.Elapsed);
615  }
616  return results;
617  }

675  private IList ListUsingQueryCache()
676  {
677  IQueryCache queryCache = session.Factory.GetQueryCache(cacheRegion);
678 
679  ISet<FilterKey> filterKeys = FilterKey.CreateFilterKeys(session.EnabledFilters, session.EntityMode);
680 
681  ISet<string> querySpaces = new HashSet<string>();
682  List<IType[]> resultTypesList = new List<IType[]>(Translators.Count);
683  for (int i = 0; i < Translators.Count; i++)
684  {
685  ITranslator queryTranslator = Translators[i];
686  querySpaces.UnionWith(queryTranslator.QuerySpaces);
687  resultTypesList.Add(queryTranslator.ReturnTypes);
688  }
689  int[] firstRows = new int[Parameters.Count];
690  int[] maxRows = new int[Parameters.Count];
691  for (int i = 0; i < Parameters.Count; i++)
692  {
693  RowSelection rowSelection = Parameters[i].RowSelection;
694  firstRows[i] = rowSelection.FirstRow;
695  maxRows[i] = rowSelection.MaxRows;
696  }
697 
698  MultipleQueriesCacheAssembler assembler = new MultipleQueriesCacheAssembler(resultTypesList);
699 
700  QueryKey key = new QueryKey(session.Factory, SqlString, combinedParameters, filterKeys)
701  .SetFirstRows(firstRows)
702  .SetMaxRows(maxRows);
703 
704  IList result = assembler.GetResultFromQueryCache(session, combinedParameters, querySpaces, queryCache, key);
705 
706  if (result == null)
707  {
708  log.Debug("Cache miss for multi query");
709  var list = DoList();
710  queryCache.Put(key, new ICacheAssembler[] { assembler }, new object[] { list }, false, session);
711  result = list;
712  }
713 
714  return GetResultList(result);
715  }

729  private QueryParameters CreateCombinedQueryParameters()
730  {
731  QueryParameters combinedQueryParameters = new QueryParameters();
732  combinedQueryParameters.ForceCacheRefresh = forceCacheRefresh;
733  combinedQueryParameters.NamedParameters = new Dictionary<string, TypedValue>();
734  var positionalParameterTypes = new List<IType>();
735  var positionalParameterValues = new List<object>();
736  int index = 0;
737  foreach (QueryParameters queryParameters in Parameters)
738  {
739  foreach (KeyValuePair<string, TypedValue> dictionaryEntry in queryParameters.NamedParameters)
740  {
741  combinedQueryParameters.NamedParameters.Add(dictionaryEntry.Key + index, dictionaryEntry.Value);
742  }
743  index += 1;
744  positionalParameterTypes.AddRange(queryParameters.PositionalParameterTypes);
745  positionalParameterValues.AddRange(queryParameters.PositionalParameterValues);
746  }
747  combinedQueryParameters.PositionalParameterTypes = positionalParameterTypes.ToArray();
748  combinedQueryParameters.PositionalParameterValues = positionalParameterValues.ToArray();
749  return combinedQueryParameters;
750  }
