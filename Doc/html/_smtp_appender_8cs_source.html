360  override protected void SendBuffer(LoggingEvent[] events)
361  {
362  // Note: this code already owns the monitor for this
363  // appender. This frees us from needing to synchronize again.
364  try
365  {
366  StringWriter writer = new StringWriter(System.Globalization.CultureInfo.InvariantCulture);
367 
368  string t = Layout.Header;
369  if (t != null)
370  {
371  writer.Write(t);
372  }
373 
374  for(int i = 0; i < events.Length; i++)
375  {
376  // Render the event and append the text to the buffer
377  RenderLoggingEvent(writer, events[i]);
378  }
379 
380  t = Layout.Footer;
381  if (t != null)
382  {
383  writer.Write(t);
384  }
385 
386  SendEmail(writer.ToString());
387  }
388  catch(Exception e)
389  {
390  ErrorHandler.Error("Error occurred while sending e-mail notification.", e);
391  }
392  }
393 
394  #endregion // Override implementation of BufferingAppenderSkeleton

420  virtual protected void SendEmail(string messageBody)
421  {
422 #if NET_2_0
423  // .NET 2.0 has a new API for SMTP email System.Net.Mail
424  // This API supports credentials and multiple hosts correctly.
425  // The old API is deprecated.
426 
427  // Create and configure the smtp client
428  SmtpClient smtpClient = new SmtpClient();
429  if (!String.IsNullOrEmpty(m_smtpHost))
430  {
431  smtpClient.Host = m_smtpHost;
432  }
433  smtpClient.Port = m_port;
434  smtpClient.DeliveryMethod = SmtpDeliveryMethod.Network;
435  smtpClient.EnableSsl = m_enableSsl;
436 
437  if (m_authentication == SmtpAuthentication.Basic)
438  {
439  // Perform basic authentication
440  smtpClient.Credentials = new System.Net.NetworkCredential(m_username, m_password);
441  }
442  else if (m_authentication == SmtpAuthentication.Ntlm)
443  {
444  // Perform integrated authentication (NTLM)
445  smtpClient.Credentials = System.Net.CredentialCache.DefaultNetworkCredentials;
446  }
447 
448  using (MailMessage mailMessage = new MailMessage())
449  {
450  mailMessage.Body = messageBody;
451  mailMessage.From = new MailAddress(m_from);
452  mailMessage.To.Add(m_to);
453  if (!String.IsNullOrEmpty(m_cc))
454  {
455  mailMessage.CC.Add(m_cc);
456  }
457  if (!String.IsNullOrEmpty(m_bcc))
458  {
459  mailMessage.Bcc.Add(m_bcc);
460  }
461  if (!String.IsNullOrEmpty(m_replyTo))
462  {
463  // .NET 4.0 warning CS0618: 'System.Net.Mail.MailMessage.ReplyTo' is obsolete:
464  // 'ReplyTo is obsoleted for this type. Please use ReplyToList instead which can accept multiple addresses. http://go.microsoft.com/fwlink/?linkid=14202'
465 #if !NET_4_0
466  mailMessage.ReplyTo = new MailAddress(m_replyTo);
467 #else
468  mailMessage.ReplyToList.Add(new MailAddress(m_replyTo));
469 #endif
470  }
471  mailMessage.Subject = m_subject;
472  mailMessage.Priority = m_mailPriority;
473 
474  // TODO: Consider using SendAsync to send the message without blocking. This would be a change in
475  // behaviour compared to .NET 1.x. We would need a SendCompletedCallback to log errors.
476  smtpClient.Send(mailMessage);
477  }
478 #else
479  // .NET 1.x uses the System.Web.Mail API for sending Mail
480 
481  MailMessage mailMessage = new MailMessage();
482  mailMessage.Body = messageBody;
483  mailMessage.From = m_from;
484  mailMessage.To = m_to;
485  if (m_cc != null && m_cc.Length > 0)
486  {
487  mailMessage.Cc = m_cc;
488  }
489  if (m_bcc != null && m_bcc.Length > 0)
490  {
491  mailMessage.Bcc = m_bcc;
492  }
493  mailMessage.Subject = m_subject;
494  mailMessage.Priority = m_mailPriority;
495 
496 #if NET_1_1
497  // The Fields property on the MailMessage allows the CDO properties to be set directly.
498  // This property is only available on .NET Framework 1.1 and the implementation must understand
499  // the CDO properties. For details of the fields available in CDO see:
500  //
501  // http://msdn.microsoft.com/library/default.asp?url=/library/en-us/cdosys/html/_cdosys_configuration_coclass.asp
502  //
503 
504  try
505  {
506  if (m_authentication == SmtpAuthentication.Basic)
507  {
508  // Perform basic authentication
509  mailMessage.Fields.Add("http://schemas.microsoft.com/cdo/configuration/smtpauthenticate", 1);
510  mailMessage.Fields.Add("http://schemas.microsoft.com/cdo/configuration/sendusername", m_username);
511  mailMessage.Fields.Add("http://schemas.microsoft.com/cdo/configuration/sendpassword", m_password);
512  }
513  else if (m_authentication == SmtpAuthentication.Ntlm)
514  {
515  // Perform integrated authentication (NTLM)
516  mailMessage.Fields.Add("http://schemas.microsoft.com/cdo/configuration/smtpauthenticate", 2);
517  }
518 
519  // Set the port if not the default value
520  if (m_port != 25)
521  {
522  mailMessage.Fields.Add("http://schemas.microsoft.com/cdo/configuration/smtpserverport", m_port);
523  }
524  }
525  catch(MissingMethodException missingMethodException)
526  {
527  // If we were compiled against .NET 1.1 but are running against .NET 1.0 then
528  // we will get a MissingMethodException when accessing the MailMessage.Fields property.
529 
530  ErrorHandler.Error("SmtpAppender: Authentication and server Port are only supported when running on the MS .NET 1.1 framework", missingMethodException);
531  }
532 #else
533  if (m_authentication != SmtpAuthentication.None)
534  {
535  ErrorHandler.Error("SmtpAppender: Authentication is only supported on the MS .NET 1.1 or MS .NET 2.0 builds of log4net");
536  }
537 
538  if (m_port != 25)
539  {
540  ErrorHandler.Error("SmtpAppender: Server Port is only supported on the MS .NET 1.1 or MS .NET 2.0 builds of log4net");
541  }
542 #endif // if NET_1_1
543 
544  if (m_smtpHost != null && m_smtpHost.Length > 0)
545  {
546  SmtpMail.SmtpServer = m_smtpHost;
547  }
548 
549  SmtpMail.Send(mailMessage);
550 #endif // if NET_2_0
551  }
