184  IList<JToken> ancestors = Ancestors().Reverse().ToList();
185  ancestors.Add(this);
186 
187  StringBuilder sb = new StringBuilder();
188  for (int i = 0; i < ancestors.Count; i++)
189  {
190  JToken current = ancestors[i];
191  JToken next = (i + 1 < ancestors.Count) ? ancestors[i + 1] : null;
192 
193  if (next != null)
194  {
195  switch (current.Type)
196  {
197  case JTokenType.Property:
198  JProperty property = (JProperty)current;
199 
200  if (sb.Length > 0)
201  sb.Append(".");
202 
203  sb.Append(property.Name);
204  break;
205  case JTokenType.Array:
206  case JTokenType.Constructor:
207  int index = ((IList<JToken>)current).IndexOf(next);
208 
209  sb.Append("[" + index + "]");
210  break;
211  }
212  }
213  }
214 
215  return sb.ToString();
216  }
