109  private int FindFromClauseInPart(string part)
110  {
111  int afterLastClosingParenIndex = 0;
112  int fromIndex = StringHelper.IndexOfCaseInsensitive(part, FromClauseToken);
113 
114  for (int i = 0; i < part.Length; i++)
115  {
116  if (parenNestCount == 0 && i > fromIndex)
117  {
118  break;
119  }
120 
121  if (part[i] == '(')
122  {
123  ++parenNestCount;
124  }
125  else if (part[i] == ')')
126  {
127  --parenNestCount;
128  if (parenNestCount != 0)
129  {
130  continue;
131  }
132 
133  afterLastClosingParenIndex = i + 1;
134  fromIndex = StringHelper.IndexOfCaseInsensitive(part, FromClauseToken, afterLastClosingParenIndex);
135  }
136  }
137 
138  if (afterLastClosingParenIndex == 0)
139  {
140  fromIndex = StringHelper.IndexOfCaseInsensitive(part, FromClauseToken);
141  }
142 
143  if(parenNestCount > 0)
144  return -1;
145
