122  private void CheckNaturalId(IEntityPersister persister, EntityEntry entry, object[] current, object[] loaded, EntityMode entityMode, ISessionImplementor session)
123  {
124  if (persister.HasNaturalIdentifier && entry.Status != Status.ReadOnly)
125  {
126  object[] snapshot = null;
127  IType[] types = persister.PropertyTypes;
128  int[] props = persister.NaturalIdentifierProperties;

130  for (int i = 0; i < props.Length; i++)
131  {

134  {

143  }

146  loadedVal = loaded[prop];
147  }

205  private bool ScheduleUpdate(FlushEntityEvent @event)
206  {
207  EntityEntry entry = @event.EntityEntry;
208  IEventSource session = @event.Session;
209  EntityMode entityMode = session.EntityMode;
210  object entity = @event.Entity;
211  Status status = entry.Status;
212  IEntityPersister persister = entry.Persister;
213  object[] values = @event.PropertyValues;
214 
215  if (log.IsDebugEnabled)
216  {
217  if (status == Status.Deleted)
218  {
219  if (!persister.IsMutable)
220  {
221  log.Debug("Updating immutable, deleted entity: " + MessageHelper.InfoString(persister, entry.Id, session.Factory));
222  }
223  else if (!entry.IsModifiableEntity())
224  {
225  log.Debug("Updating non-modifiable, deleted entity: " + MessageHelper.InfoString(persister, entry.Id, session.Factory));
226  }
227  else
228  {
229  log.Debug("Updating deleted entity: " + MessageHelper.InfoString(persister, entry.Id, session.Factory));
230  }
231  }
232  else
233  {
234  log.Debug("Updating entity: " + MessageHelper.InfoString(persister, entry.Id, session.Factory));
235  }
236  }
237 
238  bool intercepted;
239 
240  if (!entry.IsBeingReplicated)
241  {
242  // give the Interceptor a chance to process property values, if the properties
243  // were modified by the Interceptor, we need to set them back to the object
244  intercepted = HandleInterception(@event);
245  }
246  else
247  {
248  intercepted = false;
249  }
250 
251  Validate(entity, persister, status, entityMode);
252 
253  // increment the version number (if necessary)
254  object nextVersion = GetNextVersion(@event);
255 
256  // if it was dirtied by a collection only
257  int[] dirtyProperties = @event.DirtyProperties;
258  if (@event.DirtyCheckPossible && dirtyProperties == null)
259  {
260  if (!intercepted && !@event.HasDirtyCollection)
261  {
262  throw new AssertionFailure("dirty, but no dirty properties");
263  }
264  dirtyProperties = ArrayHelper.EmptyIntArray;
265  }
266 
267  // check nullability but do not perform command execute
268  // we'll use scheduled updates for that.
269  new Nullability(session).CheckNullability(values, persister, true);
270 
271  // schedule the update
272  // note that we intentionally do _not_ pass in currentPersistentState!
273  session.ActionQueue.AddAction(
274  new EntityUpdateAction(
275  entry.Id,
276  values,
277  dirtyProperties,
278  @event.HasDirtyCollection,
279  status == Status.Deleted && !entry.IsModifiableEntity() ? persister.GetPropertyValues(entity, entityMode) : entry.LoadedState,
280  entry.Version,
281  nextVersion,
282  entity,
283  persister,
284  session));
285 
286  return intercepted;

430  protected virtual void DirtyCheck(FlushEntityEvent @event)
431  {
432  object entity = @event.Entity;
433  object[] values = @event.PropertyValues;
434  ISessionImplementor session = @event.Session;
435  EntityEntry entry = @event.EntityEntry;
436  IEntityPersister persister = entry.Persister;
437  object id = entry.Id;
438  object[] loadedState = entry.LoadedState;
439 
440  int[] dirtyProperties = session.Interceptor.FindDirty(entity, id, values, loadedState, persister.PropertyNames, persister.PropertyTypes);
441 
442  @event.DatabaseSnapshot = null;
443 
444  bool interceptorHandledDirtyCheck;
445  bool cannotDirtyCheck;
446 
447  if (dirtyProperties == null)
448  {
449  // Interceptor returned null, so do the dirtycheck ourself, if possible
450  interceptorHandledDirtyCheck = false;
451 
452  cannotDirtyCheck = loadedState == null; // object loaded by update()
453  if (!cannotDirtyCheck)
454  {
455  // dirty check against the usual snapshot of the entity
456  dirtyProperties = persister.FindDirty(values, loadedState, entity, session);
457  }
458  else if (entry.Status == Status.Deleted && !@event.EntityEntry.IsModifiableEntity())
459  {
460  // A non-modifiable (e.g., read-only or immutable) entity needs to be have
461  // references to transient entities set to null before being deleted. No other
462  // fields should be updated.
463  if (values != entry.DeletedState )
464  {
465  throw new InvalidOperationException("Entity has status Status.Deleted but values != entry.DeletedState");
466  }
467  // Even if loadedState == null, we can dirty-check by comparing currentState and
468  // entry.getDeletedState() because the only fields to be updated are those that
469  // refer to transient entities that are being set to null.
470  // - currentState contains the entity's current property values.
471  // - entry.getDeletedState() contains the entity's current property values with
472  // references to transient entities set to null.
473  // - dirtyProperties will only contain properties that refer to transient entities
474  object[] currentState = persister.GetPropertyValues(@event.Entity, @event.Session.EntityMode);
475  dirtyProperties = persister.FindDirty(entry.DeletedState, currentState, entity, session);
476  cannotDirtyCheck = false;
477  }
478  else
479  {
480  // dirty check against the database snapshot, if possible/necessary
481  object[] databaseSnapshot = GetDatabaseSnapshot(session, persister, id);
482  if (databaseSnapshot != null)
483  {
484  dirtyProperties = persister.FindModified(databaseSnapshot, values, entity, session);
485  cannotDirtyCheck = false;
486  @event.DatabaseSnapshot = databaseSnapshot;
487  }
488  }
489  }
490  else
491  {
492  // the Interceptor handled the dirty checking
493  cannotDirtyCheck = false;
494  interceptorHandledDirtyCheck = true;
495  }
496 
497  @event.DirtyProperties = dirtyProperties;
498  @event.DirtyCheckHandledByInterceptor = interceptorHandledDirtyCheck;
499  @event.DirtyCheckPossible = !cannotDirtyCheck;
500  }
