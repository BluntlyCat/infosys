101  [SuppressMessage("Microsoft.Reliability", "CA2000:DisposeObjectsBeforeLosingScope", Justification = "this._key disposed in Dispose(bool) method.")]
102  private void Open(Stream privateKey, string passPhrase)
103  {
104  if (privateKey == null)
105  throw new ArgumentNullException("privateKey");
106 
107  Match privateKeyMatch = null;
108 
109  using (StreamReader sr = new StreamReader(privateKey))
110  {
111  var text = sr.ReadToEnd();
112  privateKeyMatch = _privateKeyRegex.Match(text);
113  }
114 
115  if (!privateKeyMatch.Success)
116  {
117  throw new SshException("Invalid private key file.");
118  }
119 
120  var keyName = privateKeyMatch.Result("${keyName}");
121  var cipherName = privateKeyMatch.Result("${cipherName}");
122  var salt = privateKeyMatch.Result("${salt}");
123  var data = privateKeyMatch.Result("${data}");
124 
125  var binaryData = System.Convert.FromBase64String(data);
126 
127  byte[] decryptedData = null;
128 
129  if (!string.IsNullOrEmpty(cipherName) && !string.IsNullOrEmpty(salt))
130  {
131  if (string.IsNullOrEmpty(passPhrase))
132  throw new SshPassPhraseNullOrEmptyException("Private key is encrypted but passphrase is empty.");
133 
134  byte[] binarySalt = new byte[salt.Length / 2];
135  for (int i = 0; i < binarySalt.Length; i++)
136  binarySalt[i] = Convert.ToByte(salt.Substring(i * 2, 2), 16);
137 
138  CipherInfo cipher = null;
139  switch (cipherName)
140  {
141  case "DES-EDE3-CBC":
142  cipher = new CipherInfo(192, (key, iv) => { return new TripleDesCipher(key, new CbcCipherMode(iv), new PKCS7Padding()); });
143  break;
144  case "DES-EDE3-CFB":
145  cipher = new CipherInfo(192, (key, iv) => { return new TripleDesCipher(key, new CfbCipherMode(iv), new PKCS7Padding()); });
146  break;
147  case "DES-CBC":
148  cipher = new CipherInfo(64, (key, iv) => { return new DesCipher(key, new CbcCipherMode(iv), new PKCS7Padding()); });
149  break;
150  case "AES-128-CBC":
151  cipher = new CipherInfo(128, (key, iv) => { return new AesCipher(key, new CbcCipherMode(iv), new PKCS7Padding()); });
152  break;
153  case "AES-192-CBC":
154  cipher = new CipherInfo(192, (key, iv) => { return new AesCipher(key, new CbcCipherMode(iv), new PKCS7Padding()); });
155  break;
156  case "AES-256-CBC":
157  cipher = new CipherInfo(256, (key, iv) => { return new AesCipher(key, new CbcCipherMode(iv), new PKCS7Padding()); });
158  break;
159  default:
160  throw new SshException(string.Format(CultureInfo.CurrentCulture, "Private key cipher \"{0}\" is not supported.", cipherName));
161  }
162 
163  decryptedData = DecryptKey(cipher, binaryData, passPhrase, binarySalt);
164  }
165  else
166  {
167  decryptedData = binaryData;
168  }
169 
170  switch (keyName)
171  {
172  case "RSA":
173  this._key = new RsaKey(decryptedData.ToArray());
174  this.HostKey = new KeyHostAlgorithm("ssh-rsa", this._key);
175  break;
176  case "DSA":
177  this._key = new DsaKey(decryptedData.ToArray());
178  this.HostKey = new KeyHostAlgorithm("ssh-dss", this._key);
179  break;
180  case "SSH2 ENCRYPTED":
181  var reader = new SshDataReader(decryptedData);
182  var magicNumber = reader.ReadUInt32();
183  if (magicNumber != 0x3f6ff9eb)
184  {
185  throw new SshException("Invalid SSH2 private key.");
186  }
187 
188  var totalLength = reader.ReadUInt32(); // Read total bytes length including magic number
189  var keyType = reader.ReadString();
190  var ssh2CipherName = reader.ReadString();
191  var blobSize = (int)reader.ReadUInt32();
192 
193  byte[] keyData = null;
194  if (ssh2CipherName == "none")
195  {
196  keyData = reader.ReadBytes(blobSize);
197  }
198  //else if (ssh2CipherName == "3des-cbc")

202  // keyData = ssh2Сipher.Decrypt(reader.ReadBytes(blobSize));
203  //}
204  else
205  {
206  throw new SshException(string.Format("Cipher method '{0}' is not supported.", cipherName));
207  }
208 
209  // TODO: Create two specific data types to avoid using SshDataReader class
210 
211  reader = new SshDataReader(keyData);
212 
213  var decryptedLength = reader.ReadUInt32();
214 
215  if (decryptedLength + 4 != blobSize)
216  throw new SshException("Invalid passphrase.");
217 
218  if (keyType == "if-modn{sign{rsa-pkcs1-sha1},encrypt{rsa-pkcs1v2-oaep}}")
219  {
220  var exponent = reader.ReadBigIntWithBits();//e
221  var d = reader.ReadBigIntWithBits();//d
222  var modulus = reader.ReadBigIntWithBits();//n
223  var inverseQ = reader.ReadBigIntWithBits();//u
224  var q = reader.ReadBigIntWithBits();//p
225  var p = reader.ReadBigIntWithBits();//q
226  this._key = new RsaKey(modulus, exponent, d, p, q, inverseQ);
227  this.HostKey = new KeyHostAlgorithm("ssh-rsa", this._key);
228  }
229  else if (keyType == "dl-modp{sign{dsa-nist-sha1},dh{plain}}")
230  {
231  var zero = reader.ReadUInt32();
232  if (zero != 0)
233  {
234  throw new SshException("Invalid private key");
235  }
236  var p = reader.ReadBigIntWithBits();
237  var g = reader.ReadBigIntWithBits();
238  var q = reader.ReadBigIntWithBits();
239  var y = reader.ReadBigIntWithBits();
240  var x = reader.ReadBigIntWithBits();
241  this._key = new DsaKey(p, q, g, y, x);
242  this.HostKey = new KeyHostAlgorithm("ssh-dss", this._key);
243  }
244  else
245  {
246  throw new NotSupportedException(string.Format("Key type '{0}' is not supported.", keyType));
247  }
248  break;
249  default:
250  throw new NotSupportedException(string.Format(CultureInfo.CurrentCulture, "Key '{0}' is not supported.", keyName));
251  }

254  private static byte[] GetCipherKey(string passphrase, int length)
255  {
256  List<byte> cipherKey = new List<byte>();
257 
258  using (var md5 = new MD5Hash())
259  {
260  byte[] passwordBytes = Encoding.UTF8.GetBytes(passphrase);
261 
262  var hash = md5.ComputeHash(passwordBytes.ToArray()).AsEnumerable();
263 
264  cipherKey.AddRange(hash);
265 
266  while (cipherKey.Count < length)
267  {
268  hash = passwordBytes.Concat(hash);
269 
270  hash = md5.ComputeHash(hash.ToArray());
271 
272  cipherKey.AddRange(hash);
273  }
274  }
275 
276  return cipherKey.Take(length).ToArray();
277  }

289  private static byte[] DecryptKey(CipherInfo cipherInfo, byte[] cipherData, string passPhrase, byte[] binarySalt)
290  {
291  if (cipherInfo == null)
292  throw new ArgumentNullException("cipherInfo");
293 
294  if (cipherData == null)
295  throw new ArgumentNullException("cipherData");
296 
297  if (binarySalt == null)
298  throw new ArgumentNullException("binarySalt");
299 
300  List<byte> cipherKey = new List<byte>();
301 
302  using (var md5 = new MD5Hash())
303  {
304  var passwordBytes = Encoding.UTF8.GetBytes(passPhrase);
305 
306  // Use 8 bytes binary salkt
307  var initVector = passwordBytes.Concat(binarySalt.Take(8));
308 
309  var hash = md5.ComputeHash(initVector.ToArray()).AsEnumerable();
310 
311  cipherKey.AddRange(hash);
312 
313  while (cipherKey.Count < cipherInfo.KeySize / 8)
314  {
315  hash = hash.Concat(initVector);
316 
317  hash = md5.ComputeHash(hash.ToArray());
318 
319  cipherKey.AddRange(hash);
320  }
321  }
322 
323  var cipher = cipherInfo.Cipher(cipherKey.ToArray(), binarySalt);
324 
325  return cipher.Decrypt(cipherData);
326  }
