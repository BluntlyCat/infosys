291  private static IBatcherFactory CreateBatcherFactory(IDictionary<string, string> properties, int batchSize, IConnectionProvider connectionProvider)
292  {
293  System.Type tBatcher = typeof (NonBatchingBatcherFactory);
294  string batcherClass = PropertiesHelper.GetString(Environment.BatchStrategy, properties, null);
295  if (string.IsNullOrEmpty(batcherClass))
296  {
297  if (batchSize > 0)
298  {
299  // try to get the BatcherFactory from the Drive if not available use NonBatchingBatcherFactory
300  IEmbeddedBatcherFactoryProvider ebfp = connectionProvider.Driver as IEmbeddedBatcherFactoryProvider;
301  if (ebfp != null && ebfp.BatcherFactoryClass != null)
302  tBatcher = ebfp.BatcherFactoryClass;
303  }
304  }
305  else
306  {
307  tBatcher = ReflectHelper.ClassForName(batcherClass);
308  }
309  log.Info("Batcher factory: " + tBatcher.AssemblyQualifiedName);
310  try
311  {
312  return (IBatcherFactory) Environment.BytecodeProvider.ObjectsFactory.CreateInstance(tBatcher);
313  }
314  catch (Exception cnfe)
315  {
316  throw new HibernateException("Could not instantiate BatcherFactory: " + batcherClass, cnfe);
317  }
318  }
