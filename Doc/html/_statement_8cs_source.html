91  public virtual bool ExecuteNext()
92  {
93  if (buffers.Count == 0)
94  return false;
95 
96  MySqlPacket packet = (MySqlPacket)buffers[0];

103  protected virtual void BindParameters()
104  {
105  MySqlParameterCollection parameters = command.Parameters;
106  int index = 0;
107 
108  while (true)
109  {
110  InternalBindParameters(ResolvedCommandText, parameters, null);
111 
112  // if we are not batching, then we are done. This is only really relevant the
113  // first time through
114  if (command.Batch == null) return;
115  while (index < command.Batch.Count)
116  {
117  MySqlCommand batchedCmd = command.Batch[index++];
118  MySqlPacket packet = (MySqlPacket)buffers[buffers.Count - 1];
119 
120  // now we make a guess if this statement will fit in our current stream
121  long estimatedCmdSize = batchedCmd.EstimatedSize();
122  if (((packet.Length - 4) + estimatedCmdSize) > Connection.driver.MaxPacketSize)
123  {
124  // it won't, so we setup to start a new run from here
125  parameters = batchedCmd.Parameters;
126  break;
127  }
128 
129  // looks like we might have room for it so we remember the current end of the stream
130  buffers.RemoveAt(buffers.Count - 1);
131  //long originalLength = packet.Length - 4;
132 
133  // and attempt to stream the next command
134  string text = ResolvedCommandText;
135  if (text.StartsWith("(", StringComparison.Ordinal))
136  packet.WriteStringNoNull(", ");
137  else
138  packet.WriteStringNoNull("; ");
139  InternalBindParameters(text, batchedCmd.Parameters, packet);
140  if ((packet.Length - 4) > Connection.driver.MaxPacketSize)
141  {
142  //TODO
143  //stream.InternalBuffer.SetLength(originalLength);
144  parameters = batchedCmd.Parameters;
145  break;
146  }
147  }
148  if (index == command.Batch.Count)
149  return;
150  }
151  }

153  private void InternalBindParameters(string sql, MySqlParameterCollection parameters,
154  MySqlPacket packet)
155  {
156  bool sqlServerMode = command.Connection.Settings.SqlServerMode;
157 
158  if (packet == null)
159  {
160  packet = new MySqlPacket(Driver.Encoding);
161  packet.Version = Driver.Version;
162  packet.WriteByte(0);
163  }
164 
165  MySqlTokenizer tokenizer = new MySqlTokenizer(sql);
166  tokenizer.ReturnComments = true;
167  tokenizer.SqlServerMode = sqlServerMode;
168 
169  int pos = 0;
170  string token = tokenizer.NextToken();
171  int parameterCount = 0;
172  while (token != null)
173  {
174  // serialize everything that came before the token (i.e. whitespace)
175  packet.WriteStringNoNull(sql.Substring(pos, tokenizer.StartIndex - pos));
176  pos = tokenizer.StopIndex;
177  if (MySqlTokenizer.IsParameter(token))
178  {
179  if ((!parameters.containsUnnamedParameters && token.Length == 1 && parameterCount > 0) || parameters.containsUnnamedParameters && token.Length > 1)
180  throw new MySqlException(Resources.MixedParameterNamingNotAllowed);
181 
182  parameters.containsUnnamedParameters = token.Length == 1;
183  if (SerializeParameter(parameters, packet, token, parameterCount))
184  token = null;
185  parameterCount++;
186  }
187  if (token != null)
188  {
189  if (sqlServerMode && tokenizer.Quoted && token.StartsWith("[", StringComparison.Ordinal))
190  token = String.Format("`{0}`", token.Substring(1, token.Length - 2));
191  packet.WriteStringNoNull(token);
192  }
193  token = tokenizer.NextToken();
194  }
195  buffers.Add(packet);
196  }

198  protected virtual bool ShouldIgnoreMissingParameter(string parameterName)
199  {
200  if (Connection.Settings.AllowUserVariables)
201  return true;
202  if (parameterName.StartsWith("@" + StoredProcedure.ParameterPrefix, StringComparison.OrdinalIgnoreCase))
203  return true;
204  if (parameterName.Length > 1 &&
205  (parameterName[1] == '`' || parameterName[1] == '\''))
206  return true;
207  return false;
208  }

219  private bool SerializeParameter(MySqlParameterCollection parameters,
220  MySqlPacket packet, string parmName, int parameterIndex)
221  {
222  MySqlParameter parameter = null;
223 
224  if (!parameters.containsUnnamedParameters)
225  parameter = parameters.GetParameterFlexible(parmName, false);
226  else
227  {
228  if (parameterIndex <= parameters.Count)
229  parameter = parameters[parameterIndex];
230  else
231  throw new MySqlException(Resources.ParameterIndexNotFound);
232  }
233 
234  if (parameter == null)
235  {
236  // if we are allowing user variables and the parameter name starts with @
237  // then we can't throw an exception
238  if (parmName.StartsWith("@", StringComparison.Ordinal) && ShouldIgnoreMissingParameter(parmName))
239  return false;
240  throw new MySqlException(
241  String.Format(Resources.ParameterMustBeDefined, parmName));
242  }
243  parameter.Serialize(packet, false, Connection.Settings);
244  return true;
245  }
