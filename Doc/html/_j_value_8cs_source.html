222  internal static int Compare(JTokenType valueType, object objA, object objB)
223  {
224  if (objA == null && objB == null)
225  return 0;
226  if (objA != null && objB == null)
227  return 1;
228  if (objA == null && objB != null)
229  return -1;
230 
231  switch (valueType)
232  {
233  case JTokenType.Integer:
234 #if !(NET20 || NET35 || SILVERLIGHT || PORTABLE40 || PORTABLE)
235  if (objA is BigInteger)
236  return CompareBigInteger((BigInteger)objA, objB);
237  if (objB is BigInteger)
238  return -CompareBigInteger((BigInteger)objB, objA);
239 #endif
240  if (objA is ulong || objB is ulong || objA is decimal || objB is decimal)
241  return Convert.ToDecimal(objA, CultureInfo.InvariantCulture).CompareTo(Convert.ToDecimal(objB, CultureInfo.InvariantCulture));
242  else if (objA is float || objB is float || objA is double || objB is double)
243  return CompareFloat(objA, objB);
244  else
245  return Convert.ToInt64(objA, CultureInfo.InvariantCulture).CompareTo(Convert.ToInt64(objB, CultureInfo.InvariantCulture));
246  case JTokenType.Float:
247 #if !(NET20 || NET35 || SILVERLIGHT || PORTABLE40 || PORTABLE)

252 #endif

254  case JTokenType.Comment:
255  case JTokenType.String:
256  case JTokenType.Raw:

258  string s2 = Convert.ToString(objB, CultureInfo.InvariantCulture);
259

261  case JTokenType.Boolean:
262  bool b1 = Convert.ToBoolean(objA, CultureInfo.InvariantCulture);
263  bool b2 = Convert.ToBoolean(objB, CultureInfo.InvariantCulture);
264 
265  return b1.CompareTo(b2);
266  case JTokenType.Date:
267 #if !NET20

269  {
