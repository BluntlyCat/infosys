504  public struct LockHintAppender
505  {
506  private const string UnescapedNameRegex = @"\w+";
507  private const string EscapedNameRegex = @"\[([^\]]|\]\])+\]";
508  private const string NameRegex = "(" + UnescapedNameRegex + "|" + EscapedNameRegex + ")";
509  private const string NameSeparatorRegex = @"\s*\.\s*";
510  private const string FromTableNameRegex = @"from\s+(" + NameRegex + "?" + NameSeparatorRegex + "){0,2}" + NameRegex;
511 
512  private static readonly Regex FromClauseTableNameRegex = new Regex(FromTableNameRegex, RegexOptions.IgnoreCase | RegexOptions.Multiline);
513 
514  private readonly MsSql2000Dialect _dialect;
515  private readonly IDictionary<string, LockMode> _aliasedLockModes;
516 
517  private readonly Regex _matchRegex;
518  private readonly Regex _unionSubclassRegex;
519 
520  public LockHintAppender(MsSql2000Dialect dialect, IDictionary<string, LockMode> aliasedLockModes)
521  {
522  _dialect = dialect;
523  _aliasedLockModes = aliasedLockModes;
524 
525  // Regex matching any alias out of those given. Aliases should contain
526  // no dangerous characters (they are identifiers) so they are not escaped.
527  var aliasesPattern = StringHelper.Join("|", aliasedLockModes.Keys);
528 
529  // Match < alias >, < alias,>, or < alias$>, the intent is to capture alias names
530  // in various kinds of "FROM table1 alias1, table2 alias2".
531  _matchRegex = new Regex(" (" + aliasesPattern + ")([, ]|$)");
532  _unionSubclassRegex = new Regex(@"from\s+\(((?:.|\r|\n)*)\)(?:\s+as)?\s+(?<alias>" + aliasesPattern + ")", RegexOptions.IgnoreCase | RegexOptions.Multiline);
533  }
534 
535  public SqlString AppendLockHint(SqlString sql)
536  {
537  var result = new SqlStringBuilder();
538 
539  foreach (object part in sql)
540  {
541  if (part == Parameter.Placeholder)
542  {
543  result.Add((Parameter)part);
544  continue;
545 }
546 
547  result.Add(ProcessUnionSubclassCase((string)part) ?? _matchRegex.Replace((string)part, ReplaceMatch));
548  }
549 
550  return result.ToSqlString();
551  }
552 
553  private string ProcessUnionSubclassCase(string part)
554  {
555  var unionMatch = _unionSubclassRegex.Match(part);
556  if (!unionMatch.Success)
557  {
558  return null;
559  }
560 
561  var alias = unionMatch.Groups["alias"].Value;
562  var lockMode = _aliasedLockModes[alias];
563  var @this = this;
564  var replacement = FromClauseTableNameRegex.Replace(unionMatch.Value, m => @this._dialect.AppendLockHint(lockMode, m.Value));
565 
566  return _unionSubclassRegex.Replace(part, replacement);
567  }
568 
569  private string ReplaceMatch(Match match)
570  {
571  string alias = match.Groups[1].Value;
572  string lockHint = _dialect.AppendLockHint(_aliasedLockModes[alias], alias);
573  return string.Concat(" ", lockHint, match.Groups[2].Value); // TODO: seems like this line is redundant
574  }
575  }
576  }
