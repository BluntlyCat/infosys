102  private static FieldInfo GetField(System.Type type, string fieldName, System.Type originalType)
103  {
104  if (type == null || type == typeof(object))
105  {
106  // the full inheritance chain has been walked and we could
107  // not find the Field
108  throw new PropertyNotFoundException(originalType, fieldName);
109  }
110 
111  FieldInfo field =
112  type.GetField(fieldName,
113  BindingFlags.Instance | BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.DeclaredOnly);
114  if (field == null)
115  {
116  // recursively call this method for the base Type
117  field = GetField(type.BaseType, fieldName, originalType);
