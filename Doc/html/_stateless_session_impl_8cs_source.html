142  public override void List(CriteriaImpl criteria, IList results)
143  {
144  using (new SessionIdLoggingContext(SessionId))
145  {
146  CheckAndUpdateSessionStatus();
147  string[] implementors = Factory.GetImplementors(criteria.EntityOrClassName);
148  int size = implementors.Length;
149 
150  CriteriaLoader[] loaders = new CriteriaLoader[size];
151  for (int i = 0; i < size; i++)
152  {
153  loaders[i] = new CriteriaLoader(GetOuterJoinLoadable(implementors[i]), Factory,
154  criteria, implementors[i], EnabledFilters);
155  }
156 
157  bool success = false;
158  try
159  {
160  for (int i = size - 1; i >= 0; i--)
161  {
162  ArrayHelper.AddAll(results, loaders[i].List(this));
163  }
164  success = true;
165  }
166  catch (HibernateException)
167  {
168  // Do not call Convert on HibernateExceptions
169  throw;
170  }
171  catch (Exception sqle)
172  {
173  throw Convert(sqle, "Unable to perform find");
174  }
175  finally
176  {
177  AfterOperation(success);
178  }
179  temporaryPersistenceContext.Clear();
180  }

473  public object Insert(string entityName, object entity)
474  {
475  using (new SessionIdLoggingContext(SessionId))
476  {
477  CheckAndUpdateSessionStatus();
478  IEntityPersister persister = GetEntityPersister(entityName, entity);
479  object id = persister.IdentifierGenerator.Generate(this, entity);
480  object[] state = persister.GetPropertyValues(entity, EntityMode.Poco);
481  if (persister.IsVersioned)
482  {
483  object versionValue = state[persister.VersionProperty];
484  bool substitute = Versioning.SeedVersion(state, persister.VersionProperty, persister.VersionType,
485  persister.IsUnsavedVersion(versionValue), this);
486  if (substitute)
487  {
488  persister.SetPropertyValues(entity, state, EntityMode.Poco);
489  }
490  }
491  if (id == IdentifierGeneratorFactory.PostInsertIndicator)
492  {
493  id = persister.Insert(state, entity, this);
494  }
495  else
496  {
497  persister.Insert(id, state, entity, this);
498  }
499  persister.SetIdentifier(entity, id, EntityMode.Poco);
500  return id;
501  }
