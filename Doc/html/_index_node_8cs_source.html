45  FromReferenceNode collectionNode = ( FromReferenceNode ) GetChild(0);
46  SessionFactoryHelperExtensions sessionFactoryHelper = SessionFactoryHelper;
47  collectionNode.ResolveIndex( this ); // Fully resolve the map reference, create implicit joins.
48 
49  IType type = collectionNode.DataType;
50  if ( !type.IsCollectionType )
51  {
52  throw new SemanticException( "The [] operator cannot be applied to type " + type);
53  }
54 
55  string collectionRole = ( ( CollectionType ) type ).Role;
56  IQueryableCollection queryableCollection = sessionFactoryHelper.RequireQueryableCollection( collectionRole );
57  if ( !queryableCollection.HasIndex )
58  {
59  throw new QueryException( "unindexed fromElement before []: " + collectionNode.Path );
60  }

62  // Generate the inner join -- The elements need to be joined to the collection they are in.
63  FromElement fromElement = collectionNode.FromElement;
64  String elementTable = fromElement.TableAlias;
65  FromClause fromClause = fromElement.FromClause;
66  String path = collectionNode.Path;
67 
68  FromElement elem = fromClause.FindCollectionJoin( path );
69  if ( elem == null )
70  {
71  FromElementFactory factory = new FromElementFactory( fromClause, fromElement, path );
72  elem = factory.CreateCollectionElementsJoin( queryableCollection, elementTable );
73  if ( Log.IsDebugEnabled )
74  {
75  Log.Debug( "No FROM element found for the elements of collection join path " + path
76  + ", created " + elem );
77  }
78  }
79  else
80  {
81  if ( Log.IsDebugEnabled )
82  {
83  Log.Debug( "FROM element found for collection join path " + path );
84  }
85  }
86 
87  // The 'from element' that represents the elements of the collection.
88  FromElement = fromElement;
89 
90  // Add the condition to the join sequence that qualifies the indexed element.
91  IASTNode selector = GetChild(1);
92  if ( selector == null )
93  {
94  throw new QueryException( "No index value!" );
95  }
96 
97  // Sometimes use the element table alias, sometimes use the... umm... collection table alias (many to many)
98  String collectionTableAlias = elementTable;
99  if ( elem.CollectionTableAlias != null )
100  {
101  collectionTableAlias = elem.CollectionTableAlias;
102  }
103 
104  // TODO: get SQL rendering out of here, create an AST for the join expressions.
105  // Use the SQL generator grammar to generate the SQL text for the index expression.
106  JoinSequence joinSequence = fromElement.JoinSequence;
107  string[] indexCols = queryableCollection.IndexColumnNames;
108  if ( indexCols.Length != 1 )
109  {
110  throw new QueryException( "composite-index appears in []: " + collectionNode.Path );
111  }
112 
113  SqlGenerator gen = new SqlGenerator(SessionFactoryHelper.Factory, new CommonTreeNodeStream(selector));
114 
115  try
116  {
117  gen.simpleExpr(); //TODO: used to be exprNoParens! was this needed?
118  }
119  catch ( RecognitionException e )
120  {
121  throw new QueryException( e.Message, e );
122  }
123 
124  string selectorExpression = gen.GetSQL().ToString();
125 
126  joinSequence.AddCondition(new SqlString(collectionTableAlias + '.' + indexCols[0] + " = " + selectorExpression ));
127  //joinSequence.AddCondition(collectionTableAlias, new string[] { indexCols[0] }, selectorExpression, false);
128 
129  IList<IParameterSpecification> paramSpecs = gen.GetCollectedParameters();
130  if ( paramSpecs != null )
131  {
132  switch ( paramSpecs.Count )
133  {
134  case 0 :
135  // nothing to do
136  break;
137  case 1 :
138  IParameterSpecification paramSpec = paramSpecs[0];
139  paramSpec.ExpectedType = queryableCollection.IndexType;
140  fromElement.SetIndexCollectionSelectorParamSpec( paramSpec );
141  break;
142  default:
143  fromElement.SetIndexCollectionSelectorParamSpec(
144  new AggregatedIndexCollectionSelectorParameterSpecifications( paramSpecs )
145  );
146  break;
147  }
148  }
149 
150  // Now, set the text for this node. It should be the element columns.
151  String[] elementColumns = queryableCollection.GetElementColumnNames( elementTable );
152  Text = elementColumns[0];
153

155  }
