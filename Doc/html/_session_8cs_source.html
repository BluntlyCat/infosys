420  public void Connect()
421  {
422  // TODO: Add exception documentation for Proxy.
423  if (this.ConnectionInfo == null)
424  {
425  throw new ArgumentNullException("connectionInfo");
426  }
427 
428  if (this.IsConnected)
429  return;
430 
431  try
432  {
433  _authenticationConnection.Wait();
434 
435  if (this.IsConnected)
436  return;
437 
438  lock (this)
439  {
440  // If connected don't connect again
441  if (this.IsConnected)
442  return;
443 
444  // Build list of available messages while connecting
445  this._messagesMetadata = GetMessagesMetadata();
446 
447  switch (this.ConnectionInfo.ProxyType)
448  {
449  case ProxyTypes.None:
450  this.SocketConnect(this.ConnectionInfo.Host, this.ConnectionInfo.Port);
451  break;
452  case ProxyTypes.Socks4:
453  this.SocketConnect(this.ConnectionInfo.ProxyHost, this.ConnectionInfo.ProxyPort);
454  this.ConnectSocks4();
455  break;
456  case ProxyTypes.Socks5:
457  this.SocketConnect(this.ConnectionInfo.ProxyHost, this.ConnectionInfo.ProxyPort);
458  this.ConnectSocks5();
459  break;
460  case ProxyTypes.Http:
461  this.SocketConnect(this.ConnectionInfo.ProxyHost, this.ConnectionInfo.ProxyPort);
462  this.ConnectHttp();
463  break;
464  default:
465  break;
466  }
467 
468 
469  Match versionMatch = null;
470 
471  // Get server version from the server,
472  // ignore text lines which are sent before if any
473  while (true)
474  {
475  string serverVersion = string.Empty;
476  this.SocketReadLine(ref serverVersion);
477 
478  this.ServerVersion = serverVersion;
479  if (string.IsNullOrEmpty(this.ServerVersion))
480  {
481  throw new InvalidOperationException("Server string is null or empty.");
482  }
483 
484  versionMatch = _serverVersionRe.Match(this.ServerVersion);
485 
486  if (versionMatch.Success)
487  {
488  break;
489  }
490  }
491 
492  // Set connection versions
493  this.ConnectionInfo.ServerVersion = this.ServerVersion;
494  this.ConnectionInfo.ClientVersion = this.ClientVersion;
495 
496  // Get server SSH version
497  var version = versionMatch.Result("${protoversion}");
498 
499  var softwareName = versionMatch.Result("${softwareversion}");
500 
501  this.Log(string.Format("Server version '{0}' on '{1}'.", version, softwareName));
502 
503  if (!(version.Equals("2.0") || version.Equals("1.99")))
504  {
505  throw new SshConnectionException(string.Format(CultureInfo.CurrentCulture, "Server version '{0}' is not supported.", version), DisconnectReason.ProtocolVersionNotSupported);
506  }
507 
508  this.SocketWrite(Encoding.UTF8.GetBytes(string.Format(CultureInfo.InvariantCulture, "{0}\x0D\x0A", this.ClientVersion)));
509 
510  // Register Transport response messages
511  this.RegisterMessage("SSH_MSG_DISCONNECT");
512  this.RegisterMessage("SSH_MSG_IGNORE");
513  this.RegisterMessage("SSH_MSG_UNIMPLEMENTED");
514  this.RegisterMessage("SSH_MSG_DEBUG");
515  this.RegisterMessage("SSH_MSG_SERVICE_ACCEPT");
516  this.RegisterMessage("SSH_MSG_KEXINIT");
517  this.RegisterMessage("SSH_MSG_NEWKEYS");
518 
519  // Some server implementations might sent this message first, prior establishing encryption algorithm
520  this.RegisterMessage("SSH_MSG_USERAUTH_BANNER");
521 
522  // Start incoming request listener
523  this._messageListenerCompleted = new ManualResetEvent(false);
524 
525  this.ExecuteThread(() =>
526  {
527  try
528  {
529  this.MessageListener();
530  }
531  finally
532  {
533  this._messageListenerCompleted.Set();
534  }
535  });
536 
537  // Wait for key exchange to be completed
538  this.WaitHandle(this._keyExchangeCompletedWaitHandle);
539 
540  // If sessionId is not set then its not connected
541  if (this.SessionId == null)
542  {
543  this.Disconnect();
544  return;
545  }
546 
547  // Request user authorization service
548  this.SendMessage(new ServiceRequestMessage(ServiceName.UserAuthentication));
549 
550  // Wait for service to be accepted
551  this.WaitHandle(this._serviceAccepted);
552 
553  if (string.IsNullOrEmpty(this.ConnectionInfo.Username))
554  {
555  throw new SshException("Username is not specified.");
556  }
557 
558  this._isAuthenticated = this.ConnectionInfo.Authenticate(this);
559 
560  if (!this._isAuthenticated)
561  {
562  throw new SshAuthenticationException("User cannot be authenticated.");
563  }
564 
565  // Register Connection messages
566  this.RegisterMessage("SSH_MSG_GLOBAL_REQUEST");
567  this.RegisterMessage("SSH_MSG_REQUEST_SUCCESS");
568  this.RegisterMessage("SSH_MSG_REQUEST_FAILURE");

653  internal void SendMessage(Message message)
654  {
655  if (this._socket == null || !this._socket.CanWrite())
656  throw new SshConnectionException("Client not connected.");
657 
658  if (this._keyExchangeInProgress && !(message is IKeyExchangedAllowed))
659  {
660  // Wait for key exchange to be completed
661  this.WaitHandle(this._keyExchangeCompletedWaitHandle);
662  }
663 
664  this.Log(string.Format("SendMessage to server '{0}': '{1}'.", message.GetType().Name, message.ToString()));
665 
666  // Messages can be sent by different thread so we need to synchronize it
667  var paddingMultiplier = this._clientCipher == null ? (byte)8 : Math.Max((byte)8, this._serverCipher.MinimumSize); // Should be recalculate base on cipher min length if cipher specified
668 
669  var messageData = message.GetBytes();
670 
671  if (messageData.Length > Session.MAXIMUM_PAYLOAD_SIZE)
672  {
673  throw new InvalidOperationException(string.Format(CultureInfo.CurrentCulture, "Payload cannot be more then {0} bytes.", Session.MAXIMUM_PAYLOAD_SIZE));
674  }
675 
676  if (this._clientCompression != null)
677  {
678  messageData = this._clientCompression.Compress(messageData);
679  }
680 
681  var packetLength = messageData.Length + 4 + 1; // add length bytes and padding byte
682  byte paddingLength = (byte)((-packetLength) & (paddingMultiplier - 1));
683  if (paddingLength < paddingMultiplier)
684  {
685  paddingLength += paddingMultiplier;
686  }
687 
688  // Build Packet data
689  var packetData = new byte[4 + 1 + messageData.Length + paddingLength];
690 
691  // Add packet length
692  ((uint)packetData.Length - 4).GetBytes().CopyTo(packetData, 0);
693 
694  // Add packet padding length
695  packetData[4] = paddingLength;
696 
697  // Add packet payload
698  messageData.CopyTo(packetData, 4 + 1);
699 
700  // Add random padding
701  var paddingBytes = new byte[paddingLength];
702  _randomizer.GetBytes(paddingBytes);
703  paddingBytes.CopyTo(packetData, 4 + 1 + messageData.Length);
704 
705  // Lock handling of _outboundPacketSequence since it must be sent sequently to server
706  lock (this._socketLock)
707  {
708  if (this._socket == null || !this._socket.Connected)
709  throw new SshConnectionException("Client not connected.");
710 
711  // Calculate packet hash
712  var hashData = new byte[4 + packetData.Length];
713  this._outboundPacketSequence.GetBytes().CopyTo(hashData, 0);
714  packetData.CopyTo(hashData, 4);
715 
716  // Encrypt packet data
717  if (this._clientCipher != null)
718  {
719  packetData = this._clientCipher.Encrypt(packetData);
720  }
721 
722  if (packetData.Length > Session.MAXIMUM_PACKET_SIZE)
723  {
724  throw new InvalidOperationException(string.Format(CultureInfo.CurrentCulture, "Packet is too big. Maximum packet size is {0} bytes.", Session.MAXIMUM_PACKET_SIZE));
725  }
726 
727  if (this._clientMac == null)
728  {
729  this.SocketWrite(packetData);
730  }
731  else
732  {
733  var hash = this._clientMac.ComputeHash(hashData.ToArray());
734 
735  var data = new byte[packetData.Length + this._clientMac.HashSize / 8];
736  packetData.CopyTo(data, 0);
737  hash.CopyTo(data, packetData.Length);
738 
739  this.SocketWrite(data);
740  }
741 
742  this._outboundPacketSequence++;
743 
744  Monitor.Pulse(this._socketLock);
745  }
746  }

796  private Message ReceiveMessage()
797  {
798  if (!this._socket.Connected)
799  return null;
800 
801  // No lock needed since all messages read by only one thread
802  var blockSize = this._serverCipher == null ? (byte)8 : Math.Max((byte)8, this._serverCipher.MinimumSize);
803 
804  // Read packet length first
805  var firstBlock = this.Read(blockSize);
806 
807  if (this._serverCipher != null)
808  {
809  firstBlock = this._serverCipher.Decrypt(firstBlock);
810  }
811 
812  var packetLength = (uint)(firstBlock[0] << 24 | firstBlock[1] << 16 | firstBlock[2] << 8 | firstBlock[3]);
813 
814  // Test packet minimum and maximum boundaries
815  if (packetLength < Math.Max((byte)16, blockSize) - 4 || packetLength > Session.MAXIMUM_PACKET_SIZE - 4)
816  throw new SshConnectionException(string.Format(CultureInfo.CurrentCulture, "Bad packet length {0}", packetLength), DisconnectReason.ProtocolError);
817 
818  // Read rest of the packet data
819  int bytesToRead = (int)(packetLength - (blockSize - 4));
820 
821  var data = new byte[bytesToRead + blockSize];
822 
823  firstBlock.CopyTo(data, 0);
824 
825  byte[] serverHash = null;
826  if (this._serverMac != null)
827  {
828  serverHash = new byte[this._serverMac.HashSize / 8];
829  bytesToRead += serverHash.Length;
830  }
831 
832  if (bytesToRead > 0)
833  {
834  var nextBlocks = this.Read(bytesToRead);
835 
836  if (serverHash != null)
837  {
838  Buffer.BlockCopy(nextBlocks, nextBlocks.Length - serverHash.Length, serverHash, 0, serverHash.Length);
839  nextBlocks = nextBlocks.Take(nextBlocks.Length - serverHash.Length).ToArray();
840  }
841 
842  if (nextBlocks.Length > 0)
843  {
844  if (this._serverCipher != null)
845  {
846  nextBlocks = this._serverCipher.Decrypt(nextBlocks);
847  }
848  nextBlocks.CopyTo(data, blockSize);
849  }
850  }
851 
852  var paddingLength = data[4];
853 
854  var messagePayload = new byte[packetLength - paddingLength - 1];
855  Buffer.BlockCopy(data, 5, messagePayload, 0, messagePayload.Length);
856 
857  if (this._serverDecompression != null)
858  {
859  messagePayload = this._serverDecompression.Decompress(messagePayload);
860  }
861 
862  // Validate message against MAC
863  if (this._serverMac != null)
864  {
865  var clientHashData = new byte[4 + data.Length];
866  var lengthBytes = this._inboundPacketSequence.GetBytes();
867 
868  lengthBytes.CopyTo(clientHashData, 0);
869  data.CopyTo(clientHashData, 4);
870 
871  // Calculate packet hash
872  var clientHash = this._serverMac.ComputeHash(clientHashData);
873 
874  if (!serverHash.SequenceEqual(clientHash))
875  {
876  throw new SshConnectionException("MAC error", DisconnectReason.MacError);
877  }
878  }
879 
880  this._inboundPacketSequence++;
881 
882  return this.LoadMessage(messagePayload);
883  }

1683  {
1684  case 0x00:
1685  break;
1686  case 0x02:

1700

1702  this.SocketWrite(username);

1705

1707  throw new ProxyException("Proxy password is too long.");
1708

1710  this.SocketWriteByte((byte)password.Length);
1711 
1712  // Send username
1713  this.SocketWrite(password);
1714

1716 
1717  if (serverVersion != 1)
1718  throw new ProxyException("SOCKS5: Server authentication version is not valid.");
1719

1721  if (statusCode != 0)
1722  throw new ProxyException("SOCKS5: Username/Password authentication failed.");
1723 
1724  break;
1725  case 0xFF:
1726  throw new ProxyException("SOCKS5: No acceptable authentication methods were offered.");
1727  default:
1728  break;
1729  }
1730 
1731  // Send socks version number
1732  this.SocketWriteByte(0x05);
1733 
1734  // Send command code
1735  this.SocketWriteByte(0x01); // establish a TCP/IP stream connection
1736 
1737  // Send reserved, must be 0x00
1738  this.SocketWriteByte(0x00);
1739

1741

1743  if (ip.AddressFamily == AddressFamily.InterNetwork)
1744  {
1745  this.SocketWriteByte(0x01);
1746  var address = ip.GetAddressBytes();
1747  this.SocketWrite(address);
1748  }
1749  else if (ip.AddressFamily == AddressFamily.InterNetworkV6)

1753  this.SocketWrite(address);
1754  }
1755  else
1756  {
1757  throw new ProxyException(string.Format("SOCKS5: IP address '{0}' is not supported.", ip));
1758  }
1759 
1760  // Send port
1761  this.SocketWriteByte((byte)(this.ConnectionInfo.Port / 0xFF));
1762  this.SocketWriteByte((byte)(this.ConnectionInfo.Port % 0xFF));
1763 
1764  // Read Server SOCKS5 version
1765  if (this.SocketReadByte() != 5)

1825  private void ConnectHttp()
1826  {
1827  var httpResponseRe = new Regex(@"HTTP/(?<version>\d[.]\d) (?<statusCode>\d{3}) (?<reasonPhrase>.+)$");
1828  var httpHeaderRe = new Regex(@"(?<fieldName>[^\[\]()<>@,;:\""/?={} \t]+):(?<fieldValue>.+)?");
1829 
1830  var encoding = new Renci.SshNet.Common.ASCIIEncoding();
1831 
1832  this.SocketWrite(encoding.GetBytes(string.Format("CONNECT {0}:{1} HTTP/1.0\r\n", this.ConnectionInfo.Host, this.ConnectionInfo.Port)));
1833 
1834  // Sent proxy authorization is specified
1835  if (!string.IsNullOrEmpty(this.ConnectionInfo.ProxyUsername))
1836  {
1837  var authorization = string.Format("Proxy-Authorization: Basic {0}\r\n",
1838  Convert.ToBase64String(encoding.GetBytes(string.Format("{0}:{1}", this.ConnectionInfo.ProxyUsername, this.ConnectionInfo.ProxyPassword)))
1839  );
1840  this.SocketWrite(encoding.GetBytes(authorization));
1841  }
1842 
1843  this.SocketWrite(encoding.GetBytes("\r\n"));
1844 
1845  HttpStatusCode statusCode = HttpStatusCode.OK;
1846  var response = string.Empty;
1847  var contentLength = 0;
1848 
1849  while (true)
1850  {
1851  this.SocketReadLine(ref response);
1852 
1853  var match = httpResponseRe.Match(response);
1854 
1855  if (match.Success)
1856  {
1857  statusCode = (HttpStatusCode)int.Parse(match.Result("${statusCode}"));
1858  continue;
1859  }
1860 
1861  // continue on parsing message headers coming from the server
1862  match = httpHeaderRe.Match(response);
1863  if (match.Success)
1864  {
1865  var fieldName = match.Result("${fieldName}");
1866  if (fieldName.Equals("Content-Length", StringComparison.InvariantCultureIgnoreCase))
1867  {
1868  contentLength = int.Parse(match.Result("${fieldValue}"));
1869  }
1870  continue;
1871  }
1872 
1873  // Read response body if specified
1874  if (string.IsNullOrEmpty(response) && contentLength > 0)
1875  {
1876  var contentBody = new byte[contentLength];
1877  this.SocketRead(contentLength, ref contentBody);
1878  }
1879 
1880  switch (statusCode)
1881  {
1882  case HttpStatusCode.OK:
1883  break;
1884  default:
1885  throw new ProxyException(string.Format("HTTP: Status code {0}, \"{1}\"", statusCode, statusCode));
1886  }
1887  }
