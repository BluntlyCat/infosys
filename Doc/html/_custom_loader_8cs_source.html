98  resultColumnProcessors.Add(new NonScalarResultColumnProcessor(returnableCounter++));
99  nonScalarReturnList.Add(rtn);
100  collectionowners.Add(-1);

102  specifiedAliases.Add(collRtn.Alias);
103  collectionaliases.Add(collRtn.CollectionAliases);
104  // determine if the collection elements are entities...
105  IType elementType = persister.ElementType;
106  if (elementType.IsEntityType)
107  {
108  IQueryable elementPersister = (IQueryable) ((EntityType) elementType).GetAssociatedJoinable(factory);
109  entitypersisters.Add(elementPersister);
110  entityowners.Add(-1);
111  entityaliases.Add(collRtn.ElementEntityAliases);

124  string entityName = fetchedType.GetAssociatedEntityName(Factory);
125  IQueryable persister = (IQueryable) factory.GetEntityPersister(entityName);
126  entitypersisters.Add(persister);
127  nonScalarReturnList.Add(rtn);
128  specifiedAliases.Add(fetchRtn.Alias);
129  entityaliases.Add(fetchRtn.EntityAliases);
130  querySpaces.UnionWith(persister.QuerySpaces);
131  }
132  else if (rtn is CollectionFetchReturn)
133  {
134  CollectionFetchReturn fetchRtn = (CollectionFetchReturn) rtn;
135  NonScalarReturn ownerDescriptor = fetchRtn.Owner;
136  int ownerIndex = nonScalarReturnList.IndexOf(ownerDescriptor);
137  collectionowners.Add(ownerIndex);
138  lockmodes.Add(fetchRtn.LockMode);
139  IQueryable ownerPersister = DetermineAppropriateOwnerPersister(ownerDescriptor);
140  string role = ownerPersister.EntityName + '.' + fetchRtn.OwnerProperty;

143  nonScalarReturnList.Add(rtn);
144  specifiedAliases.Add(fetchRtn.Alias);

146  // determine if the collection elements are entities...
147  IType elementType = persister.ElementType;

150  IQueryable elementPersister = (IQueryable) ((EntityType) elementType).GetAssociatedJoinable(factory);
151  entitypersisters.Add(elementPersister);
152  entityowners.Add(ownerIndex);
153  entityaliases.Add(fetchRtn.ElementEntityAliases);
154  querySpaces.UnionWith(elementPersister.QuerySpaces);
155  }
156  }
157  else

159  throw new HibernateException("unexpected custom query return type : " + rtn.GetType().FullName);
160  }
161  }

205  private IQueryable DetermineAppropriateOwnerPersister(NonScalarReturn ownerDescriptor)
206  {
207  string entityName = null;
208  RootReturn odrr = ownerDescriptor as RootReturn;
209  if (odrr != null)
210  {
211  entityName = odrr.EntityName;
212  }
213  else if (ownerDescriptor is CollectionReturn)
214  {
215  CollectionReturn collRtn = (CollectionReturn) ownerDescriptor;
216  string role = collRtn.OwnerEntityName + "." + collRtn.OwnerProperty;
217  ICollectionPersister persister = Factory.GetCollectionPersister(role);
218  EntityType ownerType = (EntityType) persister.ElementType;
219  entityName = ownerType.GetAssociatedEntityName(Factory);
220  }
221  else if (ownerDescriptor is FetchReturn)
222  {
223  FetchReturn fetchRtn = (FetchReturn) ownerDescriptor;
224  IQueryable persister = DetermineAppropriateOwnerPersister(fetchRtn.Owner);
225  IType ownerType = persister.GetPropertyType(fetchRtn.OwnerProperty);
226  if (ownerType.IsEntityType)
227  {
228  entityName = ((EntityType) ownerType).GetAssociatedEntityName(Factory);
229  }
230  else if (ownerType.IsCollectionType)
231  {
232  IType ownerCollectionElementType = ((CollectionType) ownerType).GetElementType(Factory);
233  if (ownerCollectionElementType.IsEntityType)
234  {
235  entityName = ((EntityType) ownerCollectionElementType).GetAssociatedEntityName(Factory);
236  }
237  }
238  }
239 
240  if (entityName == null)
241  {
242  throw new HibernateException("Could not determine fetch owner : " + ownerDescriptor);
243  }
244 
245  return (IQueryable) Factory.GetEntityPersister(entityName);
246  }

378  public object BuildResultRow(object[] data, IDataReader resultSet, bool hasTransformer, ISessionImplementor session)
379  {
380  object[] resultRow;
381  // NH Different behavior (patched in NH-1612 to solve Hibernate issue HHH-2831).
382  if (!hasScalars && (hasTransformer || data.Length == 0))
383  {
384  resultRow = data;
385  }
386  else
387  {
388  // build an array with indices equal to the total number
389  // of actual returns in the result Hibernate will return
390  // for this query (scalars + non-scalars)
391  resultRow = new object[columnProcessors.Length];
392  for (int i = 0; i < columnProcessors.Length; i++)
393  {
394  resultRow[i] = columnProcessors[i].Extract(data, resultSet, session);
395  }
396  }
397 
398  return (hasTransformer) ? resultRow : (resultRow.Length == 1) ? resultRow[0] : resultRow;
399  }
