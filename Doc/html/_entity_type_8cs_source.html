157  protected internal object GetIdentifier(object value, ISessionImplementor session)
158  {
159  if (IsNotEmbedded(session))
160  {
161  return value;
162  }
163 
164  if (IsReferenceToPrimaryKey)
165  {
166  return ForeignKeys.GetEntityIdentifierIfNotUnsaved(GetAssociatedEntityName(), value, session); //tolerates nulls
167  }
168  else if (value == null)
169  {
170  return null;
171  }
172  else
173  {
174  IEntityPersister entityPersister = session.Factory.GetEntityPersister(GetAssociatedEntityName());
175  object propertyValue = entityPersister.GetPropertyValue(value, uniqueKeyPropertyName, session.EntityMode);
176 
177  // We now have the value of the property-ref we reference. However,
178  // we need to dig a little deeper, as that property might also be
179  // an entity type, in which case we need to resolve its identitifier
180  IType type = entityPersister.GetPropertyType(uniqueKeyPropertyName);
181  if (type.IsEntityType)
182  {
183  propertyValue = ((EntityType)type).GetIdentifier(propertyValue, session);
184  }
185 
186  return propertyValue;
187  }

256  public override object Replace(object original, object target, ISessionImplementor session, object owner, IDictionary copyCache)
257  {
258  if (original == null)
259  {
260  return null;
261  }
262  object cached = copyCache[original];
263  if (cached != null)
264  {
265  return cached;
266  }
267  else
268  {
269  if (original == target)
270  {
271  return target;
272  }
273  if (session.GetContextEntityIdentifier(original) == null && ForeignKeys.IsTransient(associatedEntityName, original, false, session))
274  {
275  object copy = session.Factory.GetEntityPersister(associatedEntityName).Instantiate(null, session.EntityMode);
276  //TODO: should this be Session.instantiate(Persister, ...)?
277  copyCache.Add(original, copy);
278  return copy;
279  }
280  else
281  {
282  object id = GetIdentifier(original, session);
283  if (id == null)
284  {
285  throw new AssertionFailure("non-transient entity has a null id");
286  }
287  id = GetIdentifierOrUniqueKeyType(session.Factory).Replace(id, null, session, owner, copyCache);
288  return ResolveIdentifier(id, session, owner);
289  }
