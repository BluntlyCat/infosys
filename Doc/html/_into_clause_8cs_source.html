152  private void VisitPropertySpecNodes(IASTNode propertyNode, ICollection<IType> types)
153  {
154  if (propertyNode == null)
155  {
156  return;
157  }
158 
159  // TODO : we really need to be able to deal with component paths here also;
160  // this is difficult because the hql-sql grammar expects all those node types
161  // to be FromReferenceNodes. One potential fix here would be to convert the
162  // IntoClause to just use a FromClause/FromElement combo (as a child of the
163  // InsertStatement) and move all this logic into the InsertStatement. That's
164  // probably the easiest approach (read: least amount of changes to the grammar
165  // and code), but just doesn't feel right as then an insert would contain
166  // 2 from-clauses
167  string name = propertyNode.Text;
168 
169  if (IsSuperclassProperty(name))
170  {
171  throw new QueryException("INSERT statements cannot refer to superclass/joined properties [" + name + "]");
172  }
173 
174  if (name == _persister.IdentifierPropertyName)
175  {
176  _explicitIdInsertion = true;
177  }
178 
179  if (_persister.IsVersioned)
180  {
181  if (name == _persister.PropertyNames[_persister.VersionProperty])
182  {
183  _explicitVersionInsertion = true;
184  }
185  }
186 
187  string[] columnNames = _persister.ToColumns(name);
188  RenderColumns(columnNames);
189  types.Add(_persister.ToType(name));
190 
191  // visit width-first, then depth
192  VisitPropertySpecNodes(propertyNode.NextSibling, types);
193  VisitPropertySpecNodes(propertyNode.GetFirstChild(), types);
194  }

204  private bool IsSuperclassProperty(string propertyName)
205  {
206  // really there are two situations where it should be ok to allow the insertion
207  // into properties defined on a superclass:
208  // 1) union-subclass with an abstract root entity
209  // 2) discrim-subclass

211  // #1 is handled already because of the fact that
212  // UnionSubclassPersister alreay always returns 0
213  // for this call...

226  private bool AreCompatible(IType target, IType source)
227  {
228  if (target.Equals(source))
229  {
230  // if the types report logical equivalence, return true...
231  return true;
232  }
233 
234  // otherwise, perform a "deep equivalence" check...
235 
236  if (!target.ReturnedClass.IsAssignableFrom(source.ReturnedClass))
237  {
238  return false;
239  }
240 
241  SqlType[] targetDatatypes = target.SqlTypes(SessionFactoryHelper.Factory);
242  SqlType[] sourceDatatypes = source.SqlTypes(SessionFactoryHelper.Factory);
243 
244  if (targetDatatypes.Length != sourceDatatypes.Length)
245  {
246  return false;
247  }
248 
249  for (int i = 0; i < targetDatatypes.Length; i++)
250  {
251  if (!AreSqlTypesCompatible(targetDatatypes[i], sourceDatatypes[i]))
252  {
253  return false;
254  }
255  }
256 
257  return true;
258  }
