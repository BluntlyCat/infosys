145  public void Upload(Stream source, string path)
146  {
147  using (var input = new PipeStream())
148  using (var channel = this.Session.CreateChannel<ChannelSession>())
149  {
150  channel.DataReceived += delegate(object sender, Common.ChannelDataEventArgs e)
151  {
152  input.Write(e.Data, 0, e.Data.Length);
153  input.Flush();
154  };
155 
156  channel.Open();
157 
158  var pathParts = path.Split('\\', '/');
159 
160  // Send channel command request
161  channel.SendExecRequest(string.Format("scp -rt \"{0}\"", pathParts[0]));
162  this.CheckReturnCode(input);
163 
164  // Prepare directory structure
165  for (int i = 0; i < pathParts.Length - 1; i++)
166  {
167  this.InternalSetTimestamp(channel, input, DateTime.UtcNow, DateTime.UtcNow);
168  this.SendData(channel, string.Format("D0755 0 {0}\n", pathParts[i]));
169  this.CheckReturnCode(input);
170  }
171 
172  this.InternalUpload(channel, input, source, pathParts.Last());
173 
174  // Finish directory structure
175  for (int i = 0; i < pathParts.Length - 1; i++)
176  {
177  this.SendData(channel, "E\n");
178  this.CheckReturnCode(input);
179  }
180 
181  channel.Close();
182  }

193  public void Download(string filename, Stream destination)
194  {
195  if (filename.IsNullOrWhiteSpace())
196  throw new ArgumentException("filename");
197 
198  if (destination == null)
199  throw new ArgumentNullException("destination");
200 
201  using (var input = new PipeStream())
202  using (var channel = this.Session.CreateChannel<ChannelSession>())
203  {
204  channel.DataReceived += delegate(object sender, Common.ChannelDataEventArgs e)
205  {
206  input.Write(e.Data, 0, e.Data.Length);
207  input.Flush();
208  };
209 
210  channel.Open();
211 
212  // Send channel command request
213  channel.SendExecRequest(string.Format("scp -f \"{0}\"", filename));
214  this.SendConfirmation(channel); // Send reply
215 
216  var message = ReadString(input);
217  var match = _fileInfoRe.Match(message);
218 
219  if (match.Success)
220  {
221  // Read file
222  this.SendConfirmation(channel); // Send reply
223 
224  var mode = match.Result("${mode}");
225  var length = long.Parse(match.Result("${length}"));
226  var fileName = match.Result("${filename}");
227 
228  this.InternalDownload(channel, input, destination, fileName, length);
229  }
230  else
231  {
232  this.SendConfirmation(channel, 1, string.Format("\"{0}\" is not valid protocol message.", message));
233  }
234 
235  channel.Close();
236  }

275  private void InternalDownload(ChannelSession channel, Stream input, Stream output, string filename, long length)
276  {
277  var buffer = new byte[Math.Min(length, this.BufferSize)];
278  var needToRead = length;
279 
280  do
281  {
282  var read = input.Read(buffer, 0, (int)Math.Min(needToRead, this.BufferSize));
283 
284  output.Write(buffer, 0, read);
285 
286  this.RaiseDownloadingEvent(filename, length, length - needToRead);
287 
288  needToRead -= read;
289  }
290  while (needToRead > 0);
291 
292  output.Flush();
293 
294  // Raise one more time when file downloaded
295  this.RaiseDownloadingEvent(filename, length, length - needToRead);
296 
297  // Send confirmation byte after last data byte was read
298  this.SendConfirmation(channel);
299 
300  this.CheckReturnCode(input);
301  }
