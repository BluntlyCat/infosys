172  override protected void SendBuffer(LoggingEvent[] events)
173  {
174  // Note: this code already owns the monitor for this
175  // appender. This frees us from needing to synchronize again.
176  try
177  {
178  string filePath = null;
179  StreamWriter writer = null;
180 
181  // Impersonate to open the file
182  using(SecurityContext.Impersonate(this))
183  {
184  filePath = Path.Combine(m_pickupDir, SystemInfo.NewGuid().ToString("N"));
185  writer = File.CreateText(filePath);
186  }
187 
188  if (writer == null)
189  {
190  ErrorHandler.Error("Failed to create output file for writing ["+filePath+"]", null, ErrorCode.FileOpenFailure);
191  }
192  else
193  {
194  using(writer)
195  {
196  writer.WriteLine("To: " + m_to);
197  writer.WriteLine("From: " + m_from);
198  writer.WriteLine("Subject: " + m_subject);
199  writer.WriteLine("");
200 
201  string t = Layout.Header;
202  if (t != null)
203  {
204  writer.Write(t);
205  }
206 
207  for(int i = 0; i < events.Length; i++)
208  {
209  // Render the event and append the text to the buffer
210  RenderLoggingEvent(writer, events[i]);
211  }
212 
213  t = Layout.Footer;
214  if (t != null)
215  {
216  writer.Write(t);
217  }
218 
219  writer.WriteLine("");
220  writer.WriteLine(".");
221  }
222  }
223  }
224  catch(Exception e)
225  {
226  ErrorHandler.Error("Error occurred while sending e-mail notification.", e);
227  }
228  }
229 
230  #endregion Override implementation of BufferingAppenderSkeleton
