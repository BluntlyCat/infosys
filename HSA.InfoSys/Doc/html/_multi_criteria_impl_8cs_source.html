92  private IList ListUsingQueryCache()
93  {
94  IQueryCache queryCache = session.Factory.GetQueryCache(cacheRegion);
95 
96  ISet<FilterKey> filterKeys = FilterKey.CreateFilterKeys(session.EnabledFilters, session.EntityMode);
97 
98  ISet<string> querySpaces = new HashSet<string>();
99  List<IType[]> resultTypesList = new List<IType[]>();
100  int[] maxRows = new int[loaders.Count];
101  int[] firstRows = new int[loaders.Count];
102  for (int i = 0; i < loaders.Count; i++)
103  {
104  querySpaces.UnionWith(loaders[i].QuerySpaces);
105  resultTypesList.Add(loaders[i].ResultTypes);
106  firstRows[i] = parameters[i].RowSelection.FirstRow;
107  maxRows[i] = parameters[i].RowSelection.MaxRows;
108  }
109 
110  MultipleQueriesCacheAssembler assembler = new MultipleQueriesCacheAssembler(resultTypesList);
111  QueryParameters combinedParameters = CreateCombinedQueryParameters();
112  QueryKey key = new QueryKey(session.Factory, SqlString, combinedParameters, filterKeys)
113  .SetFirstRows(firstRows)
114  .SetMaxRows(maxRows);
115 
116  IList result =
117  assembler.GetResultFromQueryCache(session,
118  combinedParameters,
119  querySpaces,
120  queryCache,
121  key);
122 
123  if (result == null)
124  {
125  log.Debug("Cache miss for multi criteria query");
126  IList list = DoList();
127  queryCache.Put(key, new ICacheAssembler[] { assembler }, new object[] { list }, combinedParameters.NaturalKeyLookup, session);
128  result = list;
129  }
130 
131  return GetResultList(result);
132  }

193  private void GetResultsFromDatabase(IList results)
194  {
195  bool statsEnabled = session.Factory.Statistics.IsStatisticsEnabled;
196  var stopWatch = new Stopwatch();
197  if (statsEnabled)
198  {
199  stopWatch.Start();
200  }
201  int rowCount = 0;
202 
203  try
204  {
205  using (var reader = resultSetsCommand.GetReader(null))
206  {
207  var hydratedObjects = new List<object>[loaders.Count];
208  List<EntityKey[]>[] subselectResultKeys = new List<EntityKey[]>[loaders.Count];
209  bool[] createSubselects = new bool[loaders.Count];
210  for (int i = 0; i < loaders.Count; i++)
211  {
212  CriteriaLoader loader = loaders[i];
213  int entitySpan = loader.EntityPersisters.Length;
214  hydratedObjects[i] = entitySpan == 0 ? null : new List<object>(entitySpan);
215  EntityKey[] keys = new EntityKey[entitySpan];
216  QueryParameters queryParameters = parameters[i];
217  IList tmpResults = new List<object>();
218 
219  RowSelection selection = parameters[i].RowSelection;
220  createSubselects[i] = loader.IsSubselectLoadingEnabled;
221  subselectResultKeys[i] = createSubselects[i] ? new List<EntityKey[]>() : null;
222  int maxRows = Loader.Loader.HasMaxRows(selection) ? selection.MaxRows : int.MaxValue;
223  if (!dialect.SupportsLimitOffset || !loader.UseLimit(selection, dialect))
224  {
225  Loader.Loader.Advance(reader, selection);
226  }
227  int count;
228  for (count = 0; count < maxRows && reader.Read(); count++)
229  {
230  rowCount++;
231 
232  object o =
233  loader.GetRowFromResultSet(reader, session, queryParameters, loader.GetLockModes(queryParameters.LockModes),
234  null, hydratedObjects[i], keys, true);
235  if (createSubselects[i])
236  {
237  subselectResultKeys[i].Add(keys);
238  keys = new EntityKey[entitySpan]; //can't reuse in this case
239  }
240  tmpResults.Add(o);
241  }
242 
243  results.Add(tmpResults);
244  reader.NextResult();
245  }
246 
247  for (int i = 0; i < loaders.Count; i++)
248  {
249  CriteriaLoader loader = loaders[i];
250  loader.InitializeEntitiesAndCollections(hydratedObjects[i], reader, session, false);
251 
252  if (createSubselects[i])
253  {
254  loader.CreateSubselects(subselectResultKeys[i], parameters[i], session);
255  }
256  }
257  }
258  }
259  catch (Exception sqle)
260  {
261  var message = string.Format("Failed to execute multi criteria: [{0}]", resultSetsCommand.Sql);
262  log.Error(message, sqle);
263  throw ADOExceptionHelper.Convert(session.Factory.SQLExceptionConverter, sqle, "Failed to execute multi criteria", resultSetsCommand.Sql);
264  }
265  if (statsEnabled)
266  {
267  stopWatch.Stop();
268  session.Factory.StatisticsImplementor.QueryExecuted(string.Format("{0} queries (MultiCriteria)", loaders.Count), rowCount, stopWatch.Elapsed);
269  }
270  }

272  private void CreateCriteriaLoaders()
273  {
274  //a criteria can use more than a single query (polymorphic queries), need to have a
275  //way to correlate a loader to a result index
276  int criteriaIndex = 0;
277  foreach (CriteriaImpl criteria in criteriaQueries)
278  {
279  string[] implementors = factory.GetImplementors(criteria.EntityOrClassName);
280  int size = implementors.Length;
281 
282  ISet<string> spaces = new HashSet<string>();
283 
284  for (int i = 0; i < size; i++)
285  {
286  CriteriaLoader loader = new CriteriaLoader(
287  session.GetOuterJoinLoadable(implementors[i]),
288  factory,
289  criteria,
290  implementors[i],
291  session.EnabledFilters
292  );
293  loaders.Add(loader);
294  loaderCriteriaMap.Add(criteriaIndex);
295  spaces.UnionWith(loader.QuerySpaces);
296  }
297  criteriaIndex += 1;
298  }

431  private QueryParameters CreateCombinedQueryParameters()
432  {
433  QueryParameters combinedQueryParameters = new QueryParameters();
434  combinedQueryParameters.ForceCacheRefresh = forceCacheRefresh;
435  combinedQueryParameters.NamedParameters = new Dictionary<string, TypedValue>();
436  var positionalParameterTypes = new List<IType>();
437  var positionalParameterValues = new List<object>();
438  int index = 0;
439  foreach (QueryParameters queryParameters in parameters)
440  {
441  foreach (KeyValuePair<string, TypedValue> dictionaryEntry in queryParameters.NamedParameters)
442  {
443  combinedQueryParameters.NamedParameters.Add(dictionaryEntry.Key + index, dictionaryEntry.Value);
444  }
445  index += 1;
446  positionalParameterTypes.AddRange(queryParameters.PositionalParameterTypes);
447  positionalParameterValues.AddRange(queryParameters.PositionalParameterValues);
448  }
449  combinedQueryParameters.PositionalParameterTypes = positionalParameterTypes.ToArray();
450  combinedQueryParameters.PositionalParameterValues = positionalParameterValues.ToArray();
451  return combinedQueryParameters;
452  }
