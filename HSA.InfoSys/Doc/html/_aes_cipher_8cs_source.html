668  private uint[] GenerateWorkingKey(bool isEncryption, byte[] key)
669  {
670  int KC = key.Length / 4; // key length in words
671 
672  if (((KC != 4) && (KC != 6) && (KC != 8)) || ((KC * 4) != key.Length))
673  throw new ArgumentException("Key length not 128/192/256 bits.");
674 
675  _rounds = KC + 6; // This is not always true for the generalized Rijndael that allows larger block sizes
676  uint[] W = new uint[(_rounds + 1) * 4]; // 4 words in a block
677 
678  //
679  // copy the key into the round key array
680  //
681 
682  int t = 0;
683 
684  for (int i = 0; i < key.Length; t++)
685  {
686  W[(t >> 2) * 4 + (t & 3)] = LittleEndianToUInt32(key, i);
687  i += 4;
688  }
689 
690  //
691  // while not enough round key material calculated
692  // calculate new values
693  //
694  int k = (_rounds + 1) << 2;
695  for (int i = KC; (i < k); i++)
696  {
697  uint temp = W[((i - 1) >> 2) * 4 + ((i - 1) & 3)];
698  if ((i % KC) == 0)
699  {
700  temp = SubWord(Shift(temp, 8)) ^ rcon[(i / KC) - 1];
701  }
702  else if ((KC > 6) && ((i % KC) == 4))
703  {
704  temp = SubWord(temp);
705  }
706 
707  W[(i >> 2) * 4 + (i & 3)] = W[((i - KC) >> 2) * 4 + ((i - KC) & 3)] ^ temp;
708  }
709 
710  if (!isEncryption)
711  {
712  for (int j = 1; j < _rounds; j++)
713  {
714  for (int i = 0; i < 4; i++)
715  {
716  W[j * 4 + i] = InvMcol(W[j * 4 + i]);
717  }
718  }
719  }
720 
721  return W;
722  }
