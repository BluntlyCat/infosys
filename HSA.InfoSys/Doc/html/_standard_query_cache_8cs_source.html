58  public bool Put(QueryKey key, ICacheAssembler[] returnTypes, IList result, bool isNaturalKeyLookup, ISessionImplementor session)
59  {
60  if (isNaturalKeyLookup && result.Count == 0)
61  return false;
62 
63  long ts = session.Timestamp;
64 
65  if (Log.IsDebugEnabled)
66  Log.DebugFormat("caching query results in region: '{0}'; {1}", _regionName, key);
67 
68  IList cacheable = new List<object>(result.Count + 1) {ts};
69  for (int i = 0; i < result.Count; i++)
70  {
71  if (returnTypes.Length == 1 && !key.HasResultTransformer)
72  {
73  cacheable.Add(returnTypes[0].Disassemble(result[i], session, null));
74  }
75  else
76  {
77  cacheable.Add(TypeHelper.Disassemble((object[]) result[i], returnTypes, null, session, null));
78  }
79  }
80 
81  _queryCache.Put(key, cacheable);
82 
83  return true;

87  {
88  if (Log.IsDebugEnabled)
89  Log.DebugFormat("checking cached query results in region: '{0}'; {1}", _regionName, key);
90 
91  var cacheable = (IList)_queryCache.Get(key);
92  if (cacheable == null)
93  {
94  Log.DebugFormat("query results were not found in cache: {0}", key);
95  return null;
96  }
97 
98  var timestamp = (long)cacheable[0];
99 
100  if (Log.IsDebugEnabled)
101  Log.DebugFormat("Checking query spaces for up-to-dateness [{0}]", StringHelper.CollectionToString(spaces));
102 
103  if (!isNaturalKeyLookup && !IsUpToDate(spaces, timestamp))
104  {
105  Log.DebugFormat("cached query results were not up to date for: {0}", key);
106  return null;
107  }
108 
109  Log.DebugFormat("returning cached query results for: {0}", key);
110  for (int i = 1; i < cacheable.Count; i++)
111  {
112  if (returnTypes.Length == 1 && !key.HasResultTransformer)
113  {
114  returnTypes[0].BeforeAssemble(cacheable[i], session);
115  }
116  else
117  {
118  TypeHelper.BeforeAssemble((object[])cacheable[i], returnTypes, session);
119  }
120  }
121 
122  IList result = new List<object>(cacheable.Count - 1);
123  for (int i = 1; i < cacheable.Count; i++)
124  {
125  try
126  {
127  if (returnTypes.Length == 1 && !key.HasResultTransformer)
128  {
129  result.Add(returnTypes[0].Assemble(cacheable[i], session, null));
130  }
131  else
132  {
133  result.Add(TypeHelper.Assemble((object[])cacheable[i], returnTypes, session, null));
134  }
135  }
136  catch (UnresolvableObjectException)
137  {
138  if (isNaturalKeyLookup)
139  {
140  //TODO: not really completely correct, since
141  // the UnresolvableObjectException could occur while resolving
142  // associations, leaving the PC in an inconsistent state
143  Log.Debug("could not reassemble cached result set");
144  _queryCache.Remove(key);
145  return null;
146  }
147 
148  throw;
149  }
