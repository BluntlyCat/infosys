105  public void CascadeOn(IEntityPersister persister, object parent, object anything)
106  {
107  if (persister.HasCascades || action.RequiresNoCascadeChecking)
108  {
109  log.Info("processing cascade " + action + " for: " + persister.EntityName);
110 
111  IType[] types = persister.PropertyTypes;
112  CascadeStyle[] cascadeStyles = persister.PropertyCascadeStyles;
113  EntityMode entityMode = eventSource.EntityMode;
114  bool hasUninitializedLazyProperties = persister.HasUninitializedLazyProperties(parent, entityMode);
115  for (int i = 0; i < types.Length; i++)
116  {
117  CascadeStyle style = cascadeStyles[i];
118  if (hasUninitializedLazyProperties && persister.PropertyLaziness[i] && !action.PerformOnLazyProperty)
119  {
120  //do nothing to avoid a lazy property initialization
121  continue;
122  }
123 
124  if (style.DoCascade(action))
125  {
126  CascadeProperty(parent, persister.GetPropertyValue(parent, i, entityMode), types[i], style, anything, false);
127  }
128  else if (action.RequiresNoCascadeChecking)
129  {
130  action.NoCascade(eventSource, persister.GetPropertyValue(parent, i, entityMode), parent, persister, i);
131  }
132  }
133 
134  log.Info("done processing cascade " + action + " for: " + persister.EntityName);
135  }
136  }

139  private void CascadeProperty(object parent, object child, IType type, CascadeStyle style, object anything, bool isCascadeDeleteEnabled)
140  {
141  if (child != null)
142  {

190  private void CascadeCollection(object parent, object child, CascadeStyle style, object anything, CollectionType type)
191  {
192  ICollectionPersister persister = eventSource.Factory.GetCollectionPersister(type.Role);
193  IType elemType = persister.ElementType;
194 
195  CascadePoint oldCascadeTo = point;
196  if (point == CascadePoint.AfterInsertBeforeDelete)
197  point = CascadePoint.AfterInsertBeforeDeleteViaCollection;
198 
199  //cascade to current collection elements
200  if (elemType.IsEntityType || elemType.IsAnyType || elemType.IsComponentType)
201  CascadeCollectionElements(parent, child, type, style, elemType, anything, persister.CascadeDeleteEnabled);
202 
203  point = oldCascadeTo;
204  }

207  private void CascadeToOne(object parent, object child, IType type, CascadeStyle style, object anything, bool isCascadeDeleteEnabled)
208  {
209  string entityName = type.IsEntityType ? ((EntityType)type).GetAssociatedEntityName() : null;
210  if (style.ReallyDoCascade(action))
211  {
212  //not really necessary, but good for consistency...
213  eventSource.PersistenceContext.AddChildParent(child, parent);
214  try
215  {
216  action.Cascade(eventSource, child, entityName, anything, isCascadeDeleteEnabled);
217  }
218  finally
219  {
220  eventSource.PersistenceContext.RemoveChildParent(child);
221  }
222  }
223  }

226  private void CascadeCollectionElements(object parent, object child, CollectionType collectionType, CascadeStyle style, IType elemType, object anything, bool isCascadeDeleteEnabled)
227  {
228  // we can't cascade to non-embedded elements
229  bool embeddedElements = eventSource.EntityMode != EntityMode.Xml
230  || ((EntityType) collectionType.GetElementType(eventSource.Factory)).IsEmbeddedInXML;
231 
232  bool reallyDoCascade = style.ReallyDoCascade(action) && embeddedElements
233  && child != CollectionType.UnfetchedCollection;
234 
235  if (reallyDoCascade)
236  {
237  log.Info("cascade " + action + " for collection: " + collectionType.Role);
238 
239  foreach (object o in action.GetCascadableChildrenIterator(eventSource, collectionType, child))
240  CascadeProperty(parent, o, elemType, style, anything, isCascadeDeleteEnabled);
241 
242  log.Info("done cascade " + action + " for collection: " + collectionType.Role);
243  }
244 
245  var childAsPersColl = child as IPersistentCollection;
246  bool deleteOrphans = style.HasOrphanDelete && action.DeleteOrphans && elemType.IsEntityType
247  && childAsPersColl != null; //a newly instantiated collection can't have orphans
248 
249  if (deleteOrphans)
250  {
251  // handle orphaned entities!!
252  log.Info("deleting orphans for collection: " + collectionType.Role);
253 
254  // we can do the cast since orphan-delete does not apply to:
255  // 1. newly instantiated collections
256  // 2. arrays (we can't track orphans for detached arrays)
257  string entityName = collectionType.GetAssociatedEntityName(eventSource.Factory);
258  DeleteOrphans(entityName, childAsPersColl);
259 
260  log.Info("done deleting orphans for collection: " + collectionType.Role);
261  }
262  }

265  private void DeleteOrphans(string entityName, IPersistentCollection pc)
266  {
267  //TODO: suck this logic into the collection!
268  ICollection orphans;
269  if (pc.WasInitialized)
270  {
271  CollectionEntry ce = eventSource.PersistenceContext.GetCollectionEntry(pc);
272  orphans = ce == null ? CollectionHelper.EmptyCollection : ce.GetOrphans(entityName, pc);
273  }
274  else
275  {
276  orphans = pc.GetQueuedOrphans(entityName);
277  }
278 
279  foreach (object orphan in orphans)
280  {
281  if (orphan != null)
282  {
283  log.Info("deleting orphaned entity instance: " + entityName);
284 
285  eventSource.Delete(entityName, orphan, false, null);
286  }
