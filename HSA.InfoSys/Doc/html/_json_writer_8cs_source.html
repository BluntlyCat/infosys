77  internal static State[][] BuildStateArray()
78  {
79  var allStates = StateArrayTempate.ToList();
80  var errorStates = StateArrayTempate[0];
81  var valueStates = StateArrayTempate[7];
82 
83  foreach (JsonToken valueToken in EnumUtils.GetValues(typeof(JsonToken)))
84  {
85  if (allStates.Count <= (int)valueToken)
86  {
87  switch (valueToken)
88  {
89  case JsonToken.Integer:
90  case JsonToken.Float:
91  case JsonToken.String:
92  case JsonToken.Boolean:
93  case JsonToken.Null:
94  case JsonToken.Undefined:
95  case JsonToken.Date:
96  case JsonToken.Bytes:
97  allStates.Add(valueStates);
98  break;
99  default:
100  allStates.Add(errorStates);
101  break;
102  }
103  }
104  }
105 
106  return allStates.ToArray();
107  }

450  internal void WriteToken(JsonReader reader, bool writeChildren, bool writeDateConstructorAsDate)
451  {
452  int initialDepth;
453 
454  if (reader.TokenType == JsonToken.None)
455  initialDepth = -1;
456  else if (!IsStartToken(reader.TokenType))
457  initialDepth = reader.Depth + 1;
458  else
459  initialDepth = reader.Depth;
460 
461  WriteToken(reader, initialDepth, writeChildren, writeDateConstructorAsDate);
462  }
463 
464  internal void WriteToken(JsonReader reader, int initialDepth, bool writeChildren, bool writeDateConstructorAsDate)
465  {
466  do
467  {
468  switch (reader.TokenType)
469  {
470  case JsonToken.None:
471  // read to next
472  break;
473  case JsonToken.StartObject:
474  WriteStartObject();
475  break;
476  case JsonToken.StartArray:
477  WriteStartArray();
478  break;
479  case JsonToken.StartConstructor:
480  string constructorName = reader.Value.ToString();
481  // write a JValue date when the constructor is for a date
482  if (writeDateConstructorAsDate && string.Equals(constructorName, "Date", StringComparison.Ordinal))
483  WriteConstructorDate(reader);
484  else
485  WriteStartConstructor(reader.Value.ToString());
486  break;
487  case JsonToken.PropertyName:
488  WritePropertyName(reader.Value.ToString());
489  break;
490  case JsonToken.Comment:
491  WriteComment((reader.Value != null) ? reader.Value.ToString() : null);
492  break;
493  case JsonToken.Integer:
494 #if !(NET20 || NET35 || SILVERLIGHT || PORTABLE || PORTABLE40)
495  if (reader.Value is BigInteger)
496  {
497  WriteValue((BigInteger)reader.Value);
498  }
499  else
500 #endif
501  {
502  WriteValue(Convert.ToInt64(reader.Value, CultureInfo.InvariantCulture));
503  }
504  break;
505  case JsonToken.Float:
506  object value = reader.Value;
507 
508  if (value is decimal)
509  WriteValue((decimal)value);
510  else if (value is double)
511  WriteValue((double)value);
512  else if (value is float)
513  WriteValue((float)value);
514  else
515  WriteValue(Convert.ToDouble(value, CultureInfo.InvariantCulture));
516  break;
517  case JsonToken.String:
518  WriteValue(reader.Value.ToString());
519  break;
520  case JsonToken.Boolean:
521  WriteValue(Convert.ToBoolean(reader.Value, CultureInfo.InvariantCulture));
522  break;
523  case JsonToken.Null:
524  WriteNull();
525  break;
526  case JsonToken.Undefined:
527  WriteUndefined();
528  break;
529  case JsonToken.EndObject:
530  WriteEndObject();
531  break;
532  case JsonToken.EndArray:
533  WriteEndArray();
534  break;
535  case JsonToken.EndConstructor:
536  WriteEndConstructor();
537  break;
538  case JsonToken.Date:
539 #if !NET20
540  if (reader.Value is DateTimeOffset)
541  WriteValue((DateTimeOffset)reader.Value);
542  else
543 #endif
544  WriteValue(Convert.ToDateTime(reader.Value, CultureInfo.InvariantCulture));
545  break;
546  case JsonToken.Raw:
547  WriteRawValue((reader.Value != null) ? reader.Value.ToString() : null);
548  break;
549  case JsonToken.Bytes:
550  WriteValue((byte[])reader.Value);
551  break;
552  default:
553  throw MiscellaneousUtils.CreateArgumentOutOfRangeException("TokenType", reader.TokenType, "Unexpected token type.");
554  }
555  }
556  while (
557  // stop if we have reached the end of the token being read
558  initialDepth - 1 < reader.Depth - (IsEndToken(reader.TokenType) ? 1 : 0)
559  && writeChildren
560  && reader.Read());
561  }

563  private void WriteConstructorDate(JsonReader reader)
564  {
565  if (!reader.Read())
566  throw JsonWriterException.Create(this, "Unexpected end when reading date constructor.", null);
567  if (reader.TokenType != JsonToken.Integer)

579  }

648  private void AutoCompleteClose(JsonContainerType type)
649  {
650  // write closing symbol and calculate new state
651  int levelsToComplete = 0;
652 
653  if (_currentPosition.Type == type)
654  {
655  levelsToComplete = 1;
656  }
657  else
658  {
659  int top = Top - 2;
660  for (int i = top; i >= 0; i--)
661  {
662  int currentLevel = top - i;
663 
664  if (_stack[currentLevel].Type == type)
665  {
666  levelsToComplete = i + 2;
667  break;
668  }
669  }
670  }
671 
672  if (levelsToComplete == 0)
673  throw JsonWriterException.Create(this, "No token to close.", null);
674 
675  for (int i = 0; i < levelsToComplete; i++)
676  {
677  JsonToken token = GetCloseTokenForType(Pop());
678 
679  if (_currentState == State.Property)
680  WriteNull();
681 
682  if (_formatting == Formatting.Indented)
683  {
684  if (_currentState != State.ObjectStart && _currentState != State.ArrayStart)
685  WriteIndent();
686  }
687 
688  WriteEnd(token);
689 
690  JsonContainerType currentLevelType = Peek();
691 
692  switch (currentLevelType)
693  {
694  case JsonContainerType.Object:
695  _currentState = State.Object;
696  break;
697  case JsonContainerType.Array:
698  _currentState = State.Array;
699  break;
700  case JsonContainerType.Constructor:
701  _currentState = State.Array;
702  break;
703  case JsonContainerType.None:
704  _currentState = State.Start;
705  break;
706  default:
707  throw JsonWriterException.Create(this, "Unknown JsonType: " + currentLevelType, null);
708  }
709  }
710  }

741  internal void AutoComplete(JsonToken tokenBeingWritten)
742  {
743  // gets new state based on the current state and what is being written
744  State newState = StateArray[(int)tokenBeingWritten][(int)_currentState];
745 
746  if (newState == State.Error)
747  throw JsonWriterException.Create(this, "Token {0} in state {1} would result in an invalid JSON object.".FormatWith(CultureInfo.InvariantCulture, tokenBeingWritten.ToString(), _currentState.ToString()), null);
748 
749  if ((_currentState == State.Object || _currentState == State.Array || _currentState == State.Constructor) && tokenBeingWritten != JsonToken.Comment)
750  {
751  WriteValueDelimiter();
752  }
753 
754  if (_formatting == Formatting.Indented)
755  {
756  if (_currentState == State.Property)
757  WriteIndentSpace();
758 
759  // don't indent a property when it is the first token to be written (i.e. at the start)
760  if ((_currentState == State.Array || _currentState == State.ArrayStart || _currentState == State.Constructor || _currentState == State.ConstructorStart)
761  || (tokenBeingWritten == JsonToken.PropertyName && _currentState != State.Start))
762  WriteIndent();
763  }
764 
765  _currentState = newState;
766  }
