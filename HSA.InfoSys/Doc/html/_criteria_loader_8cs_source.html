118  protected override SqlString ApplyLocks(SqlString sqlSelectString, IDictionary<string, LockMode> lockModes,
119  Dialect.Dialect dialect)
120  {
121  if (lockModes == null || lockModes.Count == 0)
122  {
123  return sqlSelectString;
124  }
125 
126  Dictionary<string, LockMode> aliasedLockModes = new Dictionary<string, LockMode>();
127  Dictionary<string, string[]> keyColumnNames = dialect.ForUpdateOfColumns ? new Dictionary<string, string[]>() : null;
128  string[] drivingSqlAliases = Aliases;
129  for (int i = 0; i < drivingSqlAliases.Length; i++)
130  {
131  LockMode lockMode;
132  if (lockModes.TryGetValue(drivingSqlAliases[i], out lockMode))
133  {
134  ILockable drivingPersister = (ILockable) EntityPersisters[i];
135  string rootSqlAlias = drivingPersister.GetRootTableAlias(drivingSqlAliases[i]);
136  aliasedLockModes[rootSqlAlias] = lockMode;
137  if (keyColumnNames != null)
138  {
139  keyColumnNames[rootSqlAlias] = drivingPersister.RootTableIdentifierColumnNames;
140  }
141  }
142  }
143 
144  return dialect.ApplyLocksToSql(sqlSelectString, lockModes, keyColumnNames);
145  }
