208  public static long ToFileSize(string argValue, long defaultValue)
209  {
210  if (argValue == null)
211  {
212  return defaultValue;
213  }
214 
215  string s = argValue.Trim().ToUpper(CultureInfo.InvariantCulture);
216  long multiplier = 1;
217  int index;
218 
219  if ((index = s.IndexOf("KB")) != -1)
220  {
221  multiplier = 1024;
222  s = s.Substring(0, index);
223  }
224  else if ((index = s.IndexOf("MB")) != -1)
225  {
226  multiplier = 1024 * 1024;
227  s = s.Substring(0, index);
228  }
229  else if ((index = s.IndexOf("GB")) != -1)
230  {
231  multiplier = 1024 * 1024 * 1024;
232  s = s.Substring(0, index);
233  }
234  if (s != null)
235  {
236  // Try again to remove whitespace between the number and the size specifier
237  s = s.Trim();
238 
239  long longVal;
240  if (SystemInfo.TryParse(s, out longVal))
241  {
242  return longVal * multiplier;
243  }
244  else
245  {
246  LogLog.Error(declaringType, "OptionConverter: ["+ s +"] is not in the correct file size syntax.");
247  }
248  }
249  return defaultValue;
250  }

344  public static bool CanConvertTypeTo(Type sourceType, Type targetType)
345  {
346  if (sourceType == null || targetType == null)
347  {
348  return false;
349  }
350 
351  // Check if we can assign directly from the source type to the target type
352  if (targetType.IsAssignableFrom(sourceType))
353  {
354  return true;
355  }
356 
357  // Look for a To converter
358  IConvertTo tcSource = ConverterRegistry.GetConvertTo(sourceType, targetType);
359  if (tcSource != null)
360  {

538  public static string SubstituteVariables(string value, System.Collections.IDictionary props)
539  {
540  StringBuilder buf = new StringBuilder();
541 
542  int i = 0;
543  int j, k;
544 
545  while(true)
546  {
547  j = value.IndexOf(DELIM_START, i);
548  if (j == -1)
549  {
550  if (i == 0)
551  {
552  return value;
553  }
554  else
555  {
556  buf.Append(value.Substring(i, value.Length - i));
557  return buf.ToString();
558  }
559  }
560  else
561  {
562  buf.Append(value.Substring(i, j - i));
563  k = value.IndexOf(DELIM_STOP, j);
564  if (k == -1)
565  {
566  throw new LogException("[" + value + "] has no closing brace. Opening brace at position [" + j + "]");
567  }
568  else
569  {
570  j += DELIM_START_LEN;
571  string key = value.Substring(j, k - j);
572 
573  string replacement = props[key] as string;
574 
575  if (replacement != null)
576  {
577  buf.Append(replacement);
578  }
579  i = k + DELIM_STOP_LEN;
580  }
581  }
582  }
583  }

597  private static object ParseEnum(System.Type enumType, string value, bool ignoreCase)
598  {
599 #if !NETCF
600  return Enum.Parse(enumType, value, ignoreCase);
601 #else
602  FieldInfo[] fields = enumType.GetFields(BindingFlags.Public | BindingFlags.Static);
603 
604  string[] names = value.Split(new char[] {','});
605  for (int i = 0; i < names.Length; ++i)
606  {
607  names[i] = names [i].Trim();
608  }
609 
610  long retVal = 0;
611 
612  try
613  {
614  // Attempt to convert to numeric type
615  return Enum.ToObject(enumType, Convert.ChangeType(value, typeof(long), CultureInfo.InvariantCulture));
616  }
617  catch {}
618 
619  foreach (string name in names)
620  {
621  bool found = false;
622  foreach(FieldInfo field in fields)
623  {
624  if (String.Compare(name, field.Name, ignoreCase) == 0)
625  {
626  retVal |= ((IConvertible) field.GetValue(null)).ToInt64(CultureInfo.InvariantCulture);
627  found = true;
628  break;
629  }
630  }
631  if (!found)
632  {
633  throw new ArgumentException("Failed to lookup member [" + name + "] from Enum type [" + enumType.Name + "]");
634  }
635  }
636  return Enum.ToObject(enumType, retVal);
637 #endif
638  }
