260  public override SqlString GetLimitString(SqlString sql, SqlString offset, SqlString limit)
261  {
262  sql = sql.Trim();
263  bool isForUpdate = false;
264  if (sql.EndsWithCaseInsensitive(" for update"))
265  {
266  sql = sql.Substring(0, sql.Length - 11);
267  isForUpdate = true;
268  }
269 
270  string selectColumns = ExtractColumnOrAliasNames(sql);
271 
272  var pagingSelect = new SqlStringBuilder(sql.Count + 10);
273  if (offset != null)
274  {
275  pagingSelect.Add("select " + selectColumns + " from ( select row_.*, rownum rownum_ from ( ");
276  }
277  else
278  {
279  pagingSelect.Add("select " + selectColumns + " from ( ");
280  }
281  pagingSelect.Add(sql);
282  if (offset != null && limit != null)
283  {
284  pagingSelect.Add(" ) row_ where rownum <=").Add(limit).Add(") where rownum_ >").Add(offset);
285  }
286  else if (limit != null)
287  {
288  pagingSelect.Add(" ) where rownum <=").Add(limit);
289  }
290  else
291  {
292  // offset is specified, but limit is not.
293  pagingSelect.Add(" ) row_ ) where rownum_ >").Add(offset);
294  }
295 
296  if (isForUpdate)
297  {
298  pagingSelect.Add(" for update");
299  }
300 
301  return pagingSelect.ToSqlString();
302  }
