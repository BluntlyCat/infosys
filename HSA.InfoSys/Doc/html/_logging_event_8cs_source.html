742  public string ThreadName
743  {
744  get
745  {
746  if (m_data.ThreadName == null && this.m_cacheUpdatable)
747  {
748 #if NETCF
749  // Get thread ID only
750  m_data.ThreadName = SystemInfo.CurrentThreadId.ToString(System.Globalization.NumberFormatInfo.InvariantInfo);
751 #else
752  m_data.ThreadName = System.Threading.Thread.CurrentThread.Name;
753  if (m_data.ThreadName == null || m_data.ThreadName.Length == 0)
754  {
755  // The thread name is not available. Therefore we
756  // go the the AppDomain to get the ID of the
757  // current thread. (Why don't Threads know their own ID?)
758  try
759  {
760  m_data.ThreadName = SystemInfo.CurrentThreadId.ToString(System.Globalization.NumberFormatInfo.InvariantInfo);
761  }
762  catch(System.Security.SecurityException)
763  {
764  // This security exception will occur if the caller does not have
765  // some undefined set of SecurityPermission flags.
766  LogLog.Debug(declaringType, "Security exception while trying to get current thread ID. Error Ignored. Empty thread name.");
767 
768  // As a last resort use the hash code of the Thread object
769  m_data.ThreadName = System.Threading.Thread.CurrentThread.GetHashCode().ToString(System.Globalization.CultureInfo.InvariantCulture);

1219  protected void FixVolatileData(FixFlags flags)
1220  {
1221  object forceCreation = null;
1222 
1223  //Unlock the cache so that new values can be stored
1224  //This may not be ideal if we are no longer in the correct context
1225  //and someone calls fix.
1226  m_cacheUpdatable=true;
1227 
1228  // determine the flags that we are actually fixing
1229  FixFlags updateFlags = (FixFlags)((flags ^ m_fixFlags) & flags);
1230 
1231  if (updateFlags > 0)
1232  {
1233  if ((updateFlags & FixFlags.Message) != 0)
1234  {
1235  // Force the message to be rendered
1236  forceCreation = this.RenderedMessage;
1237 
1238  m_fixFlags |= FixFlags.Message;
1239  }
1240  if ((updateFlags & FixFlags.ThreadName) != 0)
1241  {
1242  // Grab the thread name
1243  forceCreation = this.ThreadName;
1244 
1245  m_fixFlags |= FixFlags.ThreadName;
1246  }
1247 
1248  if ((updateFlags & FixFlags.LocationInfo) != 0)
1249  {
1250  // Force the location information to be loaded
1251  forceCreation = this.LocationInformation;
1252 
1253  m_fixFlags |= FixFlags.LocationInfo;
1254  }
1255  if ((updateFlags & FixFlags.UserName) != 0)
1256  {
1257  // Grab the user name
1258  forceCreation = this.UserName;
1259 
1260  m_fixFlags |= FixFlags.UserName;
1261  }
1262  if ((updateFlags & FixFlags.Domain) != 0)
1263  {
1264  // Grab the domain name
1265  forceCreation = this.Domain;
1266 
1267  m_fixFlags |= FixFlags.Domain;
1268  }
1269  if ((updateFlags & FixFlags.Identity) != 0)
1270  {
1271  // Grab the identity
1272  forceCreation = this.Identity;
1273 
1274  m_fixFlags |= FixFlags.Identity;
1275  }
1276 
1277  if ((updateFlags & FixFlags.Exception) != 0)
1278  {
1279  // Force the exception text to be loaded
1280  forceCreation = GetExceptionString();
1281 
1282  m_fixFlags |= FixFlags.Exception;
1283  }
1284 
1285  if ((updateFlags & FixFlags.Properties) != 0)
1286  {
1287  CacheProperties();
1288 
1289  m_fixFlags |= FixFlags.Properties;
1290  }
1291  }
1292 
1293  // avoid warning CS0219
1294  if (forceCreation != null)
1295  {
1296  }
1297 
1298  //Finaly lock everything we've cached.
1299  m_cacheUpdatable=false;
1300  }

1332  private void CacheProperties()
1333  {
1334  if (m_data.Properties == null && this.m_cacheUpdatable)
1335  {
1336  if (m_compositeProperties == null)
1337  {
1338  CreateCompositeProperties();
1339  }
1340 
1341  PropertiesDictionary flattenedProperties = m_compositeProperties.Flatten();
1342 
1343  PropertiesDictionary fixedProperties = new PropertiesDictionary();
1344 
1345  // Validate properties
1346  foreach(DictionaryEntry entry in flattenedProperties)
1347  {
1348  string key = entry.Key as string;
1349 
1350  if (key != null)
1351  {
1352  object val = entry.Value;
1353 
1354  // Fix any IFixingRequired objects
1355  IFixingRequired fixingRequired = val as IFixingRequired;
1356  if (fixingRequired != null)
1357  {
1358  val = fixingRequired.GetFixedObject();
1359  }
1360 
1361  // Strip keys with null values
1362  if (val != null)
1363  {
