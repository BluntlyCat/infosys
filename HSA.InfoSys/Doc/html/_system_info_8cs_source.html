626  public static Type GetTypeFromString(Assembly relativeAssembly, string typeName, bool throwOnError, bool ignoreCase)
627  {
628  // Check if the type name specifies the assembly name
629  if(typeName.IndexOf(',') == -1)
630  {
631  //LogLog.Debug(declaringType, "SystemInfo: Loading type ["+typeName+"] from assembly ["+relativeAssembly.FullName+"]");
632 #if NETCF
633  return relativeAssembly.GetType(typeName, throwOnError);
634 #else
635  // Attempt to lookup the type from the relativeAssembly
636  Type type = relativeAssembly.GetType(typeName, false, ignoreCase);
637  if (type != null)
638  {
639  // Found type in relative assembly
640  //LogLog.Debug(declaringType, "SystemInfo: Loaded type ["+typeName+"] from assembly ["+relativeAssembly.FullName+"]");
641  return type;
642  }
643 
644  Assembly[] loadedAssemblies = null;
645  try
646  {
647  loadedAssemblies = AppDomain.CurrentDomain.GetAssemblies();
648  }
649  catch(System.Security.SecurityException)
650  {
651  // Insufficient permissions to get the list of loaded assemblies
652  }
653 
654  if (loadedAssemblies != null)
655  {
656  // Search the loaded assemblies for the type
657  foreach (Assembly assembly in loadedAssemblies)
658  {
659  type = assembly.GetType(typeName, false, ignoreCase);
660  if (type != null)
661  {
662  // Found type in loaded assembly
663  LogLog.Debug(declaringType, "Loaded type ["+typeName+"] from assembly ["+assembly.FullName+"] by searching loaded assemblies.");
664  return type;
665  }
666  }
667  }
668 
669  // Didn't find the type
670  if (throwOnError)
671  {
672  throw new TypeLoadException("Could not load type ["+typeName+"]. Tried assembly ["+relativeAssembly.FullName+"] and all loaded assemblies");
673  }
674  return null;
675 #endif
676  }
677  else
678  {
679  // Includes explicit assembly name
680  //LogLog.Debug(declaringType, "SystemInfo: Loading type ["+typeName+"] from global Type");
681 
682 #if NETCF
683  // In NETCF 2 and 3 arg versions seem to behave differently
684  // https://issues.apache.org/jira/browse/LOG4NET-113
685  return Type.GetType(typeName, throwOnError);
686 #else
687  return Type.GetType(typeName, throwOnError, ignoreCase);
688 #endif
689  }
690  }

932  public static string ConvertToFullPath(string path)
933  {
934  if (path == null)
935  {
936  throw new ArgumentNullException("path");
937  }
938 
939  string baseDirectory = "";
940  try
941  {
942  string applicationBaseDirectory = SystemInfo.ApplicationBaseDirectory;
943  if (applicationBaseDirectory != null)
944  {
945  // applicationBaseDirectory may be a URI not a local file path
946  Uri applicationBaseDirectoryUri = new Uri(applicationBaseDirectory);
947  if (applicationBaseDirectoryUri.IsFile)
948  {
949  baseDirectory = applicationBaseDirectoryUri.LocalPath;
950  }
951  }
952  }
953  catch
954  {
955  // Ignore URI exceptions & SecurityExceptions from SystemInfo.ApplicationBaseDirectory
956  }
957 
958  if (baseDirectory != null && baseDirectory.Length > 0)
959  {
960  // Note that Path.Combine will return the second path if it is rooted
961  return Path.GetFullPath(Path.Combine(baseDirectory, path));
962  }

1152  public static Guid NewGuid()
1153  {
1154  IntPtr hCryptProv = IntPtr.Zero;
1155  Guid guid = Guid.Empty;
1156 
1157  try
1158  {
1159  // holds random bits for guid
1160  byte[] bits = new byte[Const.ByteArraySize];
1161 
1162  // get crypto provider handle
1163  if (!WinApi.CryptAcquireContext(ref hCryptProv, null, null,
1164  WinApi.PROV_RSA_FULL, WinApi.CRYPT_VERIFYCONTEXT))
1165  {
1166  throw new SystemException(
1167  "Failed to acquire cryptography handle.");
1168  }
1169 
1170  // generate a 128 bit (16 byte) cryptographically random number
1171  if (!WinApi.CryptGenRandom(hCryptProv, bits.Length, bits))
1172  {
1173  throw new SystemException(
1174  "Failed to generate cryptography random bytes.");
1175  }
1176 
1177  // set the variant
1178  bits[Const.VariantByte] &= Const.VariantByteMask;
1179  bits[Const.VariantByte] |=
1180  ((int)GuidVariant.Standard << Const.VariantByteShift);
1181 
1182  // set the version
1183  bits[Const.VersionByte] &= Const.VersionByteMask;
1184  bits[Const.VersionByte] |=
1185  ((int)GuidVersion.Random << Const.VersionByteShift);
1186 
1187  // create the new System.Guid object
1188  guid = new Guid(bits);
1189  }
1190  finally
1191  {
1192  // release the crypto provider handle
1193  if (hCryptProv != IntPtr.Zero)
1194  WinApi.CryptReleaseContext(hCryptProv, 0);
1195  }
1196 
1197  return guid;
1198  }
