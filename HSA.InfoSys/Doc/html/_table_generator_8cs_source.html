212  public override object DoWorkInCurrentTransaction(ISessionImplementor session, IDbConnection conn,
213  IDbTransaction transaction)
214  {
215  long result;
216  int rows;
217  do
218  {
219  //the loop ensure atomicitiy of the
220  //select + uspdate even for no transaction
221  //or read committed isolation level (needed for .net?)
222 
223  IDbCommand qps = conn.CreateCommand();
224  IDataReader rs = null;
225  qps.CommandText = query;
226  qps.CommandType = CommandType.Text;
227  qps.Transaction = transaction;
228  PersistentIdGeneratorParmsNames.SqlStatementLogger.LogCommand("Reading high value:", qps, FormatStyle.Basic);
229  try
230  {
231  rs = qps.ExecuteReader();
232  if (!rs.Read())
233  {
234  string err;
235  if (string.IsNullOrEmpty(whereClause))
236  {
237  err = "could not read a hi value - you need to populate the table: " + tableName;
238  }
239  else
240  {
241  err = string.Format("could not read a hi value from table '{0}' using the where clause ({1})- you need to populate the table.", tableName, whereClause);
242  }
243  log.Error(err);
244  throw new IdentifierGenerationException(err);
245  }
246  result = Convert.ToInt64(columnType.Get(rs, 0));
247  }
248  catch (Exception e)
249  {
250  log.Error("could not read a hi value", e);
251  throw;
252  }
253  finally
254  {
255  if (rs != null)
256  {
257  rs.Close();
258  }
259  qps.Dispose();
260  }
261 
262  IDbCommand ups = session.Factory.ConnectionProvider.Driver.GenerateCommand(CommandType.Text, updateSql,
263  parameterTypes);
264  ups.Connection = conn;
265  ups.Transaction = transaction;
266 
267  try
268  {
269  columnType.Set(ups, result + 1, 0);
270  columnType.Set(ups, result, 1);
271 
272  PersistentIdGeneratorParmsNames.SqlStatementLogger.LogCommand("Updating high value:", ups, FormatStyle.Basic);
273 
274  rows = ups.ExecuteNonQuery();
275  }
276  catch (Exception e)
277  {
278  log.Error("could not update hi value in: " + tableName, e);
279  throw;
280  }
281  finally
282  {
283  ups.Dispose();
284  }
285  }
286  while (rows == 0);
287 
288  return result;
289  }
290  }
