375  public object[] GetNaturalIdSnapshot(object id, IEntityPersister persister)
376  {
377  if (!persister.HasNaturalIdentifier)
378  {
379  return null;
380  }
381 
382  // if the natural-id is marked as non-mutable, it is not retrieved during a
383  // normal database-snapshot operation...
384  int[] props = persister.NaturalIdentifierProperties;
385  bool[] updateable = persister.PropertyUpdateability;
386  bool allNatualIdPropsAreUpdateable = true;
387  for (int i = 0; i < props.Length; i++)
388  {
389  if (!updateable[props[i]])
390  {
391  allNatualIdPropsAreUpdateable = false;
392  break;
393  }
394  }
395 
396  if (allNatualIdPropsAreUpdateable)
397  {
398  // do this when all the properties are updateable since there is
399  // a certain likelihood that the information will already be
400  // snapshot-cached.
401  object[] entitySnapshot = GetDatabaseSnapshot(id, persister);
402  if (entitySnapshot == NoRow)
403  {
404  return null;
405  }
406  object[] naturalIdSnapshot = new object[props.Length];
407  for (int i = 0; i < props.Length; i++)
408  {
409  naturalIdSnapshot[i] = entitySnapshot[props[i]];
410  }
411  return naturalIdSnapshot;
412  }
413  else
414  {
415  return persister.GetNaturalIdentifierSnapshot(id, session);

944  public void InitializeNonLazyCollections()
945  {
946  if (loadCounter == 0)
947  {
948  log.Debug("initializing non-lazy collections");
949  //do this work only at the very highest level of the load
950  loadCounter++; //don't let this method be called recursively
951  try
952  {
953  while (nonlazyCollections.Count > 0)
954  {
955  //note that each iteration of the loop may add new elements
956  IPersistentCollection tempObject = nonlazyCollections[nonlazyCollections.Count - 1];
957  nonlazyCollections.RemoveAt(nonlazyCollections.Count - 1);
958  tempObject.ForceInitialization();
959  }
960  }
961  finally
962  {
963  loadCounter--;
964  ClearNullProperties();
965  }
966  }

1093  public object GetOwnerId(string entityName, string propertyName, object childEntity, IDictionary mergeMap)
1094  {
1095  string collectionRole = entityName + '.' + propertyName;
1096  IEntityPersister persister = session.Factory.GetEntityPersister(entityName);
1097  ICollectionPersister collectionPersister = session.Factory.GetCollectionPersister(collectionRole);
1098 
1099  object parent = parentsByChild[childEntity];
1100  if (parent != null)
1101  {
1102  var entityEntry = (EntityEntry) entityEntries[parent];
1103  //there maybe more than one parent, filter by type
1104  if (persister.IsSubclassEntityName(entityEntry.EntityName) && IsFoundInParent(propertyName, childEntity, persister, collectionPersister, parent))
1105  {
1106  return GetEntry(parent).Id;
1107  }
1108  parentsByChild.Remove(childEntity); // remove wrong entry
1109  }
1110 
1111  // iterate all the entities currently associated with the persistence context.
1112  foreach (DictionaryEntry entry in entityEntries)
1113  {
1114  var entityEntry = (EntityEntry) entry.Value;
1115  // does this entity entry pertain to the entity persister in which we are interested (owner)?
1116  if (persister.IsSubclassEntityName(entityEntry.EntityName))
1117  {
1118  object entityEntryInstance = entry.Key;
1119 
1120  //check if the managed object is the parent
1121  bool found = IsFoundInParent(propertyName, childEntity, persister, collectionPersister, entityEntryInstance);
1122 
1123  if (!found && mergeMap != null)
1124  {
1125  //check if the detached object being merged is the parent
1126  object unmergedInstance = mergeMap[entityEntryInstance];
1127  object unmergedChild = mergeMap[childEntity];
1128  if (unmergedInstance != null && unmergedChild != null)
1129  {
1130  found = IsFoundInParent(propertyName, unmergedChild, persister, collectionPersister, unmergedInstance);
1131  }
1132  }
1133 
1134  if (found)
1135  {
1136  return entityEntry.Id;
1137  }
1138  }
1139  }
1140 
1141  // if we get here, it is possible that we have a proxy 'in the way' of the merge map resolution...
1142  // NOTE: decided to put this here rather than in the above loop as I was nervous about the performance
1143  // of the loop-in-loop especially considering this is far more likely the 'edge case'
1144  if (mergeMap != null)
1145  {
1146  foreach (DictionaryEntry mergeMapEntry in mergeMap)
1147  {
1148  var proxy = mergeMapEntry.Key as INHibernateProxy;
1149  if (proxy != null)
1150  {
1151  if (persister.IsSubclassEntityName(proxy.HibernateLazyInitializer.EntityName))
1152  {
1153  bool found = IsFoundInParent(propertyName, childEntity, persister, collectionPersister, mergeMap[proxy]);
1154  if (!found)
1155  {
1156  found = IsFoundInParent(propertyName, mergeMap[childEntity], persister, collectionPersister, mergeMap[proxy]);
1157  }
1158  if (found)
1159  {
1160  return proxy.HibernateLazyInitializer.Identifier;
1161  }
1162  }
1163  }
1164  }
1165  }
1166 
1167  return null;

1397  {
1398  log.Debug("Deserialization callback persistent-context");
1399 
1400  // during deserialization, we need to reconnect all proxies and
1401  // collections to this session, as well as the EntityEntry and
1402  // CollectionEntry instances; these associations are transient
1403  // because serialization is used for different things.
1404  parentsByChild = IdentityMap.Instantiate(InitCollectionSize);
1405 
1406  // OnDeserialization() must be called manually on all Dictionaries and Hashtables,
1407  // otherwise they are still empty at this point (the .NET deserialization code calls
1408  // OnDeserialization() on them AFTER it calls the current method).
1409  entitiesByKey.OnDeserialization(sender);
1410  entitiesByUniqueKey.OnDeserialization(sender);
1411  ((IDeserializationCallback)entityEntries).OnDeserialization(sender);
1412  proxiesByKey.OnDeserialization(sender);
1413  entitySnapshotsByKey.OnDeserialization(sender);
1414  ((IDeserializationCallback)arrayHolders).OnDeserialization(sender);
1415  ((IDeserializationCallback)collectionEntries).OnDeserialization(sender);
1416  collectionsByKey.OnDeserialization(sender);
1417 
1418  // If nullifiableEntityKeys is once used in the current method, HashSets will need
1419  // an OnDeserialization() method.
1420  //nullifiableEntityKeys.OnDeserialization(sender);
1421 
1422  if (unownedCollections != null)
1423  {
1424  unownedCollections.OnDeserialization(sender);
1425  }
1426 
1427  // TODO NH: "reconnect" EntityKey with session.factory and create a test for serialization of StatefulPersistenceContext
1428  foreach (DictionaryEntry collectionEntry in collectionEntries)
1429  {
1430  try
1431  {
1432  ((IPersistentCollection)collectionEntry.Key).SetCurrentSession(session);
1433  CollectionEntry ce = (CollectionEntry)collectionEntry.Value;
1434  if (ce.Role != null)
1435  {
1436  ce.AfterDeserialize(Session.Factory);
1437  }
1438 
1439  }
1440  catch (HibernateException he)
1441  {
1442  throw new InvalidOperationException(he.Message);
1443  }
1444  }
1445 
1446  List<EntityKey> keysToRemove = new List<EntityKey>();
1447  foreach (KeyValuePair<EntityKey, INHibernateProxy> p in proxiesByKey)
1448  {
1449  if (p.Value != null)
1450  {
1451  (p.Value).HibernateLazyInitializer.Session = session;

1456  keysToRemove.Add(p.Key);

1460  proxiesByKey.Remove(keysToRemove[i]);
1461 
1462  foreach (EntityEntry e in entityEntries.Values)
1463  {
1464  try
1465  {
1466  e.Persister = session.Factory.GetEntityPersister(e.EntityName);
1467  }

1471  }
1472  }
