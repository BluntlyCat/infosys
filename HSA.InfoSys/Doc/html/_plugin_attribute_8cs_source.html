140  public IPlugin CreatePlugin()
141  {
142  Type pluginType = m_type;
143  if (m_type == null)
144  {
145  // Get the plugin object type from the string type name
146  pluginType = SystemInfo.GetTypeFromString(m_typeName, true, true);
147  }
148 
149  // Check that the type is a plugin
150  if (!(typeof(IPlugin).IsAssignableFrom(pluginType)))
151  {
152  throw new LogException("Plugin type [" + pluginType.FullName + "] does not implement the log4net.IPlugin interface");
153  }
154 
155  // Create an instance of the plugin using the default constructor
156  IPlugin plugin = (IPlugin)Activator.CreateInstance(pluginType);
157 
158  return plugin;
159  }
