143  private void SetNextCrawl()
144  {
145  lock (this.crawlLock)
146  {
147  try
148  {
149  if (!this.isCrawling)
150  {
151  var newSettings = this.dbManager.GetNutchClientSettings();
152 
153  if (newSettings.Equals(this.settings) == false)
154  {
155  Log.InfoFormat(
156  Properties.Resources.NUTCH_CONTROLLER_UPDATE_SETTINGS,
157  this.settings,
158  newSettings);
159 
160  this.settings = newSettings;
161  this.InitializeClients();
162  }
163 
164  if (this.settings.IsDefault() == false)
165  {
166  this.InitializeNextCrawl();
167 
168  foreach (var client in this.nutchClients)
169  {
170  client.InitializeClient(this.settings);
171 
172  if (client.IsClientUsable && client.URLs.Count > 0)
173  {
174  Log.DebugFormat(
175  Properties.Resources.NUTCH_CONTROLLER_SET_PENDING_CRAWL,
176  client.Hostname,
177  client.URLs.ElementsToString());
178 
179  var invokeCrawl = new InvokeCrawl(client.StartCrawl);
180 
181  var callback = new AsyncCallback(
182  c =>
183  {
184  this.ServiceMutex.WaitOne();
185 
186  if (c.IsCompleted)
187  {
188  this.runningCrawls--;
189  Log.InfoFormat(Properties.Resources.NUTCH_CONTROLLER_CRAWL_FINISHED, this.runningCrawls);
190 
191  if (this.runningCrawls == 0)
192  {
193  this.ResetCrawls();
194  this.isCrawling = false;
195 
196  Log.Info(Properties.Resources.NUTCH_CONTROLLER_CRAWLS_FINISHED);
197  }
198  }
199 
200  this.ServiceMutex.ReleaseMutex();
201  });
202 
203  this.runningCrawls++;
204  invokeCrawl.BeginInvoke(this.settings, this.dbManager, callback, this);
205  }
206  else if (client.URLs.Count == 0)
207  {
208  Log.WarnFormat(
209  Properties.Resources.NUTCH_CONTROLLER_DO_NOT_CRAWL_ON_HOST_NO_URLS,
210  client.Hostname,
211  client);
212  }
213  else
214  {
215  Log.WarnFormat(Properties.Resources.NUTCH_CONTROLLER_DO_NOT_CRAWL_ON_HOST, client);
216  }
217  }
218 
219  this.isCrawling = this.runningCrawls > 0;
220  }
221  else
222  {
224  Log.WarnFormat(Properties.Resources.NUTCH_CONTROLLER_NO_SETTINGS);
225  }
226  }
227  }
228  catch (NoNutchClientUsableException nuce)
229  {
230  Log.FatalFormat(Properties.Resources.NUTCH_CONTROLLER_NO_USABLE_CLIENT, nuce);
231  this.StopService(true);
232  }
233  }

319  private void InitializeNextCrawl()
320  {
321  this.ServiceMutex.WaitOne();
322 
323  Log.Info(Properties.Resources.NUTCH_CONTROLLER_INIT_NEXT_CRAWL);
324 
325  int index = 0;
326 
327  foreach (var client in this.nutchClients)
328  {
329  Log.Info(Properties.Resources.NUTCH_CONTROLLER_CHECK_CLIENT_USAGE);
330  client.CheckClientForUsage(this.settings);
331  }
332 
333  this.URLs = this.dbManager.GetAllUrls();
334 
335  foreach (var url in this.URLs)
336  {
337  bool urlAdded = false;
338 
339  while (!urlAdded)
340  {
341  var client = this.nutchClients[index % this.nutchClients.Count];
342 
343  if (client.IsClientUsable)
344  {
345  client.URLs.Add(url);
346  urlAdded = true;
347  Log.InfoFormat(Properties.Resources.NUTCH_CONTROLLER_ADD_URL, url, client.Hostname);
348  }
349  else
350  {
351  Log.WarnFormat(Properties.Resources.NUTCH_CONTROLLER_CLIENT_NOT_USABLE, client.Hostname);
352  }
353 
354  index++;
355 
356  if (index % this.nutchClients.Count == 0 && !urlAdded)
357  {
358  throw new NoNutchClientUsableException(this.GetType().Name);
359  }
360  }
361  }
362 
363  foreach (var client in this.nutchClients)
364  {
365  if (client.IsClientUsable && client.URLs.Count > 0)
366  {
367  Log.InfoFormat(Properties.Resources.NUTCH_CONTROLLER_SET_CRAWL_PROCESS, client.Hostname);
368  client.SetCrawlProcess(this.settings);
369  }
370  }
371 
372  this.ServiceMutex.ReleaseMutex();
373  }
