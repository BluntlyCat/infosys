69  public static void WriteEscapedXmlString(XmlWriter writer, string textData, string invalidCharReplacement)
70  {
71  string stringData = MaskXmlInvalidCharacters(textData, invalidCharReplacement);
72  // Write either escaped text or CDATA sections
73 
74  int weightCData = 12 * (1 + CountSubstrings(stringData, CDATA_END));
75  int weightStringEscapes = 3*(CountSubstrings(stringData, "<") + CountSubstrings(stringData, ">")) + 4*CountSubstrings(stringData, "&");
76 
77  if (weightStringEscapes <= weightCData)
78  {
79  // Write string using string escapes
80  writer.WriteString(stringData);
81  }
82  else
83  {
84  // Write string using CDATA section
85 
86  int end = stringData.IndexOf(CDATA_END);
87 
88  if (end < 0)
89  {
90  writer.WriteCData(stringData);
91  }
92  else
93  {
94  int start = 0;
95  while (end > -1)
96  {
97  writer.WriteCData(stringData.Substring(start, end - start));
98  if (end == stringData.Length - 3)
99  {
100  start = stringData.Length;
101  writer.WriteString(CDATA_END);
102  break;
103  }
104  else
105  {
106  writer.WriteString(CDATA_UNESCAPABLE_TOKEN);
107  start = end + 2;
108  end = stringData.IndexOf(CDATA_END, start);
109  }
110  }
111 
112  if (start < stringData.Length)
113  {
114  writer.WriteCData(stringData.Substring(start));
115  }
116  }
117  }
118  }

156  private static int CountSubstrings(string text, string substring)
157  {
158  int count = 0;
159  int offset = 0;
160  int length = text.Length;
161  int substringLength = substring.Length;
162 
163  if (length == 0)
164  {
165  return 0;
166  }
167  if (substringLength == 0)
168  {
169  return 0;
170  }
171 
172  while(offset < length)
173  {
174  int index = text.IndexOf(substring, offset);
175 
176  if (index == -1)
177  {
178  break;
179  }
180 
181  count++;
182  offset = index + substringLength;
183  }
184  return count;
185  }
