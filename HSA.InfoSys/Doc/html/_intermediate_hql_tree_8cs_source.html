14  /* NOTE:
15  * Because common understanding of our users, we are flatting the behavior of Skip and Take methods.
16  * In RAM a query like primeNumbers.Skip(2).Take(6).Where(x=> x > 10) has a completely different results than primeNumbers.Where(x=> x > 10).Skip(2).Take(6) that has
17  * different results than primeNumbers.Take(6).Where(x=> x > 10).Skip(2) and so on.
18  * We are flatting/override even the double-usage of Skip and Take in the same query as: primeNumbers.Skip(2).Where(x=> x > 10).Take(6).Skip(3)
19  * We ***shouldn't*** change the behavior of the query just because we are translating it in SQL.
20  */

128  private void ExecuteAddTakeClause(HqlExpression toTake)
129  {
130  if (toTake == null)
131  {
132  return;
133  }
134 
135  var hqlQuery = _root.NodesPreOrder.OfType<HqlQuery>().First();
136  var takeRoot = hqlQuery.Children.OfType<HqlTake>().FirstOrDefault();
137 
138  // were present we ignore the new value
139  if (takeRoot == null)
140  {
141  //We should check the value instead delegate the behavior to the result SQL-> MSDN: If count is less than or equal to zero, source is not enumerated and an empty IEnumerable<T> is returned.
142  takeRoot = TreeBuilder.Take(toTake);
143  hqlQuery.AddChild(takeRoot);
144  }
145  }

147  private void ExecuteAddSkipClause(HqlExpression toSkip)
148  {
149  if (toSkip == null)
150  {
151  return;
152  }
153  // We should check the value instead delegate the behavior to the result SQL-> MSDN: If count is less than or equal to zero, all elements of source are yielded.
154  var hqlQuery = _root.NodesPreOrder.OfType<HqlQuery>().First();
155  var skipRoot = hqlQuery.Children.OfType<HqlSkip>().FirstOrDefault();
156  if (skipRoot == null)
157  {
158  skipRoot = TreeBuilder.Skip(toSkip);
159  hqlQuery.AddChild(skipRoot);
160  }

172  public void AddWhereClause(HqlBooleanExpression where)
173  {
174  var currentWhere = _root.NodesPreOrder.OfType<HqlWhere>().FirstOrDefault();
175  if (currentWhere == null)

177  currentWhere = TreeBuilder.Where(where);
