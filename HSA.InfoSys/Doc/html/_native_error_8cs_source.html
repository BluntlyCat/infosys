163  [System.Security.Permissions.SecurityPermission(System.Security.Permissions.SecurityAction.Demand, UnmanagedCode = true)]
164 #endif
165  public static string GetErrorMessage(int messageId)
166  {
167  // Win32 constants
168  int FORMAT_MESSAGE_ALLOCATE_BUFFER = 0x00000100; // The function should allocates a buffer large enough to hold the formatted message
169  int FORMAT_MESSAGE_IGNORE_INSERTS = 0x00000200; // Insert sequences in the message definition are to be ignored
170  int FORMAT_MESSAGE_FROM_SYSTEM = 0x00001000; // The function should search the system message-table resource(s) for the requested message
171 
172  string msgBuf = ""; // buffer that will receive the message
173  IntPtr sourcePtr = new IntPtr(); // Location of the message definition, will be ignored
174  IntPtr argumentsPtr = new IntPtr(); // Pointer to array of values to insert, not supported as it requires unsafe code
175 
176  if (messageId != 0)
177  {
178  // If the function succeeds, the return value is the number of TCHARs stored in the output buffer, excluding the terminating null character
179  int messageSize = FormatMessage(
180  FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,
181  ref sourcePtr,
182  messageId,
183  0,
184  ref msgBuf,
185  255,
186  argumentsPtr);
187 
188  if (messageSize > 0)
189  {
190  // Remove trailing null-terminating characters (\r\n) from the message
191  msgBuf = msgBuf.TrimEnd(new char[] {'\r', '\n'});
192  }
193  else
194  {
195  // A message could not be located.
196  msgBuf = null;
197  }
198  }
199  else
200  {
201  msgBuf = null;
202  }
203 
204  return msgBuf;
205  }
