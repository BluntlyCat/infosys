50  public MarshalByRefLogImpl(ILogger logger)
51  {
52  m_logger = logger;
53 
54  // Listen for changes to the repository
55  logger.Repository.ConfigurationChanged += new LoggerRepositoryConfigurationChangedEventHandler(LoggerRepositoryConfigurationChanged);
56 
57  // load the current levels
58  ReloadLevels(logger.Repository);
59  }
60 
61  #endregion Public Instance Constructors
62 
63  private void ReloadLevels(ILoggerRepository repository)
64  {
65  LevelMap levelMap = repository.LevelMap;
66 
67  m_levelDebug = levelMap.LookupWithDefault(Level.Debug);
68  m_levelInfo = levelMap.LookupWithDefault(Level.Info);
69  m_levelWarn = levelMap.LookupWithDefault(Level.Warn);
70  m_levelError = levelMap.LookupWithDefault(Level.Error);
71  m_levelFatal = levelMap.LookupWithDefault(Level.Fatal);
72  }
73 
74  private void LoggerRepositoryConfigurationChanged(object sender, EventArgs e)
75  {
76  ILoggerRepository repository = sender as ILoggerRepository;
77  if (repository != null)
78  {
79  ReloadLevels(repository);
80  }
81  }
