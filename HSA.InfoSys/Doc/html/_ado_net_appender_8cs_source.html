452  override protected void SendBuffer(LoggingEvent[] events)
453  {
454  if (m_reconnectOnError && (m_dbConnection == null || m_dbConnection.State != ConnectionState.Open))
455  {
456  LogLog.Debug(declaringType, "Attempting to reconnect to database. Current Connection State: " + ((m_dbConnection==null)?SystemInfo.NullText:m_dbConnection.State.ToString()) );
457 
458  InitializeDatabaseConnection();
459  InitializeDatabaseCommand();
460  }
461 
462  // Check that the connection exists and is open
463  if (m_dbConnection != null && m_dbConnection.State == ConnectionState.Open)
464  {
465  if (m_useTransactions)
466  {
467  // Create transaction
468  // NJC - Do this on 2 lines because it can confuse the debugger
469  IDbTransaction dbTran = null;
470  try
471  {
472  dbTran = m_dbConnection.BeginTransaction();
473 
474  SendBuffer(dbTran, events);
475 
476  // commit transaction
477  dbTran.Commit();
478  }
479  catch(Exception ex)
480  {
481  // rollback the transaction
482  if (dbTran != null)
483  {
484  try
485  {
486  dbTran.Rollback();
487  }
488  catch(Exception)
489  {
490  // Ignore exception
491  }
492  }
493 
494  // Can't insert into the database. That's a bad thing
495  ErrorHandler.Error("Exception while writing to database", ex);
496  }
497  }
498  else
499  {
500  // Send without transaction
501  SendBuffer(null, events);
502  }
503  }
504  }

541  virtual protected void SendBuffer(IDbTransaction dbTran, LoggingEvent[] events)
542  {
543  if (m_usePreparedCommand)
544  {
545  // Send buffer using the prepared command object
546 
547  if (m_dbCommand != null)
548  {
549  if (dbTran != null)
550  {
551  m_dbCommand.Transaction = dbTran;
552  }
553 
554  // run for all events
555  foreach(LoggingEvent e in events)
556  {
557  // Set the parameter values
558  foreach(AdoNetAppenderParameter param in m_parameters)
559  {
560  param.FormatValue(m_dbCommand, e);
561  }
562 
563  // Execute the query
564  m_dbCommand.ExecuteNonQuery();
565  }
566  }
567  }
568  else
569  {
570  // create a new command
571  using(IDbCommand dbCmd = m_dbConnection.CreateCommand())
572  {
573  if (dbTran != null)
574  {
575  dbCmd.Transaction = dbTran;
576  }
577 
578  // run for all events
579  foreach(LoggingEvent e in events)
580  {
581  // Get the command text from the Layout
582  string logStatement = GetLogStatement(e);
583 
584  LogLog.Debug(declaringType, "LogStatement ["+logStatement+"]");
585 
586  dbCmd.CommandText = logStatement;
587  dbCmd.ExecuteNonQuery();
588  }
589  }
590  }

644  virtual protected string ResolveConnectionString(out string connectionStringContext)
645  {
646  if (m_connectionString != null && m_connectionString.Length > 0)

648  connectionStringContext = "ConnectionString";

652 #if NET_2_0

664  }
665  }

668  if (m_appSettingsKey != null && m_appSettingsKey.Length > 0)
669  {
670  connectionStringContext = "AppSettingsKey";
671  string appSettingsConnectionString = SystemInfo.GetAppSetting(m_appSettingsKey);
672  if (appSettingsConnectionString == null || appSettingsConnectionString.Length == 0)

679  connectionStringContext = "Unable to resolve connection string from ConnectionString, ConnectionStrings, or AppSettings.";

718  private void InitializeDatabaseCommand()

735  catch (Exception e)

738 
739  DisposeCommand(true);

741

785  private void InitializeDatabaseConnection()
786  {
787  string connectionStringContext = "Unable to determine connection string context.";
788  string resolvedConnectionString = string.Empty;
789 
790  try

798  m_dbConnection = CreateConnection(ResolveConnectionType(), resolvedConnectionString);
799

802  // Open the database connection
803  m_dbConnection.Open();
804  }
805  }
806  catch (Exception e)
807  {
808  // Sadly, your connection string is bad.
809  ErrorHandler.Error("Could not open database connection [" + resolvedConnectionString + "]. Connection string context [" + connectionStringContext + "].", e);
810 
811  m_dbConnection = null;
812  }
