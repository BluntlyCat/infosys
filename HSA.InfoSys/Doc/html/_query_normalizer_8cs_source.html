138  private void CollapseValueLists(List<Token> tok)
139  {
140  int pos = -1;

142  {

150  private void CollapseValueList(List<Token> tok, ref int pos)
151  {
152  List<int> parenIndices = new List<int>();
153 
154  // this while loop will find all closing parens in this value list
155  while (true)
156  {
157  // find the close ')'
158  while (++pos < tok.Count)
159  {
160  if (tok[pos].Type == TokenType.Symbol && tok[pos].Text == ")")
161  break;
162  if (pos == tok.Count - 1)
163  break;
164  }
165  parenIndices.Add(pos);
166 
167  // now find the next "real" token
168  while (++pos < tok.Count)
169  if (tok[pos].IsRealToken) break;
170  if (pos == tok.Count) break;
171 
172  if (tok[pos].Text != ",")
173  {
174  pos--;
175  break;
176  }
177  }
178 
179  // if we only have 1 value then we don't collapse
180  if (parenIndices.Count < 2) return;
181  int index = parenIndices[0];
182  tok[++index] = new Token(TokenType.Whitespace, " ");
183  tok[++index] = new Token(TokenType.Comment, "/* , ... */");
184  index++;
185 
186  // now mark all the other tokens as no output
187  while (index <= parenIndices[parenIndices.Count - 1])
188  tok[index++].Output = false;
189  }

212  private void CollapseInList(List<Token> tok, ref int pos)
213  {
214  Token t = GetNextRealToken(tok, ref pos);
215  // Debug.Assert(t.Text == "(");
216  if (t == null)
217  return;
218 
219  // if the first token is a keyword then we likely have a
220  // SELECT .. IN (SELECT ...)
221  t = GetNextRealToken(tok, ref pos);
222  if (t == null || t.Type == TokenType.Keyword) return;
223 
224  int start = pos;
225  // first find all the tokens that make up the in list
226  while (++pos < tok.Count)
227  {
228  t = tok[pos];
229  if (t.Type == TokenType.CommandComment) return;
230  if (!t.IsRealToken) continue;
231  if (t.Text == "(") return;
232  if (t.Text == ")") break;
233  }
234  int stop = pos;
235 
236  for (int i = stop; i > start; i--)
237  tok.RemoveAt(i);
238  tok.Insert(++start, new Token(TokenType.Whitespace, " "));
239  tok.Insert(++start, new Token(TokenType.Comment, "/* , ... */"));
240  tok.Insert(++start, new Token(TokenType.Whitespace, " "));
241  tok.Insert(++start, new Token(TokenType.Symbol, ")"));
242  }

244  private void TokenizeSql(string sql)
245  {
246  pos = 0;
247 
248  while (pos < sql.Length)
249  {
250  char c = sql[pos];
251  if (LetterStartsComment(c) && ConsumeComment())
252  continue;
253  if (Char.IsWhiteSpace(c))
254  ConsumeWhitespace();
255  else if (c == '\'' || c == '\"' || c == '`')
256  ConsumeQuotedToken(c);
257  else if (!IsSpecialCharacter(c))
258  ConsumeUnquotedToken();
259  else
260  ConsumeSymbol();
261  }
262  }

276  private bool ConsumeComment()
277  {
278  char c = fullSql[pos];
279  // make sure the comment starts correctly
280  if (c == '/' && ((pos + 1) >= fullSql.Length || fullSql[pos + 1] != '*')) return false;
281  if (c == '-' && ((pos + 2) >= fullSql.Length || fullSql[pos + 1] != '-' || fullSql[pos + 2] != ' ')) return false;
282 
283  string endingPattern = "\n";
284  if (c == '/')
285  endingPattern = "*/";
286 
287  int startingIndex = pos;
288 
289  int index = fullSql.IndexOf(endingPattern, pos);
290  if (index == -1)
291  index = fullSql.Length - 1;
292  else
293  index += endingPattern.Length;
294  string comment = fullSql.Substring(pos, index - pos);
295  if (comment.StartsWith("/*!", StringComparison.Ordinal))
296  tokens.Add(new Token(TokenType.CommandComment, comment));
297  pos = index;
298  return true;
299  }

307  private void ConsumeQuotedToken(char c)
308  {
309  bool escaped = false;
310  int start = pos;
311  pos++;
312  while (pos < fullSql.Length)
313  {
314  char x = fullSql[pos];
315 
316  if (x == c && !escaped) break;
317 
318  if (escaped)
319  escaped = false;
320  else if (x == '\\')
321  escaped = true;
322  pos++;
323  }
324  pos++;
325  if (c == '\'')
326  tokens.Add(new Token(TokenType.String, "?"));
327  else
328  tokens.Add(new Token(TokenType.Identifier, fullSql.Substring(start, pos - start)));
329  }
330 
331  private void ConsumeUnquotedToken()
332  {
333  int startPos = pos;
334  while (pos < fullSql.Length && !IsSpecialCharacter(fullSql[pos]))
335  pos++;
336  string word = fullSql.Substring(startPos, pos - startPos);
337  double v;
338  if (Double.TryParse(word, out v))
339  tokens.Add(new Token(TokenType.Number, "?"));
340  else
341  {
342  Token t = new Token(TokenType.Identifier, word);
343  if (IsKeyword(word))
344  {
345  t.Type = TokenType.Keyword;
346  t.Text = t.Text.ToUpperInvariant();
347  }
348  tokens.Add(t);
349  }
350  }

353  {
354  tokens.Add(new Token(TokenType.Whitespace, " "));
355  while (pos < fullSql.Length && Char.IsWhiteSpace(fullSql[pos]))
