183  // Resolve Session ID
184  var sessionId = this.Session.SessionId ?? this.ExchangeHash;
185 
186  // Calculate server to client initial IV
187  var serverVector = this.Hash(this.GenerateSessionKey(this.SharedKey, this.ExchangeHash, 'B', sessionId));
