43  // largest prime smaller than 65536
44  private const int BASE = 65521;
45  // NMAX is the largest n such that 255n(n+1)/2 + (n+1)(BASE-1) <= 2^32-1
46  private const int NMAX = 5552;
47 
48  internal static long adler32(long adler, byte[] buf, int index, int len)
49  {
50  if (buf == null) { return 1L; }
51 
52  long s1 = adler & 0xffff;
53  long s2 = (adler >> 16) & 0xffff;
54  int k;
55 
56  while (len > 0)
57  {
58  k = len < NMAX ? len : NMAX;
59  len -= k;
60  while (k >= 16)
61  {
62  s1 += buf[index++] & 0xff; s2 += s1;
63  s1 += buf[index++] & 0xff; s2 += s1;
64  s1 += buf[index++] & 0xff; s2 += s1;

73  s1 += buf[index++] & 0xff; s2 += s1;
