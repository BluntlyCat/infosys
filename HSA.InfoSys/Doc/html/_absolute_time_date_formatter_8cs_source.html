101  virtual public void FormatDate(DateTime dateToFormat, TextWriter writer)
102  {
103  // Calculate the current time precise only to the second
104  long currentTimeToTheSecond = (dateToFormat.Ticks - (dateToFormat.Ticks % TimeSpan.TicksPerSecond));
105 
106  // Compare this time with the stored last time
107  // If we are in the same second then append
108  // the previously calculated time string
109  if (s_lastTimeToTheSecond != currentTimeToTheSecond)
110  {
111  // lock so that only one thread can use the buffer and
112  // update the s_lastTimeToTheSecond and s_lastTimeString
113 
114  // PERF: Try removing this lock and using a new StringBuilder each time
115  lock(s_lastTimeBuf)
116  {
117  if (s_lastTimeToTheSecond != currentTimeToTheSecond)
118  {
119  // We are in a new second.
120  s_lastTimeBuf.Length = 0;
121 
122  // Calculate the new string for this second
123  FormatDateWithoutMillis(dateToFormat, s_lastTimeBuf);
124 
125  // Render the string buffer to a string
126  string currentDateWithoutMillis = s_lastTimeBuf.ToString();
127 
128 #if NET_1_1
129  // Ensure that the above string is written into the variable NOW on all threads.
130  // This is only required on multiprocessor machines with weak memeory models
131  System.Threading.Thread.MemoryBarrier();
132 #endif
133  // Store the time as a string (we only have to do this once per second)
134  s_lastTimeString = currentDateWithoutMillis;
135  s_lastTimeToTheSecond = currentTimeToTheSecond;
136  }
137  }
138  }
139  writer.Write(s_lastTimeString);
140 
141  // Append the current millisecond info
142  writer.Write(',');
143  int millis = dateToFormat.Millisecond;
144  if (millis < 100)
145  {
146  writer.Write('0');
147  }
148  if (millis < 10)
149  {
150  writer.Write('0');
151  }
152  writer.Write(millis);
153  }
