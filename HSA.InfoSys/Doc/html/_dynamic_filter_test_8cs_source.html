21  public void SecondLevelCachedCollectionsFiltering()
22  {
23  TestData testData = new TestData(this);
24  testData.Prepare();
25 
26  ISession session = OpenSession();
27 
28  // Force a collection into the second level cache, with its non-filtered elements
29  Salesperson sp = (Salesperson) session.Load(typeof(Salesperson), testData.steveId);
30  NHibernateUtil.Initialize(sp.Orders);
31  ICollectionPersister persister = ((ISessionFactoryImplementor) sessions)
32  .GetCollectionPersister(typeof(Salesperson).FullName + ".Orders");
33  Assert.IsTrue(persister.HasCache, "No cache for collection");
34  CacheKey cacheKey =
35  new CacheKey(testData.steveId, persister.KeyType, persister.Role, EntityMode.Poco, (ISessionFactoryImplementor) sessions);
36  CollectionCacheEntry cachedData = (CollectionCacheEntry)persister.Cache.Cache.Get(cacheKey);
37  Assert.IsNotNull(cachedData, "collection was not in cache");
38 
39  session.Close();
40 
41  session = OpenSession();
42  session.EnableFilter("fulfilledOrders").SetParameter("asOfDate", testData.lastMonth);
43  sp = (Salesperson) session.CreateQuery("from Salesperson as s where s.id = :id")
44  .SetInt64("id", testData.steveId)
45  .UniqueResult();
46  Assert.AreEqual(1, sp.Orders.Count, "Filtered-collection not bypassing 2L-cache");
47 
48  CollectionCacheEntry cachedData2 = (CollectionCacheEntry)persister.Cache.Cache.Get(cacheKey);
49  Assert.IsNotNull(cachedData2, "collection no longer in cache!");
50  Assert.AreSame(cachedData, cachedData2, "Different cache values!");
51 
52  session.Close();
53 
54  session = OpenSession();
55  session.EnableFilter("fulfilledOrders").SetParameter("asOfDate", testData.lastMonth);
56  sp = (Salesperson) session.Load(typeof(Salesperson), testData.steveId);
57  Assert.AreEqual(1, sp.Orders.Count, "Filtered-collection not bypassing 2L-cache");
58 
59  session.Close();
60 
61  // Finally, make sure that the original cached version did not get over-written
62  session = OpenSession();
63  sp = (Salesperson) session.Load(typeof(Salesperson), testData.steveId);
64  Assert.AreEqual(2, sp.Orders.Count, "Actual cached version got over-written");
65 
66  session.Close();
67  testData.Release();
68  }
