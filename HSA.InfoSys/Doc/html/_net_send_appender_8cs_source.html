310  [System.Security.Permissions.SecurityPermission(System.Security.Permissions.SecurityAction.Demand, UnmanagedCode = true)]
311  protected override void Append(LoggingEvent loggingEvent)
312  {
313  NativeError nativeError = null;
314 
315  // Render the event in the callers security context
316  string renderedLoggingEvent = RenderLoggingEvent(loggingEvent);
317 
318  using(m_securityContext.Impersonate(this))
319  {
320  // Send the message
321  int returnValue = NetMessageBufferSend(this.Server, this.Recipient, this.Sender, renderedLoggingEvent, renderedLoggingEvent.Length * Marshal.SystemDefaultCharSize);
322 
323  // Log the error if the message could not be sent
324  if (returnValue != 0)
325  {
326  // Lookup the native error
327  nativeError = NativeError.GetError(returnValue);
328  }
329  }
330 
331  if (nativeError != null)
332  {
333  // Handle the error over to the ErrorHandler
334  ErrorHandler.Error(nativeError.ToString() + " (Params: Server=" + this.Server + ", Recipient=" + this.Recipient + ", Sender=" + this.Sender + ")");
335  }
336  }
