420  protected void SetQueryProperties(IQuery q)
421  {
422  q.SetMaxResults(selection.MaxRows)
423  .SetFirstResult(selection.FirstRow)
424  .SetCacheable(cacheable)
425  .SetReadOnly(readOnly)
426  .SetTimeout(selection.Timeout)
427  .SetFlushMode(flushMode)
428  .SetFetchSize(selection.FetchSize);
429  if (!string.IsNullOrEmpty(comment))
430  q.SetComment(comment);
431  if (!string.IsNullOrEmpty(cacheRegion))
432  q.SetCacheRegion(cacheRegion);
433  if (resultTransformer != null)
434  q.SetResultTransformer(resultTransformer);
435  if (cacheMode.HasValue)
436  q.SetCacheMode(cacheMode.Value);
437  foreach (KeyValuePair<string, LockMode> mode in lockModes)
438  q.SetLockMode(mode.Key, mode.Value);
439 
440  // Set AbstractQueryImpl property before set parameters
441  var aqi = q as AbstractQueryImpl;
442  if (aqi != null)
443  aqi.SetIgnoreUknownNamedParameters(shouldIgnoredUnknownNamedParameters);
444 
445  // Even if the probably that somebody use a mixed technique to set parameters
446  // (from POCO using SetProperties and using named parameter setters) here is a possible
447  // difference between IQuery and DetachedQuery behaviour.
448  // In IQuery we don't know who override a param value; in DetachedQuery the direct use of
449  // a named parameter setter override the param value set by SetProperties(POCO)
450  foreach (var obj in optionalUntypeParams)
451  q.SetProperties(obj);
452 
453  // Set untyped positional parameters
454  foreach (var pup in posUntypeParams)
455  q.SetParameter(pup.Key, pup.Value);
456 
457  // Set untyped named parameters
458  foreach (var nup in namedUntypeParams)
459  q.SetParameter(nup.Key, nup.Value);
460 
461  // Set untyped named parameters list
462  foreach (var nulp in namedUntypeListParams)
463  q.SetParameterList(nulp.Key, nulp.Value);
464 
465  // Set typed positional parameters
466  foreach (var pp in posParams)
467  q.SetParameter(pp.Key, pp.Value.Value, pp.Value.Type);
468 
469  // Set typed named parameters
470  foreach (var np in namedParams)
471  q.SetParameter(np.Key, np.Value.Value, np.Value.Type);
472 
473  // Set typed named parameters List
474  foreach (var nlp in namedListParams)
475  q.SetParameterList(nlp.Key, (IEnumerable) nlp.Value.Value, nlp.Value.Type);
476  }
