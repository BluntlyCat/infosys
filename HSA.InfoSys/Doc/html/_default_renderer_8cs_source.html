148  public void RenderObject(RendererMap rendererMap, object obj, TextWriter writer)
149  {
150  if (rendererMap == null)
151  {
152  throw new ArgumentNullException("rendererMap");
153  }
154 
155  if (obj == null)
156  {
157  writer.Write(SystemInfo.NullText);
158  return;
159  }
160 
161  Array objArray = obj as Array;
162  if (objArray != null)
163  {
164  RenderArray(rendererMap, objArray, writer);
165  return;
166  }
167 
168  // Test if we are dealing with some form of collection object
169  IEnumerable objEnumerable = obj as IEnumerable;
170  if (objEnumerable != null)
171  {
172  // Get a collection interface if we can as its .Count property may be more
173  // performant than getting the IEnumerator object and trying to advance it.
174  ICollection objCollection = obj as ICollection;
175  if (objCollection != null && objCollection.Count == 0)
176  {
177  writer.Write("{}");
178  return;
179  }
180 
181  // This is a special check to allow us to get the enumerator from the IDictionary
182  // interface as this guarantees us DictionaryEntry objects. Note that in .NET 2.0
183  // the generic IDictionary<> interface enumerates KeyValuePair objects rather than
184  // DictionaryEntry ones. However the implementation of the plain IDictionary
185  // interface on the generic Dictionary<> still returns DictionaryEntry objects.
186  IDictionary objDictionary = obj as IDictionary;
187  if (objDictionary != null)
188  {
189  RenderEnumerator(rendererMap, objDictionary.GetEnumerator(), writer);
190  return;
191  }
192 
193  RenderEnumerator(rendererMap, objEnumerable.GetEnumerator(), writer);
194  return;
195  }
196 
197  IEnumerator objEnumerator = obj as IEnumerator;
198  if (objEnumerator != null)
199  {
200  RenderEnumerator(rendererMap, objEnumerator, writer);
201  return;
202  }
203 
204  if (obj is DictionaryEntry)
205  {
206  RenderDictionaryEntry(rendererMap, (DictionaryEntry)obj, writer);
207  return;
208  }
209 
210  string str = obj.ToString();
211  writer.Write( (str==null) ? SystemInfo.NullText : str );
212  }

235  private void RenderArray(RendererMap rendererMap, Array array, TextWriter writer)
236  {
237  if (array.Rank != 1)
238  {
239  writer.Write(array.ToString());
240  }
241  else
242  {
243  writer.Write(array.GetType().Name + " {");
244  int len = array.Length;
245 
246  if (len > 0)
247  {
248  rendererMap.FindAndRender(array.GetValue(0), writer);
249  for(int i=1; i<len; i++)
250  {
251  writer.Write(", ");
252  rendererMap.FindAndRender(array.GetValue(i), writer);
253  }
254  }
255  writer.Write("}");
256  }
257  }

273  private void RenderEnumerator(RendererMap rendererMap, IEnumerator enumerator, TextWriter writer)
274  {
275  writer.Write("{");
276 
277  if (enumerator != null && enumerator.MoveNext())
278  {
279  rendererMap.FindAndRender(enumerator.Current, writer);
280 
281  while (enumerator.MoveNext())
282  {
283  writer.Write(", ");
284  rendererMap.FindAndRender(enumerator.Current, writer);
285  }
286  }
287 
288  writer.Write("}");
289  }
