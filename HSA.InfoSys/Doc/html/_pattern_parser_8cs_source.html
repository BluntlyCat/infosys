189  private void ParseInternal(string pattern, string[] matches)
190  {
191  int offset = 0;
192  while(offset < pattern.Length)
193  {
194  int i = pattern.IndexOf('%', offset);
195  if (i < 0 || i == pattern.Length - 1)
196  {
197  ProcessLiteral(pattern.Substring(offset));
198  offset = pattern.Length;
199  }
200  else
201  {
202  if (pattern[i+1] == '%')
203  {
204  // Escaped
205  ProcessLiteral(pattern.Substring(offset, i - offset + 1));
206  offset = i + 2;
207  }
208  else
209  {
210  ProcessLiteral(pattern.Substring(offset, i - offset));
211  offset = i + 1;
212 
213  FormattingInfo formattingInfo = new FormattingInfo();
214 
215  // Process formatting options
216 
217  // Look for the align flag
218  if (offset < pattern.Length)
219  {
220  if (pattern[offset] == '-')
221  {
222  // Seen align flag
223  formattingInfo.LeftAlign = true;
224  offset++;
225  }
226  }
227  // Look for the minimum length
228  while (offset < pattern.Length && char.IsDigit(pattern[offset]))
229  {
230  // Seen digit
231  if (formattingInfo.Min < 0)
232  {
233  formattingInfo.Min = 0;
234  }
235  formattingInfo.Min = (formattingInfo.Min * 10) + int.Parse(pattern[offset].ToString(CultureInfo.InvariantCulture), System.Globalization.NumberFormatInfo.InvariantInfo);
236  offset++;
237  }
238  // Look for the separator between min and max
239  if (offset < pattern.Length)
240  {
241  if (pattern[offset] == '.')
242  {
243  // Seen separator
244  offset++;
245  }
246  }
247  // Look for the maximum length
248  while (offset < pattern.Length && char.IsDigit(pattern[offset]))
249  {
250  // Seen digit
251  if (formattingInfo.Max == int.MaxValue)
252  {
253  formattingInfo.Max = 0;
254  }
255  formattingInfo.Max = (formattingInfo.Max * 10) + int.Parse(pattern[offset].ToString(CultureInfo.InvariantCulture), System.Globalization.NumberFormatInfo.InvariantInfo);
256  offset++;
257  }
258 
259  int remainingStringLength = pattern.Length - offset;
260 
261  // Look for pattern
262  for(int m=0; m<matches.Length; m++)
263  {
264  if (matches[m].Length <= remainingStringLength)
265  {
266  if (String.Compare(pattern, offset, matches[m], 0, matches[m].Length, false, System.Globalization.CultureInfo.InvariantCulture) == 0)
267  {
268  // Found match
269  offset = offset + matches[m].Length;
270 
271  string option = null;
272 
273  // Look for option
274  if (offset < pattern.Length)
275  {
276  if (pattern[offset] == '{')
277  {
278  // Seen option start
279  offset++;
280 
281  int optEnd = pattern.IndexOf('}', offset);
282  if (optEnd < 0)
283  {
284  // error
285  }
286  else
287  {
288  option = pattern.Substring(offset, optEnd - offset);
289  offset = optEnd + 1;
290  }
291  }
292  }
293 
294  ProcessConverter(matches[m], option, formattingInfo);
295  break;
296  }
297  }
298  }
299  }
300  }
301  }
302  }

323  private void ProcessConverter(string converterName, string option, FormattingInfo formattingInfo)
324  {
325  LogLog.Debug(declaringType, "Converter ["+converterName+"] Option ["+option+"] Format [min="+formattingInfo.Min+",max="+formattingInfo.Max+",leftAlign="+formattingInfo.LeftAlign+"]");
326 
327  // Lookup the converter type
328  ConverterInfo converterInfo = (ConverterInfo)m_patternConverters[converterName];
329  if (converterInfo == null)
330  {
331  LogLog.Error(declaringType, "Unknown converter name ["+converterName+"] in conversion pattern.");
332  }
333  else
334  {
335  // Create the pattern converter
336  PatternConverter pc = null;
337  try
338  {
339  pc = (PatternConverter)Activator.CreateInstance(converterInfo.Type);
340  }
341  catch(Exception createInstanceEx)
342  {
343  LogLog.Error(declaringType, "Failed to create instance of Type [" + converterInfo.Type.FullName + "] using default constructor. Exception: " + createInstanceEx.ToString());
344  }
345 
346  // formattingInfo variable is an instance variable, occasionally reset
347  // and used over and over again
348  pc.FormattingInfo = formattingInfo;
349  pc.Option = option;
350  pc.Properties = converterInfo.Properties;
351 
352  IOptionHandler optionHandler = pc as IOptionHandler;
353  if (optionHandler != null)
354  {
355  optionHandler.ActivateOptions();
356  }
357 
358  AddConverter(pc);
359  }
360  }

367  private void AddConverter(PatternConverter pc)
368  {
369  // Add the pattern converter to the list.
370 
371  if (m_head == null)
372  {
373  m_head = m_tail = pc;
374  }
375  else
376  {
377  // Set the next converter on the tail
378  // Update the tail reference
379  // note that a converter may combine the 'next' into itself
380  // and therefore the tail would not change!
381  m_tail = m_tail.SetNext(pc);
382  }
