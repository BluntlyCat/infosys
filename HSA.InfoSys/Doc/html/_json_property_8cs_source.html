64  private void CalculateSkipPropertyNameEscape()
65  {
66  if (_propertyName == null)
67  {
68  _skipPropertyNameEscape = false;
69  }
70  else
71  {
72  _skipPropertyNameEscape = true;
73  foreach (char c in _propertyName)
74  {
75  if (!char.IsLetterOrDigit(c) && c != '_' && c != '@')
76  {
77  _skipPropertyNameEscape = false;
78  break;
79  }
80  }
81  }
