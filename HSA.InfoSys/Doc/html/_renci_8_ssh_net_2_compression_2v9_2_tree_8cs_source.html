120  public void BuildTree(Deflate s)
121  {
122  short[] tree = _dynamicTree;
123  short[] stree = _staticTree.TreeData;
124  int elems = _staticTree.Elements;
125  int n, m; // iterate over heap elements
126  int max_code = -1; // largest code with non zero frequency
127  int node; // new node being created
128 
129  // Construct the initial heap, with least frequent element in
130  // heap[1]. The sons of heap[n] are heap[2*n] and heap[2*n+1].
131  // heap[0] is not used.
132  s.heap_len = 0;
133  s.heap_max = HEAP_SIZE;
134 
135  for (n = 0; n < elems; n++)
136  {
137  if (tree[n * 2] != 0)
138  {
139  s.heap[++s.heap_len] = max_code = n;
140  s.depth[n] = 0;
141  }
142  else
143  {
144  tree[n * 2 + 1] = 0;
145  }
146  }
147 
148  // The pkzip format requires that at least one distance code exists,
149  // and that at least one bit should be sent even if there is only one
150  // possible code. So to avoid special checks later on we force at least
151  // two codes of non zero frequency.
152  while (s.heap_len < 2)
153  {
154  node = s.heap[++s.heap_len] = (max_code < 2 ? ++max_code : 0);
155  tree[node * 2] = 1;
156  s.depth[node] = 0;
157  s.opt_len--; if (stree != null) s.static_len -= stree[node * 2 + 1];
158  // node is 0 or 1 so it does not have extra bits
159  }
160  this.LargestCode = max_code;
161 
162  // The elements heap[heap_len/2+1 .. heap_len] are leaves of the tree,
163  // establish sub-heaps of increasing lengths:
164 
165  for (n = s.heap_len / 2; n >= 1; n--)
166  s.pqdownheap(tree, n);
167 
168  // Construct the Huffman tree by repeatedly combining the least two
169  // frequent nodes.
170 
171  node = elems; // next internal node of the tree
172  do
173  {
174  // n = node of least frequency
175  n = s.heap[1];
176  s.heap[1] = s.heap[s.heap_len--];
177  s.pqdownheap(tree, 1);
178  m = s.heap[1]; // m = node of next least frequency
179 
180  s.heap[--s.heap_max] = n; // keep the nodes sorted by frequency
181  s.heap[--s.heap_max] = m;
182 
183  // Create a new node father of n and m
184  tree[node * 2] = (short)(tree[n * 2] + tree[m * 2]);
185  s.depth[node] = (byte)(System.Math.Max(s.depth[n], s.depth[m]) + 1);
186  tree[n * 2 + 1] = tree[m * 2 + 1] = (short)node;
187 
188  // and insert the new node in the heap
189  s.heap[1] = node++;
190  s.pqdownheap(tree, 1);
191  }
192  while (s.heap_len >= 2);
193 
194  s.heap[--s.heap_max] = s.heap[1];
195 
196  // At this point, the fields freq and dad are set. We can now
197  // generate the bit lengths.
198 
199  ComputeBitLength(s);
200 
201  // The field len is now set, we can generate the bit codes
202  GenerateCodes(tree, max_code, s.bl_count);
203  }

209  private void ComputeBitLength(Deflate s)
210  {
211  short[] tree = _dynamicTree;
212  short[] stree = _staticTree.TreeData;
213  int[] extra = _staticTree.ExtraBits;
214  int based = _staticTree.ExtraBase;
215  int max_length = _staticTree.MaxLength;
216  int h; // heap index
217  int n, m; // iterate over the tree elements
218  int bits; // bit length
219  int xbits; // extra bits
220  short f; // frequency
221  int overflow = 0; // number of elements with bit length too large
222 
223  for (bits = 0; bits <= MAX_BITS; bits++) s.bl_count[bits] = 0;
224 
225  // In a first pass, compute the optimal bit lengths (which may
226  // overflow in the case of the bit length tree).
227  tree[s.heap[s.heap_max] * 2 + 1] = 0; // root of the heap
228 
229  for (h = s.heap_max + 1; h < HEAP_SIZE; h++)
230  {
231  n = s.heap[h];
232  bits = tree[tree[n * 2 + 1] * 2 + 1] + 1;
233  if (bits > max_length) { bits = max_length; overflow++; }
234  tree[n * 2 + 1] = (short)bits;
235  // We overwrite tree[n*2+1] which is no longer needed
236 
237  if (n > LargestCode) continue; // not a leaf node
238 
239  s.bl_count[bits]++;
240  xbits = 0;
241  if (n >= based) xbits = extra[n - based];
242  f = tree[n * 2];
243  s.opt_len += f * (bits + xbits);
244  if (stree != null) s.static_len += f * (stree[n * 2 + 1] + xbits);
245  }
246  if (overflow == 0) return;
247 
248  // This happens for example on obj2 and pic of the Calgary corpus
249  // Find the first bit length which could increase:
250  do
251  {
252  bits = max_length - 1;
253  while (s.bl_count[bits] == 0) bits--;
254  s.bl_count[bits]--; // move one leaf down the tree
255  s.bl_count[bits + 1] += 2; // move one overflow item as its brother
256  s.bl_count[max_length]--;
257  // The brother of the overflow item also moves one step up,
258  // but this does not affect bl_count[max_length]
259  overflow -= 2;
260  }
261  while (overflow > 0);
262 
263  for (bits = max_length; bits != 0; bits--)
264  {
265  n = s.bl_count[bits];
266  while (n != 0)
267  {
268  m = s.heap[--h];
269  if (m > LargestCode) continue;
270  if (tree[m * 2 + 1] != bits)
271  {
272  s.opt_len += (int)(((long)bits - (long)tree[m * 2 + 1]) * (long)tree[m * 2]);
273  tree[m * 2 + 1] = (short)bits;
274  }
275  n--;
276  }
277  }
278  }

286  private static void GenerateCodes(short[] tree, int max_code, short[] bl_count)
287  {
288  short[] next_code = new short[MAX_BITS + 1]; // next code value for each bit length
289  short code = 0; // running code value
290  int bits; // bit index
291  int n; // code index
292 
293  // The distribution counts are first used to generate the code values
294  // without bit reversal.
295  for (bits = 1; bits <= MAX_BITS; bits++)
296  {
297  next_code[bits] = code = (short)((code + bl_count[bits - 1]) << 1);
298  }
299 
300  // Check that the bit counts in bl_count are consistent. The last code
301  // must be all ones.
302  //Assert (code + bl_count[MAX_BITS]-1 == (1<<MAX_BITS)-1,
303  // "inconsistent bit counts");
304  //Tracev((stderr,"\ngen_codes: max_code %d ", max_code));
305 
306  for (n = 0; n <= max_code; n++)
307  {
308  int len = tree[n * 2 + 1];
309  if (len == 0) continue;
310  // Now reverse the bits
311  tree[n * 2] = (short)(BiReverse(next_code[len]++, len));
312  }
313  }

321  private static int BiReverse(int code, int len)
322  {
323  int res = 0;
324  do
325  {
326  res |= code & 1;
327  code >>= 1;
328  res <<= 1;
329  }
330  while (--len > 0);
331  return res >> 1;
332  }
