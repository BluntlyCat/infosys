80  static LoggerManager()
81  {
82  try
83  {
84  // Register the AppDomain events, note we have to do this with a
85  // method call rather than directly here because the AppDomain
86  // makes a LinkDemand which throws the exception during the JIT phase.
87  RegisterAppDomainEvents();
88  }
89  catch(System.Security.SecurityException)
90  {
91  LogLog.Debug(declaringType, "Security Exception (ControlAppDomain LinkDemand) while trying "+
92  "to register Shutdown handler with the AppDomain. LoggerManager.Shutdown() "+
93  "will not be called automatically when the AppDomain exits. It must be called "+
94  "programmatically.");
95  }
96 
97  // Dump out our assembly version into the log if debug is enabled
98  LogLog.Debug(declaringType, GetVersionInfo());
99 
100  // Set the default repository selector
101 #if NETCF
102  s_repositorySelector = new CompactRepositorySelector(typeof(log4net.Repository.Hierarchy.Hierarchy));
103 #else
104 
105  // Look for the RepositorySelector type specified in the AppSettings 'log4net.RepositorySelector'
106  string appRepositorySelectorTypeName = SystemInfo.GetAppSetting("log4net.RepositorySelector");
107  if (appRepositorySelectorTypeName != null && appRepositorySelectorTypeName.Length > 0)
108  {
109  // Resolve the config string into a Type
110  Type appRepositorySelectorType = null;
111  try
112  {
113  appRepositorySelectorType = SystemInfo.GetTypeFromString(appRepositorySelectorTypeName, false, true);
114  }
115  catch(Exception ex)
116  {
117  LogLog.Error(declaringType, "Exception while resolving RepositorySelector Type ["+appRepositorySelectorTypeName+"]", ex);
118  }
119 
120  if (appRepositorySelectorType != null)
121  {
122  // Create an instance of the RepositorySelectorType
123  object appRepositorySelectorObj = null;
124  try
125  {
126  appRepositorySelectorObj = Activator.CreateInstance(appRepositorySelectorType);
127  }
128  catch(Exception ex)
129  {
130  LogLog.Error(declaringType, "Exception while creating RepositorySelector ["+appRepositorySelectorType.FullName+"]", ex);
131  }
132 
133  if (appRepositorySelectorObj != null && appRepositorySelectorObj is IRepositorySelector)
134  {
135  s_repositorySelector = (IRepositorySelector)appRepositorySelectorObj;
136  }
137  else
138  {
139  LogLog.Error(declaringType, "RepositorySelector Type ["+appRepositorySelectorType.FullName+"] is not an IRepositorySelector");
140  }
141  }
142  }
143 
144  // Create the DefaultRepositorySelector if not configured above
145  if (s_repositorySelector == null)
146  {
147  s_repositorySelector = new DefaultRepositorySelector(typeof(log4net.Repository.Hierarchy.Hierarchy));
148  }
149 #endif
150  }

164  private static void RegisterAppDomainEvents()
165  {
166 #if !NETCF
167  // ProcessExit seems to be fired if we are part of the default domain
168  AppDomain.CurrentDomain.ProcessExit += new EventHandler(OnProcessExit);
169 
170  // Otherwise DomainUnload is fired
171  AppDomain.CurrentDomain.DomainUnload += new EventHandler(OnDomainUnload);
172 #endif
173  }

797  private static string GetVersionInfo()
798  {
799  System.Text.StringBuilder sb = new System.Text.StringBuilder();
800 
801  // Grab the currently executing assembly
802  Assembly myAssembly = Assembly.GetExecutingAssembly();
803 
804  // Build Up message
805  sb.Append("log4net assembly [").Append(myAssembly.FullName).Append("]. ");
806  sb.Append("Loaded from [").Append(SystemInfo.AssemblyLocationInfo(myAssembly)).Append("]. ");
807  sb.Append("(.NET Runtime [").Append(Environment.Version.ToString()).Append("]");
808 #if (!SSCLI)
809  sb.Append(" on ").Append(Environment.OSVersion.ToString());
810 #endif
811  sb.Append(")");
812  return sb.ToString();
813  }
