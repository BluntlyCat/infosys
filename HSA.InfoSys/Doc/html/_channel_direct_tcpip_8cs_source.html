67  public void Bind()
68  {
69  // Cannot bind if channel is not open
70  if (!this.IsOpen)
71  return;
72 
73  // Start reading data from the port and send to channel
74  Exception exception = null;
75 
76  try
77  {
78  var buffer = new byte[this.PacketSize - 9];
79 
80  while (this._socket != null && this._socket.CanRead())
81  {
82  try
83  {
84  var read = 0;
85  this.InternalSocketReceive(buffer, ref read);
86  if (read > 0)
87  {
88  this.SendMessage(new ChannelDataMessage(this.RemoteChannelNumber, buffer.Take(read).ToArray()));
89  }
90  else
91  {
92  break;
93  }
94  }
95  catch (SocketException exp)
96  {
97  if (exp.SocketErrorCode == SocketError.WouldBlock ||
98  exp.SocketErrorCode == SocketError.IOPending ||
99  exp.SocketErrorCode == SocketError.NoBufferSpaceAvailable)
100  {
101  // socket buffer is probably empty, wait and try again
102  Thread.Sleep(30);
103  }
104  else if (exp.SocketErrorCode == SocketError.ConnectionAborted || exp.SocketErrorCode == SocketError.ConnectionReset)
105  {
106  break;
107  }
108  else
109  throw; // throw any other error
110  }
111  }
112  }
113  catch (Exception exp)
114  {
115  exception = exp;
116  }
117 
118  // Channel was open and we MUST receive EOF notification,
119  // data transfer can take longer then connection specified timeout
120  // If listener thread is finished then socket was closed
121  System.Threading.WaitHandle.WaitAny(new WaitHandle[] { this._channelEof });
122 
123  // Close socket if still open
124  if (this._socket != null)
125  {
126  this._socket.Dispose();
127  this._socket = null;
128  }
129 
130  if (exception != null)
131  throw exception;
