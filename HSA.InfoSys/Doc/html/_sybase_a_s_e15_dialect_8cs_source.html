275  public override SqlString ApplyLocksToSql(SqlString sql, IDictionary<string, LockMode> aliasedLockModes, IDictionary<string, string[]> keyColumnNames)
276  {
277  // TODO: merge additional lockoptions support in Dialect.applyLocksToSql
278 
279  var buffer = new StringBuilder(sql.ToString());
280  int correction = 0;
281 
282  foreach (KeyValuePair<string, LockMode> entry in aliasedLockModes)
283  {
284  LockMode mode = entry.Value;
285 
286  if (mode.GreaterThan(LockMode.Read))
287  {
288  string alias = entry.Key;
289  int start = -1;
290  int end = -1;
291 
292  if (sql.EndsWith(" " + alias))
293  {
294  start = (sql.Length - alias.Length) + correction;
295  end = start + alias.Length;
296  }
297  else
298  {
299  int position = sql.IndexOfCaseInsensitive(" " + alias + " ");
300 
301  if (position <= -1)
302  position = sql.IndexOfCaseInsensitive(" " + alias + ",");
303 
304  if (position > -1)
305  {
306  start = position + correction + 1;
307  end = start + alias.Length;
308  }
309  }
310 
311  if (start > -1)
312  {
313  string lockHint = AppendLockHint(mode, alias);
314  buffer.Remove(start, end - start + 1);
315  buffer.Insert(start, lockHint);
316  correction += (lockHint.Length - alias.Length);
317  }
318  }
319  }
320  return new SqlString(buffer.ToString());
321  }
