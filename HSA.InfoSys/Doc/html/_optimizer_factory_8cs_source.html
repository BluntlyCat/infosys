37  private static IOptimizer BuildOptimizer(string type, System.Type returnClass, int incrementSize)
38  {
39  if (string.IsNullOrEmpty(type))
40  {
41  throw new ArgumentNullException("type");
42  }
43  if (returnClass == null)
44  {
45  throw new ArgumentNullException("returnClass");
46  }
47  string optimizerClassName;
48  switch (type)
49  {
50  case None:
51  optimizerClassName = typeof(NoopOptimizer).FullName;
52  break;
53  case HiLo:
54  optimizerClassName = typeof(HiLoOptimizer).FullName;
55  break;
56  case Pool:
57  optimizerClassName = typeof(PooledOptimizer).FullName;
58  break;
59  case PoolLo:
60  optimizerClassName = typeof(PooledLoOptimizer).FullName;
61  break;
62  default:
63  optimizerClassName = type;
64  break;
65  }
66 
67  try
68  {
69  System.Type optimizerClass = ReflectHelper.ClassForName(optimizerClassName);
70  ConstructorInfo ctor = optimizerClass.GetConstructor(CtorSignature);
71 
72  if (ctor == null)
73  throw new HibernateException("Optimizer does not have expected contructor");
74 
75  return (IOptimizer)ctor.Invoke(new object[] { returnClass, incrementSize });
76  }
77  catch (Exception)
78  {
79  Log.Error("Unable to instantiate id generator optimizer."); // FIXME: Review log message.
80  }
81 
82  // the default...
83  return new NoopOptimizer(returnClass, incrementSize);
84  }

310  {

312  if (_value < 1)

324  {
325  _hiValue = _value;

327  }
328  }
