320  private string GetWholeAssociationPath(CriteriaImpl.Subcriteria subcriteria)
321  {
322  string path = subcriteria.Path;
323 
324  // some messy, complex stuff here, since createCriteria() can take an
325  // aliased path, or a path rooted at the creating criteria instance
326  ICriteria parent = null;
327  if (path.IndexOf('.') > 0)
328  {
329  // if it is a compound path
330  string testAlias = StringHelper.Root(path);
331  if (!testAlias.Equals(subcriteria.Alias))
332  {
333  // and the qualifier is not the alias of this criteria
334  // -> check to see if we belong to some criteria other
335  // than the one that created us
336  aliasCriteriaMap.TryGetValue(testAlias, out parent);
337  }
338  }
339  if (parent == null)
340  {
341  // otherwise assume the parent is the the criteria that created us
342  parent = subcriteria.Parent;
343  }
344  else
345  {
346  path = StringHelper.Unroot(path);
347  }
348 
349  if (parent.Equals(rootCriteria))
350  {
351  // if its the root criteria, we are done
352  return path;
353  }
354  else
355  {
356  // otherwise, recurse
357  return GetWholeAssociationPath((CriteriaImpl.Subcriteria)parent) + '.' + path;
358  }

390  private Persister.Entity.IJoinable GetPathJoinable(string path)
391  {
392  NHibernate_Persister_Entity.IJoinable last = (NHibernate_Persister_Entity.IJoinable)Factory.GetEntityPersister(rootEntityName);
393  NHibernate_Persister_Entity.IPropertyMapping lastEntity = (NHibernate_Persister_Entity.IPropertyMapping)last;
394 
395  string componentPath = "";
396 
397  StringTokenizer tokens = new StringTokenizer(path, ".", false);
398  foreach (string token in tokens)
399  {
400  componentPath += token;
401  IType type = lastEntity.ToType(componentPath);
402  if (type.IsAssociationType)
403  {
404  if(type.IsCollectionType)
405  {
406  // ignore joinables for composite collections
407  var collectionType = (CollectionType)type;
408  var persister = Factory.GetCollectionPersister(collectionType.Role);
409  if(persister.ElementType.IsEntityType==false)
410  return null;
411  }
412  IAssociationType atype = (IAssociationType)type;
413 
414  last = atype.GetAssociatedJoinable(Factory);
415  lastEntity = (NHibernate_Persister_Entity.IPropertyMapping)Factory.GetEntityPersister(atype.GetAssociatedEntityName(Factory));
416  componentPath = "";
417  }
418  else if (type.IsComponentType)
419  {
420  componentPath += '.';
421  }
422  else
423  {
424  throw new QueryException("not an association: " + componentPath);
425  }
426  }

430  private ICriteriaInfoProvider GetPathInfo(string path)
431  {
432  StringTokenizer tokens = new StringTokenizer(path, ".", false);
433  string componentPath = string.Empty;
434 
435  // start with the 'rootProvider'
436  ICriteriaInfoProvider provider;
437  if (nameCriteriaInfoMap.TryGetValue(rootEntityName, out provider) == false)
438  throw new ArgumentException("Could not find ICriteriaInfoProvider for: " + path);
439 
440 
441  foreach (string token in tokens)
442  {
443  componentPath += token;
444  logger.DebugFormat("searching for {0}", componentPath);
445  IType type = provider.GetType(componentPath);
446  if (type.IsAssociationType)
447  {
448  // CollectionTypes are always also AssociationTypes - but there's not always an associated entity...
449  IAssociationType atype = (IAssociationType)type;
450 
451  CollectionType ctype = type.IsCollectionType ? (CollectionType)type : null;
452  IType elementType = (ctype != null) ? ctype.GetElementType(sessionFactory) : null;
453  // is the association a collection of components or value-types? (i.e a colloction of valued types?)
454  if (ctype != null && elementType.IsComponentType)
455  {
456  provider = new ComponentCollectionCriteriaInfoProvider(helper.GetCollectionPersister(ctype.Role));
457  }
458  else if (ctype != null && !elementType.IsEntityType)
459  {
460  provider = new ScalarCollectionCriteriaInfoProvider(helper, ctype.Role);
461  }
462  else
463  {
464  provider = new EntityCriteriaInfoProvider((NHibernate_Persister_Entity.IQueryable)sessionFactory.GetEntityPersister(
465  atype.GetAssociatedEntityName(
466  sessionFactory)
467  ));
468  }
469 
470  componentPath = string.Empty;
471  }
472  else if (type.IsComponentType)
473  {
474  componentPath += '.';
475  }
476  else
477  {
478  throw new QueryException("not an association: " + componentPath);
479  }
480  }
481 
482  logger.DebugFormat("returning entity name={0} for path={1} class={2}",
483  provider.Name, path, provider.GetType().Name);
484  return provider;
485  }

767  public SqlString GetHavingCondition(IDictionary<string, IFilter> enabledFilters)
768  {
769  SqlStringBuilder condition = new SqlStringBuilder(30);
770  bool first = true;
771  foreach (CriteriaImpl.CriterionEntry entry in rootCriteria.IterateExpressionEntries())
772  {
773  if (HasGroupedOrAggregateProjection(entry.Criterion.GetProjections()))
774  {
775  if (!first)
776  {
777  condition.Add(" and ");
778  }
779  first = false;
780  SqlString sqlString = entry.Criterion.ToSqlString(entry.Criteria, this, enabledFilters);
781  condition.Add(sqlString);
782  }
783  }
784  return condition.ToSqlString();
