318  private static IBytecodeProvider BuildBytecodeProvider(string providerName)
319  {
320  switch (providerName)
321  {
322  case "codedom":
323  return new Bytecode.CodeDom.BytecodeProviderImpl();
324  case "lcg":
325  return new Bytecode.Lightweight.BytecodeProviderImpl();
326  case "null":
327  return new NullBytecodeProvider();
328  default:
329  log.Info("custom bytecode provider [" + providerName + "]");
330  return CreateCustomBytecodeProvider(providerName);
