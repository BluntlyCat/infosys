715  public Logger GetLogger(string name, ILoggerFactory factory)
716  {
717  if (name == null)
718  {
719  throw new ArgumentNullException("name");
720  }
721  if (factory == null)
722  {
723  throw new ArgumentNullException("factory");
724  }
725 
726  LoggerKey key = new LoggerKey(name);
727 
728  // Synchronize to prevent write conflicts. Read conflicts (in
729  // GetEffectiveLevel() method) are possible only if variable
730  // assignments are non-atomic.
731 
732  lock(m_ht)
733  {
734  Logger logger = null;
735 
736  Object node = m_ht[key];
737  if (node == null)
738  {
739  logger = factory.CreateLogger(this, name);
740  logger.Hierarchy = this;
741  m_ht[key] = logger;
742  UpdateParents(logger);
743  OnLoggerCreationEvent(logger);
744  return logger;
745  }
746 
747  Logger nodeLogger = node as Logger;
748  if (nodeLogger != null)
749  {
750  return nodeLogger;
751  }
752 
753  ProvisionNode nodeProvisionNode = node as ProvisionNode;
754  if (nodeProvisionNode != null)
755  {
756  logger = factory.CreateLogger(this, name);
757  logger.Hierarchy = this;
758  m_ht[key] = logger;
759  UpdateChildren(nodeProvisionNode, logger);
760  UpdateParents(logger);
761  OnLoggerCreationEvent(logger);
762  return logger;
763  }
764 
765  // It should be impossible to arrive here but let's keep the compiler happy.
766  return null;
767  }

829  private void UpdateParents(Logger log)
830  {
831  string name = log.Name;
832  int length = name.Length;
833  bool parentFound = false;
834 
835  // if name = "w.x.y.z", loop through "w.x.y", "w.x" and "w", but not "w.x.y.z"
836  for(int i = name.LastIndexOf('.', length-1); i >= 0; i = name.LastIndexOf('.', i-1))
837  {
838  string substr = name.Substring(0, i);
839 
840  LoggerKey key = new LoggerKey(substr); // simple constructor
841  Object node = m_ht[key];
842  // Create a provision node for a future parent.
843  if (node == null)
844  {
845  ProvisionNode pn = new ProvisionNode(log);
846  m_ht[key] = pn;
847  }
848  else
849  {
850  Logger nodeLogger = node as Logger;
851  if (nodeLogger != null)
852  {
853  parentFound = true;
854  log.Parent = nodeLogger;
855  break; // no need to update the ancestors of the closest ancestor
856  }
857  else
858  {
859  ProvisionNode nodeProvisionNode = node as ProvisionNode;
860  if (nodeProvisionNode != null)
861  {
862  nodeProvisionNode.Add(log);
863  }
864  else
865  {
866  LogLog.Error(declaringType, "Unexpected object type ["+node.GetType()+"] in ht.", new LogException());
867  }
868  }
869  }
870  }
871 
872  // If we could not find any existing parents, then link with root.
873  if (!parentFound)
874  {
875  log.Parent = this.Root;
876  }
877  }

903  private static void UpdateChildren(ProvisionNode pn, Logger log)
904  {
905  for(int i = 0; i < pn.Count; i++)
906  {
907  Logger childLogger = (Logger)pn[i];
908 
909  // Unless this child already points to a correct (lower) parent,
910  // make log.Parent point to childLogger.Parent and childLogger.Parent to log.
911  if (!childLogger.Parent.Name.StartsWith(log.Name))
912  {
913  log.Parent = childLogger.Parent;
914  childLogger.Parent = log;
915  }
916  }

931  internal void AddLevel(LevelEntry levelEntry)
932  {
933  if (levelEntry == null) throw new ArgumentNullException("levelEntry");
934  if (levelEntry.Name == null) throw new ArgumentNullException("levelEntry.Name");

943  }
