474  [DisplayName("program_name")]
475  public string ProgramName
476  {
477  get
478  {
479  string name = string.Empty;
480  name = Environment.CommandLine;
481  try
482  {
483  string path = Environment.CommandLine.Substring(0, Environment.CommandLine.IndexOf("\" ")).Trim('"');
484  name = System.IO.Path.GetFileName(path);
485  if (Assembly.GetEntryAssembly() != null)
486  name = Assembly.GetEntryAssembly().ManifestModule.Name;
487  }
488  catch (Exception ex) { System.Diagnostics.Debug.WriteLine(ex.ToString()); }
489  return name;
490  }
491  }

499  try

502  var collection = searcher.Get();
503  foreach (var mgtObj in collection)
504  {
505  os = mgtObj.GetPropertyValue("Caption").ToString();
506  break;
507  }
508  }
509  catch (Exception ex) { System.Diagnostics.Debug.WriteLine(ex.ToString()); }
510 
511  return os;
