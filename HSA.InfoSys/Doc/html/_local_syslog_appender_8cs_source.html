343  public override void ActivateOptions()
344  {
345  base.ActivateOptions();
346 
347  m_levelMapping.ActivateOptions();
348 
349  string identString = m_identity;
350  if (identString == null)
351  {
352  // Set to app name by default
353  identString = SystemInfo.ApplicationFriendlyName;
354  }
355 
356  // create the native heap ansi string. Note this is a copy of our string
357  // so we do not need to hold on to the string itself, holding on to the
358  // handle will keep the heap ansi string alive.
359  m_handleToIdentity = Marshal.StringToHGlobalAnsi(identString);
360 
361  // open syslog
362  openlog(m_handleToIdentity, 1, m_facility);
363  }

385  protected override void Append(LoggingEvent loggingEvent)

455  virtual protected SyslogSeverity GetSeverity(Level level)
456  {
457  LevelSeverity levelSeverity = m_levelMapping.Lookup(level) as LevelSeverity;
458  if (levelSeverity != null)
459  {
460  return levelSeverity.Severity;
461  }
462 
463  //
464  // Fallback to sensible default values
465  //
466 
467  if (level >= Level.Alert)
468  {
469  return SyslogSeverity.Alert;
470  }
471  else if (level >= Level.Critical)
472  {
473  return SyslogSeverity.Critical;
474  }
475  else if (level >= Level.Error)
476  {
477  return SyslogSeverity.Error;
478  }
479  else if (level >= Level.Warn)
480  {
481  return SyslogSeverity.Warning;
482  }
483  else if (level >= Level.Notice)
484  {
485  return SyslogSeverity.Notice;
486  }
487  else if (level >= Level.Info)
488  {
489  return SyslogSeverity.Informational;
490  }
491  // Default setting
492  return SyslogSeverity.Debug;
493  }
