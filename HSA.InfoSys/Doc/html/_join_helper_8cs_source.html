152  public override string GetTableName(IAssociationType type)
153  {
154  if (type.UseLHSPrimaryKey)
155  {
156  return Persister.TableName;
157  }
158  else
159  {
160  string propertyName = type.LHSPropertyName;
161  if (propertyName == null)
162  {
163  //if there is no property-ref, assume the join
164  //is to the subclass table (ie. the table of the
165  //subclass that the association belongs to)
166  return Persister.GetSubclassPropertyTableName(propertyIdx);
167  }
168  else
169  {
170  //handle a property-ref
171  string propertyRefTable = Persister.GetPropertyTableName(propertyName);
172  if (propertyRefTable == null)
173  {
174  //it is possible that the tree-walking in OuterJoinLoader can get to
175  //an association defined by a subclass, in which case the property-ref
176  //might refer to a property defined on a subclass of the current class
177  //in this case, the table name is not known - this temporary solution
178  //assumes that the property-ref refers to a property of the subclass
179  //table that the association belongs to (a reasonable guess)
180  //TODO: fix this, add: IOuterJoinLoadable.getSubclassPropertyTableName(string propertyName)
181  propertyRefTable = Persister.GetSubclassPropertyTableName(propertyIdx);
182  }
183  return propertyRefTable;
184  }
185  }
