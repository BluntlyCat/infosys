168  {
169  this.AdjustByteCounts();
170 
171  //
172  // expand 16 word block into 80 word blocks.
173  //
174  for (int ti = 16; ti <= 79; ++ti)
175  {
176  X[ti] = Sigma1(X[ti - 2]) + X[ti - 7] + Sigma0(X[ti - 15]) + X[ti - 16];
177  }

181  //

183  ulong b = H2;
184  ulong c = H3;
185  ulong d = H4;
186  ulong e = H5;
187  ulong f = H6;
188  ulong g = H7;
189  ulong h = H8;

192  for (int i = 0; i < 10; i++)
