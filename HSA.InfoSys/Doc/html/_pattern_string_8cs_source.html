395  private PatternParser CreatePatternParser(string pattern)
396  {
397  PatternParser patternParser = new PatternParser(pattern);
398 
399  // Add all the builtin patterns
400  foreach(DictionaryEntry entry in s_globalRulesRegistry)
401  {
402  ConverterInfo converterInfo = new ConverterInfo();
403  converterInfo.Name = (string)entry.Key;
404  converterInfo.Type = (Type)entry.Value;
405  patternParser.PatternConverters.Add(entry.Key, converterInfo);
406  }
407  // Add the instance patterns
408  foreach(DictionaryEntry entry in m_instanceRulesRegistry)
409  {
410  patternParser.PatternConverters[entry.Key] = entry.Value;
411  }
412 
413  return patternParser;
414  }

425  public void Format(TextWriter writer)
426  {
427  if (writer == null)
428  {
429  throw new ArgumentNullException("writer");
430  }
431 
432  PatternConverter c = m_head;
433 
434  // loop through the chain of pattern converters
435  while(c != null)
436  {
437  c.Format(writer, null);
438  c = c.Next;
439  }
440  }
