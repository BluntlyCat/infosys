79  public static void DeriveParameters(MySqlCommand command)
80  {
81  if (command.CommandType != CommandType.StoredProcedure)
82  throw new InvalidOperationException(Resources.CanNotDeriveParametersForTextCommands);
83 
84  // retrieve the proc definition from the cache.
85  string spName = command.CommandText;
86  if (spName.IndexOf(".") == -1)
87  spName = command.Connection.Database + "." + spName;
88 
89  try
90  {
91  DataSet ds = command.Connection.ProcedureCache.GetProcedure(command.Connection, spName, null);
92  DataTable parameters = ds.Tables["Procedure Parameters"];
93  DataTable procTable = ds.Tables["Procedures"];
94  command.Parameters.Clear();
95  foreach (DataRow row in parameters.Rows)
96  {
97  MySqlParameter p = new MySqlParameter();
98  p.ParameterName = String.Format("@{0}", row["PARAMETER_NAME"]);
99  if (row["ORDINAL_POSITION"].Equals(0) && p.ParameterName == "@")
100  p.ParameterName = "@RETURN_VALUE";
101  p.Direction = GetDirection(row);
102  bool unsigned = StoredProcedure.GetFlags(row["DTD_IDENTIFIER"].ToString()).IndexOf("UNSIGNED") != -1;
103  bool real_as_float = procTable.Rows[0]["SQL_MODE"].ToString().IndexOf("REAL_AS_FLOAT") != -1;
104  p.MySqlDbType = MetaData.NameToType(row["DATA_TYPE"].ToString(),
105  unsigned, real_as_float, command.Connection);
106  if (!row["CHARACTER_MAXIMUM_LENGTH"].Equals(DBNull.Value))
107  p.Size = (int)row["CHARACTER_MAXIMUM_LENGTH"];
108  if (!row["NUMERIC_PRECISION"].Equals(DBNull.Value))
109  p.Precision = Convert.ToByte(row["NUMERIC_PRECISION"]);
110  if (!row["NUMERIC_SCALE"].Equals(DBNull.Value))
111  p.Scale = Convert.ToByte(row["NUMERIC_SCALE"]);
112  if (p.MySqlDbType == MySqlDbType.Set || p.MySqlDbType == MySqlDbType.Enum)
113  p.PossibleValues = GetPossibleValues(row);
114  command.Parameters.Add(p);
115  }
116  }
117  catch (InvalidOperationException ioe)
118  {
119  throw new MySqlException(Resources.UnableToDeriveParameters, ioe);
120  }
121  }

123  private static List<string> GetPossibleValues(DataRow row)
124  {
125  string[] types = new string[] { "ENUM", "SET" };
126  string dtdIdentifier = row["DTD_IDENTIFIER"].ToString().Trim();
127 
128  int index = 0;
129  for (; index < 2; index++)
130  if (dtdIdentifier.StartsWith(types[index], StringComparison.OrdinalIgnoreCase ))
131  break;
132  if (index == 2) return null;
133  dtdIdentifier = dtdIdentifier.Substring(types[index].Length).Trim();
134  dtdIdentifier = dtdIdentifier.Trim('(', ')').Trim();
135 
136  List<string> values = new List<string>();
137  MySqlTokenizer tokenzier = new MySqlTokenizer(dtdIdentifier);
138  string token = tokenzier.NextToken();
139  int start = tokenzier.StartIndex;
140  while (true)
141  {
142  if (token == null || token == ",")
143  {
144  int end = dtdIdentifier.Length - 1;
145  if (token == ",")
146  end = tokenzier.StartIndex;
147 
148  string value = dtdIdentifier.Substring(start, end - start).Trim('\'', '\"').Trim();
149  values.Add(value);
150  start = tokenzier.StopIndex;
151  }
152  if (token == null) break;
153  token = tokenzier.NextToken();
154  }
155  return values;
156  }

158  private static ParameterDirection GetDirection(DataRow row)
159  {
160  string mode = row["PARAMETER_MODE"].ToString();
161  int ordinal = Convert.ToInt32(row["ORDINAL_POSITION"]);
162 
163  if (0 == ordinal)
164  return ParameterDirection.ReturnValue;
165  else if (mode == "IN")
166  return ParameterDirection.Input;
167  else if (mode == "OUT")
168  return ParameterDirection.Output;
169  return ParameterDirection.InputOutput;
170  }

199  public override string QuoteIdentifier(string unquotedIdentifier)
200  {
201  if (unquotedIdentifier == null) throw new
202  ArgumentNullException("unquotedIdentifier");
203 
204  // don't quote again if it is already quoted
205  if (unquotedIdentifier.StartsWith(QuotePrefix) &&
206  unquotedIdentifier.EndsWith(QuoteSuffix))
207  return unquotedIdentifier;
208 
209  unquotedIdentifier = unquotedIdentifier.Replace(QuotePrefix, QuotePrefix + QuotePrefix);
210 
211  return String.Format("{0}{1}{2}", QuotePrefix, unquotedIdentifier, QuoteSuffix);
212  }
213 
214  public override string UnquoteIdentifier(string quotedIdentifier)
215  {
216  if (quotedIdentifier == null) throw new
217  ArgumentNullException("quotedIdentifier");
218 
219  // don't unquote again if it is already unquoted
220  if (!quotedIdentifier.StartsWith(QuotePrefix) ||
221  !quotedIdentifier.EndsWith(QuoteSuffix))
222  return quotedIdentifier;
223 
224  if (quotedIdentifier.StartsWith(QuotePrefix))
225  quotedIdentifier = quotedIdentifier.Substring(1);
226  if (quotedIdentifier.EndsWith(QuoteSuffix))
227  quotedIdentifier = quotedIdentifier.Substring(0, quotedIdentifier.Length - 1);
228 
229  quotedIdentifier = quotedIdentifier.Replace(QuotePrefix + QuotePrefix, QuotePrefix);
230 
231  return quotedIdentifier;
232  }

252  protected override string GetParameterName(string parameterName)
253  {
254  StringBuilder sb = new StringBuilder(parameterName);
255  sb.Replace(" ", "");
256  sb.Replace("/", "_per_");
257  sb.Replace("-", "_");
258  sb.Replace(")", "_cb_");
259  sb.Replace("(", "_ob_");
260  sb.Replace("%", "_pct_");
261  sb.Replace("<", "_lt_");
262  sb.Replace(">", "_gt_");
263  sb.Replace(".", "_pt_");
264  return String.Format("@{0}", sb.ToString());
265  }
