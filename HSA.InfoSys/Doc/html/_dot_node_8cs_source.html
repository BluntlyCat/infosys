188  public override void Resolve(bool generateJoin, bool implicitJoin, string classAlias, IASTNode parent)
189  {
190  // If this dot has already been resolved, stop now.
191  if ( IsResolved )
192  {
193  return;
194  }
195 
196  IType propertyType = PrepareLhs(); // Prepare the left hand side and get the data type.
197 
198  // If there is no data type for this node, and we're at the end of the path (top most dot node), then
199  // this might be a Java constant.
200  if ( propertyType == null )
201  {
202  if ( parent == null )
203  {
204  Walker.LiteralProcessor.LookupConstant( this );
205  }
206  // If the propertyType is null and there isn't a parent, just
207  // stop now... there was a problem resolving the node anyway.
208  return;
209  }
210 
211  if ( propertyType.IsComponentType )
212  {
213  // The property is a component...
214  CheckLhsIsNotCollection();
215  DereferenceComponent( parent );
216  InitText();
217  }
218  else if ( propertyType.IsEntityType )
219  {
220  // The property is another class..
221  CheckLhsIsNotCollection();
222  DereferenceEntity( ( EntityType ) propertyType, implicitJoin, classAlias, generateJoin, parent );
223  InitText();
224  }
225  else if (propertyType.IsCollectionType)
226  {
227  // The property is a collection...
228  CheckLhsIsNotCollection();
229  DereferenceCollection((CollectionType)propertyType, implicitJoin, false, classAlias);
230  }
231  else
232  {
233  // Otherwise, this is a primitive type.
234  if (!CollectionProperties.IsAnyCollectionProperty(_propertyName))
235  {
236  CheckLhsIsNotCollection();
237  }
238  _dereferenceType = DerefPrimitive;
239  InitText();
240  }
241

361  private void DereferenceEntity(EntityType entityType, bool implicitJoin, string classAlias, bool generateJoin, IASTNode parent)
362  {
363  CheckForCorrelatedSubquery( "dereferenceEntity" );
364 
365  // three general cases we check here as to whether to render a physical SQL join:
366  // 1) is our parent a DotNode as well? If so, our property reference is
367  // being further de-referenced...
368  // 2) is this a DML statement
369  // 3) we were asked to generate any needed joins (generateJoins==true) *OR*
370  // we are currently processing a select or from clause
371  // (an additional check is the REGRESSION_STYLE_JOIN_SUPPRESSION check solely intended for the test suite)
372  //
373  // The REGRESSION_STYLE_JOIN_SUPPRESSION is an additional check
374  // intended solely for use within the test suite. This forces the
375  // implicit join resolution to behave more like the classic parser.
376  // The underlying issue is that classic translator is simply wrong
377  // about its decisions on whether or not to render an implicit join
378  // into a physical SQL join in a lot of cases. The piece it generally
379  // tends to miss is that INNER joins effect the results by further
380  // restricting the data set! A particular manifestation of this is
381  // the fact that the classic translator will skip the physical join
382  // for ToOne implicit joins *if the query is shallow*; the result
383  // being that Query.list() and Query.iterate() could return
384  // different number of results!
385 
386  DotNode parentAsDotNode = null;
387  string property = _propertyName;
388  bool joinIsNeeded;
389 
390  if ( IsDotNode( parent ) )
391  {
392  // our parent is another dot node, meaning we are being further dereferenced.
393  // thus we need to generate a join unless the parent refers to the associated
394  // entity's PK (because 'our' table would know the FK).
395  parentAsDotNode = ( DotNode ) parent;
396  property = parentAsDotNode._propertyName;
397  joinIsNeeded = generateJoin && !IsReferenceToPrimaryKey( parentAsDotNode._propertyName, entityType );
398  }
399  else if ( ! Walker.IsSelectStatement )
400  {
401  // in non-select queries, the only time we should need to join is if we are in a subquery from clause
402  joinIsNeeded = Walker.CurrentStatementType == HqlSqlWalker.SELECT && Walker.IsInFrom;
403  }
404  else if ( REGRESSION_STYLE_JOIN_SUPPRESSION ) {
405  // this is the regression style determination which matches the logic of the classic translator
406  joinIsNeeded = generateJoin && ( !Walker.IsInSelect || !Walker.IsShallowQuery);
407  }
408  else
409  {
410  joinIsNeeded = generateJoin || ( Walker.IsInSelect || Walker.IsInFrom );
411  }
412 
413  if ( joinIsNeeded )
414  {
415  DereferenceEntityJoin( classAlias, entityType, implicitJoin, parent );
416  }
417  else
418  {
419  DereferenceEntityIdentifier( property, parentAsDotNode );
420  }
421  }

699  {

701  {
