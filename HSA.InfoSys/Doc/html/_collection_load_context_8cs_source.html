70  public IPersistentCollection GetLoadingCollection(ICollectionPersister persister, object key)
71  {
72  EntityMode em = loadContexts.PersistenceContext.Session.EntityMode;
73 
74  CollectionKey collectionKey = new CollectionKey(persister, key, em);
75  if (log.IsDebugEnabled)
76  {
77  log.Debug("starting attempt to find loading collection [" + MessageHelper.InfoString(persister.Role, key) + "]");
78  }
79  LoadingCollectionEntry loadingCollectionEntry = loadContexts.LocateLoadingCollectionEntry(collectionKey);
80  if (loadingCollectionEntry == null)
81  {
82  // look for existing collection as part of the persistence context
83  IPersistentCollection collection = loadContexts.PersistenceContext.GetCollection(collectionKey);
84  if (collection != null)
85  {
86  if (collection.WasInitialized)
87  {
88  log.Debug("collection already initialized; ignoring");
89  return null; // ignore this row of results! Note the early exit
90  }
91  else
92  {
93  // initialize this collection
94  log.Debug("collection not yet initialized; initializing");
95  }
96  }
97  else
98  {
99  object owner = loadContexts.PersistenceContext.GetCollectionOwner(key, persister);
100  bool newlySavedEntity = owner != null && loadContexts.PersistenceContext.GetEntry(owner).Status != Status.Loading && em != EntityMode.Xml;
101  if (newlySavedEntity)
102  {
103  // important, to account for newly saved entities in query
104  // todo : some kind of check for new status...
105  log.Debug("owning entity already loaded; ignoring");
106  return null;
107  }
108  else
109  {
110  // create one
111  if (log.IsDebugEnabled)
112  {
113  log.Debug("instantiating new collection [key=" + key + ", rs=" + resultSet + "]");
114  }
115  collection = persister.CollectionType.Instantiate(loadContexts.PersistenceContext.Session, persister, key);
116  }
117  }
118  collection.BeforeInitialize(persister, -1);
119  collection.BeginRead();
120  localLoadingCollectionKeys.Add(collectionKey);
121  loadContexts.RegisterLoadingCollectionXRef(collectionKey, new LoadingCollectionEntry(resultSet, persister, key, collection));
122  return collection;
123  }
124  else
125  {
126  if (loadingCollectionEntry.ResultSet == resultSet)
127  {
128  log.Debug("found loading collection bound to current result set processing; reading row");
129  return loadingCollectionEntry.Collection;
130  }
131  else
132  {
133  // ignore this row, the collection is in process of
134  // being loaded somewhere further "up" the stack
135  log.Debug("collection is already being initialized; ignoring row");
136  return null;
137  }
138  }

147  public void EndLoadingCollections(ICollectionPersister persister)
148  {
149  if (!loadContexts.HasLoadingCollectionEntries && (localLoadingCollectionKeys.Count == 0))
150  {
151  return;
152  }
153 
154  // in an effort to avoid concurrent-modification-exceptions (from
155  // potential recursive calls back through here as a result of the
156  // eventual call to PersistentCollection#endRead), we scan the
157  // internal loadingCollections map for matches and store those matches
158  // in a temp collection. the temp collection is then used to "drive"
159  // the #endRead processing.
160  List<CollectionKey> toRemove = new List<CollectionKey>();
161  List<LoadingCollectionEntry> matches =new List<LoadingCollectionEntry>();
162  foreach (CollectionKey collectionKey in localLoadingCollectionKeys)
163  {
164  ISessionImplementor session = LoadContext.PersistenceContext.Session;
165 
166  LoadingCollectionEntry lce = loadContexts.LocateLoadingCollectionEntry(collectionKey);
167  if (lce == null)
168  {
169  log.Warn("In CollectionLoadContext#endLoadingCollections, localLoadingCollectionKeys contained [" + collectionKey + "], but no LoadingCollectionEntry was found in loadContexts");
170  }
171  else if (lce.ResultSet == resultSet && lce.Persister == persister)
172  {
173  matches.Add(lce);
174  if (lce.Collection.Owner == null)
175  {
176  session.PersistenceContext.AddUnownedCollection(new CollectionKey(persister, lce.Key, session.EntityMode),
177  lce.Collection);
178  }
179  if (log.IsDebugEnabled)
180  {
181  log.Debug("removing collection load entry [" + lce + "]");
182  }
183 
184  // todo : i'd much rather have this done from #endLoadingCollection(CollectionPersister,LoadingCollectionEntry)...
185  loadContexts.UnregisterLoadingCollectionXRef(collectionKey);
186  toRemove.Add(collectionKey);
187  }
188  }
189  localLoadingCollectionKeys.ExceptWith(toRemove);
190 
191  EndLoadingCollections(persister, matches);
192  if ((localLoadingCollectionKeys.Count == 0))
193  {
194  // todo : hack!!!
195  // NOTE : here we cleanup the load context when we have no more local
196  // LCE entries. This "works" for the time being because really
197  // only the collection load contexts are implemented. Long term,
198  // this cleanup should become part of the "close result set"
199  // processing from the (sandbox/jdbc) jdbc-container code.
200  loadContexts.Cleanup(resultSet);
201  }
202  }

232  private void EndLoadingCollection(LoadingCollectionEntry lce, ICollectionPersister persister)
233  {
234  if (log.IsDebugEnabled)
235  {
236  log.Debug("ending loading collection [" + lce + "]");
237  }
238  ISessionImplementor session = LoadContext.PersistenceContext.Session;
239  EntityMode em = session.EntityMode;
240 
241  bool statsEnabled = session.Factory.Statistics.IsStatisticsEnabled;
242  var stopWath = new Stopwatch();
243  if (statsEnabled)
244  {
245  stopWath.Start();
246  }
247 
248  bool hasNoQueuedAdds = lce.Collection.EndRead(persister); // warning: can cause a recursive calls! (proxy initialization)
249 
250  if (persister.CollectionType.HasHolder(em))
251  {
252  LoadContext.PersistenceContext.AddCollectionHolder(lce.Collection);
253  }
254 
255  CollectionEntry ce = LoadContext.PersistenceContext.GetCollectionEntry(lce.Collection);
256  if (ce == null)
257  {
258  ce = LoadContext.PersistenceContext.AddInitializedCollection(persister, lce.Collection, lce.Key);
259  }
260  else
261  {
262  ce.PostInitialize(lce.Collection);
263  }
264 
265  bool addToCache = hasNoQueuedAdds && persister.HasCache &&
266  ((session.CacheMode & CacheMode.Put) == CacheMode.Put) && !ce.IsDoremove; // and this is not a forced initialization during flush
267 
268  if (addToCache)
269  {
270  AddCollectionToCache(lce, persister);
271  }
272 
273  if (log.IsDebugEnabled)
274  {
275  log.Debug("collection fully initialized: " + MessageHelper.InfoString(persister, lce.Key, session.Factory));
276  }
277 
278  if (statsEnabled)
279  {
280  stopWath.Stop();
281  session.Factory.StatisticsImplementor.LoadCollection(persister.Role, stopWath.Elapsed);
282  }

288  private void AddCollectionToCache(LoadingCollectionEntry lce, ICollectionPersister persister)
289  {
290  ISessionImplementor session = LoadContext.PersistenceContext.Session;
291  ISessionFactoryImplementor factory = session.Factory;
292 
293  if (log.IsDebugEnabled)
294  {
295  log.Debug("Caching collection: " + MessageHelper.InfoString(persister, lce.Key, factory));
296  }
297 
298  if (!(session.EnabledFilters.Count == 0) && persister.IsAffectedByEnabledFilters(session))
299  {
300  // some filters affecting the collection are enabled on the session, so do not do the put into the cache.
301  log.Debug("Refusing to add to cache due to enabled filters");
302  // todo : add the notion of enabled filters to the CacheKey to differentiate filtered collections from non-filtered;
303  // but CacheKey is currently used for both collections and entities; would ideally need to define two separate ones;
304  // currently this works in conjunction with the check on
305  // DefaultInitializeCollectionEventHandler.initializeCollectionFromCache() (which makes sure to not read from
306  // cache with enabled filters).
307  return; // EARLY EXIT!!!!!
308  }

311  object version;
312  if (persister.IsVersioned)
313  {
314  versionComparator = persister.OwnerEntityPersister.VersionType.Comparator;
315  object collectionOwner = LoadContext.PersistenceContext.GetCollectionOwner(lce.Key, persister);

317  }
318  else
