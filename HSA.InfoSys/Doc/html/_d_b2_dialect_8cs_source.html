212  {
213  /*
214  * "select * from (select row_number() over(orderby_clause) as rownum, "
215  * querySqlString_without select
216  * " ) as tempresult where rownum between ? and ?"
217  */
218  string rownumClause = GetRowNumber(querySqlString);
219 
220  SqlStringBuilder pagingBuilder = new SqlStringBuilder();
221  pagingBuilder
222  .Add("select * from (select ")
223  .Add(rownumClause)
224  .Add(querySqlString.Substring(7))
225  .Add(") as tempresult where rownum ");
226 
227  if (offset != null && limit != null)
228  {
229  pagingBuilder
230  .Add("between ")
231  .Add(offset)
232  .Add("+1 and ")
233  .Add(limit);
234  }
235  else if (limit != null)
236  {
237  pagingBuilder
238  .Add("<= ")
239  .Add(limit);
240  }
241  else

247  }
