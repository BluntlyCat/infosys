611  // Restore the heap property by moving down the tree starting at node k,
612  // exchanging a node with the smallest of its two sons if necessary, stopping
613  // when the heap property is re-established (each father smaller than its
614  // two sons).
615  internal void pqdownheap(short[] tree, // the tree to restore
616  int k // node to move down
617  )
618  {
619  int v = heap[k];
620  int j = k << 1; // left son of k
621  while (j <= heap_len)
622  {
623  // Set j to the smallest of the two sons:
624  if (j < heap_len &&
625  smaller(tree, heap[j + 1], heap[j], depth))
626  {
627  j++;
628  }
629  // Exit if v is smaller than both sons
630  if (smaller(tree, v, heap[j], depth)) break;
631 
632  // Exchange v with the smallest son
633  heap[k] = heap[j]; k = j;
634  // And continue down the tree, setting j to the left son of k
635  j <<= 1;
636  }
637  heap[k] = v;
638  }

744  public ZLibStatus deflateEnd()
745  {
746  if (_status != INIT_STATE && _status != BUSY_STATE && _status != FINISH_STATE)
747  {
748  return ZLibStatus.Z_STREAM_ERROR;

1843  private int longest_match(int cur_match)
1844  {
1845  int chain_length = _maxChainLength; // max hash chain length
1846  int scan = _startInsertString; // current string
1847  int match; // matched string
1848  int len; // length of current match
1849  int best_len = _previousLength; // best match length so far
1850  int limit = _startInsertString > (_windowSize - MIN_LOOKAHEAD) ?
1851  _startInsertString - (_windowSize - MIN_LOOKAHEAD) : 0;
1852  int nice_match = this.nice_match;
1853 
1854  // Stop when cur_match becomes <= limit. To simplify the code,
1855  // we prevent matches with the string of window index 0.
1856 
1857  int wmask = _windowMask;
1858 
1859  int strend = _startInsertString + MAX_MATCH;
1860  byte scan_end1 = _window[scan + best_len - 1];
1861  byte scan_end = _window[scan + best_len];
1862 
1863  // The code is optimized for HASH_BITS >= 8 and MAX_MATCH-2 multiple of 16.
1864  // It is easy to get rid of this optimization if necessary.
1865 
1866  // Do not waste too much time if we already have a good match:
1867  if (_previousLength >= good_match)
1868  {
1869  chain_length >>= 2;
1870  }
1871 
1872  // Do not look for matches beyond the end of the input. This is necessary
1873  // to make deflate deterministic.
1874  if (nice_match > _validBytesAhead) nice_match = _validBytesAhead;
1875 
1876  do
1877  {
1878  match = cur_match;
1879 
1880  // Skip to next match if the match length cannot increase
1881  // or if the match length is less than 2:
1882  if (_window[match + best_len] != scan_end ||
1883  _window[match + best_len - 1] != scan_end1 ||
1884  _window[match] != _window[scan] ||
1885  _window[++match] != _window[scan + 1]) continue;
1886 
1887  // The check at best_len-1 can be removed because it will be made
1888  // again later. (This heuristic is not always a win.)
1889  // It is not necessary to compare scan[2] and match[2] since they
1890  // are always equal when the other bytes match, given that
1891  // the hash keys are equal and that HASH_BITS >= 8.
1892  scan += 2; match++;
1893 
1894  // We check for insufficient lookahead only every 8th comparison;
1895  // the 256th check will be made at strstart+258.
1896  do
1897  {
1898  } while (_window[++scan] == _window[++match] &&
1899  _window[++scan] == _window[++match] &&
1900  _window[++scan] == _window[++match] &&
1901  _window[++scan] == _window[++match] &&
1902  _window[++scan] == _window[++match] &&
1903  _window[++scan] == _window[++match] &&
1904  _window[++scan] == _window[++match] &&
1905  _window[++scan] == _window[++match] &&
1906  scan < strend);
1907 
1908  len = MAX_MATCH - (int)(strend - scan);
1909  scan = strend - MAX_MATCH;
1910 
1911  if (len > best_len)
1912  {
1913  _startMatchString = cur_match;
1914  best_len = len;
1915  if (len >= nice_match) break;
1916  scan_end1 = _window[scan + best_len - 1];
1917  scan_end = _window[scan + best_len];
1918  }
1919 
1920  } while ((cur_match = (_previous[cur_match & wmask] & 0xffff)) > limit
1921  && --chain_length != 0);
1922 
1923  if (best_len <= _validBytesAhead) return best_len;
1924  return _validBytesAhead;
1925  }
