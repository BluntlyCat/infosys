292  private static TypeClassification GetTypeClassification(string typeName)
293  {
294  int indexOfOpenParen = typeName.IndexOf("(");
295  int indexOfComma = 0;
296  if (indexOfOpenParen >= 0)
297  {
298  indexOfComma = typeName.IndexOf(",", indexOfOpenParen);
299  }
300 
301  if (indexOfOpenParen >= 0)
302  {
303  if (indexOfComma >= 0)
304  {
305  return TypeClassification.PrecisionScale;
306  }
307  else
308  {
309  return TypeClassification.Length;
310  }
311  }
312  else
313  {
314  return TypeClassification.Plain;
315  }
316  }

328  public static IType Basic(string name)
329  {
330  string typeName;
331 
332  // Use the basic name (such as String or String(255)) to get the
333  // instance of the IType object.
334  IType returnType;
335  if (typeByTypeOfName.TryGetValue(name, out returnType))
336  {
337  return returnType;
338  }
339 
340  // if we get to here then the basic type with the length or precision/scale
341  // combination doesn't exists - so lets figure out which one we have and
342  // invoke the appropriate delegate
343  TypeClassification typeClassification = GetTypeClassification(name);
344 
345  if (typeClassification == TypeClassification.PrecisionScale)
346  {
347  //precision/scale based
348 
349  string[] parsedName = name.Split(PrecisionScaleSplit);
350  if (parsedName.Length < 4)
351  {
352  throw new ArgumentOutOfRangeException("TypeClassification.PrecisionScale", name,
353  "It is not a valid Precision/Scale name");
354  }
355 
356  typeName = parsedName[0].Trim();
357  byte precision = Byte.Parse(parsedName[1].Trim());
358  byte scale = Byte.Parse(parsedName[2].Trim());
359 
360  return BuiltInType(typeName, precision, scale);
361  }
362  else if (typeClassification == TypeClassification.Length)
363  {
364  //length based
365 
366  string[] parsedName = name.Split(LengthSplit);
367  if (parsedName.Length < 3)
368  {
369  throw new ArgumentOutOfRangeException("TypeClassification.Length", name, "It is not a valid Length name");
370  }
371 
372  typeName = parsedName[0].Trim();
373  int length = Int32.Parse(parsedName[1].Trim());
374 
375  return BuiltInType(typeName, length);
376  }
377 
378  else
379  {
380  // it is not in the basicNameMap and typeByTypeOfName
381  // nor is it a Length or Precision/Scale based type
382  // so it must be a user defined type or something else that NHibernate
383  // doesn't have built into it.
384  return null;
385  }
386  }

466  public static IType HeuristicType(string typeName, IDictionary<string, string> parameters, int? length)
467  {
468  IType type = Basic(typeName);
469 
470  if (type != null)
471  return type;
472 
473  string[] parsedTypeName;
474  TypeClassification typeClassification = GetTypeClassification(typeName);
475  if (typeClassification == TypeClassification.Length)
476  parsedTypeName = typeName.Split(LengthSplit);
477  else
478  parsedTypeName = typeClassification == TypeClassification.PrecisionScale ? typeName.Split(PrecisionScaleSplit) : new[] { typeName };
479 
480 
481  System.Type typeClass;
482  try
483  {
484  typeClass = ReflectHelper.ClassForName(parsedTypeName[0]); //typeName);
485  }
486  catch (Exception)
487  {
488  typeClass = null;
489  }
490 
491  if (typeClass == null)
492  return null;
493 
494  if (typeof(IType).IsAssignableFrom(typeClass))
495  {
496  try
497  {
498  type = (IType) Cfg.Environment.BytecodeProvider.ObjectsFactory.CreateInstance(typeClass);
499  }
500  catch (Exception e)
501  {
502  throw new MappingException("Could not instantiate IType " + typeClass.Name + ": " + e, e);
503  }
504  InjectParameters(type, parameters);
505  return type;
506  }
507  if (typeof(ICompositeUserType).IsAssignableFrom(typeClass))
508  {
509  return new CompositeCustomType(typeClass, parameters);
510  }
511  if (typeof(IUserType).IsAssignableFrom(typeClass))
512  {
513  return new CustomType(typeClass, parameters);
514  }
515  if (typeof(ILifecycle).IsAssignableFrom(typeClass))
516  {
517  return NHibernateUtil.Entity(typeClass);
518  }
519 
520  var unwrapped = typeClass.UnwrapIfNullable();
521  if (unwrapped.IsEnum)
522  {
523  try
524  {
525  return (IType) Activator.CreateInstance(typeof (EnumType<>).MakeGenericType(unwrapped));
526  }
527  catch (Exception e)
528  {
529  throw new MappingException(string.Format("Can't instantiate enum {0}; The enum can't be empty", typeClass.FullName), e);
530  }
531  }
532 
533  if (!typeClass.IsSerializable)
534  return null;
535 
536  if (typeClassification == TypeClassification.Length)
537  return GetSerializableType(typeClass, Int32.Parse(parsedTypeName[1]));
538 
539  if (length.HasValue)
540  return GetSerializableType(typeClass, length.Value);
541 
542  return GetSerializableType(typeClass);
543  }

867  public static CollectionType CustomCollection(string typeName, IDictionary<string, string> typeParameters,
868  string role, string propertyRef, bool embedded)
869  {
870  System.Type typeClass;
871  try
872  {
873  typeClass = ReflectHelper.ClassForName(typeName);
874  }
875  catch (Exception cnfe)
876  {
877  throw new MappingException("user collection type class not found: " + typeName, cnfe);
878  }
879  CustomCollectionType result = new CustomCollectionType(typeClass, role, propertyRef, embedded);
880  if (typeParameters != null)
881  {
882  InjectParameters(result.UserType, typeParameters);
883  }
884  return result;
885  }

887  public static void InjectParameters(Object type, IDictionary<string, string> parameters)
888  {
889  if (type is IParameterizedType)
890  {
891  ((IParameterizedType) type).SetParameterValues(parameters);
892  }
893  else if (parameters != null && !(parameters.Count == 0))
894  {
895  throw new MappingException("type is not parameterized: " + type.GetType().Name);
896  }
897  }
898  }
