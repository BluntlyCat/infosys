199  internal void AddToken(BsonToken token)
200  {
201  if (_parent != null)
202  {
203  if (_parent is BsonObject)
204  {
205  ((BsonObject) _parent).Add(_propertyName, token);
206  _propertyName = null;
207  }
208  else
209  {
210  ((BsonArray) _parent).Add(token);
211  }
212  }
213  else
214  {
215  if (token.Type != BsonType.Object && token.Type != BsonType.Array)
216  throw JsonWriterException.Create(this, "Error writing {0} value. BSON must start with an Object or Array.".FormatWith(CultureInfo.InvariantCulture, token.Type), null);
217 
218  _parent = token;
219  _root = token;
220  }
221  }
