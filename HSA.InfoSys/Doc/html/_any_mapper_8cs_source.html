130  public void MetaValue(object value, System.Type entityType)
131  {
132  if (value == null)
133  {
134  throw new ArgumentNullException("value");
135  }
136  if (entityType == null)
137  {
138  throw new ArgumentNullException("entityType");
139  }
140  if (value is System.Type)
141  {
142  throw new ArgumentOutOfRangeException("value", "System.Type is invalid meta-type (you don't need to set meta-values).");
143  }
144  System.Type metavalueType = value.GetType();
145  if (any.metavalue == null)
146  {
147  any.metavalue = new HbmMetaValue[0];
148  }
149  Dictionary<string, string> values = any.metavalue.ToDictionary(mv => mv.value, mv => mv.@class);
150  MetaType(metavalueType);
151  string newClassMetavalue = entityType.GetShortClassName(mapDoc);
152  string metavalueKey = value.ToString();
153  string existingClassMetavalue;
154  if (values.TryGetValue(metavalueKey, out existingClassMetavalue) && existingClassMetavalue != newClassMetavalue)
155  {
156  throw new ArgumentException(
157  string.Format(
158  "Can't set two different classes for same meta-value (meta-value='{0}' old-class:'{1}' new-class='{2}')",
159  metavalueKey, existingClassMetavalue, newClassMetavalue));
160  }
161  values[metavalueKey] = newClassMetavalue;
162  any.metavalue = values.Select(vd => new HbmMetaValue {value = vd.Key, @class = vd.Value}).ToArray();
163  }
