433  public override object DoWorkInCurrentTransaction(ISessionImplementor session, System.Data.IDbConnection conn, System.Data.IDbTransaction transaction)
434  {
435  long result;
436  int updatedRows;
437 
438  do
439  {
440  object selectedValue;
441 
442  try
443  {
444  IDbCommand selectCmd = session.Factory.ConnectionProvider.Driver.GenerateCommand(CommandType.Text, selectQuery, selectParameterTypes);
445  using (selectCmd)
446  {
447  selectCmd.Connection = conn;
448  selectCmd.Transaction = transaction;
449  string s = selectCmd.CommandText;
450  ((IDataParameter)selectCmd.Parameters[0]).Value = SegmentValue;
451  PersistentIdGeneratorParmsNames.SqlStatementLogger.LogCommand(selectCmd, FormatStyle.Basic);
452 
453  selectedValue = selectCmd.ExecuteScalar();
454  }
455 
456  if (selectedValue == null)
457  {
458  result = InitialValue;
459 
460  IDbCommand insertCmd = session.Factory.ConnectionProvider.Driver.GenerateCommand(CommandType.Text, insertQuery, insertParameterTypes);
461  using (insertCmd)
462  {
463  insertCmd.Connection = conn;
464  insertCmd.Transaction = transaction;
465 
466  ((IDataParameter)insertCmd.Parameters[0]).Value = SegmentValue;
467  ((IDataParameter)insertCmd.Parameters[1]).Value = result;
468 
469  PersistentIdGeneratorParmsNames.SqlStatementLogger.LogCommand(insertCmd, FormatStyle.Basic);
470  insertCmd.ExecuteNonQuery();
471  }
472  }
473  else
474  {
475  result = Convert.ToInt64(selectedValue);
476  }
477  }
478  catch (Exception ex)
479  {
480  log.Error("Unable to read or initialize hi value in " + TableName, ex);
481  throw;
482  }
483 
484 
485  try
486  {
487  IDbCommand updateCmd = session.Factory.ConnectionProvider.Driver.GenerateCommand(CommandType.Text, updateQuery, updateParameterTypes);
488  using (updateCmd)
489  {
490  updateCmd.Connection = conn;
491  updateCmd.Transaction = transaction;
492 
493  int increment = Optimizer.ApplyIncrementSizeToSourceValues ? IncrementSize : 1;
494  ((IDataParameter)updateCmd.Parameters[0]).Value = result + increment;
495  ((IDataParameter)updateCmd.Parameters[1]).Value = result;
496  ((IDataParameter)updateCmd.Parameters[2]).Value = SegmentValue;
497  PersistentIdGeneratorParmsNames.SqlStatementLogger.LogCommand(updateCmd, FormatStyle.Basic);
498  updatedRows = updateCmd.ExecuteNonQuery();
499  }
500  }
501  catch (Exception ex)
502  {
503  log.Error("Unable to update hi value in " + TableName, ex);
504  throw;
505  }
506  }
507  while (updatedRows == 0);
508 
509  TableAccessCount++;
510 
511  return result;
512  }
