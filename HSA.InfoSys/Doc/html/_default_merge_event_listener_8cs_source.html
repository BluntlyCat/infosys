83  public virtual void OnMerge(MergeEvent @event, IDictionary copiedAlready)
84  {
85  EventCache copyCache = (EventCache)copiedAlready;
86  IEventSource source = @event.Session;
87  object original = @event.Original;
88 
89  if (original != null)
90  {
91  object entity;
92  if (original.IsProxy())
93  {
94  ILazyInitializer li = ((INHibernateProxy)original).HibernateLazyInitializer;
95  if (li.IsUninitialized)
96  {
97  log.Debug("ignoring uninitialized proxy");
98  @event.Result = source.Load(li.EntityName, li.Identifier);
99  return; //EARLY EXIT!
100  }
101  else
102  {
103  entity = li.GetImplementation();
104  }
105  }
106  else
107  {
108  entity = original;
109  }
110 
111  if (copyCache.Contains(entity) && copyCache.IsOperatedOn(entity))
112  {
113  log.Debug("already in merge process");
114  @event.Result = entity;
115  }
116  else
117  {
118  if (copyCache.Contains(entity))
119  {
120  log.Info("already in copyCache; setting in merge process");
121  copyCache.SetOperatedOn(entity, true);
122  }
123 
124  @event.Entity = entity;
125  EntityState entityState = EntityState.Undefined;
126  if (ReferenceEquals(null, @event.EntityName))
127  {
128  @event.EntityName = source.BestGuessEntityName(entity);
129  }
130 
131  // Check the persistence context for an entry relating to this
132  // entity to be merged...
133  EntityEntry entry = source.PersistenceContext.GetEntry(entity);
134  if (entry == null)
135  {
136  IEntityPersister persister = source.GetEntityPersister(@event.EntityName, entity);
137  object id = persister.GetIdentifier(entity, source.EntityMode);
138  if (id != null)
139  {
140  EntityKey key = source.GenerateEntityKey(id, persister);
141  object managedEntity = source.PersistenceContext.GetEntity(key);
142  entry = source.PersistenceContext.GetEntry(managedEntity);
143  if (entry != null)
144  {
145  // we have specialized case of a detached entity from the
146  // perspective of the merge operation. Specifically, we
147  // have an incoming entity instance which has a corresponding
148  // entry in the current persistence context, but registered
149  // under a different entity instance
150  entityState = EntityState.Detached;
151  }
152  }
153  }
154 
155  if (entityState == EntityState.Undefined)
156  {
157  entityState = GetEntityState(entity, @event.EntityName, entry, source);
158  }
159 
160  switch (entityState)
161  {
162  case EntityState.Persistent:
163  EntityIsPersistent(@event, copyCache);
164  break;
165  case EntityState.Transient:
166  EntityIsTransient(@event, copyCache);
167  break;
168  case EntityState.Detached:
169  EntityIsDetached(@event, copyCache);
170  break;
171  default:
172  throw new ObjectDeletedException("deleted instance passed to merge", null, GetLoggableName(@event.EntityName, entity));
173  }
174  }
175  }

209  private object MergeTransientEntity(object entity, string entityName, object requestedId, IEventSource source, IDictionary copyCache)
210  {
211  IEntityPersister persister = source.GetEntityPersister(entityName, entity);
212 
213  object id = persister.HasIdentifierProperty ? persister.GetIdentifier(entity, source.EntityMode) : null;
214  object copy = null;
215 
216  if (copyCache.Contains(entity))
217  {
218  copy = copyCache[entity];
219  persister.SetIdentifier(copy, id, source.EntityMode);
220  }
221  else
222  {
223  copy = source.Instantiate(persister, id);
224  ((EventCache)copyCache).Add(entity, copy, true); // before cascade!
225  }
226 
227  // cascade first, so that all unsaved objects get their
228  // copy created before we actually copy
229  //cascadeOnMerge(event, persister, entity, copyCache, Cascades.CASCADE_BEFORE_MERGE);
230  base.CascadeBeforeSave(source, persister, entity, copyCache);
231  CopyValues(persister, entity, copy, source, copyCache, ForeignKeyDirection.ForeignKeyFromParent);
232 
233  try
234  {
235  // try saving; check for non-nullable properties that are null or transient entities before saving
236  this.SaveTransientEntity(copy, entityName, requestedId, source, copyCache);
237  }
238  catch (PropertyValueException ex)
239  {
240  string propertyName = ex.PropertyName;
241  object propertyFromCopy = persister.GetPropertyValue(copy, propertyName, source.EntityMode);
242  object propertyFromEntity = persister.GetPropertyValue(entity, propertyName, source.EntityMode);
243  IType propertyType = persister.GetPropertyType(propertyName);
244  EntityEntry copyEntry = source.PersistenceContext.GetEntry(copy);
245 
246  if (propertyFromCopy == null || !propertyType.IsEntityType)
247  {
248  log.InfoFormat("property '{0}.{1}' is null or not an entity; {1} =[{2}]", copyEntry.EntityName, propertyName, propertyFromCopy);
249  throw;
250  }
251 
252  if (!copyCache.Contains(propertyFromEntity))
253  {
254  log.InfoFormat("property '{0}.{1}' from original entity is not in copyCache; {1} =[{2}]", copyEntry.EntityName, propertyName, propertyFromEntity);
255  throw;
256  }
257 
258  if (((EventCache)copyCache).IsOperatedOn(propertyFromEntity))
259  {
260  log.InfoFormat("property '{0}.{1}' from original entity is in copyCache and is in the process of being merged; {1} =[{2}]", copyEntry.EntityName, propertyName, propertyFromEntity);
261  }
262  else
263  {
264  log.InfoFormat("property '{0}.{1}' from original entity is in copyCache and is not in the process of being merged; {1} =[{2}]", copyEntry.EntityName, propertyName, propertyFromEntity);
265  }
266 
267  // continue...; we'll find out if it ends up not getting saved later
268  }
269 
270  // cascade first, so that all unsaved objects get their
271  // copy created before we actually copy
272  base.CascadeAfterSave(source, persister, entity, copyCache);
273  CopyValues(persister, entity, copy, source, copyCache, ForeignKeyDirection.ForeignKeyToParent);
274 
275  return copy;
276  }

293  protected virtual void EntityIsDetached(MergeEvent @event, IDictionary copyCache)
294  {
295  log.Debug("merging detached instance");
296 
297  object entity = @event.Entity;
298  IEventSource source = @event.Session;
299 
300  IEntityPersister persister = source.GetEntityPersister(@event.EntityName, entity);
301  string entityName = persister.EntityName;
302 
303  object id = @event.RequestedId;
304  if (id == null)
305  {
306  id = persister.GetIdentifier(entity, source.EntityMode);
307  }
308  else
309  {
310  // check that entity id = requestedId
311  object entityId = persister.GetIdentifier(entity, source.EntityMode);
312  if (!persister.IdentifierType.IsEqual(id, entityId, source.EntityMode, source.Factory))
313  {
314  throw new HibernateException("merge requested with id not matching id of passed entity");
315  }
316  }
317 
318  string previousFetchProfile = source.FetchProfile;
319  source.FetchProfile = "merge";
320 
321  //we must clone embedded composite identifiers, or
322  //we will get back the same instance that we pass in
323  object clonedIdentifier = persister.IdentifierType.DeepCopy(id, source.EntityMode, source.Factory);
324  object result = source.Get(persister.EntityName, clonedIdentifier);
325 
326  source.FetchProfile = previousFetchProfile;
327 
328  if (result == null)
329  {
330  //TODO: we should throw an exception if we really *know* for sure
331  // that this is a detached instance, rather than just assuming
332  //throw new StaleObjectStateException(entityName, id);
333 
334  // we got here because we assumed that an instance
335  // with an assigned id was detached, when it was
336  // really persistent
337  EntityIsTransient(@event, copyCache);
338  }
339  else
340  {
341  // NH different behavior : NH-1517
342  if (InvokeUpdateLifecycle(entity, persister, source))
343  {
344  return;
345  }
346 
347  ((EventCache)copyCache).Add(entity, result, true); //before cascade!
348 
349  object target = source.PersistenceContext.Unproxy(result);
350  if (target == entity)
351  {
352  throw new AssertionFailure("entity was not detached");
353  }
354  else if (!source.GetEntityName(target).Equals(entityName))
355  {
356  throw new WrongClassException("class of the given object did not match class of persistent copy",
357  @event.RequestedId, persister.EntityName);
358  }
359  else if (IsVersionChanged(entity, source, persister, target))
360  {
361  if (source.Factory.Statistics.IsStatisticsEnabled)
362  {
363  source.Factory.StatisticsImplementor.OptimisticFailure(entityName);
364  }
365  throw new StaleObjectStateException(persister.EntityName, id);
366  }
367 
368  // cascade first, so that all unsaved objects get their
369  // copy created before we actually copy
370  CascadeOnMerge(source, persister, entity, copyCache);
371  CopyValues(persister, entity, target, source, copyCache);
372 
373  //copyValues works by reflection, so explicitly mark the entity instance dirty
374  MarkInterceptorDirty(entity, target);
375 
376  @event.Result = result;
377  }

406  private static bool IsVersionChanged(object entity, IEventSource source, IEntityPersister persister, object target)
407  {
408  if (!persister.IsVersioned)
409  {
410  return false;
411  }
412  // for merging of versioned entities, we consider the version having
413  // been changed only when:
414  // 1) the two version values are different;
415  // *AND*
416  // 2) The target actually represents database state!
417  //
418  // This second condition is a special case which allows
419  // an entity to be merged during the same transaction
420  // (though during a separate operation) in which it was
421  // originally persisted/saved
422  bool changed =
423  !persister.VersionType.IsSame(persister.GetVersion(target, source.EntityMode),
424  persister.GetVersion(entity, source.EntityMode), source.EntityMode);
425 
426  // TODO : perhaps we should additionally require that the incoming entity
427  // version be equivalent to the defined unsaved-value?
428  return changed && ExistsInDatabase(target, source, persister);
429  }

519  protected EventCache GetTransientCopyCache(MergeEvent @event, EventCache copyCache)
520  {
521  EventCache transientCopyCache = new EventCache();
522 
523  foreach(object entity in copyCache.Keys)
524  {
525  object entityCopy = copyCache[entity];
526 
527  if (entityCopy.IsProxy())
528  entityCopy = ((INHibernateProxy)entityCopy).HibernateLazyInitializer.GetImplementation();
529 
530  // NH-specific: Disregard entities that implement ILifecycle and manage their own state - they
531  // don't have an EntityEntry, and we can't determine if they are transient or not
532  if (entityCopy is ILifecycle)
533  continue;
534 
535  EntityEntry copyEntry = @event.Session.PersistenceContext.GetEntry(entityCopy);
536 
537  if (copyEntry == null)
538  {
539  // entity name will not be available for non-POJO entities
540  // TODO: cache the entity name somewhere so that it is available to this exception
541  log.InfoFormat(
542  "transient instance could not be processed by merge: {0} [{1}]",
543  @event.Session.GuessEntityName(entityCopy),
544  entity);
545 
546  // merge did not cascade to this entity; it's in copyCache because a
547  // different entity has a non-nullable reference to it;
548  // this entity should not be put in transientCopyCache, because it was
549  // not included in the merge;
550 
551  throw new TransientObjectException(
552  "object is an unsaved transient instance - save the transient instance before merging: " + @event.Session.GuessEntityName(entityCopy));
553  }
554  else if (copyEntry.Status == Status.Saving)
555  {
556  transientCopyCache.Add(entity, entityCopy, copyCache.IsOperatedOn(entity));
557  }
558  else if (copyEntry.Status != Status.Loaded && copyEntry.Status != Status.ReadOnly)
559  {
560  throw new AssertionFailure(
561  String.Format(
562  "Merged entity does not have status set to MANAGED or READ_ONLY; {0} status = {1}",
563  entityCopy,
564  copyEntry.Status));
565  }
566  }
567  return transientCopyCache;
568  }

576  protected void RetryMergeTransientEntities(MergeEvent @event, IDictionary transientCopyCache, EventCache copyCache)
577  {
578  // TODO: The order in which entities are saved may matter (e.g., a particular
579  // transient entity may need to be saved before other transient entities can
580  // be saved).
581  // Keep retrying the batch of transient entities until either:
582  // 1) there are no transient entities left in transientCopyCache
583  // or 2) no transient entities were saved in the last batch.
584  // For now, just run through the transient entities and retry the merge
585 
586  foreach(object entity in transientCopyCache.Keys)
587  {
588  object copy = transientCopyCache[entity];
589  EntityEntry copyEntry = @event.Session.PersistenceContext.GetEntry(copy);
590 
591  if (entity == @event.Entity)
592  MergeTransientEntity(entity, copyEntry.EntityName, @event.RequestedId, @event.Session, copyCache);
593  else
594  MergeTransientEntity(entity, copyEntry.EntityName, copyEntry.Id, @event.Session, copyCache);
595  }
