61  {
62  // Using strftime returns 0-padded strings. '07' <> 7, so it is better to convert to an integer.
63  RegisterFunction("second", new SQLFunctionTemplate(NHibernateUtil.Int32, "cast(strftime('%S', ?1) as int)"));
64  RegisterFunction("minute", new SQLFunctionTemplate(NHibernateUtil.Int32, "cast(strftime('%M', ?1) as int)"));
65  RegisterFunction("hour", new SQLFunctionTemplate(NHibernateUtil.Int32, "cast(strftime('%H', ?1) as int)"));
66  RegisterFunction("day", new SQLFunctionTemplate(NHibernateUtil.Int32, "cast(strftime('%d', ?1) as int)"));
67  RegisterFunction("month", new SQLFunctionTemplate(NHibernateUtil.Int32, "cast(strftime('%m', ?1) as int)"));
68  RegisterFunction("year", new SQLFunctionTemplate(NHibernateUtil.Int32, "cast(strftime('%Y', ?1) as int)"));
69  // Uses local time like MSSQL and PostgreSQL.
70  RegisterFunction("current_timestamp", new SQLFunctionTemplate(NHibernateUtil.DateTime, "datetime(current_timestamp, 'localtime')"));
71  // The System.Data.SQLite driver stores both Date and DateTime as 'YYYY-MM-DD HH:MM:SS'
72  // The SQLite date() function returns YYYY-MM-DD, which unfortunately SQLite does not consider
73  // as equal to 'YYYY-MM-DD 00:00:00'. Because of this, it is best to return the
74  // 'YYYY-MM-DD 00:00:00' format for the date function.
75  RegisterFunction("date", new SQLFunctionTemplate(NHibernateUtil.Date, "datetime(date(?1))"));
76 
77  RegisterFunction("substring", new StandardSQLFunction("substr", NHibernateUtil.String));
78  RegisterFunction("left", new SQLFunctionTemplate(NHibernateUtil.String, "substr(?1,1,?2)"));
79  RegisterFunction("trim", new AnsiTrimEmulationFunction());

186  public override string Qualify(string catalog, string schema, string table)
187  {
188  StringBuilder qualifiedName = new StringBuilder();
189  bool quoted = false;
190 
191  if (!string.IsNullOrEmpty(catalog))
192  {
193  if (catalog.StartsWith(OpenQuote.ToString()))
194  {
195  catalog = catalog.Substring(1, catalog.Length - 1);
196  quoted = true;
197  }
198  if (catalog.EndsWith(CloseQuote.ToString()))
199  {
200  catalog = catalog.Substring(0, catalog.Length - 1);
201  quoted = true;
202  }
203  qualifiedName.Append(catalog).Append(StringHelper.Underscore);
204  }
205  if (!string.IsNullOrEmpty(schema))
206  {
207  if (schema.StartsWith(OpenQuote.ToString()))
208  {
209  schema = schema.Substring(1, schema.Length - 1);
210  quoted = true;
211  }
212  if (schema.EndsWith(CloseQuote.ToString()))
213  {
214  schema = schema.Substring(0, schema.Length - 1);
215  quoted = true;
216  }
217  qualifiedName.Append(schema).Append(StringHelper.Underscore);
218  }
219 
220  if (table.StartsWith(OpenQuote.ToString()))
221  {
222  table = table.Substring(1, table.Length - 1);
223  quoted = true;
224  }
225  if (table.EndsWith(CloseQuote.ToString()))
226  {
227  table = table.Substring(0, table.Length - 1);
228  quoted = true;
229  }
230 
231  string name = qualifiedName.Append(table).ToString();
232  if (quoted)
233  return OpenQuote + name + CloseQuote;
234  return name;
235 
236  }
