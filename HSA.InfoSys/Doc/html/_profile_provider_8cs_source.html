393  public override SettingsPropertyValueCollection GetPropertyValues(
394  SettingsContext context, SettingsPropertyCollection collection)
395  {
396  SettingsPropertyValueCollection values = new SettingsPropertyValueCollection();
397 
398  if (collection.Count < 1) return values;
399 
400  string username = (string)context["UserName"];
401 
402  foreach (SettingsProperty property in collection)
403  {
404  if (property.PropertyType.IsPrimitive || property.PropertyType == typeof(string))
405  property.SerializeAs = SettingsSerializeAs.String;
406  else
407  property.SerializeAs = SettingsSerializeAs.Xml;
408 
409  values.Add(new SettingsPropertyValue(property));
410  }
411 
412  if (String.IsNullOrEmpty(username))
413  return values;
414 
415  // retrieve encoded profile data from the database
416  try
417  {
418 
419  using (MySqlConnection c = new MySqlConnection(connectionString))
420  {
421  c.Open();
422  MySqlCommand cmd = new MySqlCommand(
423  @"SELECT * FROM my_aspnet_profiles p
424  JOIN my_aspnet_users u ON u.id = p.userId
425  WHERE u.applicationId = @appId AND u.name = @name", c);
426  cmd.Parameters.AddWithValue("@appId", app.FetchId(c));
427  cmd.Parameters.AddWithValue("@name", username);
428  MySqlDataAdapter da = new MySqlDataAdapter(cmd);
429  DataTable dt = new DataTable();
430  da.Fill(dt);
431 
432  if (dt.Rows.Count > 0)
433  DecodeProfileData(dt.Rows[0], values);
434  return values;
435  }
436  }
437  catch (Exception ex)
438  {
439  throw new ProviderException(Resources.UnableToRetrieveProfileData, ex);
440  }
441  }

448  public override void SetPropertyValues(
449  SettingsContext context, SettingsPropertyValueCollection collection)
450  {
451  bool isAuthenticated = (bool)context["IsAuthenticated"];
452  string username = (string)context["UserName"];
453 
454  if (String.IsNullOrEmpty(username)) return;
455  if (collection.Count < 1) return;
456 
457  string index = String.Empty;
458  string stringData = String.Empty;
459  byte[] binaryData = null;
460  int count = EncodeProfileData(collection, isAuthenticated, ref index, ref stringData, ref binaryData);
461  if (count < 1) return;
462 
463  MySqlTransaction txn = null;
464  // save the encoded profile data to the database
465  using (MySqlConnection connection = new MySqlConnection(connectionString))
466  {
467  try
468  {
469  connection.Open();
470  txn = connection.BeginTransaction();
471 
472  // either create a new user or fetch the existing user id
473  long userId = SchemaManager.CreateOrFetchUserId(connection, username,
474  app.EnsureId(connection), isAuthenticated);
475 
476  MySqlCommand cmd = new MySqlCommand(
477  @"INSERT INTO my_aspnet_profiles
478  VALUES (@userId, @index, @stringData, @binaryData, NULL) ON DUPLICATE KEY UPDATE
479  valueindex=VALUES(valueindex), stringdata=VALUES(stringdata),
480  binarydata=VALUES(binarydata)", connection);
481  cmd.Parameters.Clear();
482  cmd.Parameters.AddWithValue("@userId", userId);
483  cmd.Parameters.AddWithValue("@index", index);
484  cmd.Parameters.AddWithValue("@stringData", stringData);
485  cmd.Parameters.AddWithValue("@binaryData", binaryData);
486  count = cmd.ExecuteNonQuery();
487  if (count == 0)
488  throw new Exception(Resources.ProfileUpdateFailed);
489  txn.Commit();
490  }
491  catch (Exception ex)
492  {
493  if (txn != null)
494  txn.Rollback();
495  throw new ProviderException(Resources.ProfileUpdateFailed, ex);
496  }
497  }

512  private void DecodeProfileData(DataRow profileRow, SettingsPropertyValueCollection values)
513  {
514  string indexData = (string)profileRow["valueindex"];
515  string stringData = (string)profileRow["stringData"];
516  byte[] binaryData = (byte[])profileRow["binaryData"];
517 
518  if (indexData == null) return;
519 
520  string[] indexes = indexData.Split(':');
521 
522  foreach (string index in indexes)
523  {
524  string[] parts = index.Split('/');
525  SettingsPropertyValue value = values[parts[0]];
526  if (value == null) continue;
527 
528  int pos = Int32.Parse(parts[2], CultureInfo.InvariantCulture);
529  int len = Int32.Parse(parts[3], CultureInfo.InvariantCulture);
530  if (len == -1)
531  {
532  value.PropertyValue = null;
533  value.IsDirty = false;
534  value.Deserialized = true;
535  }
536  else if (parts[1].Equals("0"))
537  value.SerializedValue = stringData.Substring(pos, len);
538  else
539  {
540  byte[] buf = new byte[len];
541  Buffer.BlockCopy(binaryData, pos, buf, 0, len);
542  value.SerializedValue = buf;
543  }
544  }
545  }

547  private int EncodeProfileData(SettingsPropertyValueCollection collection, bool isAuthenticated,
548  ref string index, ref string stringData, ref byte[] binaryData)
549  {
550  bool itemsToSave = false;
551 
552  // first we need to determine if there are any items that need saving
553  // this is an optimization
554  foreach (SettingsPropertyValue value in collection)
555  {
556  if (!value.IsDirty) continue;
557  if (value.Property.Attributes["AllowAnonymous"].Equals(false) &&
558  !isAuthenticated) continue;
559  itemsToSave = true;
560  break;
561  }
562  if (!itemsToSave) return 0;
563 
564  StringBuilder indexBuilder = new StringBuilder();
565  StringBuilder stringDataBuilder = new StringBuilder();
566  MemoryStream binaryBuilder = new MemoryStream();
567  int count = 0;
568 
569  // ok, we have some values that need to be saved so we go back through
570  foreach (SettingsPropertyValue value in collection)
571  {
572  // if the value has not been written to and is still using the default value
573  // no need to save it
574  if (value.UsingDefaultValue && !value.IsDirty) continue;
575 
576  // we don't save properties that require the user to be authenticated when the
577  // current user is not authenticated.
578  if (value.Property.Attributes["AllowAnonymous"].Equals(false) &&
579  !isAuthenticated) continue;
580 
581  count++;
582  object propValue = value.SerializedValue;
583  if ((value.Deserialized && value.PropertyValue == null) ||
584  value.SerializedValue == null)
585  indexBuilder.AppendFormat("{0}//0/-1:", value.Name);
586  else if (propValue is string)
587  {
588  indexBuilder.AppendFormat("{0}/0/{1}/{2}:", value.Name,
589  stringDataBuilder.Length, (propValue as string).Length);
590  stringDataBuilder.Append(propValue);
591  }
592  else
593  {
594  byte[] binaryValue = (byte[])propValue;
595  indexBuilder.AppendFormat("{0}/1/{1}/{2}:", value.Name,
596  binaryBuilder.Position, binaryValue.Length);
597  binaryBuilder.Write(binaryValue, 0, binaryValue.Length);
598  }
599  }
600  index = indexBuilder.ToString();
601  stringData = stringDataBuilder.ToString();
602  binaryData = binaryBuilder.ToArray();
603  return count;
604  }

607  private ProfileInfoCollection GetProfiles(
608  ProfileAuthenticationOption authenticationOption,
609  string usernameToMatch, DateTime userInactiveSinceDate,
610  int pageIndex, int pageSize, out int totalRecords)
611  {
612  List<string> whereClauses = new List<string>();
613 
614  using (MySqlConnection c = new MySqlConnection(connectionString))
615  {
616  c.Open();
617 
618  MySqlCommand cmd = new MySqlCommand(
619  @"SELECT p.*, u.name, u.isAnonymous, u.lastActivityDate,
620  LENGTH(p.stringdata) + LENGTH(p.binarydata) AS profilesize
621  FROM my_aspnet_profiles p
622  JOIN my_aspnet_users u ON u.id = p.userId
623  WHERE u.applicationId = @appId", c);
624  cmd.Parameters.AddWithValue("@appId", app.FetchId(c));
625 
626  if (usernameToMatch != null)
627  {
628  cmd.CommandText += " AND u.name LIKE @userName";
629  cmd.Parameters.AddWithValue("@userName", usernameToMatch);
630  }
631  if (userInactiveSinceDate != DateTime.MinValue)
632  {
633  cmd.CommandText += " AND u.lastActivityDate < @lastActivityDate";
634  cmd.Parameters.AddWithValue("@lastActivityDate", userInactiveSinceDate);
635  }
636  if (authenticationOption == ProfileAuthenticationOption.Anonymous)
637  cmd.CommandText += " AND u.isAnonymous = 1";
638  else if (authenticationOption == ProfileAuthenticationOption.Authenticated)
639  cmd.CommandText += " AND u.isAnonymous = 0";
640 
641  cmd.CommandText += String.Format(" LIMIT {0},{1}", pageIndex * pageSize, pageSize);
642 
643  ProfileInfoCollection pic = new ProfileInfoCollection();
644  using (MySqlDataReader reader = cmd.ExecuteReader())
645  {
646  while (reader.Read())
647  {
648  ProfileInfo pi = new ProfileInfo(
649  reader.GetString("name"),
650  reader.GetBoolean("isAnonymous"),
651  reader.GetDateTime("lastActivityDate"),
652  reader.GetDateTime("lastUpdatedDate"),
653  reader.GetInt32("profilesize"));
654  pic.Add(pi);
655  }
656  }
657  cmd.CommandText = "SELECT FOUND_ROWS()";
658  totalRecords = Convert.ToInt32(cmd.ExecuteScalar());
659  return pic;
660  }
661  }
