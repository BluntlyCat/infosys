124  public string GetAlias(Dialect.Dialect dialect)
125  {
126  string alias = name;
127  string _unique = uniqueInteger.ToString() + '_';
128  int lastLetter = StringHelper.LastIndexOfLetter(name);
129  if (lastLetter == -1)
130  {
131  alias = "column";
132  }
133  else if (lastLetter < name.Length - 1)
134  {
135  alias = name.Substring(0, lastLetter + 1);
136  }
137  if (alias.Length > dialect.MaxAliasLength)
138  {
139  alias = alias.Substring(0, dialect.MaxAliasLength - _unique.Length);
140  }
141  bool useRawName = name.Equals(alias) &&
142  !quoted &&
143  !StringHelper.EqualsCaseInsensitive(name, "rowid");
144 
145  if (useRawName)
146  {
147  return alias;
148  }
149  else
150  {
151  return alias + _unique;
152  }
153  }
