250  protected virtual void RegisterCastTypes()
251  {
252  // According to the MySql documentation (http://dev.mysql.com/doc/refman/4.1/en/cast-functions.html)
253  // only a few values are supported for the cast target type: BINARY, CHAR, DATE, DATETIME,
254  // SIGNED, TIME, and UNSIGNED. So we must limit our possible cast types to these
255 
256  // The Dialect.GetCastTypeName() method uses the default length, precision, and
257  // scale values, so there's no need to consider those values here either, just use the defaults
258  RegisterCastType(DbType.AnsiString, "CHAR");
259  RegisterCastType(DbType.AnsiStringFixedLength, "CHAR");
260  RegisterCastType(DbType.String, "CHAR");
261  RegisterCastType(DbType.StringFixedLength, "CHAR");
262  RegisterCastType(DbType.Binary, "BINARY");
263  RegisterCastType(DbType.Int16, "SIGNED");
264  RegisterCastType(DbType.Int32, "SIGNED");
265  RegisterCastType(DbType.Int64, "SIGNED");
266  RegisterCastType(DbType.UInt16, "UNSIGNED");
267  RegisterCastType(DbType.UInt32, "UNSIGNED");
268  RegisterCastType(DbType.UInt64, "UNSIGNED");
269  RegisterCastType(DbType.Guid, "CHAR(40)");
270  RegisterCastType(DbType.Time, "TIME");
271  RegisterCastType(DbType.Date, "DATE");
272  RegisterCastType(DbType.DateTime, "DATETIME");
