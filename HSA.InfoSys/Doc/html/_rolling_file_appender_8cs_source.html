879  private void InitializeFromOneFile(string baseFile, string curFileName)
880  {
881  if (curFileName.StartsWith(Path.GetFileNameWithoutExtension(baseFile)) == false)
882  {
883  // This is not a log file, so ignore
884  return;
885  }
886  if (curFileName.Equals(baseFile))
887  {
888  // Base log file is not an incremented logfile (.1 or .2, etc)
889  return;
890  }
891 
892  /*
893  if (m_staticLogFileName)
894  {
895  int endLength = curFileName.Length - index;
896  if (baseFile.Length + endLength != curFileName.Length)
897  {
898  // file is probably scheduledFilename + .x so I don't care
899  return;
900  }
901  }
902  */
903 
904  // Only look for files in the current roll point
905  if (m_rollDate && !m_staticLogFileName)
906  {
907  if (! curFileName.StartsWith(CombinePath(baseFile, m_dateTime.Now.ToString(m_datePattern, System.Globalization.DateTimeFormatInfo.InvariantInfo))))
908  {
909  LogLog.Debug(declaringType, "Ignoring file ["+curFileName+"] because it is from a different date period");
910  return;
911  }
912  }
913 
914  try
915  {
916  // Bump the counter up to the highest count seen so far
917  int backup = GetBackUpIndex(curFileName);
918 
919  // caution: we might get a false positive when certain
920  // date patterns such as yyyyMMdd are used...those are
921  // valid number but aren't the kind of back up index
922  // we're looking for
923  if (backup > m_curSizeRollBackups)
924  {
925  if (0 == m_maxSizeRollBackups)
926  {
927  // Stay at zero when zero backups are desired
928  }
929  else if (-1 == m_maxSizeRollBackups)
930  {
931  // Infinite backups, so go as high as the highest value
932  m_curSizeRollBackups = backup;
933  }
934  else
935  {
936  // Backups limited to a finite number
937  if (m_countDirection >= 0)
938  {
939  // Go with the highest file when counting up
940  m_curSizeRollBackups = backup;
941  }
942  else
943  {
944  // Clip to the limit when counting down
945  if (backup <= m_maxSizeRollBackups)
946  {
947  m_curSizeRollBackups = backup;
948  }
949  }
950  }
951  LogLog.Debug(declaringType, "File name [" + curFileName + "] moves current count to [" + m_curSizeRollBackups + "]");
952  }
953  }
954  catch(FormatException)
955  {
956  //this happens when file.log -> file.log.yyyy-MM-dd which is normal
957  //when staticLogFileName == false
958  LogLog.Debug(declaringType, "Encountered a backup file not ending in .x ["+curFileName+"]");
959  }
960  }

1025  private RollPoint ComputeCheckPeriod(string datePattern)
1026  {
1027  // s_date1970 is 1970-01-01 00:00:00 this is UniversalSortableDateTimePattern
1028  // (based on ISO 8601) using universal time. This date is used for reference
1029  // purposes to calculate the resolution of the date pattern.
1030 
1031  // Get string representation of base line date
1032  string r0 = s_date1970.ToString(datePattern, System.Globalization.DateTimeFormatInfo.InvariantInfo);
1033 
1034  // Check each type of rolling mode starting with the smallest increment.
1035  for(int i = (int)RollPoint.TopOfMinute; i <= (int)RollPoint.TopOfMonth; i++)
1036  {
1037  // Get string representation of next pattern
1038  string r1 = NextCheckDate(s_date1970, (RollPoint)i).ToString(datePattern, System.Globalization.DateTimeFormatInfo.InvariantInfo);
1039 
1040  LogLog.Debug(declaringType, "Type = ["+i+"], r0 = ["+r0+"], r1 = ["+r1+"]");
1041 
1042  // Check if the string representations are different
1043  if (r0 != null && r1 != null && !r0.Equals(r1))
1044  {
1045  // Found highest precision roll point
1046  return (RollPoint)i;
1047  }
1048  }
1049 
1050  return RollPoint.InvalidRollPoint; // Deliberately head for trouble...
1051  }

1286

1288 
1289  // Try to move the file to temp name.
1290  // If the file is locked we may still be able to move it
1291  string tempFileName = fileName + "." + Environment.TickCount + ".DeletePending";
1292  try
1293  {
1294  using(SecurityContext.Impersonate(this))
1295  {
1296  System.IO.File.Move(fileName, tempFileName);
1297  }
1298  fileToDelete = tempFileName;
1299  }
1300  catch(Exception moveEx)
1301  {
1302  LogLog.Debug(declaringType, "Exception while moving file to be deleted [" + fileName + "] -> [" + tempFileName + "]", moveEx);
1303  }
1304 
1305  // Try to delete the file (either the original or the moved file)
1306  try
1307  {
1308  using(SecurityContext.Impersonate(this))
1309  {
1310  System.IO.File.Delete(fileToDelete);
1311  }
1312  LogLog.Debug(declaringType, "Deleted file [" + fileName + "]");
1313  }
1314  catch(Exception deleteEx)
1315  {
1316  if (fileToDelete == fileName)
1317  {
1318  // Unable to move or delete the file
1319  ErrorHandler.Error("Exception while deleting file [" + fileToDelete + "]", deleteEx, ErrorCode.GenericFailure);
1320  }
1321  else

1403  protected void RollOverRenameFiles(string baseFileName)
1404  {
1405  // If maxBackups <= 0, then there is no file renaming to be done.
1406  if (m_maxSizeRollBackups != 0)
1407  {
1408  if (m_countDirection < 0)
1409  {
1410  // Delete the oldest file, to keep Windows happy.
1411  if (m_curSizeRollBackups == m_maxSizeRollBackups)
1412  {
1413  DeleteFile(CombinePath(baseFileName, "." + m_maxSizeRollBackups));
1414  m_curSizeRollBackups--;
1415  }
1416 
1417  // Map {(maxBackupIndex - 1), ..., 2, 1} to {maxBackupIndex, ..., 3, 2}
1418  for (int i = m_curSizeRollBackups; i >= 1; i--)
1419  {
1420  RollFile((CombinePath(baseFileName, "." + i)), (CombinePath(baseFileName, "." + (i + 1))));
1421  }
1422 
1423  m_curSizeRollBackups++;
1424 
1425  // Rename fileName to fileName.1
1426  RollFile(baseFileName, CombinePath(baseFileName, ".1"));
1427  }
1428  else
1429  {
1430  //countDirection >= 0
1431  if (m_curSizeRollBackups >= m_maxSizeRollBackups && m_maxSizeRollBackups > 0)
1432  {
1433  //delete the first and keep counting up.
1434  int oldestFileIndex = m_curSizeRollBackups - m_maxSizeRollBackups;
1435 
1436  // If static then there is 1 file without a number, therefore 1 less archive
1437  if (m_staticLogFileName)
1438  {
1439  oldestFileIndex++;
1440  }
1441 
1442  // If using a static log file then the base for the numbered sequence is the baseFileName passed in
1443  // If not using a static log file then the baseFileName will already have a numbered postfix which
1444  // we must remove, however it may have a date postfix which we must keep!

1452  }
1453  }

1457  }
1458 
1459  if (m_staticLogFileName)
1460  {
1461  m_curSizeRollBackups++;

1489  protected DateTime NextCheckDate(DateTime currentDateTime, RollPoint rollPoint)

1499  current = current.AddSeconds(-current.Second);

1509

1519  else
1520  {
1521  current = current.AddHours(-current.Hour);
1522  current = current.AddDays(1);
1523  }
1524  break;
1525 
1526  case RollPoint.TopOfDay:
1527  current = current.AddMilliseconds(-current.Millisecond);
1528  current = current.AddSeconds(-current.Second);
1529  current = current.AddMinutes(-current.Minute);
1530  current = current.AddHours(-current.Hour);

1533
