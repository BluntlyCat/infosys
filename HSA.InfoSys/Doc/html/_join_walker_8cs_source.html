151  private void AddAssociationToJoinTree(IAssociationType type, string[] aliasedLhsColumns, string alias,
152  string path, int currentDepth, JoinType joinType)
153  {
154  IJoinable joinable = type.GetAssociatedJoinable(Factory);
155

157 
158  OuterJoinableAssociation assoc =
159  new OuterJoinableAssociation(type, alias, aliasedLhsColumns, subalias, joinType, GetWithClause(path), Factory, enabledFilters);

179  private static int[] GetTopologicalSortOrder(List<DependentAlias> fields)
180  {
181  TopologicalSorter g = new TopologicalSorter(fields.Count);
182  Dictionary<string, int> _indexes = new Dictionary<string, int>();
183 
184  // add vertices
185  for (int i = 0; i < fields.Count; i++)
186  {
187  _indexes[fields[i].Alias.ToLower()] = g.AddVertex(i);
188  }
189 
190  // add edges
191  for (int i = 0; i < fields.Count; i++)
192  {
193  var dependentAlias = fields[i];
194  if (dependentAlias.DependsOn != null)
195  {
196  for (int j = 0; j < dependentAlias.DependsOn.Length; j++)
197  {
198  var dependentField = dependentAlias.DependsOn[j].ToLower();
199  int end;
200  if (_indexes.TryGetValue(dependentField, out end))
201  {
202  g.AddEdge(i, end);
203  }
204  }
205  }
206  }
207 
208  return g.Sort();
209  }

214  private void AddAssociation(string subalias, OuterJoinableAssociation association)
215  {
216  subalias = subalias.ToLower();
217 
218  var dependencies = new List<string>();
219  var on = association.On.ToString();
220  if (!String.IsNullOrEmpty(on))
221  {
222  foreach (Match match in aliasRegex.Matches(on))
223  {
224  string alias = match.Groups[1].Value;
225  if (alias == subalias) continue;
226  dependencies.Add(alias.ToLower());
227  }
228  }
229 
230  _dependentAliases.Add(new DependentAlias
231  {
232  Alias = subalias,
233  DependsOn = dependencies.ToArray()
234  });
235 
236  associations.Add(association);
237  }

259  private void WalkCollectionTree(IQueryableCollection persister, string alias, string path, int currentDepth)
260  {
261  if (persister.IsOneToMany)
262  {
263  WalkEntityTree((IOuterJoinLoadable)persister.ElementPersister, alias, path, currentDepth);
264  }
265  else
266  {
267  IType type = persister.ElementType;
268  if (type.IsAssociationType)
269  {
270  // a many-to-many
271  // decrement currentDepth here to allow join across the association table
272  // without exceeding MAX_FETCH_DEPTH (i.e. the "currentDepth - 1" bit)
273  IAssociationType associationType = (IAssociationType)type;
274  string[] aliasedLhsColumns = persister.GetElementColumnNames(alias);
275  string[] lhsColumns = persister.ElementColumnNames;
276 
277  // if the current depth is 0, the root thing being loaded is the
278  // many-to-many collection itself. Here, it is alright to use
279  // an inner join...
280  bool useInnerJoin = currentDepth == 0;
281 
282  JoinType joinType =
283  GetJoinType(associationType, persister.FetchMode, path, persister.TableName, lhsColumns, !useInnerJoin,
284  currentDepth - 1, null);
285 
286  AddAssociationToJoinTreeIfNecessary(associationType, aliasedLhsColumns, alias, path, currentDepth - 1, joinType);
287  }
288  else if (type.IsComponentType)
289  {
290  WalkCompositeElementTree((IAbstractComponentType)type, persister.ElementColumnNames, persister, alias, path,
291  currentDepth);
292  }
293  }

377  IQueryableCollection persister, string alias, string path, int currentDepth)
378  {
379  IType[] types = compositeType.Subtypes;
380  string[] propertyNames = compositeType.PropertyNames;
381  int begin = 0;
382  for (int i = 0; i < types.Length; i++)
383  {

385  string[] lhsColumns = ArrayHelper.Slice(cols, begin, length);
386 
387  if (types[i].IsAssociationType)
388  {
389  IAssociationType associationType = types[i] as IAssociationType;
390

395  bool[] propertyNullability = compositeType.PropertyNullability;
396 
397  JoinType joinType =

399  propertyNullability == null || propertyNullability[i], currentDepth, compositeType.GetCascadeStyle(i));

631  protected JoinFragment MergeOuterJoins(IList<OuterJoinableAssociation> associations)
632  {
633  IList<OuterJoinableAssociation> sortedAssociations = new List<OuterJoinableAssociation>();
634 
635  var indices = GetTopologicalSortOrder(_dependentAliases);
636  for (int index = indices.Length - 1; index >= 0; index--)
637  {
638  sortedAssociations.Add(associations[indices[index]]);
639  }
640 
641  JoinFragment outerjoin = Dialect.CreateOuterJoinFragment();
642 
643  OuterJoinableAssociation last = null;
644  foreach (OuterJoinableAssociation oj in sortedAssociations)
645  {
646  if (last != null && last.IsManyToManyWith(oj))
647  {
648  oj.AddManyToManyJoin(outerjoin, (IQueryableCollection) last.Joinable);
649  }
650  else
651  {
652  oj.AddJoins(outerjoin);
653  // NH Different behavior : NH1179 and NH1293
654  // Apply filters in Many-To-One association
655  if (enabledFiltersForManyToOne.Count > 0)
656  {
657  string manyToOneFilterFragment = oj.Joinable.FilterFragment(oj.RHSAlias, enabledFiltersForManyToOne);
658  bool joinClauseDoesNotContainsFilterAlready =
659  outerjoin.ToFromFragmentString.IndexOfCaseInsensitive(manyToOneFilterFragment) == -1;
660  if (joinClauseDoesNotContainsFilterAlready)
661  {
662  outerjoin.AddCondition(manyToOneFilterFragment);
663  }
664  }
665  }
666  last = oj;
667  }
668 
669  return outerjoin;
670  }

762  protected SqlStringBuilder WhereString(string alias, string[] columnNames, int batchSize)
763  {
764  if (columnNames.Length == 1)
765  {
766  // if not a composite key, use "foo in (?, ?, ?)" for batching
767  // if no batch, and not a composite key, use "foo = ?"
768  string tableAlias = GenerateAliasForColumn(alias, columnNames[0]);
769  InFragment inf = new InFragment().SetColumn(tableAlias, columnNames[0]);
770 
771  for (int i = 0; i < batchSize; i++)
772  inf.AddValue(Parameter.Placeholder);
773 
774  return new SqlStringBuilder(inf.ToFragmentString());
775  }
776  else
777  {
778  var fragments = new ConditionalFragment[batchSize];
779  for (int i = 0; i < batchSize; i++)
780  {
781  fragments[i] = new ConditionalFragment()
782  .SetTableAlias(alias)
783  .SetCondition(columnNames, Parameter.GenerateParameters(columnNames.Length));
784  }
785 
786  var whereString = new SqlStringBuilder();
787 
788  if (fragments.Length == 1)
789  {
790  // if no batch, use "foo = ? and bar = ?"
791  whereString.Add(fragments[0].ToSqlStringFragment());
792  }
793  else
794  {
795  // if batching, use "( (foo = ? and bar = ?) or (foo = ? and bar = ?) )"
796  var df = new DisjunctionFragment(fragments);
797
