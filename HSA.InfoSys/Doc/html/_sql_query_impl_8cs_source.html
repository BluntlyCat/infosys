270  protected internal override void VerifyParameters()
271  {
272  base.VerifyParameters();
273  bool noReturns = queryReturns == null || queryReturns.Count == 0;
274  if (noReturns)
275  {
276  autoDiscoverTypes = noReturns;
277  }
278  else
279  {
280  foreach (INativeSQLQueryReturn rtn in queryReturns)
281  {
282  if (rtn is NativeSQLQueryScalarReturn)
283  {
284  NativeSQLQueryScalarReturn scalar = (NativeSQLQueryScalarReturn) rtn;
285  if (scalar.Type == null)
286  {
287  autoDiscoverTypes = true;
