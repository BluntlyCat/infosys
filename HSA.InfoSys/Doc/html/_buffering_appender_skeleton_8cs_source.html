302  public virtual void Flush(bool flushLossyBuffer)
303  {
304  // This method will be called outside of the AppenderSkeleton DoAppend() method
305  // therefore it needs to be protected by its own lock. This will block any
306  // Appends while the buffer is flushed.
307  lock(this)
308  {
309  if (m_cb != null && m_cb.Length > 0)
310  {
311  if (m_lossy)
312  {
313  // If we are allowed to eagerly flush from the lossy buffer
314  if (flushLossyBuffer)
315  {
316  if (m_lossyEvaluator != null)
317  {
318  // Test the contents of the buffer against the lossy evaluator
319  LoggingEvent[] bufferedEvents = m_cb.PopAll();
320  ArrayList filteredEvents = new ArrayList(bufferedEvents.Length);
321 
322  foreach(LoggingEvent loggingEvent in bufferedEvents)
323  {
324  if (m_lossyEvaluator.IsTriggeringEvent(loggingEvent))
325  {
326  filteredEvents.Add(loggingEvent);
327  }
328  }
329 
330  // Send the events that meet the lossy evaluator criteria
331  if (filteredEvents.Count > 0)
332  {
333  SendBuffer((LoggingEvent[])filteredEvents.ToArray(typeof(LoggingEvent)));
334  }
335  }
336  else
337  {
338  // No lossy evaluator, all buffered events are discarded
339  m_cb.Clear();
340  }
341  }
342  }
343  else
344  {
345  // Not lossy, send whole buffer
346  SendFromBuffer(null, m_cb);
347  }
348  }
349  }
350  }

444  override protected void Append(LoggingEvent loggingEvent)
445  {
446  // If the buffer size is set to 1 or less then the buffer will be
447  // sent immediately because there is not enough space in the buffer
448  // to buffer up more than 1 event. Therefore as a special case
449  // we don't use the buffer at all.
450  if (m_cb == null || m_bufferSize <= 1)
451  {
452  // Only send the event if we are in non lossy mode or the event is a triggering event
453  if ((!m_lossy) ||
454  (m_evaluator != null && m_evaluator.IsTriggeringEvent(loggingEvent)) ||
455  (m_lossyEvaluator != null && m_lossyEvaluator.IsTriggeringEvent(loggingEvent)))
456  {
457  if (m_eventMustBeFixed)
458  {
459  // Derive class expects fixed events
460  loggingEvent.Fix = this.Fix;
461  }
462 
463  // Not buffering events, send immediately
464  SendBuffer(new LoggingEvent[] { loggingEvent } );
465  }
466  }
467  else
468  {
469  // Because we are caching the LoggingEvent beyond the
470  // lifetime of the Append() method we must fix any
471  // volatile data in the event.
472  loggingEvent.Fix = this.Fix;
473 
474  // Add to the buffer, returns the event discarded from the buffer if there is no space remaining after the append
475  LoggingEvent discardedLoggingEvent = m_cb.Append(loggingEvent);
476 
477  if (discardedLoggingEvent != null)
478  {
479  // Buffer is full and has had to discard an event
480  if (!m_lossy)
481  {
482  // Not lossy, must send all events
483  SendFromBuffer(discardedLoggingEvent, m_cb);
484  }
485  else
486  {
487  // Check if the discarded event should not be logged
488  if (m_lossyEvaluator == null || !m_lossyEvaluator.IsTriggeringEvent(discardedLoggingEvent))
489  {
490  // Clear the discarded event as we should not forward it
491  discardedLoggingEvent = null;
492  }
493 
494  // Check if the event should trigger the whole buffer to be sent
495  if (m_evaluator != null && m_evaluator.IsTriggeringEvent(loggingEvent))
496  {
497  SendFromBuffer(discardedLoggingEvent, m_cb);
498  }
499  else if (discardedLoggingEvent != null)
500  {
501  // Just send the discarded event
502  SendBuffer(new LoggingEvent[] { discardedLoggingEvent } );
503  }
504  }
505  }
506  else
507  {
508  // Buffer is not yet full
509 
510  // Check if the event should trigger the whole buffer to be sent
511  if (m_evaluator != null && m_evaluator.IsTriggeringEvent(loggingEvent))
512  {
513  SendFromBuffer(null, m_cb);
514  }
515  }
516  }
517  }

533  virtual protected void SendFromBuffer(LoggingEvent firstLoggingEvent, CyclicBuffer buffer)
534  {
535  LoggingEvent[] bufferEvents = buffer.PopAll();
536 
537  if (firstLoggingEvent == null)
538  {
539  SendBuffer(bufferEvents);
540  }
541  else if (bufferEvents.Length == 0)
542  {
543  SendBuffer(new LoggingEvent[] { firstLoggingEvent } );
544  }
545  else
546  {
547  // Create new array with the firstLoggingEvent at the head
548  LoggingEvent[] events = new LoggingEvent[bufferEvents.Length + 1];
549  Array.Copy(bufferEvents, 0, events, 1, bufferEvents.Length);
550  events[0] = firstLoggingEvent;
551 
552  SendBuffer(events);
553  }
554  }
