43  public override void WriteJson(JsonWriter writer, object value, JsonSerializer serializer)
44  {
45  if (value == null)
46  {
47  writer.WriteNull();
48  }
49  else if (value is Version)
50  {
51  writer.WriteValue(value.ToString());
52  }
53  else
54  {
55  throw new JsonSerializationException("Expected Version object value");
56  }
57  }

67  public override object ReadJson(JsonReader reader, Type objectType, object existingValue, JsonSerializer serializer)
68  {
69  if (reader.TokenType == JsonToken.Null)
70  {
71  return null;
72  }
73  else
74  {
75  if (reader.TokenType == JsonToken.String)
76  {
77  try
78  {
79  Version v = new Version((string) reader.Value);
80  return v;
81  }
82  catch (Exception ex)
83  {
84  throw JsonSerializationException.Create(reader, "Error parsing version string: {0}".FormatWith(CultureInfo.InvariantCulture, reader.Value), ex);
85  }
86  }
87  else
88  {
89  throw JsonSerializationException.Create(reader, "Unexpected token or value when parsing version. Token: {0}, Value: {1}".FormatWith(CultureInfo.InvariantCulture, reader.TokenType, reader.Value));
90  }
91  }
92  }
