172  private string InvokeSolrQuery(string solrQuery)
173  {
174  var url = string.Format("http://{0}:{1}{2}", this.Host, this.Port, solrQuery);
175  var request = WebRequest.Create(url);
176 
177  request.Method = "GET";
178 
179  var response = request.GetResponse();
180  var data = response.GetResponseStream();
181 
182  if (data != null)
183  {
184  var reader = new StreamReader(data);
185 
186  var responseFromSolr = reader.ReadToEnd();
187 
188  reader.Close();
189  data.Close();
190  response.Close();
191 
192  return responseFromSolr;
193  }
194 
195  return string.Empty;
196  }

231  private SolrResultPot ParseToResult(string jsonResult)
232  {
233  var json = JsonConvert.DeserializeObject(jsonResult) as JObject;
234 
235  var resultPot = new SolrResultPot(this.ComponentGUID);
236 
237  if (json != null)
238  {
239  var response = json["response"];
240  var docs = response["docs"];
241 
242  foreach (var doc in docs)
243  {
244  var result = new Result();
245 
246  var content = this.GetJsonValue(doc, "content").Replace(".", ".\r\n");
247 
248  result.ContentHash = content.GetHashCode();
249  result.Content = content.Length > 300 ? string.Format(
250  "{0}...",
251  content.Substring(0, 300)) : content;
252 
253  result.URL = this.GetJsonValue(doc, "url");
254  result.Title = this.RemoveSpecialChars(this.GetJsonValue(doc, "title"));
255  result.ComponentGUID = this.ComponentGUID;
256 
257  try
258  {
259  result.Time = DateTime.Parse(this.GetJsonValue(doc, "tstamp"));
260  }
261  catch
262  {
263  result.Time = DateTime.Now;
264  }
265 
266  resultPot.Results.Add(result);
267 
268  Log.InfoFormat("Search resualt was added [{0}]", result.Title);
269  }
270  }
271 
272  return resultPot;
273  }

304  private string RemoveSpecialChars(string text)

306  string tmp = string.Empty;
307 
308  text = text.Replace("\r", string.Empty);
