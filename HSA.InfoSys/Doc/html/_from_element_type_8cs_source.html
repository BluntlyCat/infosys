136  public string RenderIdentifierSelect(int size, int k)
137  {
138  CheckInitialized();
139 
140  // Render the identifier select fragment using the table alias.
141  if (_fromElement.FromClause.IsSubQuery)
142  {
143  var queryable = Queryable;
144  if (queryable == null)
145  return string.Empty;
146 
147  // TODO: Replace this with a more elegant solution.
148  string[] idColumnNames = queryable.IdentifierColumnNames;
149 
150  var buf = new StringBuilder();
151  for (int i = 0; i < idColumnNames.Length; i++)
152  {
153  buf.Append(_fromElement.TableAlias).Append('.').Append(idColumnNames[i]);
154  if (i != idColumnNames.Length - 1) buf.Append(", ");
155  }
156  return buf.ToString();
157  }
158  else
159  {
160  var queryable = Queryable;
161  if (queryable == null)
162  throw new QueryException("not an entity");
163 
164  string fragment = queryable.IdentifierSelectFragment(TableAlias, GetSuffix(size, k));
165  return TrimLeadingCommaAndSpaces(fragment);
166  }

257  {

259

262  // Not a collection?
263  return (IPropertyMapping)_persister; // Return the entity property mapping.

266  // If the property is a special collection property name, return a CollectionPropertyMapping.

268  {
269  if (_collectionPropertyMapping == null)

273  return _collectionPropertyMapping;

300  public virtual IType GetPropertyType(string propertyName, string propertyPath)
301  {
302  CheckInitialized();
303 
304  IType type = null;
305 
306  // If this is an entity and the property is the identifier property, then use getIdentifierType().
307  // Note that the propertyName.equals( propertyPath ) checks whether we have a component
308  // key reference, where the component class property name is the same as the
309  // entity id property name; if the two are not equal, this is the case and
310  // we'd need to "fall through" to using the property mapping.
311  if (_persister != null && (propertyName == propertyPath) && propertyName == _persister.IdentifierPropertyName)
312  {
313  type = _persister.IdentifierType;
314  }
315  else
316  { // Otherwise, use the property mapping.
317  IPropertyMapping mapping = GetPropertyMapping(propertyName);
318  type = mapping.ToType(propertyPath);
319  }
320  if (type == null)
321  {
322  throw new MappingException("Property " + propertyName + " does not exist in " +
323  ((_queryableCollection == null) ? "class" : "collection") + " "
324  + ((_queryableCollection == null) ? _fromElement.ClassName : _queryableCollection.Role));
325  }
326  return type;
327  }

360  public string[] ToColumns(string tableAlias, string path, bool inSelect, bool forceAlias)
361  {
362  CheckInitialized();
363  IPropertyMapping propertyMapping = GetPropertyMapping(path);
364 
365  // If this from element is a collection and the path is a collection property (maxIndex, etc.) then
366  // generate a sub-query.
367  if (!inSelect && _queryableCollection != null && CollectionProperties.IsCollectionProperty(path))
368  {
369  IDictionary<string, IFilter> enabledFilters = _fromElement.Walker.EnabledFilters;
370 
371  string subquery = CollectionSubqueryFactory.CreateCollectionSubquery(
372  _joinSequence,
373  enabledFilters,
374  propertyMapping.ToColumns(tableAlias, path)
375  );
376  if (Log.IsDebugEnabled)
377  {
378  Log.Debug("toColumns(" + tableAlias + "," + path + ") : subquery = " + subquery);
379  }
380  return new [] { "(" + subquery + ")" };
381  }
382  else
383  {
384  if (forceAlias)
385  {
386  return propertyMapping.ToColumns(tableAlias, path);
387  }
388  else if (_fromElement.Walker.StatementType == HqlSqlWalker.SELECT)
389  {
390  return propertyMapping.ToColumns(tableAlias, path);
391  }
392  else if (_fromElement.Walker.CurrentClauseType == HqlSqlWalker.SELECT)
393  {
394  return propertyMapping.ToColumns(tableAlias, path);
395  }
396  else if (_fromElement.Walker.IsSubQuery)
397  {
398  // for a subquery, the alias to use depends on a few things (we
399  // already know this is not an overall SELECT):
400  // 1) if this FROM_ELEMENT represents a correlation to the
401  // outer-most query
402  // A) if the outer query represents a multi-table
403  // persister, we need to use the given alias
404  // in anticipation of one of the multi-table
405  // executors being used (as this subquery will
406  // actually be used in the "id select" phase
407  // of that multi-table executor)
408  // B) otherwise, we need to use the persister's
409  // table name as the column qualification
410  // 2) otherwise (not correlated), use the given alias
411  if (IsCorrelation)
412  {
413  if (IsMultiTable)
414  {
415  return propertyMapping.ToColumns(tableAlias, path);
416  }
417  else
418  {
419  return propertyMapping.ToColumns(ExtractTableName(), path);
420  }
421  }
422  else
423  {
424  return propertyMapping.ToColumns(tableAlias, path);
425  }
426  }
427  else
428  {
429  string[] columns = propertyMapping.ToColumns(path);
430  Log.Info("Using non-qualified column reference [" + path + " -> (" + ArrayHelper.ToString(columns) + ")]");
431  return columns;
432  }
433  }
