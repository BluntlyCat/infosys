120  public override int Read(byte[] b, int off, int len)
121  {
122  if (len==0)
123  return 0;
124 
125  z.next_out = b;
126  z.next_out_index = off;
127  z.avail_out = len;
128 
129  int err;
130  do
131  {
132  if (z.avail_in == 0 && !nomoreinput)
133  {
134  // if buffer is empty and more input is available, refill it
135  z.next_in_index = 0;
136  z.avail_in = input.Read(buf, 0, buf.Length); //(bufsize<z.avail_out ? bufsize : z.avail_out));
137 
138  if (z.avail_in <= 0)
139  {
140  z.avail_in = 0;
141  nomoreinput = true;
142  }
143  }
144 
145  err = compress
146  ? z.deflate(flushLevel)
147  : z.inflate(flushLevel);
148 
149  if (nomoreinput && err == JZlib.Z_BUF_ERROR)
150  return 0;
151  if (err != JZlib.Z_OK && err != JZlib.Z_STREAM_END)
152  // TODO
153 // throw new ZStreamException((compress ? "de" : "in") + "flating: " + z.msg);
154  throw new IOException((compress ? "de" : "in") + "flating: " + z.msg);
155  if ((nomoreinput || err == JZlib.Z_STREAM_END) && z.avail_out == len)
156  return 0;
157  }
158  while(z.avail_out == len && err == JZlib.Z_OK);
159  //Console.Error.WriteLine("("+(len-z.avail_out)+")");
160  return len - z.avail_out;
161  }
