276  override public void ActivateOptions()
277  {
278  try
279  {
280  base.ActivateOptions();
281 
282  if (m_securityContext == null)
283  {
284  m_securityContext = SecurityContextProvider.DefaultProvider.CreateSecurityContext(this);
285  }
286 
287  bool sourceAlreadyExists = false;
288  string currentLogName = null;
289 
290  using (SecurityContext.Impersonate(this))
291  {
292  sourceAlreadyExists = EventLog.SourceExists(m_applicationName);
293  if (sourceAlreadyExists) {
294  currentLogName = EventLog.LogNameFromSourceName(m_applicationName, m_machineName);
295  }
296  }
297 
298  if (sourceAlreadyExists && currentLogName != m_logName)
299  {
300  LogLog.Debug(declaringType, "Changing event source [" + m_applicationName + "] from log [" + currentLogName + "] to log [" + m_logName + "]");
301  }
302  else if (!sourceAlreadyExists)
303  {
304  LogLog.Debug(declaringType, "Creating event source Source [" + m_applicationName + "] in log " + m_logName + "]");
305  }
306 
307  string registeredLogName = null;
308 
309  using (SecurityContext.Impersonate(this))
310  {
311  if (sourceAlreadyExists && currentLogName != m_logName)
312  {
313  //
314  // Re-register this to the current application if the user has changed
315  // the application / logfile association
316  //
317  EventLog.DeleteEventSource(m_applicationName, m_machineName);
318  CreateEventSource(m_applicationName, m_logName, m_machineName);
319 
320  registeredLogName = EventLog.LogNameFromSourceName(m_applicationName, m_machineName);
321  }
322  else if (!sourceAlreadyExists)
323  {
324  CreateEventSource(m_applicationName, m_logName, m_machineName);
325 
326  registeredLogName = EventLog.LogNameFromSourceName(m_applicationName, m_machineName);
327  }
328  }
329 
330  m_levelMapping.ActivateOptions();
331 
332  LogLog.Debug(declaringType, "Source [" + m_applicationName + "] is registered to log [" + registeredLogName + "]");
333  }
334  catch (System.Security.SecurityException ex)
335  {
336  ErrorHandler.Error("Caught a SecurityException trying to access the EventLog. Most likely the event source "
337  + m_applicationName
338  + " doesn't exist and must be created by a local administrator. Will disable EventLogAppender."
339  + " See http://logging.apache.org/log4net/release/faq.html#trouble-EventLog",
340  ex);
341  Threshold = Level.Off;
342  }
343  }

382  override protected void Append(LoggingEvent loggingEvent)
383  {
384  //
385  // Write the resulting string to the event log system
386  //
387  int eventID = m_eventId;
388 
389  // Look for the EventID property
390  object eventIDPropertyObj = loggingEvent.LookupProperty("EventID");
391  if (eventIDPropertyObj != null)
392  {
393  if (eventIDPropertyObj is int)
394  {
395  eventID = (int)eventIDPropertyObj;
396  }
397  else
398  {
399  string eventIDPropertyString = eventIDPropertyObj as string;
400  if (eventIDPropertyString == null)
401  {
402  eventIDPropertyString = eventIDPropertyObj.ToString();
403  }
404  if (eventIDPropertyString != null && eventIDPropertyString.Length > 0)
405  {
406  // Read the string property into a number
407  int intVal;
408  if (SystemInfo.TryParse(eventIDPropertyString, out intVal))
409  {
410  eventID = intVal;
411  }
412  else
413  {
414  ErrorHandler.Error("Unable to parse event ID property [" + eventIDPropertyString + "].");
415  }
416  }
417  }
418  }
419 
420  short category = m_category;
421  // Look for the Category property
422  object categoryPropertyObj = loggingEvent.LookupProperty("Category");
423  if (categoryPropertyObj != null)
424  {
425  if (categoryPropertyObj is short)
426  {
427  category = (short) categoryPropertyObj;
428  }
429  else
430  {
431  string categoryPropertyString = categoryPropertyObj as string;
432  if (categoryPropertyString == null)
433  {
434  categoryPropertyString = categoryPropertyObj.ToString();
435  }
436  if (categoryPropertyString != null && categoryPropertyString.Length > 0)
437  {
438  // Read the string property into a number
439  short shortVal;
440  if (SystemInfo.TryParse(categoryPropertyString, out shortVal))
441  {
442  category = shortVal;
443  }
444  else
445  {
446  ErrorHandler.Error("Unable to parse event category property [" + categoryPropertyString + "].");
447  }
448  }
449  }
450  }
451 
452  // Write to the event log
453  try
454  {
455  string eventTxt = RenderLoggingEvent(loggingEvent);
456 
457  // There is a limit of 32K characters for an event log message
458  if (eventTxt.Length > 32000)
459  {
460  eventTxt = eventTxt.Substring(0, 32000);
461  }
462 
463  EventLogEntryType entryType = GetEntryType(loggingEvent.Level);
464 
465  using(SecurityContext.Impersonate(this))
466  {
467  EventLog.WriteEntry(m_applicationName, eventTxt, entryType, eventID, category);
468  }
469  }
470  catch(Exception ex)
471  {
472  ErrorHandler.Error("Unable to write to event log [" + m_logName + "] using source [" + m_applicationName + "]", ex);
473  }
474  }

505  virtual protected EventLogEntryType GetEntryType(Level level)
506  {
507  // see if there is a specified lookup.
508  Level2EventLogEntryType entryType = m_levelMapping.Lookup(level) as Level2EventLogEntryType;
509  if (entryType != null)
510  {
511  return entryType.EventLogEntryType;
512  }
513 
514  // Use default behavior
515 
516  if (level >= Level.Error)
517  {
518  return EventLogEntryType.Error;
519  }
520  else if (level == Level.Warn)
521  {
522  return EventLogEntryType.Warning;
523  }
524 
525  // Default setting
526  return EventLogEntryType.Information;
527  }
