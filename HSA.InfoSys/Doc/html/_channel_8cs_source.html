421  protected void SendMessage(ChannelDataMessage message)
422  {
423  // Send channel messages only while channel is open
424  if (!this.IsOpen)
425  return;
426 
427  var messageLength = message.Data.Length;
428  do
429  {
430  lock (this._serverWindowSizeLock)
431  {
432  var serverWindowSize = this.ServerWindowSize;
433  if (serverWindowSize < messageLength)
434  {
435  // Wait for window to be big enough for this message
436  this._channelServerWindowAdjustWaitHandle.Reset();
437  }
438  else
439  {
440  this.ServerWindowSize -= (uint)messageLength;
441  break;
442  }
443  }
444 
445  // Wait for window to change
446  this.WaitHandle(this._channelServerWindowAdjustWaitHandle);
447 
448  } while (true);
449 
450  this._session.SendMessage(message);
451  }

499  protected virtual void Close(bool wait)
500  {
501  // Send message to close the channel on the server
502  // Ignore sending close message when client not connected
503  if (!_closeMessageSent && this.IsConnected)
504  {
505  lock (this)
506  {
507  if (!_closeMessageSent)
508  {
509  this.SendMessage(new ChannelCloseMessage(this.RemoteChannelNumber));
510  this._closeMessageSent = true;
511  }
512  }
513  }
514

518  this._session.WaitHandle(this._channelClosedWaitHandle);
