217  public void RemoveUnusedCommandParameters(IDbCommand cmd, SqlString sqlString)
218  {
219  if (!UseNamedPrefixInSql)
220  return; // Applicable only to named parameters
221 
222  var formatter = GetSqlStringFormatter();
223  formatter.Format(sqlString);
224  var assignedParameterNames = new HashSet<string>(formatter.AssignedParameterNames);
225 
226  cmd.Parameters
227  .Cast<IDbDataParameter>()
228  .Select(p => p.ParameterName)
229  .Where(p => !assignedParameterNames.Contains(UseNamedPrefixInParameter ? p : FormatNameForSql(p)))
230  .ToList()
231  .ForEach(unusedParameterName => cmd.Parameters.RemoveAt(unusedParameterName));
232  }
