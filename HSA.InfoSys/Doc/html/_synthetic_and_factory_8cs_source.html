32  JoinFragment joinFragment,
33  SqlString whereFragment,
34  QueryNode query,
35  FromElement fromElement,
36  HqlSqlWalker hqlSqlWalker)
37  {
38  if (whereFragment == null)
39  {
40  return;
41  }
42 
43  if (!fromElement.UseWhereFragment && !joinFragment.HasThetaJoins)
44  {
45  return;
46  }
47 
48  whereFragment = whereFragment.Trim();
49  if (StringHelper.IsEmpty(whereFragment.ToString()))
50  {
51  return;
52  }
53 
54  // Forcefully remove leading ands from where fragments; the grammar will
55  // handle adding them
56  if (whereFragment.StartsWithCaseInsensitive("and"))
57  {
58  whereFragment = whereFragment.Substring(4);
59  }
60 
61  log.Debug("Using unprocessed WHERE-fragment [" + whereFragment +"]");
62 
63  SqlFragment fragment = (SqlFragment) Create(HqlSqlWalker.SQL_TOKEN, whereFragment.ToString());
64 
65  fragment.SetJoinFragment(joinFragment);
66  fragment.FromElement = fromElement;
67 
68  if (fromElement.IndexCollectionSelectorParamSpec != null)
69  {
70  fragment.AddEmbeddedParameter(fromElement.IndexCollectionSelectorParamSpec);
71  fromElement.IndexCollectionSelectorParamSpec = null;
72  }
73 
74  if (hqlSqlWalker.IsFilter())
75  {
76  //if (whereFragment.IndexOfCaseInsensitive("?") >= 0)
77  if (whereFragment.ToString().IndexOf("?") >= 0)
78  {
79  IType collectionFilterKeyType = hqlSqlWalker.SessionFactoryHelper
80  .RequireQueryableCollection(hqlSqlWalker.CollectionFilterRole)
81  .KeyType;
82  CollectionFilterKeyParameterSpecification paramSpec = new CollectionFilterKeyParameterSpecification(
83  hqlSqlWalker.CollectionFilterRole,
84  collectionFilterKeyType,
85  0
86  );
87  fragment.AddEmbeddedParameter(paramSpec);
88  }
89  }
90 
91  JoinProcessor.ProcessDynamicFilterParameters(
92  whereFragment,
93  fragment,
94  hqlSqlWalker
95  );
96 
97  log.Debug("Using processed WHERE-fragment [" + fragment.Text + "]");
98 
99  // Filter conditions need to be inserted before the HQL where condition and the
100  // theta join node. This is because org.hibernate.loader.Loader binds the filter parameters first,
101  // then it binds all the HQL query parameters, see org.hibernate.loader.Loader.processFilterParameters().
102  if (fragment.FromElement.IsFilter || fragment.HasFilterCondition)
103  {
104  if (_filters == null)
105  {
106  // Find or create the WHERE clause
107  IASTNode where = (IASTNode) query.WhereClause;
108  // Create a new FILTERS node as a parent of all filters
109  _filters = Create(HqlSqlWalker.FILTERS, "{filter conditions}");
110  // Put the FILTERS node before the HQL condition and theta joins
111  where.InsertChild(0, _filters);
112  }
113 
114  // add the current fragment to the FILTERS node
115  _filters.AddChild(fragment);
116  }
117  else
118  {
119  if (_thetaJoins == null)
120  {
121  // Find or create the WHERE clause
122  IASTNode where = (IASTNode) query.WhereClause;
123 
124  // Create a new THETA_JOINS node as a parent of all filters
125  _thetaJoins = Create(HqlSqlWalker.THETA_JOINS, "{theta joins}");
126 
127  // Put the THETA_JOINS node before the HQL condition, after the filters.
128  if (_filters == null)
129  {
130  where.InsertChild(0, _thetaJoins);
131  }
132  else
133  {
134  _filters.AddSibling(_thetaJoins);
135  }
136  }
137 
138  // add the current fragment to the THETA_JOINS node
139  _thetaJoins.AddChild(fragment);
140  }
141  }

148  public virtual void AddDiscriminatorWhereFragment(IRestrictableStatement statement, IQueryable persister, IDictionary<string, IFilter> enabledFilters, string alias)
149  {
150  string whereFragment = persister.FilterFragment(alias, enabledFilters).Trim();
151  if (string.Empty.Equals(whereFragment))
152  {
153  return;
154  }
155  if (whereFragment.StartsWith("and"))
156  {
157  whereFragment = whereFragment.Substring(4);
158  }
159 
160  // Need to parse off the column qualifiers; this is assuming (which is true as of now)
161  // that this is only used from update and delete HQL statement parsing
162  whereFragment = StringHelper.Replace(whereFragment, persister.GenerateFilterConditionAlias(alias) + ".", "");
163 
164  // Note: this simply constructs a "raw" SQL_TOKEN representing the
165  // where fragment and injects this into the tree. This "works";
166  // however it is probably not the best long-term solution.
167  //
168  // At some point we probably want to apply an additional grammar to
169  // properly tokenize this where fragment into constituent parts
170  // focused on the operators embedded within the fragment.
171  IASTNode discrimNode = Create(HqlSqlWalker.SQL_TOKEN, whereFragment);
172 
173  if (statement.WhereClause.ChildCount == 0)
174  {
175  statement.WhereClause.SetFirstChild(discrimNode);
176  }
177  else
178  {
179  IASTNode and = Create(HqlSqlWalker.AND, "{and}");
180  IASTNode currentFirstChild = statement.WhereClause.GetFirstChild();
181  and.SetFirstChild(discrimNode);
182  and.AddChild(currentFirstChild);
183  statement.WhereClause.SetFirstChild(and);
184  }
185  }
