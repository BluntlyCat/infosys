28  {

31  IEntityPersister persister;

33  {

41

44

53  if (persister.IdentifierType.IsComponentType)
54  {
55  // skip this check for composite-ids relating to dom4j entity-mode;
56  // alternatively, we could add a check to make sure the incoming id value is
57  // an instance of Element...
58  }
59  else
60  {
61  System.Type idClass = persister.IdentifierType.ReturnedClass;
62  if (idClass != null && !idClass.IsInstanceOfType(@event.EntityId))
63  {
64  throw new TypeMismatchException("Provided id of the wrong type. Expected: " + idClass + ", got " + @event.EntityId.GetType());
65  }
66  }
67 
68  EntityKey keyToLoad = source.GenerateEntityKey(@event.EntityId, persister);
69  try
70  {
71  if (loadType.IsNakedEntityReturned)
72  {
73  //do not return a proxy!
74  //(this option indicates we are initializing a proxy)
75  @event.Result = Load(@event, persister, keyToLoad, loadType);
76  }
77  else
78  {
79  //return a proxy if appropriate
80  if (@event.LockMode == LockMode.None)

99  protected virtual object Load(LoadEvent @event, IEntityPersister persister, EntityKey keyToLoad, LoadType options)
100  {
101  if (@event.InstanceToLoad != null)
102  {
103  if (@event.Session.PersistenceContext.GetEntry(@event.InstanceToLoad) != null)
104  {
105  throw new PersistentObjectException("attempted to load into an instance that was already associated with the session: " + MessageHelper.InfoString(persister, @event.EntityId, @event.Session.Factory));
106  }
107  persister.SetIdentifier(@event.InstanceToLoad, @event.EntityId, @event.Session.EntityMode);
108  }
109 
110  object entity = DoLoad(@event, persister, keyToLoad, options);
111 
112  bool isOptionalInstance = @event.InstanceToLoad != null;
113 
114  if (!options.IsAllowNulls || isOptionalInstance)
115  {
116  if (entity == null)
117  {
118  @event.Session.Factory.EntityNotFoundDelegate.HandleEntityNotFound(@event.EntityClassName, @event.EntityId);
119  }
120  }
121 
122  if (isOptionalInstance && entity != @event.InstanceToLoad)
123  {
124  throw new NonUniqueObjectException(@event.EntityId, @event.EntityClassName);
125  }
126 
127  return entity;
128  }

135  protected virtual object ProxyOrLoad(LoadEvent @event, IEntityPersister persister, EntityKey keyToLoad, LoadType options)
136  {
137  if (log.IsDebugEnabled)

140  }
141 
142  if (!persister.HasProxy)
143  {
144  // this class has no proxies (so do a shortcut)
145  return Load(@event, persister, keyToLoad, options);

176  private object ReturnNarrowedProxy(LoadEvent @event, IEntityPersister persister, EntityKey keyToLoad, LoadType options, IPersistenceContext persistenceContext, object proxy)
177  {
178  log.Debug("entity proxy found in session cache");
179  var castedProxy = (INHibernateProxy) proxy;
180  ILazyInitializer li = castedProxy.HibernateLazyInitializer;
181  if (li.Unwrap)
182  {
183  return li.GetImplementation();
184  }
185  object impl = null;
186  if (!options.IsAllowProxyCreation)
187  {
188  impl = Load(@event, persister, keyToLoad, options);
189  // NH Different behavior : NH-1252
190  if (impl == null && !options.IsAllowNulls)
191  {
192  @event.Session.Factory.EntityNotFoundDelegate.HandleEntityNotFound(persister.EntityName, keyToLoad.Identifier);
193  }
194  }
195  if (impl == null && !options.IsAllowProxyCreation && options.ExactPersister)
196  {
197  // NH Different behavior : NH-1252
198  return null;
199  }
200  return persistenceContext.NarrowProxy(castedProxy, persister, keyToLoad, impl);

208  private object CreateProxyIfNecessary(LoadEvent @event, IEntityPersister persister, EntityKey keyToLoad, LoadType options, IPersistenceContext persistenceContext)
209  {
210  object existing = persistenceContext.GetEntity(keyToLoad);
211  if (existing != null)
212  {
213  // return existing object or initialized proxy (unless deleted)
214  log.Debug("entity found in session cache");
215  if (options.IsCheckDeleted)
216  {
217  EntityEntry entry = persistenceContext.GetEntry(existing);
218  Status status = entry.Status;
219  if (status == Status.Deleted || status == Status.Gone)
220  {
221  return null;
222  }
223  }
224  return existing;
225  }
226  else
227  {
228  log.Debug("creating new proxy for entity");
229  // return new uninitialized proxy
230  object proxy = persister.CreateProxy(@event.EntityId, @event.Session);
231  persistenceContext.BatchFetchQueue.AddBatchLoadableEntityKey(keyToLoad);
232  persistenceContext.AddProxy(keyToLoad, (INHibernateProxy)proxy);
233  ((INHibernateProxy)proxy)
234  .HibernateLazyInitializer
235  .ReadOnly = @event.Session.DefaultReadOnly || !persister.IsMutable;
236  return proxy;

378  {

456  private object AssembleCacheEntry(CacheEntry entry, object id, IEntityPersister persister, LoadEvent @event)
457  {
458  object optionalObject = @event.InstanceToLoad;
459  IEventSource session = @event.Session;
460  ISessionFactoryImplementor factory = session.Factory;
461 
462  if (log.IsDebugEnabled)
463  {
464  log.Debug("assembling entity from second-level cache: " + MessageHelper.InfoString(persister, id, factory));
465  }
466 
467  IEntityPersister subclassPersister = factory.GetEntityPersister(entry.Subclass);
468  object result = optionalObject ?? session.Instantiate(subclassPersister, id);
469 
470  // make it circular-reference safe
471  EntityKey entityKey = session.GenerateEntityKey(id, subclassPersister);
472  TwoPhaseLoad.AddUninitializedCachedEntity(entityKey, result, subclassPersister, LockMode.None, entry.AreLazyPropertiesUnfetched, entry.Version, session);
473 
474  IType[] types = subclassPersister.PropertyTypes;
475  object[] values = entry.Assemble(result, id, subclassPersister, session.Interceptor, session); // intializes result by side-effect
476  TypeHelper.DeepCopy(values, types, subclassPersister.PropertyUpdateability, values, session);
477 
478  object version = Versioning.GetVersion(values, subclassPersister);
479  if (log.IsDebugEnabled)
480  {
481  log.Debug("Cached Version: " + version);
482  }
483 
484  IPersistenceContext persistenceContext = session.PersistenceContext;
485  bool isReadOnly = session.DefaultReadOnly;
486 
487  if (persister.IsMutable)
488  {
489  object proxy = persistenceContext.GetProxy(entityKey);
490  if (proxy != null)
491  {
492  // this is already a proxy for this impl
493  // only set the status to read-only if the proxy is read-only
494  isReadOnly = ((INHibernateProxy)proxy).HibernateLazyInitializer.ReadOnly;
495  }
496  }
497  else
498  isReadOnly = true;
499 
500  persistenceContext.AddEntry(
501  result,
502  isReadOnly ? Status.ReadOnly : Status.Loaded,
503  values,
504  null,
505  id,
506  version,
507  LockMode.None,
508  true,
509  subclassPersister,
510  false,
511  entry.AreLazyPropertiesUnfetched);
512 
513  subclassPersister.AfterInitialize(result, entry.AreLazyPropertiesUnfetched, session);
514  persistenceContext.InitializeNonLazyCollections();
515  // upgrade the lock if necessary:
516  //lock(result, lockMode);
517 
518  //PostLoad is needed for EJB3
519  //TODO: reuse the PostLoadEvent...
520  PostLoadEvent postLoadEvent = new PostLoadEvent(session);
521  postLoadEvent.Entity = result;
522  postLoadEvent.Id = id;
523  postLoadEvent.Persister = persister;
524

526  for (int i = 0; i < listeners.Length; i++)

528  listeners[i].OnPostLoad(postLoadEvent);
