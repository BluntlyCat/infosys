151  static private void InternalConfigure(ILoggerRepository repository)
152  {
153  LogLog.Debug(declaringType, "configuring repository [" + repository.Name + "] using .config file section");
154 
155  try
156  {
157  LogLog.Debug(declaringType, "Application config file is [" + SystemInfo.ConfigurationFileLocation + "]");
158  }
159  catch
160  {
161  // ignore error
162  LogLog.Debug(declaringType, "Application config file location unknown");
163  }
164 
165 #if NETCF
166  // No config file reading stuff. Just go straight for the file
167  Configure(repository, new FileInfo(SystemInfo.ConfigurationFileLocation));
168 #else
169  try
170  {
171  XmlElement configElement = null;
172 #if NET_2_0
173  configElement = System.Configuration.ConfigurationManager.GetSection("log4net") as XmlElement;
174 #else
175  configElement = System.Configuration.ConfigurationSettings.GetConfig("log4net") as XmlElement;
176 #endif
177  if (configElement == null)
178  {
179  // Failed to load the xml config using configuration settings handler
180  LogLog.Error(declaringType, "Failed to find configuration section 'log4net' in the application's .config file. Check your .config file for the <log4net> and <configSections> elements. The configuration section should look like: <section name=\"log4net\" type=\"log4net.Config.Log4NetConfigurationSectionHandler,log4net\" />");
181  }
182  else
183  {
184  // Configure using the xml loaded from the config file
185  InternalConfigureFromXml(repository, configElement);
186  }
187  }
188  catch(System.Configuration.ConfigurationException confEx)
189  {
190  if (confEx.BareMessage.IndexOf("Unrecognized element") >= 0)
191  {
192  // Looks like the XML file is not valid
193  LogLog.Error(declaringType, "Failed to parse config file. Check your .config file is well formed XML.", confEx);
194  }
195  else
196  {
197  // This exception is typically due to the assembly name not being correctly specified in the section type.
198  string configSectionStr = "<section name=\"log4net\" type=\"log4net.Config.Log4NetConfigurationSectionHandler," + Assembly.GetExecutingAssembly().FullName + "\" />";
199  LogLog.Error(declaringType, "Failed to parse config file. Is the <configSections> specified as: " + configSectionStr, confEx);
200  }
201  }
202 #endif
203  }

522  static private void InternalConfigure(ILoggerRepository repository, FileInfo configFile)
523  {
524  LogLog.Debug(declaringType, "configuring repository [" + repository.Name + "] using file [" + configFile + "]");
525 
526  if (configFile == null)
527  {
528  LogLog.Error(declaringType, "Configure called with null 'configFile' parameter");
529  }
530  else
531  {
532  // Have to use File.Exists() rather than configFile.Exists()
533  // because configFile.Exists() caches the value, not what we want.
534  if (File.Exists(configFile.FullName))
535  {
536  // Open the file for reading
537  FileStream fs = null;
538 
539  // Try hard to open the file
540  for(int retry = 5; --retry >= 0; )
541  {
542  try
543  {
544  fs = configFile.Open(FileMode.Open, FileAccess.Read, FileShare.Read);
545  break;
546  }
547  catch(IOException ex)
548  {
549  if (retry == 0)
550  {
551  LogLog.Error(declaringType, "Failed to open XML config file [" + configFile.Name + "]", ex);
552 
553  // The stream cannot be valid
554  fs = null;
555  }
556  System.Threading.Thread.Sleep(250);
557  }
558  }
559 
560  if (fs != null)
561  {
562  try
563  {
564  // Load the configuration from the stream
565  InternalConfigure(repository, fs);
566  }
567  finally
568  {
569  // Force the file closed whatever happens
570  fs.Close();
571  }
572  }
573  }
574  else
575  {
576  LogLog.Debug(declaringType, "config file [" + configFile.FullName + "] not found. Configuration unchanged.");
577  }
578  }
579  }

611  static private void InternalConfigure(ILoggerRepository repository, Uri configUri)
612  {
613  LogLog.Debug(declaringType, "configuring repository [" + repository.Name + "] using URI ["+configUri+"]");
614 
615  if (configUri == null)
616  {
617  LogLog.Error(declaringType, "Configure called with null 'configUri' parameter");
618  }
619  else
620  {
621  if (configUri.IsFile)
622  {
623  // If URI is local file then call Configure with FileInfo
624  InternalConfigure(repository, new FileInfo(configUri.LocalPath));
625  }
626  else
627  {
628  // NETCF dose not support WebClient
629  WebRequest configRequest = null;
630 
631  try
632  {
633  configRequest = WebRequest.Create(configUri);
634  }
635  catch(Exception ex)
636  {
637  LogLog.Error(declaringType, "Failed to create WebRequest for URI ["+configUri+"]", ex);
638  }
639 
640  if (configRequest != null)
641  {
642 #if !NETCF_1_0
643  // authentication may be required, set client to use default credentials
644  try
645  {
646  configRequest.Credentials = CredentialCache.DefaultCredentials;
647  }
648  catch
649  {
650  // ignore security exception
651  }
652 #endif
653  try
654  {
655  WebResponse response = configRequest.GetResponse();
656  if (response != null)
657  {
658  try
659  {
660  // Open stream on config URI
661  using(Stream configStream = response.GetResponseStream())
662  {
663  InternalConfigure(repository, configStream);
664  }
665  }
666  finally
667  {
668  response.Close();
669  }
670  }
671  }
672  catch(Exception ex)
673  {
674  LogLog.Error(declaringType, "Failed to request config from URI ["+configUri+"]", ex);
675  }
676  }
677  }
678  }

711  static private void InternalConfigure(ILoggerRepository repository, Stream configStream)
712  {
713  LogLog.Debug(declaringType, "configuring repository [" + repository.Name + "] using stream");
714 
715  if (configStream == null)
716  {
717  LogLog.Error(declaringType, "Configure called with null 'configStream' parameter");
718  }
719  else
720  {
721  // Load the config file into a document
722  XmlDocument doc = new XmlDocument();
723  try
724  {
725 #if (NETCF)
726  // Create a text reader for the file stream
727  XmlTextReader xmlReader = new XmlTextReader(configStream);
728 #elif NET_2_0
729  // Allow the DTD to specify entity includes
730  XmlReaderSettings settings = new XmlReaderSettings();
731  // .NET 4.0 warning CS0618: 'System.Xml.XmlReaderSettings.ProhibitDtd'
732  // is obsolete: 'Use XmlReaderSettings.DtdProcessing property instead.'
733 #if !NET_4_0
734  settings.ProhibitDtd = false;
735 #else
736  settings.DtdProcessing = DtdProcessing.Parse;
737 #endif
738 
739  // Create a reader over the input stream
740  XmlReader xmlReader = XmlReader.Create(configStream, settings);
741 #else
742  // Create a validating reader around a text reader for the file stream
743  XmlValidatingReader xmlReader = new XmlValidatingReader(new XmlTextReader(configStream));
744 
745  // Specify that the reader should not perform validation, but that it should
746  // expand entity refs.
747  xmlReader.ValidationType = ValidationType.None;
748  xmlReader.EntityHandling = EntityHandling.ExpandEntities;
749 #endif
750 
751  // load the data into the document
752  doc.Load(xmlReader);
753  }
754  catch(Exception ex)
755  {
756  LogLog.Error(declaringType, "Error while loading XML configuration", ex);
757 
758  // The document is invalid
759  doc = null;
760  }
761 
762  if (doc != null)
763  {
764  LogLog.Debug(declaringType, "loading XML configuration");
765 
766  // Configure using the 'log4net' element
767  XmlNodeList configNodeList = doc.GetElementsByTagName("log4net");
768  if (configNodeList.Count == 0)
769  {
770  LogLog.Debug(declaringType, "XML configuration does not contain a <log4net> element. Configuration Aborted.");
771  }
772  else if (configNodeList.Count > 1)
773  {
774  LogLog.Error(declaringType, "XML configuration contains [" + configNodeList.Count + "] <log4net> elements. Only one is allowed. Configuration Aborted.");
775  }
776  else
777  {
778  InternalConfigureFromXml(repository, configNodeList[0] as XmlElement);
779  }
780  }
781  }

864  static private void InternalConfigureAndWatch(ILoggerRepository repository, FileInfo configFile)
865  {
866  LogLog.Debug(declaringType, "configuring repository [" + repository.Name + "] using file [" + configFile + "] watching for file updates");
867 
868  if (configFile == null)
869  {
870  LogLog.Error(declaringType, "ConfigureAndWatch called with null 'configFile' parameter");
871  }
872  else
873  {
874  // Configure log4net now
875  InternalConfigure(repository, configFile);
876 
877  try
878  {
879  lock (m_repositoryName2ConfigAndWatchHandler)
880  {
881  // support multiple repositories each having their own watcher
882  ConfigureAndWatchHandler handler =
883  (ConfigureAndWatchHandler)m_repositoryName2ConfigAndWatchHandler[repository.Name];
884 
885  if (handler != null)
886  {
887  m_repositoryName2ConfigAndWatchHandler.Remove(repository.Name);
888  handler.Dispose();
889  }
890 
891  // Create and start a watch handler that will reload the
892  // configuration whenever the config file is modified.
893  handler = new ConfigureAndWatchHandler(repository, configFile);
894  m_repositoryName2ConfigAndWatchHandler[repository.Name] = handler;
895  }
896  }
897  catch(Exception ex)
898  {
899  LogLog.Error(declaringType, "Failed to initialize configuration file watcher for file ["+configFile.FullName+"]", ex);
900  }
901  }
902  }

967  [System.Security.SecuritySafeCritical]

969  public ConfigureAndWatchHandler(ILoggerRepository repository, FileInfo configFile)
970  {
971  m_repository = repository;
972  m_configFile = configFile;
973 
974  // Create a new FileSystemWatcher and set its properties.
975  m_watcher = new FileSystemWatcher();
976 
977  m_watcher.Path = m_configFile.DirectoryName;
978  m_watcher.Filter = m_configFile.Name;
979 
980  // Set the notification filters
981  m_watcher.NotifyFilter = NotifyFilters.CreationTime | NotifyFilters.LastWrite | NotifyFilters.FileName;
982 
983  // Add event handlers. OnChanged will do for all event handlers that fire a FileSystemEventArgs
984  m_watcher.Changed += new FileSystemEventHandler(ConfigureAndWatchHandler_OnChanged);
985  m_watcher.Created += new FileSystemEventHandler(ConfigureAndWatchHandler_OnChanged);
986  m_watcher.Deleted += new FileSystemEventHandler(ConfigureAndWatchHandler_OnChanged);
987  m_watcher.Renamed += new RenamedEventHandler(ConfigureAndWatchHandler_OnRenamed);
988 
989  // Begin watching.
990  m_watcher.EnableRaisingEvents = true;
991 
992  // Create the timer that will be used to deliver events. Set as disabled
993  m_timer = new Timer(new TimerCallback(OnWatchedFileChange), null, Timeout.Infinite, Timeout.Infinite);
994  }

1077  static private void InternalConfigureFromXml(ILoggerRepository repository, XmlElement element)
1078  {
1079  if (element == null)
1080  {
1081  LogLog.Error(declaringType, "ConfigureFromXml called with null 'element' parameter");
1082  }
1083  else if (repository == null)
1084  {
1085  LogLog.Error(declaringType, "ConfigureFromXml called with null 'repository' parameter");
1086  }
1087  else

1089  LogLog.Debug(declaringType, "Configuring Repository [" + repository.Name + "]");
1090 
1091  IXmlRepositoryConfigurator configurableRepository = repository as IXmlRepositoryConfigurator;
1092  if (configurableRepository == null)

1099  // this isolates the xml config data from the rest of

1103 
1104  // Pass the configurator the config element
