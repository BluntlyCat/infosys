218  override protected void FormatXml(XmlWriter writer, LoggingEvent loggingEvent)
219  {
220  writer.WriteStartElement(m_elmEvent);
221  writer.WriteAttributeString(ATTR_LOGGER, loggingEvent.LoggerName);
222 
223 #if NET_2_0 || NETCF_2_0 || MONO_2_0
224  writer.WriteAttributeString(ATTR_TIMESTAMP, XmlConvert.ToString(loggingEvent.TimeStamp, XmlDateTimeSerializationMode.Local));
225 #else
226  writer.WriteAttributeString(ATTR_TIMESTAMP, XmlConvert.ToString(loggingEvent.TimeStamp));
227 #endif
228 
229  writer.WriteAttributeString(ATTR_LEVEL, loggingEvent.Level.DisplayName);
230  writer.WriteAttributeString(ATTR_THREAD, loggingEvent.ThreadName);
231 
232  if (loggingEvent.Domain != null && loggingEvent.Domain.Length > 0)
233  {
234  writer.WriteAttributeString(ATTR_DOMAIN, loggingEvent.Domain);
235  }
236  if (loggingEvent.Identity != null && loggingEvent.Identity.Length > 0)
237  {
238  writer.WriteAttributeString(ATTR_IDENTITY, loggingEvent.Identity);
239  }
240  if (loggingEvent.UserName != null && loggingEvent.UserName.Length > 0)
241  {
242  writer.WriteAttributeString(ATTR_USERNAME, loggingEvent.UserName);
243  }
244 
245  // Append the message text
246  writer.WriteStartElement(m_elmMessage);
247  if (!this.Base64EncodeMessage)
248  {
249  Transform.WriteEscapedXmlString(writer, loggingEvent.RenderedMessage, this.InvalidCharReplacement);
250  }
251  else
252  {
253  byte[] messageBytes = Encoding.UTF8.GetBytes(loggingEvent.RenderedMessage);
254  string base64Message = Convert.ToBase64String(messageBytes, 0, messageBytes.Length);
255  Transform.WriteEscapedXmlString(writer, base64Message,this.InvalidCharReplacement);
256  }
257  writer.WriteEndElement();
258 
259  PropertiesDictionary properties = loggingEvent.GetProperties();
260 
261  // Append the properties text
262  if (properties.Count > 0)
263  {
264  writer.WriteStartElement(m_elmProperties);
265  foreach(System.Collections.DictionaryEntry entry in properties)
266  {
267  writer.WriteStartElement(m_elmData);
268  writer.WriteAttributeString(ATTR_NAME, Transform.MaskXmlInvalidCharacters((string)entry.Key,this.InvalidCharReplacement));
269 
270  // Use an ObjectRenderer to convert the object to a string
271  string valueStr =null;
272  if (!this.Base64EncodeProperties)
273  {
274  valueStr = Transform.MaskXmlInvalidCharacters(loggingEvent.Repository.RendererMap.FindAndRender(entry.Value),this.InvalidCharReplacement);
275  }
276  else
277  {
278  byte[] propertyValueBytes = Encoding.UTF8.GetBytes(loggingEvent.Repository.RendererMap.FindAndRender(entry.Value));
279  valueStr = Convert.ToBase64String(propertyValueBytes, 0, propertyValueBytes.Length);
280  }
281  writer.WriteAttributeString(ATTR_VALUE, valueStr);
282 
283  writer.WriteEndElement();
284  }
285  writer.WriteEndElement();
286  }
287 
288  string exceptionStr = loggingEvent.GetExceptionString();
289  if (exceptionStr != null && exceptionStr.Length > 0)
290  {
291  // Append the stack trace line
292  writer.WriteStartElement(m_elmException);
293  Transform.WriteEscapedXmlString(writer, exceptionStr,this.InvalidCharReplacement);
294  writer.WriteEndElement();
295  }
296 
297  if (LocationInfo)
298  {
299  LocationInfo locationInfo = loggingEvent.LocationInformation;
300 
301  writer.WriteStartElement(m_elmLocation);
302  writer.WriteAttributeString(ATTR_CLASS, locationInfo.ClassName);
303  writer.WriteAttributeString(ATTR_METHOD, locationInfo.MethodName);
304  writer.WriteAttributeString(ATTR_FILE, locationInfo.FileName);
305  writer.WriteAttributeString(ATTR_LINE, locationInfo.LineNumber);
306  writer.WriteEndElement();
307  }
308 
309  writer.WriteEndElement();
310  }
