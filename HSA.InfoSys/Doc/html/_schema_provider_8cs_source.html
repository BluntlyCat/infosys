63  public virtual DataTable GetDatabases(string[] restrictions)
64  {
65  Regex regex = null;
66  int caseSetting = Int32.Parse(connection.driver.Property("lower_case_table_names"));
67 
68  string sql = "SHOW DATABASES";
69 
70  // if lower_case_table_names is zero, then case lookup should be sensitive
71  // so we can use LIKE to do the matching.
72  if (caseSetting == 0)
73  {
74  if (restrictions != null && restrictions.Length >= 1)
75  sql = sql + " LIKE '" + restrictions[0] + "'";
76  }
77  else if (restrictions != null && restrictions.Length >= 1 && restrictions[0] != null)
78  regex = new Regex(restrictions[0], RegexOptions.IgnoreCase);
79 
80  MySqlDataAdapter da = new MySqlDataAdapter(sql, connection);
81  DataTable dt = new DataTable();
82  da.Fill(dt);
83 
84  DataTable table = new DataTable("Databases");
85  table.Columns.Add("CATALOG_NAME", typeof(string));
86  table.Columns.Add("SCHEMA_NAME", typeof(string));
87 
88  foreach (DataRow row in dt.Rows)
89  {
90  if (caseSetting != 0 && regex != null &&
91  !regex.Match(row[0].ToString()).Success) continue;
92 
93  DataRow newRow = table.NewRow();
94  newRow[1] = row[0];
95  table.Rows.Add(newRow);
96  }
97 
98  return table;
99  }

198  private void LoadTableColumns(DataTable dt, string schema,
199  string tableName, string columnRestriction)
200  {
201  string sql = String.Format("SHOW FULL COLUMNS FROM `{0}`.`{1}`",
202  schema, tableName);
203  MySqlCommand cmd = new MySqlCommand(sql, connection);
204 
205  int pos = 1;
206  using (MySqlDataReader reader = cmd.ExecuteReader())
207  {
208  while (reader.Read())
209  {
210  string colName = reader.GetString(0);
211  if (columnRestriction != null && colName != columnRestriction)
212  continue;
213  DataRow row = dt.NewRow();
214  row["TABLE_CATALOG"] = DBNull.Value;
215  row["TABLE_SCHEMA"] = schema;
216  row["TABLE_NAME"] = tableName;
217  row["COLUMN_NAME"] = colName;
218  row["ORDINAL_POSITION"] = pos++;
219  row["COLUMN_DEFAULT"] = reader.GetValue(5);
220  row["IS_NULLABLE"] = reader.GetString(3);
221  row["DATA_TYPE"] = reader.GetString(1);
222  row["CHARACTER_MAXIMUM_LENGTH"] = DBNull.Value;
223  row["CHARACTER_OCTET_LENGTH"] = DBNull.Value;
224  row["NUMERIC_PRECISION"] = DBNull.Value;
225  row["NUMERIC_SCALE"] = DBNull.Value;
226  row["CHARACTER_SET_NAME"] = reader.GetValue(2);
227  row["COLLATION_NAME"] = row["CHARACTER_SET_NAME"];
228  row["COLUMN_TYPE"] = reader.GetString(1);
229  row["COLUMN_KEY"] = reader.GetString(4);
230  row["EXTRA"] = reader.GetString(6);
231  row["PRIVILEGES"] = reader.GetString(7);
232  row["COLUMN_COMMENT"] = reader.GetString(8);
233  ParseColumnRow(row);
234  dt.Rows.Add(row);
235  }
236  }

239  private static void ParseColumnRow(DataRow row)
240  {
241  // first parse the character set name
242  string charset = row["CHARACTER_SET_NAME"].ToString();
243  int index = charset.IndexOf('_');
244  if (index != -1)
245  row["CHARACTER_SET_NAME"] = charset.Substring(0, index);
246 
247  // now parse the data type
248  string dataType = row["DATA_TYPE"].ToString();
249  index = dataType.IndexOf('(');
250  if (index == -1)
251  return;
252  row["DATA_TYPE"] = dataType.Substring(0, index);
253  int stop = dataType.IndexOf(')', index);
254  string dataLen = dataType.Substring(index + 1, stop - (index + 1));
255  string lowerType = row["DATA_TYPE"].ToString().ToLower();
256  if (lowerType == "char" || lowerType == "varchar")
257  row["CHARACTER_MAXIMUM_LENGTH"] = dataLen;
258  else if (lowerType == "real" || lowerType == "decimal")
259  {
260  string[] lenparts = dataLen.Split(new char[] { ',' });
261  row["NUMERIC_PRECISION"] = lenparts[0];
262  if (lenparts.Length == 2)
263  row["NUMERIC_SCALE"] = lenparts[1];
264  }
265  }

267  public virtual DataTable GetIndexes(string[] restrictions)
268  {
269  DataTable dt = new DataTable("Indexes");
270  dt.Columns.Add("INDEX_CATALOG", typeof(string));
271  dt.Columns.Add("INDEX_SCHEMA", typeof(string));
272  dt.Columns.Add("INDEX_NAME", typeof(string));
273  dt.Columns.Add("TABLE_NAME", typeof(string));
274  dt.Columns.Add("UNIQUE", typeof(bool));
275  dt.Columns.Add("PRIMARY", typeof(bool));
276  dt.Columns.Add("TYPE", typeof(string));
277  dt.Columns.Add("COMMENT", typeof(string));
278 
279  // Get the list of tables first
280  int max = restrictions == null ? 4 : restrictions.Length;
281  string[] tableRestrictions = new string[Math.Max(max, 4)];
282  if (restrictions != null)
283  restrictions.CopyTo(tableRestrictions, 0);
284  tableRestrictions[3] = "BASE TABLE";
285  DataTable tables = GetTables(tableRestrictions);
286 
287  foreach (DataRow table in tables.Rows)
288  {
289  string sql = String.Format("SHOW INDEX FROM `{0}`.`{1}`",
290  MySqlHelper.DoubleQuoteString((string)table["TABLE_SCHEMA"]),
291  MySqlHelper.DoubleQuoteString((string)table["TABLE_NAME"]));
292  MySqlDataAdapter da = new MySqlDataAdapter(sql, connection);
293  DataTable indexes = new DataTable();
294  da.Fill(indexes);
295  foreach (DataRow index in indexes.Rows)
296  {
297  long seq_index = (long)index["SEQ_IN_INDEX"];
298  if (seq_index != 1) continue;
299  if (restrictions != null && restrictions.Length == 4 &&
300  restrictions[3] != null &&
301  !index["KEY_NAME"].Equals(restrictions[3])) continue;
302  DataRow row = dt.NewRow();
303  row["INDEX_CATALOG"] = null;
304  row["INDEX_SCHEMA"] = table["TABLE_SCHEMA"];
305  row["INDEX_NAME"] = index["KEY_NAME"];
306  row["TABLE_NAME"] = index["TABLE"];
307  row["UNIQUE"] = (long)index["NON_UNIQUE"] == 0;
308  row["PRIMARY"] = index["KEY_NAME"].Equals("PRIMARY");
309  row["TYPE"] = index["INDEX_TYPE"];
310  row["COMMENT"] = index["COMMENT"];
311  dt.Rows.Add(row);
312  }
313  }
314 
315  return dt;
316  }

462  private void GetForeignKeysOnTable(DataTable fkTable, DataRow tableToParse,
463  string filterName, bool includeColumns)
464  {
465  string sqlMode = GetSqlMode();
466 
467  if (filterName != null)
468  filterName = filterName.ToLower(CultureInfo.InvariantCulture);
469 
470  string sql = string.Format("SHOW CREATE TABLE `{0}`.`{1}`",
471  tableToParse["TABLE_SCHEMA"], tableToParse["TABLE_NAME"]);
472  string lowerBody = null, body = null;
473  MySqlCommand cmd = new MySqlCommand(sql, connection);
474  using (MySqlDataReader reader = cmd.ExecuteReader())
475  {
476  reader.Read();
477  body = reader.GetString(1);
478  lowerBody = body.ToLower(CultureInfo.InvariantCulture);
479  }
480 
481  MySqlTokenizer tokenizer = new MySqlTokenizer(lowerBody);
482  tokenizer.AnsiQuotes = sqlMode.IndexOf("ANSI_QUOTES") != -1;
483  tokenizer.BackslashEscapes = sqlMode.IndexOf("NO_BACKSLASH_ESCAPES") != -1;
484 
485  while (true)
486  {
487  string token = tokenizer.NextToken();
488  // look for a starting contraint
489  while (token != null && (token != "constraint" || tokenizer.Quoted))
490  token = tokenizer.NextToken();
491  if (token == null) break;
492 
493  ParseConstraint(fkTable, tableToParse, tokenizer, includeColumns);
494  }
495  }

497  private static void ParseConstraint(DataTable fkTable, DataRow table,
498  MySqlTokenizer tokenizer, bool includeColumns)
499  {
500  string name = tokenizer.NextToken();
501  DataRow row = fkTable.NewRow();
502 
503  // make sure this constraint is a FK
504  string token = tokenizer.NextToken();
505  if (token != "foreign" || tokenizer.Quoted)
506  return;
507  tokenizer.NextToken(); // read off the 'KEY' symbol
508  tokenizer.NextToken(); // read off the '(' symbol
509 
510  row["CONSTRAINT_CATALOG"] = table["TABLE_CATALOG"];
511  row["CONSTRAINT_SCHEMA"] = table["TABLE_SCHEMA"];
512  row["TABLE_CATALOG"] = table["TABLE_CATALOG"];
513  row["TABLE_SCHEMA"] = table["TABLE_SCHEMA"];
514  row["TABLE_NAME"] = table["TABLE_NAME"];
515  row["REFERENCED_TABLE_CATALOG"] = null;
516  row["CONSTRAINT_NAME"] = name.Trim(new char[] { '\'', '`' });
517 
518  ArrayList srcColumns = includeColumns ? ParseColumns(tokenizer) : null;
519 
520  // now look for the references section
521  while (token != "references" || tokenizer.Quoted)
522  token = tokenizer.NextToken();
523  string target1 = tokenizer.NextToken();
524  string target2 = tokenizer.NextToken();
525  if (target2.StartsWith(".", StringComparison.Ordinal))
526  {
527  row["REFERENCED_TABLE_SCHEMA"] = target1;
528  row["REFERENCED_TABLE_NAME"] = target2.Substring(1).Trim(new char[] { '\'', '`' });
529  tokenizer.NextToken(); // read off the '('
530  }
531  else
532  {
533  row["REFERENCED_TABLE_SCHEMA"] = table["TABLE_SCHEMA"];
534  row["REFERENCED_TABLE_NAME"] = target1.Substring(1).Trim(new char[] { '\'', '`' }); ;
535  }
536 
537  // if we are supposed to include columns, read the target columns
538  ArrayList targetColumns = includeColumns ? ParseColumns(tokenizer) : null;
539 
540  if (includeColumns)
541  ProcessColumns(fkTable, row, srcColumns, targetColumns);
542  else
543  fkTable.Rows.Add(row);
544  }
545 
546  private static ArrayList ParseColumns(MySqlTokenizer tokenizer)
547  {
548  ArrayList sc = new ArrayList();
549  string token = tokenizer.NextToken();
550  while (token != ")")
551  {
552  if (token != ",")
553  sc.Add(token);
554  token = tokenizer.NextToken();
555  }
556  return sc;
557  }

559  private static void ProcessColumns(DataTable fkTable, DataRow row,
560  ArrayList srcColumns, ArrayList targetColumns)
561  {
562  for (int i = 0; i < srcColumns.Count; i++)
563  {
564  DataRow newRow = fkTable.NewRow();
565  newRow.ItemArray = row.ItemArray;
566  newRow["COLUMN_NAME"] = (string)srcColumns[i];
567  newRow["ORDINAL_POSITION"] = i;
568  newRow["REFERENCED_COLUMN_NAME"] = (string)targetColumns[i];
569  fkTable.Rows.Add(newRow);
570  }
571  }
