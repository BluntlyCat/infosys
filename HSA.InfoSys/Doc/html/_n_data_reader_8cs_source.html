489  private readonly IList<string> fieldIndexToName = new List<string>();

501  internal NResult(IDataReader reader, bool isMidstream)
502  {
503  schemaTable = reader.GetSchemaTable();
504 
505  List<object[]> recordsList = new List<object[]>();
506  int rowIndex = 0;
507 
508  // if we are in the middle of processing the reader then don't bother
509  // to move to the next record - just use the current one.
510  while (isMidstream || reader.Read())
511  {
512  if (rowIndex == 0)
513  {
514  for (int i = 0; i < reader.FieldCount; i++)
515  {
516  string fieldName = reader.GetName(i);
517  fieldNameToIndex[fieldName] = i;
518  fieldIndexToName.Add(fieldName);
519  fieldTypes.Add(reader.GetFieldType(i));
520  fieldDataTypeNames.Add(reader.GetDataTypeName(i));
521  }
522 
523  colCount = reader.FieldCount;
524  }
525 
526  rowIndex++;
527 
528  object[] colValues = new object[reader.FieldCount];
529  reader.GetValues(colValues);
530  recordsList.Add(colValues);
531 
532  // we can go back to reading a reader like normal and don't need
533  // to consider where we started from.
534  isMidstream = false;
535  }
536 
537  records = recordsList.ToArray();
538  }

590  public int GetOrdinal(string colName)
591  {
592  // Martijn Boland, 20041106: perform a case-sensitive search first and if that returns
593  // null, perform a case-insensitive search (as being described in the IDataRecord
594  // interface, see http://msdn.microsoft.com/library/default.asp?url=/library/en-us/cpref/html/frlrfSystemDataIDataRecordClassItemTopic1.asp.
595  // This is necessary for databases that don't preserve the case of field names when
596  // they are created without quotes (e.g. DB2, PostgreSQL).
597  int value;
598  if (fieldNameToIndex.TryGetValue(colName, out value))
599  return value;
600 
601  foreach (KeyValuePair<string, int> pair in fieldNameToIndex)
602  {
603  if (StringHelper.EqualsCaseInsensitive(pair.Key, colName))
604  {
605  return pair.Value;
606  }
607  }
608 
609  throw new IndexOutOfRangeException(String.Format("No column with the specified name was found: {0}.", colName));
610  }
