177  private void BindJoin(HbmJoin joinMapping, Join join, IDictionary<string, MetaAttribute> inheritedMetas)
178  {
179  PersistentClass persistentClass = join.PersistentClass;
180 
181  // TABLENAME
182  string schema = joinMapping.schema ?? mappings.SchemaName;
183  string catalog = joinMapping.catalog ?? mappings.CatalogName;
184 
185  string action = "all"; // joinMapping.schemaaction ?? "all";
186 
187  string tableName = joinMapping.table;
188  Table table = mappings.AddTable(schema, catalog, GetClassTableName(persistentClass, tableName), joinMapping.Subselect, false, action);
189  join.Table = table;
190 
191  join.IsSequentialSelect = joinMapping.fetch == HbmJoinFetch.Select;
192  join.IsInverse = joinMapping.inverse;
193  join.IsOptional = joinMapping.optional;
194 
195  log.InfoFormat("Mapping class join: {0} -> {1}", persistentClass.EntityName, join.Table.Name);
196 
197  // KEY
198  SimpleValue key = new DependantValue(table, persistentClass.Identifier);
199  key.ForeignKeyName = joinMapping.key.foreignkey;
200  join.Key = key;
201  key.IsCascadeDeleteEnabled = joinMapping.key.ondelete == HbmOndelete.Cascade;
202  new ValuePropertyBinder(key, Mappings).BindSimpleValue(joinMapping.key, persistentClass.EntityName, false);
203 
204  join.CreatePrimaryKey(dialect);
205  join.CreateForeignKey();
206 
207  // PROPERTIES
208  new PropertiesBinder(Mappings, persistentClass, dialect).Bind(joinMapping.Properties, join.Table,
209  inheritedMetas, p => { },
210  join.AddProperty);
211 
212  // CUSTOM SQL

343  protected void BindAnyMeta(IAnyMapping anyMapping, Any model)
344  {
345  if(string.IsNullOrEmpty(anyMapping.MetaType))
346  {
347  return;
348  }
349  model.MetaType = anyMapping.MetaType;
350  var metaValues = anyMapping.MetaValues;
351  if (metaValues.Count == 0)
352  {
353  return;
354  }
355  IDictionary<object, string> values = new Dictionary<object, string>();
356  IType metaType = TypeFactory.HeuristicType(model.MetaType);
357 
358  foreach (var metaValue in metaValues)
359  {
360  try
361  {
362  object value = ((IDiscriminatorType)metaType).StringToObject(metaValue.value);
363  string entityName = GetClassName(metaValue.@class, mappings);
364  values[value] = entityName;
365  }
366  catch (InvalidCastException)
367  {
368  throw new MappingException("meta-type was not an IDiscriminatorType: " + metaType.Name);
369  }
370  catch (HibernateException he)
371  {
372  throw new MappingException("could not interpret meta-value", he);
373  }
374  catch (TypeLoadException cnfe)
375  {
376  throw new MappingException("meta-value class not found", cnfe);
377  }
378  }
379  model.MetaValues = values.Count > 0 ? values : null;
380  }
