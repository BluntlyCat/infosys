105  protected virtual object SaveWithGeneratedId(object entity, string entityName, object anything, IEventSource source, bool requiresImmediateIdAccess)
106  {
107  IEntityPersister persister = source.GetEntityPersister(entityName, entity);
108  object generatedId = persister.IdentifierGenerator.Generate(source, entity);
109  if (generatedId == null)
110  {
111  throw new IdentifierGenerationException("null id generated for:" + entity.GetType());

113  else if (generatedId == IdentifierGeneratorFactory.ShortCircuitIndicator)

117  else if (generatedId == IdentifierGeneratorFactory.PostInsertIndicator)

119  return PerformSave(entity, null, persister, true, anything, source, requiresImmediateIdAccess);

126  persister.IdentifierType.ToLoggableString(generatedId, source.Factory),

420  else

422  //the object is transient or detached
