139  override protected void FormatXml(XmlWriter writer, LoggingEvent loggingEvent)
140  {
141  // Translate logging events for log4j
142 
143  // Translate hostname property
144  if (loggingEvent.LookupProperty(LoggingEvent.HostNameProperty) != null &&
145  loggingEvent.LookupProperty("log4jmachinename") == null)
146  {
147  loggingEvent.GetProperties()["log4jmachinename"] = loggingEvent.LookupProperty(LoggingEvent.HostNameProperty);
148  }
149 
150  // translate appdomain name
151  if (loggingEvent.LookupProperty("log4japp") == null &&
152  loggingEvent.Domain != null &&
153  loggingEvent.Domain.Length > 0)
154  {
155  loggingEvent.GetProperties()["log4japp"] = loggingEvent.Domain;
156  }
157 
158  // translate identity name
159  if (loggingEvent.Identity != null &&
160  loggingEvent.Identity.Length > 0 &&
161  loggingEvent.LookupProperty(LoggingEvent.IdentityProperty) == null)
162  {
163  loggingEvent.GetProperties()[LoggingEvent.IdentityProperty] = loggingEvent.Identity;
164  }
165 
166  // translate user name
167  if (loggingEvent.UserName != null &&
168  loggingEvent.UserName.Length > 0 &&
169  loggingEvent.LookupProperty(LoggingEvent.UserNameProperty) == null)
170  {
171  loggingEvent.GetProperties()[LoggingEvent.UserNameProperty] = loggingEvent.UserName;
172  }
173 
174  // Write the start element
175  writer.WriteStartElement("log4j:event");
176  writer.WriteAttributeString("logger", loggingEvent.LoggerName);
177 
178  // Calculate the timestamp as the number of milliseconds since january 1970
179  //
180  // We must convert the TimeStamp to UTC before performing any mathematical
181  // operations. This allows use to take into account discontinuities
182  // caused by daylight savings time transitions.
183  TimeSpan timeSince1970 = loggingEvent.TimeStamp.ToUniversalTime() - s_date1970;
184 
185  writer.WriteAttributeString("timestamp", XmlConvert.ToString((long)timeSince1970.TotalMilliseconds));
186  writer.WriteAttributeString("level", loggingEvent.Level.DisplayName);
187  writer.WriteAttributeString("thread", loggingEvent.ThreadName);
188 
189  // Append the message text
190  writer.WriteStartElement("log4j:message");
191  Transform.WriteEscapedXmlString(writer, loggingEvent.RenderedMessage,this.InvalidCharReplacement);
192  writer.WriteEndElement();
193 
194  object ndcObj = loggingEvent.LookupProperty("NDC");
195  if (ndcObj != null)
196  {
197  string valueStr = loggingEvent.Repository.RendererMap.FindAndRender(ndcObj);
198 
199  if (valueStr != null && valueStr.Length > 0)
200  {
201  // Append the NDC text
202  writer.WriteStartElement("log4j:NDC");
203  Transform.WriteEscapedXmlString(writer, valueStr,this.InvalidCharReplacement);
204  writer.WriteEndElement();
205  }
206  }
207 
208  // Append the properties text
209  PropertiesDictionary properties = loggingEvent.GetProperties();
210  if (properties.Count > 0)
211  {
212  writer.WriteStartElement("log4j:properties");
213  foreach(System.Collections.DictionaryEntry entry in properties)
214  {
215  writer.WriteStartElement("log4j:data");
216  writer.WriteAttributeString("name", (string)entry.Key);
217 
218  // Use an ObjectRenderer to convert the object to a string
219  string valueStr = loggingEvent.Repository.RendererMap.FindAndRender(entry.Value);
220  writer.WriteAttributeString("value", valueStr);
221 
222  writer.WriteEndElement();
223  }
224  writer.WriteEndElement();
225  }
226 
227  string exceptionStr = loggingEvent.GetExceptionString();
228  if (exceptionStr != null && exceptionStr.Length > 0)
229  {
230  // Append the stack trace line
231  writer.WriteStartElement("log4j:throwable");
232  Transform.WriteEscapedXmlString(writer, exceptionStr,this.InvalidCharReplacement);
233  writer.WriteEndElement();
234  }
235 
236  if (LocationInfo)
237  {
238  LocationInfo locationInfo = loggingEvent.LocationInformation;
239 
240  writer.WriteStartElement("log4j:locationInfo");
241  writer.WriteAttributeString("class", locationInfo.ClassName);
242  writer.WriteAttributeString("method", locationInfo.MethodName);
243  writer.WriteAttributeString("file", locationInfo.FileName);
244  writer.WriteAttributeString("line", locationInfo.LineNumber);
245  writer.WriteEndElement();
246  }
247 
248  writer.WriteEndElement();
249  }
250  }
