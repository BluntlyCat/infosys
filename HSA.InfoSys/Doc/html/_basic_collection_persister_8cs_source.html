159  protected override int DoUpdateRows(object id, IPersistentCollection collection, ISessionImplementor session)
160  {
161  if (ArrayHelper.IsAllFalse(elementColumnIsSettable)) return 0;
162 
163  try
164  {
165  IDbCommand st = null;
166  IExpectation expectation = Expectations.AppropriateExpectation(UpdateCheckStyle);
167  //bool callable = UpdateCallable;
168  bool useBatch = expectation.CanBeBatched;
169  IEnumerable entries = collection.Entries(this);
170  int i = 0;
171  int count = 0;
172  foreach (object entry in entries)
173  {
174  if (collection.NeedsUpdating(entry, i, ElementType))
175  {
176  int offset = 0;
177  if (useBatch)
178  {
179  if (st == null)
180  {
181  st =
182  session.Batcher.PrepareBatchCommand(SqlUpdateRowString.CommandType, SqlUpdateRowString.Text,
183  SqlUpdateRowString.ParameterTypes);
184  }
185  }
186  else
187  {
188  st =
189  session.Batcher.PrepareCommand(SqlUpdateRowString.CommandType, SqlUpdateRowString.Text,
190  SqlUpdateRowString.ParameterTypes);
191  }
192 
193  try
194  {
195  //offset += expectation.Prepare(st, Factory.ConnectionProvider.Driver);
196 
197  int loc = WriteElement(st, collection.GetElement(entry), offset, session);
198  if (hasIdentifier)
199  {
200  WriteIdentifier(st, collection.GetIdentifier(entry, i), loc, session);
201  }
202  else
203  {
204  loc = WriteKey(st, id, loc, session);
205  if (HasIndex && !indexContainsFormula)
206  {
207  WriteIndexToWhere(st, collection.GetIndex(entry, i, this), loc, session);
208  }
209  else
210  {
211  WriteElementToWhere(st, collection.GetSnapshotElement(entry, i), loc, session);
212  }
213  }
214 
215  if (useBatch)
216  {
217  session.Batcher.AddToBatch(expectation);
218  }
219  else
220  {
221  expectation.VerifyOutcomeNonBatched(session.Batcher.ExecuteNonQuery(st), st);
222  }
223  }
224  catch (Exception e)
225  {
226  if (useBatch)
227  {
228  session.Batcher.AbortBatch(e);
229  }
230  throw;
231  }
232  finally
233  {
234  if (!useBatch)
235  {
236  session.Batcher.CloseCommand(st, null);
237  }
238  }
239  count++;
240  }
241  i++;
242  }
243  return count;
244  }
245  catch (DbException sqle)
246  {
247  throw ADOExceptionHelper.Convert(SQLExceptionConverter, sqle,
248  "could not update collection rows: " + MessageHelper.InfoString(this, id),
249  SqlUpdateRowString.Text);
250  }
251  }
