233  public virtual void Configure(MySqlConnection connection)
234  {
235  bool firstConfigure = false;
236 
237  // if we have not already configured our server variables
238  // then do so now
239  if (serverProps == null)
240  {
241  firstConfigure = true;
242 
243  // if we are in a pool and the user has said it's ok to cache the
244  // properties, then grab it from the pool
245  try
246  {
247  if (Pool != null && Settings.CacheServerProperties)
248  {
249  if (Pool.ServerProperties == null)
250  Pool.ServerProperties = LoadServerProperties(connection);
251  serverProps = Pool.ServerProperties;
252  }
253  else
254  serverProps = LoadServerProperties(connection);
255 
256  LoadCharacterSets(connection);
257  }
258  catch (MySqlException ex)
259  {
260  // expired password capability
261  if (ex.Number == 1820)
262  {
263  IsPasswordExpired = true;
264  return;
265  }
266  throw;
267  }
268  }
269 
270 
271 #if AUTHENTICATED
272  string licenseType = serverProps["license"];
273  if (licenseType == null || licenseType.Length == 0 ||
274  licenseType != "commercial")
275  throw new MySqlException( "This client library licensed only for use with commercially-licensed MySQL servers." );
276 #endif
277  // if the user has indicated that we are not to reset
278  // the connection and this is not our first time through,
279  // then we are done.
280  if (!Settings.ConnectionReset && !firstConfigure) return;
281 
282  string charSet = connectionString.CharacterSet;
283  if (charSet == null || charSet.Length == 0)
284  {
285  if (serverCharSetIndex >= 0)
286  charSet = (string)charSets[serverCharSetIndex];
287  else
288  charSet = serverCharSet;
289  }
290 
291  if (serverProps.Contains("max_allowed_packet"))
292  maxPacketSize = Convert.ToInt64(serverProps["max_allowed_packet"]);
293 
294  // now tell the server which character set we will send queries in and which charset we
295  // want results in
296  MySqlCommand charSetCmd = new MySqlCommand("SET character_set_results=NULL",
297  connection);
298  charSetCmd.InternallyCreated = true;
299  object clientCharSet = serverProps["character_set_client"];
300  object connCharSet = serverProps["character_set_connection"];
301  if ((clientCharSet != null && clientCharSet.ToString() != charSet) ||
302  (connCharSet != null && connCharSet.ToString() != charSet))
303  {
304  MySqlCommand setNamesCmd = new MySqlCommand("SET NAMES " + charSet, connection);
305  setNamesCmd.InternallyCreated = true;
306  setNamesCmd.ExecuteNonQuery();
307  }
308  charSetCmd.ExecuteNonQuery();
309 
310  if (charSet != null)
311  Encoding = CharSetMap.GetEncoding(Version, charSet);
312  else
313  Encoding = CharSetMap.GetEncoding(Version, "latin1");
314 
315  handler.Configure();
316  }

323  private Hashtable LoadServerProperties(MySqlConnection connection)
324  {
325  // load server properties
326  Hashtable hash = new Hashtable();
327  MySqlCommand cmd = new MySqlCommand("SHOW VARIABLES", connection);
328 
329  using (MySqlDataReader reader = cmd.ExecuteReader())
330  {
331  try
332  {
333  while (reader.Read())
334  {
335  string key = reader.GetString(0);
336  string value = reader.GetString(1);
337  hash[key] = value;
338  }
339  }
340  catch (Exception ex)
341  {
342  MySqlTrace.LogError(ThreadID, ex.Message);
343  throw;
344  }
345  }
346  // Get time zone offset as numerical value
347  timeZoneOffset = GetTimeZoneOffset( connection );
348  return hash;
349  }

351  private int GetTimeZoneOffset( MySqlConnection con )
352  {
353  MySqlCommand cmd = new MySqlCommand("select timediff( curtime(), utc_time() )", con);
354  string s = cmd.ExecuteScalar().ToString();

362  private void LoadCharacterSets(MySqlConnection connection)
363  {
364  MySqlCommand cmd = new MySqlCommand("SHOW COLLATION", connection);
365 
366  // now we load all the currently active collations
367  try
368  {
369  using (MySqlDataReader reader = cmd.ExecuteReader())
370  {
371  charSets = new Hashtable();
372  while (reader.Read())
373  {
374  charSets[Convert.ToInt32(reader["id"], NumberFormatInfo.InvariantInfo)] =
375  reader.GetString(reader.GetOrdinal("charset"));
376  }
377  }
378  }
379  catch (Exception ex)
380  {
381  MySqlTrace.LogError(ThreadID, ex.Message);
382  throw;
383  }

386  public virtual List<MySqlError> ReportWarnings(MySqlConnection connection)
387  {
388  List<MySqlError> warnings = new List<MySqlError>();
389 
390  MySqlCommand cmd = new MySqlCommand("SHOW WARNINGS", connection);
391  cmd.InternallyCreated = true;
392  using (MySqlDataReader reader = cmd.ExecuteReader())
393  {
394  while (reader.Read())
395  {
396  warnings.Add(new MySqlError(reader.GetString(0),
397  reader.GetInt32(1), reader.GetString(2)));
398  }
399  }
400 
401  MySqlInfoMessageEventArgs args = new MySqlInfoMessageEventArgs();
402  args.errors = warnings.ToArray();
403  if (connection != null)
404  connection.OnInfoMessage(args);
405  return warnings;
406  }

414  public virtual ResultSet NextResult(int statementId, bool force)
415  {
416  if (!force && !firstResult && !HasStatus(ServerStatusFlags.AnotherQuery | ServerStatusFlags.MoreResults))
417  return null;
418  firstResult = false;
419 
420  int affectedRows = -1, warnings = 0;
421  long insertedId = -1;
422  int fieldCount = GetResult(statementId, ref affectedRows, ref insertedId);
423  if (fieldCount == -1)
424  return null;
425  if (fieldCount > 0)
426  return new ResultSet(this, statementId, fieldCount);
427  else
428  return new ResultSet(affectedRows, insertedId);
429  }

518  protected virtual void Dispose(bool disposing)
519  {
520  // Avoid cyclic calls to Dispose.
521  if (disposeInProgress)
522  return;
523 
524  disposeInProgress = true;
525 
526  try
527  {
528  ResetTimeout(1000);
529  if (disposing)
530  handler.Close(isOpen);
531  // if we are pooling, then release ourselves
532  if (connectionString.Pooling)
533  MySqlPoolManager.RemoveConnection(this);
534  }
535  catch (Exception)
536  {
537  if (disposing)
538  throw;
539  }
540  finally
541  {
542  reader = null;
543  isOpen = false;
544  disposeInProgress = false;
545  }
546  }
