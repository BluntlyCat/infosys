27  public virtual void OnRefresh(RefreshEvent @event, IDictionary refreshedAlready)
28  {
29  IEventSource source = @event.Session;
30 
31  bool isTransient = !source.Contains(@event.Entity);
32  if (source.PersistenceContext.ReassociateIfUninitializedProxy(@event.Entity))
33  {
34  if (isTransient)
35  source.SetReadOnly(@event.Entity, source.DefaultReadOnly);
36  return;
37  }
38 
39  object obj = source.PersistenceContext.UnproxyAndReassociate(@event.Entity);
40 
41  if (refreshedAlready.Contains(obj))
42  {
43  log.Debug("already refreshed");
44  return;
45  }
46 
47  EntityEntry e = source.PersistenceContext.GetEntry(obj);
48  IEntityPersister persister;
49  object id;
50 
51  if (e == null)
52  {
53  persister = source.GetEntityPersister(null, obj); //refresh() does not pass an entityName
54  id = persister.GetIdentifier(obj, source.EntityMode);
55  if (log.IsDebugEnabled)
56  {
57  log.Debug("refreshing transient " + MessageHelper.InfoString(persister, id, source.Factory));
58  }
59  EntityKey key = source.GenerateEntityKey(id, persister);
60  if (source.PersistenceContext.GetEntry(key) != null)
61  {
62  throw new PersistentObjectException("attempted to refresh transient instance when persistent instance was already associated with the Session: " +
63  MessageHelper.InfoString(persister, id, source.Factory));
64  }
65  }
66  else
67  {
68  if (log.IsDebugEnabled)
69  {
70  log.Debug("refreshing " + MessageHelper.InfoString(e.Persister, e.Id, source.Factory));
71  }
72  if (!e.ExistsInDatabase)
73  {
74  throw new HibernateException("this instance does not yet exist as a row in the database");
75  }
76 
77  persister = e.Persister;
78  id = e.Id;
79  }
80 
81  // cascade the refresh prior to refreshing this entity
82  refreshedAlready[obj] = obj;
83  new Cascade(CascadingAction.Refresh, CascadePoint.BeforeRefresh, source).CascadeOn(persister, obj, refreshedAlready);
84 
85  if (e != null)
86  {
87  EntityKey key = source.GenerateEntityKey(id, persister);
88  source.PersistenceContext.RemoveEntity(key);
89  if (persister.HasCollections)
90  new EvictVisitor(source).Process(obj, persister);
91  }
92 
93  if (persister.HasCache)
94  {
95  CacheKey ck = source.GenerateCacheKey(id, persister.IdentifierType, persister.RootEntityName);
96  persister.Cache.Remove(ck);
97  }
98 
99  EvictCachedCollections(persister, id, source.Factory);
100 
101  // NH Different behavior : NH-1601
102  // At this point the entity need the real refresh, all elementes of collections are Refreshed,
103  // the collection state was evicted, but the PersistentCollection (in the entity state)
104  // is associated with a possible previous session.
105  new WrapVisitor(source).Process(obj, persister);
106 
107  string previousFetchProfile = source.FetchProfile;
108  source.FetchProfile = "refresh";
109  object result = persister.Load(id, obj, @event.LockMode, source);
110 
111  if (result != null)
112  if (!persister.IsMutable)
113  source.SetReadOnly(result, true);
114  else
115  source.SetReadOnly(result, (e == null ? source.DefaultReadOnly : e.IsReadOnly));
116 
117  source.FetchProfile = previousFetchProfile;
118 
119  // NH Different behavior : we are ignoring transient entities without throw any kind of exception
120  // because a transient entity is "self refreshed"
121  if (!ForeignKeys.IsTransient(persister.EntityName, obj, result == null, @event.Session))
122  UnresolvableObjectException.ThrowIfNull(result, id, persister.EntityName);
123  }

138  }
139  else if (types[i].IsComponentType)
140  {
141  IAbstractComponentType actype = (IAbstractComponentType)types[i];
142  EvictCachedCollections(actype.Subtypes, id, factory);
