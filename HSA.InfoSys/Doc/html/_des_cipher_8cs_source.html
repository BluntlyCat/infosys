297  protected int[] GenerateWorkingKey(bool encrypting, byte[] key)
298  {
299  this.ValidateKey();
300 
301  int[] newKey = new int[32];
302  bool[] pc1m = new bool[56];
303  bool[] pcr = new bool[56];
304 
305  for (int j = 0; j < 56; j++)
306  {
307  int l = pc1[j];
308 
309  pc1m[j] = ((key[(uint)l >> 3] & bytebit[l & 07]) != 0);
310  }
311 
312  for (int i = 0; i < 16; i++)
313  {
314  int l, m, n;
315 
316  if (encrypting)
317  {
318  m = i << 1;
319  }
320  else
321  {
322  m = (15 - i) << 1;
323  }
324 
325  n = m + 1;
326  newKey[m] = newKey[n] = 0;
327 
328  for (int j = 0; j < 28; j++)
329  {
330  l = j + totrot[i];
331  if (l < 28)
332  {
333  pcr[j] = pc1m[l];
334  }
335  else
336  {
337  pcr[j] = pc1m[l - 28];
338  }
339  }
340 
341  for (int j = 28; j < 56; j++)
342  {
343  l = j + totrot[i];
344  if (l < 56)
345  {
346  pcr[j] = pc1m[l];
347  }
348  else
349  {
350  pcr[j] = pc1m[l - 28];
351  }
352  }
353 
354  for (int j = 0; j < 24; j++)
355  {
356  if (pcr[pc2[j]])
357  {
358  newKey[m] |= bigbyte[j];
359  }
