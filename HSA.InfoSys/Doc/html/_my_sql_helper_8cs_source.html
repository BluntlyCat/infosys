244  cmd.Parameters.Add(p);

398  private static CharClass[] makeCharClassArray()
399  {
400 
401  CharClass[] a = new CharClass[65536];
402  foreach (char c in stringOfBackslashChars)
403  {
404  a[c] = CharClass.Backslash;
405  }

408  a[c] = CharClass.Quote;
409  }

413  private static bool needsQuoting(string s)

430  public static string EscapeString(string value)
431  {
432  if (!needsQuoting(value))
433  return value;
434 
435  StringBuilder sb = new StringBuilder();
436 
437  foreach (char c in value)
438  {
439  CharClass charClass = charClassArray[c];
440  if (charClass != CharClass.None)
441  {
442  sb.Append("\\");
443  }
444  sb.Append(c);
445  }
446  return sb.ToString();
447  }

449  public static string DoubleQuoteString(string value)
450  {
451  if (!needsQuoting(value))
452  return value;
453 
454  StringBuilder sb = new StringBuilder();
455  foreach (char c in value)
456  {
457  CharClass charClass = charClassArray[c];
458  if (charClass == CharClass.Quote)
459  sb.Append(c);
460  else if (charClass == CharClass.Backslash)
461  sb.Append("\\");
462  sb.Append(c);
463  }
464  return sb.ToString();
465  }
