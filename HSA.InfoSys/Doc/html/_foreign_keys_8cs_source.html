48  private object NullifyTransientReferences(object value, IType type)
49  {
50  if (value == null)
51  {
52  return null;
53  }
54  else if (type.IsEntityType)
55  {
56  EntityType entityType = (EntityType)type;
57  if (entityType.IsOneToOne)
58  {
59  return value;
60  }
61  else
62  {
63  string entityName = entityType.GetAssociatedEntityName();
64  return IsNullifiable(entityName, value) ? null : value;
65  }
66  }
67  else if (type.IsAnyType)
68  {
69  return IsNullifiable(null, value) ? null : value;
70  }
71  else if (type.IsComponentType)
72  {
73  IAbstractComponentType actype = (IAbstractComponentType)type;
74  object[] subvalues = actype.GetPropertyValues(value, session);
75  IType[] subtypes = actype.Subtypes;
76  bool substitute = false;
77  for (int i = 0; i < subvalues.Length; i++)
78  {
79  object replacement = NullifyTransientReferences(subvalues[i], subtypes[i]);
80  if (replacement != subvalues[i])
81  {
82  substitute = true;
83  subvalues[i] = replacement;
84  }
85  }
86  if (substitute)
87  actype.SetPropertyValues(value, subvalues, session.EntityMode);
88  return value;
89  }
90  else
91  {
92  return value;
93  }
94  }

99  private bool IsNullifiable(string entityName, object obj)
100  {
101 
102  //if (obj == org.hibernate.intercept.LazyPropertyInitializer_Fields.UNFETCHED_PROPERTY)
103  // return false; //this is kinda the best we can do...
104 
105  if (obj.IsProxy())
106  {
107  INHibernateProxy proxy = obj as INHibernateProxy;
108 
109  // if its an uninitialized proxy it can't be transient
110  ILazyInitializer li = proxy.HibernateLazyInitializer;
111  if (li.GetImplementation(session) == null)
112  {
113  return false;
114  // ie. we never have to null out a reference to
115  // an uninitialized proxy
116  }
117  else
118  {
119  //unwrap it
120  obj = li.GetImplementation();
121  }
122  }
123 
124  // if it was a reference to self, don't need to nullify
125  // unless we are using native id generation, in which
126  // case we definitely need to nullify
127  if (obj == self)
128  {
129  // TODO H3.2: Different behaviour
130  //return isEarlyInsert || (isDelete && session.Factory.Dialect.HasSelfReferentialForeignKeyBug);
131  return isEarlyInsert || isDelete;
132  }
133 
134  // See if the entity is already bound to this session, if not look at the
135  // entity identifier and assume that the entity is persistent if the
136  // id is not "unsaved" (that is, we rely on foreign keys to keep
137  // database integrity)
138  EntityEntry entityEntry = session.PersistenceContext.GetEntry(obj);
139  if (entityEntry == null)
140  {
141  return IsTransient(entityName, obj, null, session);
142  }
143  else
144  {
145  return entityEntry.IsNullifiable(isEarlyInsert, session);
146  }
147  }

178  public static bool IsTransient(string entityName, object entity, bool? assumed, ISessionImplementor session)
179  {
180  if (Equals(Intercept.LazyPropertyInitializer.UnfetchedProperty, entity))
181  {
182  // an unfetched association can only point to
183  // an entity that already exists in the db
184  return false;
185  }
186 
187  // let the interceptor inspect the instance to decide
188  bool? isUnsaved = session.Interceptor.IsTransient(entity);
189  if (isUnsaved.HasValue)
190  return isUnsaved.Value;
191 
192  // let the persister inspect the instance to decide
193  IEntityPersister persister = session.GetEntityPersister(entityName, entity);
194  isUnsaved = persister.IsTransient(entity, session);
195  if (isUnsaved.HasValue)
196  return isUnsaved.Value;
197 
198  // we use the assumed value, if there is one, to avoid hitting
199  // the database
200  if (assumed.HasValue)
201  return assumed.Value;
202 
203  if (persister.IdentifierGenerator is Assigned)
204  {
205  // When using assigned identifiers we cannot tell if an entity
206  // is transient or detached without querying the database.
207  // This could potentially cause Select N+1 in cascaded saves, so warn the user.
208  log.Warn("Unable to determine if " + entity.ToString()
209  + " with assigned identifier " + persister.GetIdentifier(entity, session.EntityMode)
210  + " is transient or detached; querying the database."
211  + " Use explicit Save() or Update() in session to prevent this.");
212  }
213 
214  // hit the database, after checking the session cache for a snapshot
215  System.Object[] snapshot =
216  session.PersistenceContext.GetDatabaseSnapshot(persister.GetIdentifier(entity, session.EntityMode), persister);
217  return snapshot == null;
218  }
