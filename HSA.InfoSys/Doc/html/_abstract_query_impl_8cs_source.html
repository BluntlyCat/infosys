85  protected internal virtual void VerifyParameters(bool reserveFirstParameter)
86  {
87  if (parameterMetadata.NamedParameterNames.Count != namedParameters.Count + namedParameterLists.Count)
88  {
89  var missingParams = new HashSet<string>(parameterMetadata.NamedParameterNames);
90  missingParams.ExceptWith(namedParameterLists.Keys);
91  missingParams.ExceptWith(namedParameters.Keys);
92  throw new QueryException("Not all named parameters have been set: " + CollectionPrinter.ToString(missingParams), QueryString);
93  }
94 
95  int positionalValueSpan = 0;
96  for (int i = 0; i < values.Count; i++)
97  {
98  object obj = types[i];
99  if (values[i] == UNSET_PARAMETER || obj == UNSET_TYPE)
100  {
101  if (reserveFirstParameter && i == 0)
102  {
103  continue;
104  }
105  else
106  {
107  throw new QueryException("Unset positional parameter at position: " + i, QueryString);
108  }
109  }
110  positionalValueSpan++;
111  }
112 
113  if (parameterMetadata.OrdinalParameterCount != positionalValueSpan)
114  {
115  if (reserveFirstParameter && parameterMetadata.OrdinalParameterCount - 1 != positionalValueSpan)
116  {
117  throw new QueryException(
118  "Expected positional parameter count: " + (parameterMetadata.OrdinalParameterCount - 1) + ", actual parameters: "
119  + CollectionPrinter.ToString(values), QueryString);
120  }
121  else if (!reserveFirstParameter)
122  {
123  throw new QueryException(
124  "Expected positional parameter count: " + parameterMetadata.OrdinalParameterCount + ", actual parameters: "
125  + CollectionPrinter.ToString(values), QueryString);
126  }
127  }

189  private IType GuessType(System.Type clazz)
190  {
191  if (clazz == null)
192  {
193  throw new ArgumentNullException("clazz", "The IType can not be guessed for a null value.");
194  }
195 
196  string typename = clazz.AssemblyQualifiedName;
197  IType type = TypeFactory.HeuristicType(typename);
198  bool serializable = (type != null && type is SerializableType);
199  if (type == null || serializable)
200  {
201  try
202  {
203  session.Factory.GetEntityPersister(clazz.FullName);
204  }
205  catch (MappingException)
206  {
207  if (serializable)
208  {
209  return type;
210  }
211  else
212  {
213  throw new HibernateException("Could not determine a type for class: " + typename);
214  }
215  }
216  return NHibernateUtil.Entity(clazz);
217  }
218  else
219  {
220  return type;

239  private string ExpandParameterList(string query, string name, TypedValue typedList, IDictionary<string, TypedValue> namedParamsCopy)
240  {
241  var vals = (IEnumerable) typedList.Value;
242  var type = typedList.Type;
243 
244  var typedValues = (from object value in vals
245  select new TypedValue(type, value, session.EntityMode))
246  .ToList();
247 
248  if (typedValues.Count == 1)
249  {
250  namedParamsCopy[name] = typedValues[0];
251  return query;
252  }
253 
254  var isJpaPositionalParam = parameterMetadata.GetNamedParameterDescriptor(name).JpaStyle;
255  var aliases = new string[typedValues.Count];
256  for (var index = 0; index < typedValues.Count; index++)
257  {
258  var value = typedValues[index];
259  var alias = (isJpaPositionalParam ? 'x' + name : name + StringHelper.Underscore) + index + StringHelper.Underscore;
260  namedParamsCopy[alias] = value;
261  aliases[index] = ParserHelper.HqlVariablePrefix + alias;
262  }
263 
264  var paramPrefix = isJpaPositionalParam ? StringHelper.SqlParameter : ParserHelper.HqlVariablePrefix;
265 
266  return StringHelper.Replace(query, paramPrefix + name, string.Join(StringHelper.CommaSpace, aliases), true);
267  }
