125  protected string DetermineSequenceName(IDictionary<string, string> parms, Dialect.Dialect dialect)
126  {
127  string sequenceName = PropertiesHelper.GetString(SequenceParam, parms, DefaultSequenceName);
128  if (sequenceName.IndexOf('.') < 0)
129  {
130  string schemaName;
131  string catalogName;
132  parms.TryGetValue(PersistentIdGeneratorParmsNames.Schema, out schemaName);
133  parms.TryGetValue(PersistentIdGeneratorParmsNames.Catalog, out catalogName);
134  sequenceName = dialect.Qualify(catalogName, schemaName, sequenceName);
135  }
136  else
137  {
138  // If already qualified there is not much we can do in a portable manner so we pass it
139  // through and assume the user has set up the name correctly.
140  }
141 
142  return sequenceName;
143  }
