1145  internal static void ExtractColumnOrAliasNames(SqlString select, out List<SqlString> columnsOrAliases, out Dictionary<SqlString, SqlString> aliasToColumn, out Dictionary<SqlString, SqlString> columnToAlias)
1146  {
1147  columnsOrAliases = new List<SqlString>();
1148  aliasToColumn = new Dictionary<SqlString, SqlString>();
1149  columnToAlias = new Dictionary<SqlString, SqlString>();
1150 
1151  var tokens = new QuotedAndParenthesisStringTokenizer(select).GetTokens();
1152  int index = 0;
1153  while (index < tokens.Count)
1154  {
1155  var token = tokens[index];
1156 
1157  int nextTokenIndex = index += 1;
1158 
1159  if (token.StartsWithCaseInsensitive("select"))
1160  continue;
1161 
1162  if (token.StartsWithCaseInsensitive("distinct"))
1163  continue;
1164 
1165  if (token.StartsWithCaseInsensitive(","))
1166  continue;
1167 
1168  if (token.StartsWithCaseInsensitive("from"))
1169  break;
1170 
1171  // handle composite expressions like "2 * 4 as foo"
1172  while ((nextTokenIndex < tokens.Count)
1173  && (tokens[nextTokenIndex].StartsWithCaseInsensitive("as") == false
1174  && tokens[nextTokenIndex].StartsWithCaseInsensitive("from") == false
1175  && tokens[nextTokenIndex].StartsWithCaseInsensitive(",") == false))
1176  {
1177  SqlString nextToken = tokens[nextTokenIndex];
1178  token = token.Append(nextToken);
1179  nextTokenIndex = index += 1;
1180  }
1181 
1182  // if there is no alias, the token and the alias will be the same
1183  SqlString alias = token;
1184 
1185  bool isFunctionCallOrQuotedString = token.IndexOfCaseInsensitive("'") >= 0 || token.IndexOfCaseInsensitive("(") >= 0;
1186 
1187  // this is heuristic guess, if the expression contains ' or (, it is probably
1188  // not appropriate to just slice parts off of it
1189  if (isFunctionCallOrQuotedString == false)
1190  {
1191  // its a simple column reference, so lets set the alias to the
1192  // column name minus the table qualifier if it exists
1193  int dot = token.IndexOfCaseInsensitive(".");
1194  if (dot != -1)
1195  alias = token.Substring(dot + 1);
1196  }
1197 
1198  // notice! we are checking here the existence of "as" "alias", two
1199  // tokens from the current one
1200  if (nextTokenIndex + 1 < tokens.Count)
1201  {
1202  SqlString nextToken = tokens[nextTokenIndex];
1203  if (nextToken.IndexOfCaseInsensitive("as") >= 0)
1204  {
1205  SqlString tokenAfterNext = tokens[nextTokenIndex + 1];
1206  alias = tokenAfterNext;
1207  index += 2; //skip the "as" and the alias
1208  }
1209  }
1210 
1211  columnsOrAliases.Add(alias);
1212  aliasToColumn[alias] = token;
1213  columnToAlias[token] = alias;
1214  }
1215  }

1232  IEnumerator<SqlString> IEnumerable<SqlString>.GetEnumerator()
1233  {
1234  TokenizerState state = TokenizerState.WhiteSpace;
1235  int parenthesisCount = 0;
1236  bool escapeQuote = false;
1237  char quoteType = '\'';
1238  int tokenStart = 0;
1239  int tokenLength = 0;
1240  string originalString = _original.ToString();
1241 
1242  for (int i = 0; i < originalString.Length; i++)
1243  {
1244  char ch = originalString[i];
1245  switch (state)
1246  {
1247  case TokenizerState.WhiteSpace:
1248  if (ch == '\'')
1249  {
1250  state = TokenizerState.Quoted;
1251  quoteType = '\'';
1252  tokenLength += 1;
1253  }
1254  else if (ch == '"')
1255  {
1256  state = TokenizerState.Quoted;
1257  quoteType = '"';
1258  tokenLength += 1;
1259  }
1260  else if (ch == ',')
1261  {
1262  yield return new SqlString(",");
1263  tokenStart += 1;
1264  }
1265  else if (ch == '(' || ch == '[')
1266  {
1267  state = TokenizerState.InParenthesis;
1268  tokenLength += 1;
1269  parenthesisCount = 1;
1270  }
1271  else if (char.IsWhiteSpace(ch) == false)
1272  {
1273  state = TokenizerState.Token;
1274  tokenLength += 1;
1275  }
1276  else
1277  {
1278  tokenStart += 1;
1279  }
1280  break;
1281  case TokenizerState.Quoted:
1282  if (escapeQuote)
1283  {
1284  escapeQuote = false;
1285  tokenLength += 1;
1286  }
1287  // handle escaping of ' by using '' or \'
1288  else if (ch == '\\' || (ch == quoteType && i + 1 < originalString.Length && originalString[i + 1] == quoteType))
1289  {
1290  escapeQuote = true;
1291  tokenLength += 1;
1292  }
1293  else if (ch == quoteType)
1294  {
1295  yield return _original.Substring(tokenStart, tokenLength + 1);
1296  tokenStart += tokenLength + 1;
1297  tokenLength = 0;
1298  state = TokenizerState.WhiteSpace;
1299  }
1300  else
1301  {
1302  tokenLength += 1;
1303  }
1304  break;
1305  case TokenizerState.InParenthesis:
1306  if (ch == ')' || ch == ']')
1307  {
1308  tokenLength += 1;
1309  parenthesisCount -= 1;
1310  if (parenthesisCount == 0)
1311  {
1312  yield return _original.Substring(tokenStart, tokenLength);
1313  tokenStart += tokenLength;
1314  tokenLength = 0;
1315  state = TokenizerState.WhiteSpace;
1316  }
1317  }
1318  else if (ch == '(' || ch == '[')
1319  {
1320  tokenLength += 1;
1321  parenthesisCount += 1;
1322  }
1323  else
1324  {
1325  tokenLength += 1;
1326  }
1327  break;
1328  case TokenizerState.Token:
1329  if (char.IsWhiteSpace(ch))
1330  {
1331  yield return _original.Substring(tokenStart, tokenLength);
1332  tokenStart += tokenLength + 1;
1333  tokenLength = 0;
1334  state = TokenizerState.WhiteSpace;
1335  }
1336  else if (ch == ',') // stop current token, and send the , as well
1337  {
1338  yield return _original.Substring(tokenStart, tokenLength);
1339  yield return new SqlString(",");
1340  tokenStart += tokenLength + 1;
1341  tokenLength = 0;
1342  state = TokenizerState.WhiteSpace;
1343  }
1344  else if (ch == '(' || ch == '[')
1345  {
1346  state = TokenizerState.InParenthesis;
1347  parenthesisCount = 1;
1348  tokenLength += 1;
1349  }
1350  else if (ch == '\'')
1351  {
1352  state = TokenizerState.Quoted;
1353  quoteType = '\'';
1354  tokenLength += 1;
1355  }
1356  else if (ch == '"')
1357  {
1358  state = TokenizerState.Quoted;
1359  quoteType = '"';
1360  tokenLength += 1;
1361  }
1362  else
1363  {
1364  tokenLength += 1;
1365  }
1366  break;
1367  default:
1368  throw new InvalidExpressionException("Could not understand the string " + _original);
1369  }
1370  }
1371  if (tokenLength > 0)
1372  {
1373  yield return _original.Substring(tokenStart, tokenLength);
1374  }
1375  }
