64  public void ProcessJoins(QueryNode query)
65  {
66  FromClause fromClause = query.FromClause;
67 
68  IList<IASTNode> fromElements;
69  if ( DotNode.UseThetaStyleImplicitJoins )
70  {
71  // for regression testing against output from the old parser...
72  // found it easiest to simply reorder the FromElements here into ascending order
73  // in terms of injecting them into the resulting sql ast in orders relative to those
74  // expected by the old parser; this is definitely another of those "only needed
75  // for regression purposes". The SyntheticAndFactory, then, simply injects them as it
76  // encounters them.
77  fromElements = new List<IASTNode>();
78  IList<IASTNode> t = fromClause.GetFromElements();
79 
80  for (int i = t.Count - 1; i >= 0; i--)
81  {
82  fromElements.Add(t[i]);
83  }
84  }
85  else
86  {
87  fromElements = fromClause.GetFromElements();
88  }
89 
90  // Iterate through the alias,JoinSequence pairs and generate SQL token nodes.
91  foreach (FromElement fromElement in fromElements)
92  {
93  JoinSequence join = fromElement.JoinSequence;
94 
95  join.SetSelector(new JoinSequenceSelector(_walker, fromClause, fromElement));
96 
97  AddJoinNodes( query, join, fromElement );
98  }
99  }

101  private void AddJoinNodes(QueryNode query, JoinSequence join, FromElement fromElement)
102  {
103  JoinFragment joinFragment = join.ToJoinFragment(
104  _walker.EnabledFilters,
105  fromElement.UseFromFragment || fromElement.IsDereferencedBySuperclassOrSubclassProperty,
106  fromElement.WithClauseFragment,
107  fromElement.WithClauseJoinAlias
108  );
109 
110  SqlString frag = joinFragment.ToFromFragmentString;
111  SqlString whereFrag = joinFragment.ToWhereFragmentString;
112 
113  // If the from element represents a JOIN_FRAGMENT and it is
114  // a theta-style join, convert its type from JOIN_FRAGMENT
115  // to FROM_FRAGMENT
116  if ( fromElement.Type == HqlSqlWalker.JOIN_FRAGMENT &&
117  ( join.IsThetaStyle || SqlStringHelper.IsNotEmpty( whereFrag ) ) )
118  {
119  fromElement.Type = HqlSqlWalker.FROM_FRAGMENT;
120  fromElement.JoinSequence.SetUseThetaStyle( true ); // this is used during SqlGenerator processing
121  }
122 
123  // If there is a FROM fragment and the FROM element is an explicit, then add the from part.
124  if ( fromElement.UseFromFragment /*&& StringHelper.isNotEmpty( frag )*/ )
125  {
126  SqlString fromFragment = ProcessFromFragment( frag, join ).Trim();
127  if ( log.IsDebugEnabled )
128  {
129  log.Debug( "Using FROM fragment [" + fromFragment + "]" );
130  }
131 
132  ProcessDynamicFilterParameters(fromFragment,fromElement,_walker);
133  }
134 
135  _syntheticAndFactory.AddWhereFragment(
136  joinFragment,
137  whereFrag,
138  query,
139  fromElement,
140  _walker
141  );
142  }

192  public bool IncludeSubclasses(string alias)
193  {
194  // The uber-rule here is that we need to include subclass joins if
195  // the FromElement is in any way dereferenced by a property from
196  // the subclass table; otherwise we end up with column references
197  // qualified by a non-existent table reference in the resulting SQL...
198  bool containsTableAlias = _fromClause.ContainsTableAlias( alias );
199 
200  if ( _fromElement.IsDereferencedBySubclassProperty)
201  {
202  // TODO : or should we return 'containsTableAlias'??
203  log.Info( "forcing inclusion of extra joins [alias=" + alias + ", containsTableAlias=" + containsTableAlias + "]" );
204  return true;
205  }
206  bool shallowQuery = _walker.IsShallowQuery;
207  bool includeSubclasses = _fromElement.IncludeSubclasses;
208  bool subQuery = _fromClause.IsSubQuery;
209  return includeSubclasses && containsTableAlias && !subQuery && !shallowQuery;
210  }
