62  protected override SqlString ApplyLocks(SqlString sql, IDictionary<string, LockMode> lockModes,
63  Dialect.Dialect dialect)
64  {
65  if (lockModes == null || lockModes.Count == 0)
66  {
67  return sql;
68  }
69 
70  // can't cache this stuff either (per-invocation)
71  // we are given a map of user-alias -> lock mode
72  // create a new map of sql-alias -> lock mode
73  var aliasedLockModes = new Dictionary<string, LockMode>();
74  Dictionary<string, string[]> keyColumnNames = dialect.ForUpdateOfColumns ? new Dictionary<string, string[]>() : null;
75 
76  foreach (var entry in lockModes)
77  {
78  string userAlias = entry.Key;
79  string drivingSqlAlias = _sqlAliasByEntityAlias[userAlias];
80  if (drivingSqlAlias == null)
81  {
82  throw new InvalidOperationException("could not locate alias to apply lock mode : " + userAlias);
83  }
84 
85  // at this point we have (drivingSqlAlias) the SQL alias of the driving table
86  // corresponding to the given user alias. However, the driving table is not
87  // (necessarily) the table against which we want to apply locks. Mainly,
88  // the exception case here is joined-subclass hierarchies where we instead
89  // want to apply the lock against the root table (for all other strategies,
90  // it just happens that driving and root are the same).
91  var select = (QueryNode) _queryTranslator.SqlAST;
92  var drivingPersister = (ILockable) select.FromClause.GetFromElement(userAlias).Queryable;
93  string sqlAlias = drivingPersister.GetRootTableAlias(drivingSqlAlias);
94  aliasedLockModes.Add(sqlAlias, entry.Value);
95 
96  if (keyColumnNames != null)
97  {
98  keyColumnNames.Add(sqlAlias, drivingPersister.RootTableIdentifierColumnNames);
99  }
100  }
101 
102  return dialect.ApplyLocksToSql(sql, aliasedLockModes, keyColumnNames);
103  }

145  public override LockMode[] GetLockModes(IDictionary<string, LockMode> lockModes)
146  {
147  if (lockModes == null || lockModes.Count == 0)
148  {
149  return _defaultLockModes;
150  }
151  else
152  {
153  // unfortunately this stuff can't be cached because
154  // it is per-invocation, not constant for the
155  // QueryTranslator instance
156 
157  var lockModeArray = new LockMode[_entityAliases.Length];
158  for (int i = 0; i < _entityAliases.Length; i++)
159  {
160  LockMode lockMode;
161 
162  if (!lockModes.TryGetValue(_entityAliases[i], out lockMode))

164  //NONE, because its the requested lock mode, not the actual!
165  lockMode = LockMode.None;

167 
168  lockModeArray[i] = lockMode;

199  private void Initialize(SelectClause selectClause)
200  {
201  IList<FromElement> fromElementList = selectClause.FromElementsForLoad;
202 
203  _hasScalars = selectClause.IsScalarSelect;
204  _scalarColumnNames = selectClause.ColumnNames;
205  //sqlResultTypes = selectClause.getSqlResultTypes();
206  _queryReturnTypes = selectClause.QueryReturnTypes;
207 
208  _selectNewTransformer = HolderInstantiator.CreateSelectNewTransformer(selectClause.Constructor, selectClause.IsMap, selectClause.IsList);
209  _queryReturnAliases = selectClause.QueryReturnAliases;
210 
211  IList<FromElement> collectionFromElements = selectClause.CollectionFromElements;
212  if (collectionFromElements != null && collectionFromElements.Count != 0)
213  {
214  int length = collectionFromElements.Count;
215  _collectionPersisters = new IQueryableCollection[length];
216  _collectionOwners = new int[length];
217  _collectionSuffixes = new string[length];
218 
219  for (int i = 0; i < length; i++)
220  {
221  FromElement collectionFromElement = collectionFromElements[i];
222  _collectionPersisters[i] = collectionFromElement.QueryableCollection;
223  _collectionOwners[i] = fromElementList.IndexOf(collectionFromElement.Origin);
224  // collectionSuffixes[i] = collectionFromElement.getColumnAliasSuffix();
225  // collectionSuffixes[i] = Integer.toString( i ) + "_";
226  _collectionSuffixes[i] = collectionFromElement.CollectionSuffix;
227  }
228  }
229 
230  int size = fromElementList.Count;
231  _entityPersisters = new IQueryable[size];
232  _entityEagerPropertyFetches = new bool[size];
233  _entityAliases = new String[size];
234  _sqlAliases = new String[size];
235  _sqlAliasSuffixes = new String[size];
236  _includeInSelect = new bool[size];
237  _owners = new int[size];
238  _ownerAssociationTypes = new EntityType[size];
239 
240  for (int i = 0; i < size; i++)
241  {
242  FromElement element = fromElementList[i];
243  _entityPersisters[i] = (IQueryable) element.EntityPersister;
244 
245  if (_entityPersisters[i] == null)
246  {
247  throw new InvalidOperationException("No entity persister for " + element);
248  }
249 
250  _entityEagerPropertyFetches[i] = element.IsAllPropertyFetch;
251  _sqlAliases[i] = element.TableAlias;
252  _entityAliases[i] = element.ClassAlias;
253  _sqlAliasByEntityAlias.Add(_entityAliases[i], _sqlAliases[i]);
254  // TODO should we just collect these like with the collections above?
255  _sqlAliasSuffixes[i] = (size == 1) ? "" : i + "_";
256  // sqlAliasSuffixes[i] = element.getColumnAliasSuffix();
257  _includeInSelect[i] = !element.IsFetch;
258  if (_includeInSelect[i])
259  {
260  _selectLength++;
261  }
262 
263  _owners[i] = -1; //by default
264  if (element.IsFetch)
265  {
266  if (element.IsCollectionJoin || element.QueryableCollection != null)
267  {
268  // This is now handled earlier in this method.
269  }
270  else if (element.DataType.IsEntityType)
271  {
272  var entityType = (EntityType) element.DataType;
273  if (entityType.IsOneToOne)
274  {
275  _owners[i] = fromElementList.IndexOf(element.Origin);
276  }
277  _ownerAssociationTypes[i] = entityType;
278  }
279  }
280  }
281 
282  //NONE, because its the requested lock mode, not the actual!
283  _defaultLockModes = ArrayHelper.Fill(LockMode.None, size);
284  }

322  protected override object GetResultColumnOrRow(object[] row, IResultTransformer resultTransformer, IDataReader rs,
323  ISessionImplementor session)
324  {
325  row = ToResultRow(row);
326  bool hasTransform = HasSelectNew || resultTransformer != null;
327 
328  if (_hasScalars)
329  {
330  string[][] scalarColumns = _scalarColumnNames;
331  int queryCols = _queryReturnTypes.Length;
332 
333  if (!hasTransform && queryCols == 1)
334  {
335  return _queryReturnTypes[0].NullSafeGet(rs, scalarColumns[0], session, null);
336  }
337  else
338  {
339  row = new object[queryCols];
340  for (int i = 0; i < queryCols; i++)
341  {
342  row[i] = _queryReturnTypes[i].NullSafeGet(rs, scalarColumns[i], session, null);
343  }
344  return row;
345  }
346  }
347  else if (!hasTransform)
348  {
349  return row.Length == 1 ? row[0] : row;
350  }
351  else
352  {
353  return row;
354  }

357  private object[] ToResultRow(object[] row)
358  {
359  if (_selectLength == row.Length)
360  {
361  return row;
362  }
363 
364  var result = new object[_selectLength];
365  int j = 0;
366  for (int i = 0; i < row.Length; i++)
367  {
368  if (_includeInSelect[i])
369  {
370  result[j++] = row[i];
371  }
372  }
373  return result;

394  internal IEnumerable GetEnumerable(QueryParameters queryParameters, IEventSource session)
395  {
396  CheckQuery(queryParameters);
397  bool statsEnabled = session.Factory.Statistics.IsStatisticsEnabled;
398 
399  var stopWath = new Stopwatch();
400  if (statsEnabled)
401  {
402  stopWath.Start();
403  }
404 
405  IDbCommand cmd = PrepareQueryCommand(queryParameters, false, session);
406 
407  // This IDataReader is disposed of in EnumerableImpl.Dispose
408  IDataReader rs = GetResultSet(cmd, queryParameters.HasAutoDiscoverScalarTypes, false, queryParameters.RowSelection, session);
409 
410  HolderInstantiator hi =
411  HolderInstantiator.GetHolderInstantiator(_selectNewTransformer, queryParameters.ResultTransformer, _queryReturnAliases);
412 
413  IEnumerable result =
414  new EnumerableImpl(rs, cmd, session, queryParameters.IsReadOnly(session), _queryTranslator.ReturnTypes, _queryTranslator.GetColumnNames(), queryParameters.RowSelection, hi);
415 
416  if (statsEnabled)
417  {
418  stopWath.Stop();
419  session.Factory.StatisticsImplementor.QueryExecuted("HQL: " + _queryTranslator.QueryString, 0, stopWath.Elapsed);
420  // NH: Different behavior (H3.2 use QueryLoader in AST parser) we need statistic for orginal query too.
421  // probably we have a bug some where else for statistic RowCount
422  session.Factory.StatisticsImplementor.QueryExecuted(QueryIdentifier, 0, stopWath.Elapsed);
423  }
424  return result;
