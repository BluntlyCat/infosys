123  sealed override protected void Convert(TextWriter writer, LoggingEvent loggingEvent)
124  {
125  string name = GetFullyQualifiedName(loggingEvent);
126  if (m_precision <= 0 || name == null || name.Length < 2)
127  {
128  writer.Write(name);
129  }
130  else
131  {
132  int len = name.Length;
133  string trailingDot = string.Empty;
134  if (name.EndsWith(DOT))
135  {
136  trailingDot = DOT;
137  name = name.Substring(0, len - 1);
138  len--;
139  }
140 
141  int end = name.LastIndexOf(DOT);
142  for(int i = 1; end > 0 && i < m_precision; i++)
143  {
144  end = name.LastIndexOf('.', end - 1);
145  }
146  if (end == -1)
147  {
148  writer.Write(name + trailingDot);
149  }
150  else
151  {
152  writer.Write(name.Substring(end + 1, len - end - 1) + trailingDot);
153  }
154  }
