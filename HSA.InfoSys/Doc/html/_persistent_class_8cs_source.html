847  private Property GetRecursiveProperty(string propertyPath, IEnumerable<Property> iter)
848  {
849  Property property = null;
850 
851  StringTokenizer st = new StringTokenizer(propertyPath, ".", false);
852  try
853  {
854  foreach (string element in st)
855  {
856  if (property == null)
857  {
858  // we are processing the root of the propertyPath, so we have the following
859  // considerations:
860  // 1) specifically account for identifier properties
861  // 2) specifically account for embedded composite-identifiers
862  // 3) perform a normal property lookup
863  Property identifierProperty = IdentifierProperty;
864  if (identifierProperty != null && identifierProperty.Name.Equals(element))
865  {
866  // we have a mapped identifier property and the root of
867  // the incoming property path matched that identifier
868  // property
869  property = identifierProperty;
870  }
871  else if (identifierProperty == null)
872  {
873  var component = Identifier as Component;
874  if (component != null)
875  {
876  // we have an embedded composite identifier
877  try
878  {
879  identifierProperty = GetProperty(element, component.PropertyIterator);
880  if (identifierProperty != null)
881  {
882  // the root of the incoming property path matched one
883  // of the embedded composite identifier properties
884  property = identifierProperty;
885  }
886  }
887  catch (MappingException)
888  {
889  // ignore it...
890  }
891  }
892  }
893 
894  if (property == null)
895  {
896  property = GetProperty(element, iter);
897  }
898  }
899  else
900  {
901  //flat recursive algorithm
902  property = ((Component)property.Value).GetProperty(element);
903  }
904  }
905  }
906  catch (MappingException)
907  {
908  throw new MappingException("property [" + propertyPath + "] not found on entity [" + EntityName + "]");
909  }
910 
911  return property;
912  }
