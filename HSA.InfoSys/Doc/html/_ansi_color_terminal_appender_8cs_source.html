294  override protected void Append(log4net.Core.LoggingEvent loggingEvent)
295  {
296  string loggingMessage = RenderLoggingEvent(loggingEvent);
297 
298  // see if there is a specified lookup.
299  LevelColors levelColors = m_levelMapping.Lookup(loggingEvent.Level) as LevelColors;
300  if (levelColors != null)
301  {
302  // Prepend the Ansi Color code
303  loggingMessage = levelColors.CombinedColor + loggingMessage;
304  }
305 
306  // on most terminals there are weird effects if we don't clear the background color
307  // before the new line. This checks to see if it ends with a newline, and if
308  // so, inserts the clear codes before the newline, otherwise the clear codes
309  // are inserted afterwards.
310  if (loggingMessage.Length > 1)
311  {
312  if (loggingMessage.EndsWith("\r\n") || loggingMessage.EndsWith("\n\r"))
313  {
314  loggingMessage = loggingMessage.Insert(loggingMessage.Length - 2, PostEventCodes);
315  }
316  else if (loggingMessage.EndsWith("\n") || loggingMessage.EndsWith("\r"))
317  {
318  loggingMessage = loggingMessage.Insert(loggingMessage.Length - 1, PostEventCodes);
319  }
320  else
321  {
322  loggingMessage = loggingMessage + PostEventCodes;
323  }
324  }
325  else
326  {
327  if (loggingMessage[0] == '\n' || loggingMessage[0] == '\r')
328  {
329  loggingMessage = PostEventCodes + loggingMessage;
330  }
331  else
332  {
333  loggingMessage = loggingMessage + PostEventCodes;
334  }
335  }
336 
337 #if NETCF_1_0
338  // Write to the output stream
339  Console.Write(loggingMessage);
340 #else
341  if (m_writeToErrorStream)
342  {
343  // Write to the error stream
344  Console.Error.Write(loggingMessage);
345  }
346  else
347  {
348  // Write to the output stream
349  Console.Write(loggingMessage);
350  }
351 #endif
352 
353  }

504  public override void ActivateOptions()
505  {
506  base.ActivateOptions();
507 
508  StringBuilder buf = new StringBuilder();
509 
510  // Reset any existing codes
511  buf.Append("\x1b[0;");
512 
513  // set the foreground color
514  buf.Append(30 + (int)m_foreColor);
515  buf.Append(';');
516 
517  // set the background color
518  buf.Append(40 + (int)m_backColor);
519 
520  // set the attributes
521  if ((m_attributes & AnsiAttributes.Bright) > 0)
522  {
523  buf.Append(";1");
524  }
525  if ((m_attributes & AnsiAttributes.Dim) > 0)
526  {
527  buf.Append(";2");
528  }
529  if ((m_attributes & AnsiAttributes.Underscore) > 0)
530  {
531  buf.Append(";4");
532  }
533  if ((m_attributes & AnsiAttributes.Blink) > 0)
534  {
535  buf.Append(";5");
536  }
537  if ((m_attributes & AnsiAttributes.Reverse) > 0)
538  {
539  buf.Append(";7");
540  }
541  if ((m_attributes & AnsiAttributes.Hidden) > 0)
542  {
543  buf.Append(";8");
544  }
545  if ((m_attributes & AnsiAttributes.Strikethrough) > 0)
546  {
547  buf.Append(";9");
548  }
549 
550  buf.Append('m');
551 
552  m_combinedColor = buf.ToString();
553  }
