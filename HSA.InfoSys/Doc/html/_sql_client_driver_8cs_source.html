119  protected static void SetDefaultParameterSize(IDbDataParameter dbParam, SqlType sqlType)
120  {
121  switch (dbParam.DbType)
122  {
123  case DbType.AnsiString:
124  case DbType.AnsiStringFixedLength:
125  dbParam.Size = MaxSizeForLengthLimitedAnsiString;
126  break;
127  case DbType.Binary:
128  dbParam.Size = IsBlob(dbParam, sqlType) ? MaxSizeForBlob : MaxSizeForLengthLimitedBinary;
129  break;
130  case DbType.Decimal:
131  dbParam.Precision = MaxPrecision;
132  dbParam.Scale = MaxScale;
133  break;
134  case DbType.String:
135  case DbType.StringFixedLength:
136  dbParam.Size = IsText(dbParam, sqlType) ? MaxSizeForClob : MaxSizeForLengthLimitedString;
137  break;
138  case DbType.DateTime2:
139  dbParam.Size = MaxDateTime2;
140  break;
141  case DbType.DateTimeOffset:
