146  protected byte[] ReadBytes(int length)
147  {
148  // Note that this also prevents allocating non-relevant lengths, such as if length is greater than _data.Count but less than int.MaxValue.
149  // For the nerds, the condition translates to: if (length > data.Count && length < int.MaxValue)
150  // Which probably would cause all sorts of exception, most notably OutOfMemoryException.
151  if (length > this._data.Count)
152  throw new ArgumentOutOfRangeException("length");
153 
154  var result = new byte[length];
155  this._data.CopyTo(this._readerIndex, result, 0, length);
156  this._readerIndex += length;
157  return result;
158  }
