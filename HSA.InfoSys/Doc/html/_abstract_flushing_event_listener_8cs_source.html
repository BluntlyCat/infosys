39  protected virtual void FlushEverythingToExecutions(FlushEvent @event)
40  {
41  log.Debug("flushing session");
42 
43  IEventSource session = @event.Session;
44  IPersistenceContext persistenceContext = session.PersistenceContext;
45 
46  session.Interceptor.PreFlush((ICollection) persistenceContext.EntitiesByKey.Values);
47 
48  PrepareEntityFlushes(session);
49  // we could move this inside if we wanted to
50  // tolerate collection initializations during
51  // collection dirty checking:
52  PrepareCollectionFlushes(session);
53  // now, any collections that are initialized
54  // inside this block do not get updated - they
55  // are ignored until the next flush
56 
57  persistenceContext.Flushing = true;
58  try
59  {
60  FlushEntities(@event);
61  FlushCollections(session);
62  }
63  finally
64  {
65  persistenceContext.Flushing = false;
66  }
67 
68  //some statistics
69  if (log.IsDebugEnabled)
70  {
71  StringBuilder sb = new StringBuilder(100);
72 
73  sb.Append("Flushed: ").Append(session.ActionQueue.InsertionsCount).Append(" insertions, ").Append(
74  session.ActionQueue.UpdatesCount).Append(" updates, ").Append(session.ActionQueue.DeletionsCount).Append(
75  " deletions to ").Append(persistenceContext.EntityEntries.Count).Append(" objects");
76  log.Debug(sb.ToString());

87  protected virtual void FlushCollections(IEventSource session)
88  {
89  log.Debug("Processing unreferenced collections");
90 
91  ICollection list = IdentityMap.Entries(session.PersistenceContext.CollectionEntries);
92  foreach (DictionaryEntry me in list)
93  {
94  CollectionEntry ce = (CollectionEntry) me.Value;
95  if (!ce.IsReached && !ce.IsIgnore)
96  {
97  Collections.ProcessUnreachableCollection((IPersistentCollection) me.Key, session);
98  }
99  }
100 
101  // Schedule updates to collections:
102 
103  log.Debug("Scheduling collection removes/(re)creates/updates");
104 
105  list = IdentityMap.Entries(session.PersistenceContext.CollectionEntries);
106  ActionQueue actionQueue = session.ActionQueue;
107  foreach (DictionaryEntry me in list)
108  {
109  IPersistentCollection coll = (IPersistentCollection) me.Key;
110  CollectionEntry ce = (CollectionEntry) me.Value;
111 
112  if (ce.IsDorecreate)
113  {
114  session.Interceptor.OnCollectionRecreate(coll, ce.CurrentKey);
115  actionQueue.AddAction(new CollectionRecreateAction(coll, ce.CurrentPersister, ce.CurrentKey, session));
116  }
117  if (ce.IsDoremove)
118  {
119  session.Interceptor.OnCollectionRemove(coll, ce.LoadedKey);
120  actionQueue.AddAction(
121  new CollectionRemoveAction(coll, ce.LoadedPersister, ce.LoadedKey, ce.IsSnapshotEmpty(coll), session));
122  }
123  if (ce.IsDoupdate)
124  {
125  session.Interceptor.OnCollectionUpdate(coll, ce.LoadedKey);
126  actionQueue.AddAction(
127  new CollectionUpdateAction(coll, ce.LoadedPersister, ce.LoadedKey, ce.IsSnapshotEmpty(coll), session));
128  }
129  }
130  actionQueue.SortCollectionActions();
131  }

136  protected virtual void FlushEntities(FlushEvent @event)
137  {
138  log.Debug("Flushing entities and processing referenced collections");
139 
140  // Among other things, updateReachables() will recursively load all
141  // collections that are moving roles. This might cause entities to
142  // be loaded.
143 
144  // So this needs to be safe from concurrent modification problems.
145  // It is safe because of how IdentityMap implements entrySet()
146  IEventSource source = @event.Session;
147 
148  ICollection list = IdentityMap.ConcurrentEntries(source.PersistenceContext.EntityEntries);
149  foreach (DictionaryEntry me in list)
150  {
151  // Update the status of the object and if necessary, schedule an update
152  EntityEntry entry = (EntityEntry) me.Value;
153  Status status = entry.Status;
154 
155  if (status != Status.Loading && status != Status.Gone)
156  {
157  FlushEntityEvent entityEvent = new FlushEntityEvent(source, me.Key, entry);
158  IFlushEntityEventListener[] listeners = source.Listeners.FlushEntityEventListeners;
159  foreach (IFlushEntityEventListener listener in listeners)
160  {
161  listener.OnFlushEntity(entityEvent);
162  }
163  }
164  }
165  source.ActionQueue.SortActions();
166  }

186  {
187  log.Debug("processing flush-time cascades");
188 
189  ICollection list = IdentityMap.ConcurrentEntries(session.PersistenceContext.EntityEntries);
190  //safe from concurrent modification because of how entryList() is implemented on IdentityMap
191  foreach (DictionaryEntry me in list)
192  {
193  EntityEntry entry = (EntityEntry) me.Value;
194  Status status = entry.Status;
195  if (status == Status.Loaded || status == Status.Saving || status == Status.ReadOnly)
196  {
197  CascadeOnFlush(session, entry.Persister, me.Key, Anything);

229  if (log.IsDebugEnabled)
230  {
231  log.Debug("executing flush");
232  }
233 
234  try
235  {
236  session.ConnectionManager.FlushBeginning();
237  // we need to lock the collection caches before
238  // executing entity inserts/updates in order to
239  // account for bidi associations
240  session.ActionQueue.PrepareActions();
241  session.ActionQueue.ExecuteActions();
242  }
243  catch (HibernateException he)
244  {
245  if (log.IsErrorEnabled)
246  {
247  log.Error("Could not synchronize database state with session", he);

262  protected virtual void PostFlush(ISessionImplementor session)
263  {
264  if (log.IsDebugEnabled)
265  {
266  log.Debug("post flush");
267  }
268 
269  IPersistenceContext persistenceContext = session.PersistenceContext;
270  persistenceContext.CollectionsByKey.Clear();
271  persistenceContext.BatchFetchQueue.ClearSubselects();
272  //the database has changed now, so the subselect results need to be invalidated
273 
274  // NH Different implementation: In NET an iterator is immutable;
275  // we need something to hold the persistent collection to remove, and it must be less intrusive as possible
276  IDictionary cEntries = persistenceContext.CollectionEntries;
277  List<IPersistentCollection> keysToRemove = new List<IPersistentCollection>(cEntries.Count);
278  foreach (DictionaryEntry me in cEntries)
279  {
280  CollectionEntry collectionEntry = (CollectionEntry) me.Value;
281  IPersistentCollection persistentCollection = (IPersistentCollection) me.Key;
282  collectionEntry.PostFlush(persistentCollection);
283  if (collectionEntry.LoadedPersister == null)
284  {
285  keysToRemove.Add(persistentCollection);
286  }
287  else
288  {
289  //otherwise recreate the mapping between the collection and its key
290  CollectionKey collectionKey =
291  new CollectionKey(collectionEntry.LoadedPersister, collectionEntry.LoadedKey, session.EntityMode);
292  persistenceContext.CollectionsByKey[collectionKey] = persistentCollection;
293  }
294  }
295  foreach (IPersistentCollection key in keysToRemove)
296  {
297  persistenceContext.CollectionEntries.Remove(key);
298  }
299  session.Interceptor.PostFlush((ICollection) persistenceContext.EntitiesByKey.Values);
300  }
301  }
